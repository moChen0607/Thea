65 std::string
66 Application::programPath()
67 {
68  char path[8192];
69 
70 #ifdef THEA_WINDOWS
71  {
72  GetModuleFileNameA(nullptr, path, sizeof(path));
73  }
74 #elif defined(THEA_MAC)
75  {
76  char unresolved_path[8192];
77  uint32_t size = (uint32_t)sizeof(unresolved_path);
78  if (_NSGetExecutablePath(unresolved_path, &size) != 0)
79  throw Error("Application: Executable path is longer than buffer size");
80 
81  if (!realpath(unresolved_path, path))
82  throw Error("Application: Could not resolve executable path");
83  }
84 #else
85  {
86  int ret = readlink("/proc/self/exe", path, sizeof(path));
87 
88  // In case of an error, leave the handling up to the caller
89  if (ret == -1)
90  return "";
91 
92  debugAssertM((int)sizeof(path) > ret, "Application: String too short to store current program path");
93  // Ensure proper nullptr termination
94  path[ret] = 0;
95  }
96 #endif
97 
98  return path;
99 }

122 std::string
123 Application::getPluginPath(std::string const & plugin_name, Array<std::string> const * plugin_dirs)
124 {
125  std::string app_dir = FilePath::parent(FileSystem::resolve(Application::programPath()));
126  std::string app_plugin_dir1 = FilePath::concat(FilePath::parent(app_dir), "lib"); // in parent
127  std::string app_plugin_dir2 = FilePath::concat(FilePath::parent(app_dir), "../lib"); // in grandparent
128 
129  Array<std::string> search_dirs;
130  search_dirs.push_back(app_dir); // the application directory has highest priority
131  search_dirs.push_back(app_plugin_dir1); // the parent directory is next
132  search_dirs.push_back(app_plugin_dir2); // the grandparent directory is next
133 
134  if (plugin_dirs) // user-supplied search directories are next
135  search_dirs.insert(search_dirs.end(), plugin_dirs->begin(), plugin_dirs->end());
136 
137  search_dirs.push_back("/usr/lib"); // system directories follow
138  search_dirs.push_back("/usr/local/lib");
139  search_dirs.push_back("/sw/lib"); // -- Fink
140  search_dirs.push_back("/opt/local/lib"); // -- Homebrew
141 
142  std::string dir = FilePath::parent(plugin_name);
143  std::string basename = FilePath::completeBaseName(plugin_name);
144  std::string ext = FilePath::extension(plugin_name);
145 
146 #ifndef THEA_WINDOWS
147  if (!beginsWith(basename, "lib"))
148  basename = "lib" + basename;
149 #endif
150 
151 #ifdef THEA_WINDOWS
152  if (ext != "dll")
153  {
154  if (!ext.empty()) basename = basename + "." + ext; // e.g. we got something like libGL.3
155  ext = "dll";
156  }
157 #elif THEA_MAC
158  if (ext != "dylib")
159  {
160  if (!ext.empty()) basename = basename + "." + ext;
161  ext = "dylib";
162  }
163 #else
164  if (ext != "so")
165  {
166  if (!ext.empty()) basename = basename + "." + ext;
167  ext = "so";
168  }
169 #endif
170 
171  std::string debug_filename = basename + "d." + ext;
172  std::string release_filename = basename + "." + ext;
173 
174  for (size_t i = 0; i < search_dirs.size(); ++i)
175  {
176  if (search_dirs[i].empty()) // if you need to specify the current working directory, use "."
177  continue;
178 
179  std::string debug_path = FilePath::concat(search_dirs[i], debug_filename);
180  std::string release_path = FilePath::concat(search_dirs[i], release_filename);
181 
182 #ifdef THEA_DEBUG_BUILD
183 
184  // Debug preferred over release
185  if (FileSystem::fileExists(debug_path))
186  return FileSystem::resolve(debug_path);
187  else if (FileSystem::fileExists(release_path))
188  return FileSystem::resolve(release_path);
189 
190 #else // THEA_DEBUG_BUILD
191 
192  // Release preferred over debug
193  if (FileSystem::fileExists(release_path))
194  return FileSystem::resolve(release_path);
195  else if (FileSystem::fileExists(debug_path))
196  return FileSystem::resolve(debug_path);
197 
198 #endif // THEA_DEBUG_BUILD
199  }
200 
201  return std::string();
202 }
