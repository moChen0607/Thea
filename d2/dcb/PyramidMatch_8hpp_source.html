95  template <typename T> void construct(T const * base_data, int n, Real * means, bool smooth_base_level)
96  {
97  alwaysAssertM(Math::isPowerOf2((unsigned)n), "Pyramid1D: Base size must be a power of 2");
98 
99  num_levels = Math::floorLog2((uint32)n) + 1;
100  levels.resize(num_levels);
101 
102  if (smooth_base_level)
103  {
104  levels[0].resize((size_t)n, 0);
105  Real * dst = &levels[0][0];
106 
107  for (int i = 0; i < n; ++i)
108  PyramidInternal::smoothIncrement1D(dst, n, i, means[i], base_data[i]);
109  }
110  else
111  {
112  levels[0].reserve((size_t)n);
113  levels[0].insert(levels[0].end(), base_data, base_data + n);
114  }
115 
116  createPyramid(means);
117  }
