214 template <typename T>
215 Matrix<3, 3, T>
216 rotationAxisAngle(Vector<3, T> const & axis, Real radians)
217 {
218  Vector<3, T> uaxis = axis.normalized();
219 
220  T cos_val = static_cast<T>(std::cos(radians));
221  T sin_val = static_cast<T>(std::sin(radians));
222  T one_minus_cos = 1 - cos_val;
223  T x2 = uaxis[0] * uaxis[0];
224  T y2 = uaxis[1] * uaxis[1];
225  T z2 = uaxis[2] * uaxis[2];
226  T xym = uaxis[0] * uaxis[1] * one_minus_cos;
227  T xzm = uaxis[0] * uaxis[2] * one_minus_cos;
228  T yzm = uaxis[1] * uaxis[2] * one_minus_cos;
229  T xsin = uaxis[0] * sin_val;
230  T ysin = uaxis[1] * sin_val;
231  T zsin = uaxis[2] * sin_val;
232 
233  Matrix<3, 3, T> m;
234  m(0, 0) = x2 * one_minus_cos + cos_val;
235  m(0, 1) = xym - zsin;
236  m(0, 2) = xzm + ysin;
237 
238  m(1, 0) = xym + zsin;
239  m(1, 1) = y2 * one_minus_cos + cos_val;
240  m(1, 2) = yzm - xsin;
241 
242  m(2, 0) = xzm - ysin;
243  m(2, 1) = yzm + xsin;
244  m(2, 2) = z2 * one_minus_cos + cos_val;
245 
246  return m;
247 }

394 template <typename T>
395 Matrix<3, 3, T>
396 rotationQuat(T const & x, T const & y, T const & z, T const & w)
397 {
398  // From G3D::Matrix3:
399  // Implementation from Watt and Watt, pg 362
400  // See also http://www.flipcode.com/documents/matrfaq.html#Q54
401 
402  Vector<4, T> q(x, y, z, w);
403  q.normalize();
404 
405  T xx = 2 * q.x() * q.x();
406  T xy = 2 * q.x() * q.y();
407  T xz = 2 * q.x() * q.z();
408  T xw = 2 * q.x() * q.w();
409 
410  T yy = 2 * q.y() * q.y();
411  T yz = 2 * q.y() * q.z();
412  T yw = 2 * q.y() * q.w();
413 
414  T zz = 2 * q.z() * q.z();
415  T zw = 2 * q.z() * q.w();
416 
417  Matrix<3, 3, T> ret;
418  ret << 1 - yy - zz, xy - zw, xz + yw,
419  xy + zw, 1 - xx - zz, yz - xw,
420  xz - yw, yz + xw, 1 - xx - yy;
421 
422  return ret;
423 }

433 template <typename T>
434 Matrix<3, 3, T>
435 rotationArc(Vector<3, T> const & start_dir, Vector<3, T> const & end_dir, bool normalize_dirs = true)
436 {
437  // From John Ratcliff's Code Suppository.
438  //
439  // Reference, from Stan Melax in Game Gems I
440  // Quaternion q;
441  // vector3 c = CrossProduct(v0,v1);
442  // REAL d = DotProduct(v0,v1);
443  // REAL s = (REAL)sqrt((1+d)*2);
444  // q.x = c.x / s;
445  // q.y = c.y / s;
446  // q.z = c.z / s;
447  // q.w = s / 2.0f;
448  // return q;
449 
450  Vector<3, T> u = start_dir, v = end_dir;
451  if (normalize_dirs)
452  {
453  u.normalize();
454  v.normalize();
455  }
456 
457  T d = u.dot(v);
458  if (d < -0.9999)
459  {
460  Vector<3, T> perp;
461  switch (maxAbsAxis(u))
462  {
463  case 0: perp = Vector<3, T>(u.y(), -u.x(), 0); break;
464  case 1: perp = Vector<3, T>(u.y(), -u.x(), 0); break;
465  default /* 2 */: perp = Vector<3, T>(u.z(), 0, -u.x());
466  }
467 
468  return rotationAxisAngle(perp, (Real)Math::pi());
469  }
470 
471  T s = std::sqrt((1 + d) * 2);
472  T recip = 1 / s;
473  Vector<3, T> rcc = recip * u.cross(v);
474 
475  return rotationQuat(rcc.x(), rcc.y(), rcc.z(), static_cast<T>(0.5 * s));
476 }

483 template <typename T>
484 Matrix<4, 4, T>
485 orthogonalProjection(T const & left, T const & right, T const & bottom, T const & top, T const & nearval, T const & farval,
486  bool y_increases_upwards = true)
487 {
488  // Adapted from Mesa.
489  // Note that Microsoft (http://msdn.microsoft.com/library/default.asp?url=/library/en-us/opengl/glfunc03_8qnj.asp) and
490  // Linux (http://www.xfree86.org/current/glOrtho.3.html) have different matrices shown in their documentation.
491 
492  T x = 2 / (right - left);
493  T y = 2 / (top - bottom);
494  T z = -2 / (farval - nearval);
495  T tx = -(right + left) / (right - left);
496  T ty = -(top + bottom) / (top - bottom);
497  T tz = -(farval + nearval) / (farval - nearval);
498 
499  if (!y_increases_upwards)
500  {
501  y *= -1;
502  ty *= -1;
503  }
504 
505  return (Matrix<4, 4, T>() << x, 0, 0, tx,
506  0, y, 0, ty,
507  0, 0, z, tz,
508  0, 0, 0, 1).finished();
509 }

516 template <typename T>
517 Matrix<4, 4, T>
518 perspectiveProjection(T const & left, T const & right, T const & bottom, T const & top, T const & nearval, T const & farval,
519  bool y_increases_upwards = true)
520 {
521  T x = (2 * nearval) / (right - left);
522  T y = (2 * nearval) / (top - bottom);
523  T a = (right + left) / (right - left);
524  T b = (top + bottom) / (top - bottom);
525 
526  T c, d;
527  if (Math::isInfinite(farval))
528  {
529  // Infinite view frustum
530  c = -1;
531  d = -2 * nearval;
532  }
533  else
534  {
535  c = -(farval + nearval) / (farval - nearval);
536  d = -(2 * farval * nearval) / (farval - nearval);
537  }
538 
539  if (!y_increases_upwards)
540  {
541  y *= -1;
542  b *= -1;
543  }
544 
545  return (Matrix<4, 4, T>() << x, 0, a, 0,
546  0, y, b, 0,
547  0, 0, c, d,
548  0, 0, -1, 0).finished();
549 }

561 template <typename T>
562 int
563 eigenSolve(Matrix<2, 2, T> const & m, T * eigenvalues, Vector<2, T> * eigenvectors, T const & tol = -1)
564 {
565  T a = m(0, 0), b = m(0, 1);
566  T c = m(1, 0), d = m(1, 1);
567 
568  T trace = a + d;
569  T det = a * d - b * c;
570 
571  T disc = trace * trace / 4 - det;
572  if (disc < 0)
573  return 0;
574 
575  T s = std::sqrt(disc);
576  eigenvalues[0] = trace / 2 - s;
577  eigenvalues[1] = trace / 2 + s;
578 
579  if (!Math::fuzzyEq(c, static_cast<T>(0), (tol >= 0 ? tol : Math::eps(c, static_cast<T>(0)))))
580  {
581  eigenvectors[0][0] = eigenvalues[0] - d;
582  eigenvectors[0][1] = c;
583 
584  eigenvectors[1][0] = eigenvalues[1] - d;
585  eigenvectors[1][1] = c;
586  }
587  else if (!Math::fuzzyEq(b, static_cast<T>(0), (tol >= 0 ? tol : Math::eps(b, static_cast<T>(0)))))
588  {
589  eigenvectors[0][0] = b;
590  eigenvectors[0][1] = eigenvalues[0] - a;
591 
592  eigenvectors[1][0] = b;
593  eigenvectors[1][1] = eigenvalues[1] - a;
594  }
595  else
596  {
597  eigenvectors[0][0] = 1;
598  eigenvectors[0][1] = 0;
599 
600  eigenvectors[1][0] = 0;
601  eigenvectors[1][1] = 1;
602  }
603 
604  return 2;
605 }

630 template <typename Derived>
631 std::string
632 toString(Eigen::DenseBase<Derived> const & m, intx max_rows = 4, intx max_cols = 4)
633 {
634  intx first_rows = m.rows(), last_rows = 0;
635  intx first_cols = m.cols(), last_cols = 0;
636  if (m.rows() > max_rows)
637  {
638  first_rows = max_rows / 2 + max_rows % 2;
639  last_rows = max_rows / 2;
640  }
641  if (m.cols() > max_cols)
642  {
643  first_cols = max_cols / 2 + max_cols % 2;
644  last_cols = max_cols / 2;
645  }
646 
647  std::ostringstream oss;
648  oss << '[';
649 
650  for (int i = 0; i < 2; ++i)
651  {
652  intx row_begin = (i == 0 ? 0 : m.rows() - last_rows);
653  intx row_end = (i == 0 ? first_rows : m.rows());
654 
655  for (intx r = row_begin; r < row_end; ++r)
656  {
657  for (int j = 0; j < 2; ++j)
658  {
659  intx col_begin = (j == 0 ? 0 : m.cols() - last_cols);
660  intx col_end = (j == 0 ? first_cols : m.cols());
661 
662  for (intx c = col_begin; c < col_end; ++c)
663  {
664  if (c > col_begin) oss << ' ';
665  oss << m(r, c);
666  }
667 
668  if (j == 0 && (first_cols + last_cols) < m.cols())
669  oss << " ... ";
670  }
671 
672  if (r != m.rows() - 1) oss << "; ";
673  }
674 
675  if (i == 0)
676  {
677  if ((first_rows + last_rows) < m.rows()) oss << " ... ";
678  if (last_rows > 0) oss << "; ";
679  }
680  }
681 
682  oss << ']';
683  return oss.str();
684 }
