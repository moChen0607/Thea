117  Triangle const * tarray;
118  intx tcount;
119  if (num_external_tris > 0)
120  {
121  tarray = external_tris;
122  tcount = num_external_tris;
123  }
124  else
125  {
126  if (tris.isEmpty())
127  {
128  THEA_CONSOLE << "MeshSampler: Selected 0 sample(s) uniformly by area";
129  return 0;
130  }
131  else
132  {
133  tarray = &tris.getTriangles()[0];
134  tcount = tris.numTriangles();
135  }
136  }
137 
138  if (verbose)
139  THEA_CONSOLE << "MeshSampler: Obtained triangle list";
140 
141  if (tcount <= 0)
142  {
143  THEA_CONSOLE << "MeshSampler: Selected 0 sample(s) uniformly by area";
144  return 0;
145  }
146 
147  double total_area = 0;
148  for (intx i = 0; i < tcount; ++i)
149  total_area += tarray[i].getArea();
150 
151  if (total_area <= 0)
152  {
153  THEA_CONSOLE << "MeshSampler: Selected 0 sample(s) uniformly by area";
154  return 0;
155  }
156 
157  if (verbose)
158  {
159  THEA_CONSOLE << "MeshSampler: Computed triangle areas";
160  std::cout << "MeshSampler: Selecting samples: " << std::flush;
161  }
162 
163  int prev_percent = 0;
164  if (count_mode == CountMode::EXACT)
165  {
166  Array<double> cum_areas((size_t)tcount);
167  cum_areas[0] = tarray[0].getArea();
168  for (size_t i = 1; i < cum_areas.size(); ++i)
169  cum_areas[i] = cum_areas[i - 1] + tarray[i].getArea();
170 
171  positions.resize((size_t)desired_num_samples);
172  if (face_normals) face_normals->resize(positions.size());
173  if (triangles) triangles->resize(positions.size());
174 
175  // Sample N triangles with replacement, weighted by area
176  double const * first = &cum_areas[0];
177  double const * last = first + cum_areas.size();
178  for (size_t i = 0; i < positions.size(); ++i)
179  {
180  double t = Random::common().uniform01() * total_area;
181  intx tri_index = std::min((intx)(std::lower_bound(first, last, t) - first), tcount - 1);
182 
183  positions[i] = tarray[tri_index].randomPoint();
184  if (face_normals) (*face_normals)[i] = tarray[tri_index].getNormal();
185  if (triangles) (*triangles)[i] = &tarray[tri_index];
186 
187  if (verbose)
188  {
189  int curr_percent = (int)std::floor(100 * (i / (float)desired_num_samples));
190  if (curr_percent >= prev_percent + 2)
191  {
192  for (prev_percent += 2; prev_percent <= curr_percent; prev_percent += 2)
193  {
194  if (prev_percent % 10 == 0)
195  std::cout << prev_percent << '%' << std::flush;
196  else
197  std::cout << '.' << std::flush;
198  }
199 
200  prev_percent = curr_percent;
201  }
202  }
203  }
204  }
205  else
206  {
207  int prev_percent = 0;
208  for (intx i = 0; i < tcount; ++i)
209  {
210  double frac_samples = (desired_num_samples * tarray[i].getArea()) / total_area;
211  while (frac_samples > 0)
212  {
213  // Last (possible) sample?
214  if (frac_samples < 1 && Random::common().uniform01() > frac_samples)
215  break;
216 
217  positions.push_back(tarray[i].randomPoint());
218  if (face_normals) face_normals->push_back(tarray[i].getNormal());
219  if (triangles) triangles->push_back(&tarray[i]);
220 
221  frac_samples -= 1.0;
222 
223  if (verbose)
224  {
225  int curr_percent = (int)std::floor(100 * (positions.size() / (float)desired_num_samples));
226  if (curr_percent >= prev_percent + 2 && curr_percent < 100)
227  {
228  for (prev_percent += 2; prev_percent <= curr_percent; prev_percent += 2)
229  {
230  if (prev_percent % 10 == 0)
231  std::cout << prev_percent << '%' << std::flush;
232  else
233  std::cout << '.' << std::flush;
234  }
235 
236  prev_percent = curr_percent;
237  }
238  }
239  }
240  }
241  }
242 
243  if (verbose)
244  {
245  std::cout << "done" << std::endl;
246  THEA_CONSOLE << "MeshSampler: Selected " << positions.size() << " sample(s) uniformly by area";
247  }
248 
249  return (intx)positions.size();
250  }
