192  template <typename FromT, typename ToProximityQueryStructureT, typename FromWeightFuncT>
193  AffineTransformT align(intx from_num_pts, FromT const * from, FromWeightFuncT * from_weight_func,
194  PlaneT const * from_symmetry_plane, ToProximityQueryStructureT const & to,
195  PlaneT const * to_symmetry_plane, ScalarT * error = nullptr) const
196  {
197  if (verbose)
198  THEA_CONSOLE << "ICP3(fractional_error_change = " << fractional_error_threshold
199  << ", max_iterations = " << max_iterations << ')';
200 
201  if (from_num_pts <= 0)
202  {
203  if (error) *error = 0;
204  return AffineTransformT::identity();
205  }
206 
207  Array<VectorT> from_points((size_t)from_num_pts);
208  Array<VectorT> to_points((size_t)from_num_pts);
209  for (size_t i = 0; i < from_points.size(); ++i)
210  from_points[i] = PointTraitsN<FromT, 3, ScalarT>::getPosition(from[i]);
211 
212  AffineTransformT old_tr = AffineTransformT::identity();
213  ScalarT old_error = measureError(old_tr, from_num_pts, &from_points[0], from_weight_func, to, &to_points[0]);
214  if (verbose) THEA_CONSOLE << "Initial error: " << old_error;
215 
216  if (old_error <= std::numeric_limits<ScalarT>::min())
217  {
218  if (error) *error = old_error;
219  return old_tr;
220  }
221 
222  ScalarT new_error = old_error;
223  AffineTransformT new_tr = old_tr;
224  for (intx i = 0; i < max_iterations; ++i)
225  {
226  // Align using the point mapping created by the last call to measureError()
227  AffineTransformT inc_tr = alignOneStep(from_num_pts, &from_points[0], from_weight_func, from_symmetry_plane,
228  &to_points[0], to_symmetry_plane);
229 
230  // Update the overall transform
231  new_tr = inc_tr * old_tr;
232 
233  // Compute the new error and the new mapping between points
234  if (i < max_iterations - 1 || error)
235  new_error = measureError(inc_tr, from_num_pts, &from_points[0], from_weight_func, to, &to_points[0]);
236 
237  if (i < max_iterations - 1)
238  {
239  ScalarT frac_change = (old_error - new_error) / old_error;
240 
241  if (verbose)
242  THEA_CONSOLE << "Iteration " << i << " error: " << new_error << " (fractional change: " << frac_change << ')';
243 
244  if (frac_change < fractional_error_threshold)
245  {
246  if (frac_change > 0) // we improved slightly
247  {
248  if (error) *error = new_error;
249  return new_tr;
250  }
251  else // the previous alignment was better
252  {
253  if (error) *error = old_error;
254  return old_tr;
255  }
256  }
257 
258  for (size_t j = 0; j < from_points.size(); ++j) // transform original points to prevent drift
259  from_points[j] = new_tr * PointTraitsN<FromT, 3, ScalarT>::getPosition(from[j]);
260  }
261  else
262  {
263  if (verbose)
264  THEA_CONSOLE << "Iteration " << i << " error: " << new_error;
265  }
266 
267  old_error = new_error;
268  old_tr = new_tr;
269  }
270 
271  if (error) *error = new_error;
272  return new_tr;
273  }

296  template <typename FromT, typename FromWeightFuncT>
297  AffineTransformT alignOneStep(intx num_pts, FromT const * from, FromWeightFuncT * from_weight_func,
298  PlaneT const * from_sym_plane, VectorT const * to, PlaneT const * to_sym_plane) const
299  {
300  // When both point sets have symmetry planes, we'll align the projections. This does *NOT* minimize the correct error
301  // function, but provides a simple approximation to the true solution which is ok for now.
302  bool use_symmetry = from_sym_plane && to_sym_plane;
303 
304  // Find centroids of the two sets
305  VectorT p_mean = VectorT::Zero();
306  VectorT q_mean = VectorT::Zero();
307  ScalarT sum_weights = 0;
308  for (intx i = 0; i < num_pts; ++i)
309  {
310  VectorT p = PointTraitsN<FromT, 3, ScalarT>::getPosition(from[i]);
311  VectorT q = to[i];
312  if (use_symmetry)
313  {
314  p = from_sym_plane->closestPoint(p);
315  q = to_sym_plane->closestPoint(q);
316  }
317 
318  if (from_weight_func)
319  {
320  ScalarT weight = (ScalarT)from_weight_func->getTranslationWeight(from[i]);
321  p_mean += weight * p;
322  q_mean += weight * q;
323  sum_weights += weight;
324  }
325  else
326  {
327  p_mean += p;
328  q_mean += q;
329  sum_weights += 1;
330  }
331  }
332 
333  p_mean /= sum_weights;
334  q_mean /= sum_weights;
335 
336  // Find covariance matrix
337  MatrixT cov; cov.setZero();
338  for (intx i = 0; i < num_pts; ++i)
339  {
340  VectorT p = PointTraitsN<FromT, 3, ScalarT>::getPosition(from[i]);
341  VectorT q = to[i];
342  if (use_symmetry)
343  {
344  p = from_sym_plane->closestPoint(p);
345  q = to_sym_plane->closestPoint(q);
346  }
347 
348  VectorT dp = p - p_mean;
349  VectorT dq = q - q_mean;
350 
351  if (has_up) // remove the component in the up direction
352  {
353  dp = dp - (dp.dot(up) * up);
354  dq = dq - (dq.dot(up) * up);
355  }
356 
357  if (from_weight_func)
358  {
359  ScalarT weight = (ScalarT)from_weight_func->getRotationWeight(from[i]);
360  dp *= weight;
361  dq *= weight;
362  }
363 
364  cov += (dp * dq.transpose()); // outer product
365  }
366 
367  Eigen::JacobiSVD<MatrixT> svd(cov);
368  MatrixT rot = svd.matrixU() * svd.matrixV().transpose();
369 
370  if (use_symmetry && rot.determinant() < 0) // matching two planar projections can cause flips in the symmetry plane
371  {
372  // Generate the transformation that flips in the (origin-centered) target symmetry plane
373  VectorT n = to_sym_plane->getNormal();
374  MatrixT nn = n * n.transpose(); // outer product
375  MatrixT flip = MatrixT::Identity() - static_cast<ScalarT>(2) * nn;
376  rot = flip * rot;
377  }
378 
379  VectorT trans = q_mean - rot * p_mean;
380 
381  return AffineTransformT(rot, trans);
382  }

408  template <typename FromT, typename FromWeightFuncT>
409  static ScalarT measureError(AffineTransformT const & tr,
410  intx num_pts, FromT const * from, FromWeightFuncT * from_weight_func,
411  VectorT const * to)
412  {
413  if (num_pts <= 0)
414  return 0;
415 
416  VectorT p_mean = VectorT::Zero();
417  VectorT q_mean = VectorT::Zero();
418 
419  ScalarT sum_weights = 0;
420  for (intx i = 0; i < num_pts; ++i)
421  {
422  VectorT p = PointTraitsN<FromT, 3, ScalarT>::getPosition(from[i]);
423  VectorT const & q = to[i];
424 
425  if (from_weight_func)
426  {
427  ScalarT weight = (ScalarT)from_weight_func->getTranslationWeight(from[i]);
428  p_mean += weight * p;
429  q_mean += weight * q;
430  sum_weights += weight;
431  }
432  else
433  {
434  p_mean += p;
435  q_mean += q;
436  sum_weights += 1;
437  }
438  }
439 
440  p_mean /= sum_weights;
441  q_mean /= sum_weights;
442 
443  ScalarT err = (tr * p_mean - q_mean).squaredNorm();
444 
445  for (intx i = 0; i < num_pts; ++i)
446  {
447  VectorT p = PointTraitsN<FromT, 3, ScalarT>::getPosition(from[i]);
448  VectorT q = to[i];
449 
450  VectorT dp = p - p_mean;
451  VectorT dq = q - q_mean;
452  if (from_weight_func)
453  {
454  ScalarT weight = (ScalarT)from_weight_func->getRotationWeight(from[i]);
455  err += weight * weight * (tr.getLinear() * dp - dq).squaredNorm();
456  }
457  else
458  err += (tr.getLinear() * dp - dq).squaredNorm();
459  }
460 
461  return err;
462  }
