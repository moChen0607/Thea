167  bool contains(Vector3 const & p) const
168  {
169  if (!halfedge) return false;
170 
171  // Generate a ray for the even-odd test, from p to the midpoint of the first halfedge. Ignore degenerate situations for
172  // now.
173  Vector3 u = 0.5 * (halfedge->getOrigin()->getPosition() + halfedge->getEnd()->getPosition()) - p;
174 
175  int count = 1; // first halfedge is obviously intersected, since we generated the ray through its midpoint
176  Halfedge const * e = halfedge->next();
177  while (e != halfedge)
178  {
179  Vector3 v0 = e->getOrigin()->getPosition() - p;
180  Vector3 v1 = e->getEnd()->getPosition() - p;
181 
182  // If winding order is: vector to first vertex, ray, vector to second vertex, then intersects
183  Vector3 c0 = v0.cross(u);
184  Vector3 c1 = u.cross(v1);
185  if (c0.dot(c1) > 0) // intersects, now check forward or reverse
186  {
187  // Forward if the vector to the point nearest to p on the line containing the edge makes an acute angle with u.
188  //
189  // The point p' on line v + t * e closest to point p is v + t0 * e, where t0 = e.dot(p - v) / e.dot(e)
190  // (see www.geometrictools.com/Documentation/DistancePointLine.pdf).
191  //
192  // We translate p to the origin for simpler computations.
193  Vector3 edge = v1 - v0;
194  Real t0 = -edge.dot(v0) / edge.dot(edge);
195  Vector3 u0 = v0 + t0 * edge;
196 
197  if (u0.dot(u) > 0)
198  count++;
199  }
200 
201  e = e->next();
202  }
203 
204  return (count % 2 == 1);
205  }
