119  double computeProjectedCurvature(Vector3 const & position, Vector3 const & normal, Real nbd_radius = -1) const
120  {
121  if (nbd_radius <= 0)
122  nbd_radius = 0.1f;
123 
124  nbd_radius *= this->getNormalizationScale();
125 
126  ProjectedCurvatureFunctor func(position, normal);
127  Ball3 range(position, nbd_radius);
128 
129  if (this->hasExternalKDTree())
130  this->getMutableExternalKDTree()->template processRangeUntil<IntersectionTester>(range, std::ref(func));
131  else
132  this->getMutableInternalKDTree()->template processRangeUntil<IntersectionTester>(range, std::ref(func));

134  return func.getCurvature();
