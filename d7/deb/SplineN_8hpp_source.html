101  double fitToPoints(InputIterator begin, InputIterator end,
102  T const * initial_params = nullptr,
103  T * final_params = nullptr,
104  bool fix_first_and_last = false,
105  intx max_reparam_iters = -1,
106  intx num_reparam_steps_per_iter = -1)
107  {
108  if (max_reparam_iters < 0) max_reparam_iters = (initial_params ? 0 : 3);
109  if (num_reparam_steps_per_iter < 0) num_reparam_steps_per_iter = 1; // conservative choice, following Graphics Gems
110 
111  // Initialize parameter values for the points
112  Array<double> u;
113  if (initial_params)
114  {
115  T const * up = initial_params;
116  for (InputIterator pi = begin; pi != end; ++pi, ++up)
117  u.push_back(static_cast<double>(*up));
118  }
119  else
120  this->chordLengthParametrize(begin, end, u, (double)this->minParam(), (double)this->maxParam());
121 
122  if ((intx)u.size() < this->numControls())
123  {
124  THEA_ERROR << "SplineN: Cannot fit curve to fewer data points than control vectors";
125  return -1;
126  }
127 
128  // Fit the curve iteratively, alternating between a linear least squares fit with known parameter values, and
129  // re-estimation of parameters via Newton-Raphson
130  double sqerr = -1;
131  while (true)
132  {
133  double e = llsqFit(begin, end, u, fix_first_and_last);
134  if (e < 0) // could not fit, revert to last solution
135  break;
136 
137  if (sqerr >= 0 && e > sqerr) // error increased, stop iterating
138  break;
139 
140  sqerr = e;
141 
142  // A bit wasteful to save it every iteration instead of outside the loop, but do it in case llsqFit fails and we have to
143  // revert to the last solution
144  if (final_params)
145  Algorithms::fastCopy(&u[0], &u[0] + u.size(), final_params);
146 
147  if (--max_reparam_iters < 0)
148  break;
149 
150  refineParameters(begin, end, u, num_reparam_steps_per_iter);
151  }
152 
153  return sqerr;
154  }

217  template <typename InputIterator>
218  double llsqFit(InputIterator begin, InputIterator end, Array<double> const & u, bool fix_first_and_last)
219  {
220  using namespace Algorithms;
221  typedef typename std::iterator_traits<InputIterator>::value_type PointT;
222 
223  // If the sequence is empty, fitting is not possible
224  if (begin == end)
225  {
226  THEA_ERROR << "SplineN: Cannot fit curve to empty sequence of points";
227  return -1;
228  }
229 
230  // Check if fixing first and last positions is even possible. Else, turn this feature off.
231  if (fix_first_and_last && !firstAndLastControlsArePositions())
232  {
233  THEA_WARNING << "SplineN: The beginning and end of the curve are not the first and last control vectors, hence"
234  " they cannot be fixed: this feature will be disabled";
235  fix_first_and_last = false;
236  }
237 
238  intx num_ctrls = (intx)this->numControls();
239  intx num_fixed = (fix_first_and_last ? 2 : 0);
240  intx fixed_offset = (fix_first_and_last ? 1 : 0);
241  intx num_unknown_ctrls = num_ctrls - num_fixed;
242  intx num_unknowns = (intx)(N * num_unknown_ctrls);
243  intx num_objectives = (intx)(N * std::distance(begin, end));
244 
245  VectorXd basis;
246  MatrixXd coeffs(num_objectives, num_unknowns);
247  VectorXd constants(num_objectives);
248 
249  // Cache the first and last points of the sequence
250  InputIterator first = begin, last = begin;
251  VectorT first_pos, last_pos;
252 
253  if (fix_first_and_last)
254  {
255  for (InputIterator pi = begin; pi != end; ++pi)
256  last = pi;
257 
258  first_pos = PointTraitsN<PointT, N, T>::getPosition(*first);
259  last_pos = PointTraitsN<PointT, N, T>::getPosition(*last);
260  }
261 
262  // Try to make each point in the sequence match the point on the curve with the corresponding parameters
263  coeffs.fill(0.0);
264  size_t i = 0;
265  intx obj = 0;
266  for (InputIterator pi = begin; pi != end; ++pi, ++i)
267  {
268  getBasisFunctions(u[i], basis);
269  VectorT d = PointTraitsN<PointT, N, T>::getPosition(*pi);
270  if (fix_first_and_last)
271  {
272  d = d - basis[0] * first_pos // first control vector of curve is fixed at starting point of sequence
273  - basis[basis.size() - 1] * last_pos; // last control vector is also fixed to last point of sequence
274  }
275 
276  // One scalar objective for each dimension
277  for (intx j = 0; j < N; ++j, ++obj)
278  {
279  intx offset = j * num_unknown_ctrls;
280  coeffs.row(obj).segment(offset, basis.size() - num_fixed)
281  = basis.segment(fixed_offset, basis.size() - num_fixed); // assigning col vector to row vector should be ok
282 
283  constants[obj] = d[j];
284  }
285  }
286 
287  // Solve the least-squares linear system
288  StdLinearSolver llsq(StdLinearSolver::Method::DEFAULT, StdLinearSolver::Constraint::UNCONSTRAINED);
289  if (!llsq.solve(coeffs, constants.data()))
290  {
291  THEA_ERROR << "SplineN: Could not solve linear least-squares curve fitting problem";
292  return -1;
293  }
294 
295  double const * sol = llsq.getSolution();
296 
297  // Update the control vectors
298  if (fix_first_and_last) setControl(0, first_pos);
299 
300  Array<VectorT> new_ctrls(num_unknown_ctrls);
301  for (intx i = 0; i < N; ++i)
302  {
303  intx offset = i * num_unknown_ctrls;
304  for (intx j = 0; j < num_unknown_ctrls; ++j)
305  new_ctrls[(size_t)j][i] = static_cast<T>(sol[offset + j]);
306  }
307 
308  for (intx i = 0; i < num_unknown_ctrls; ++i)
309  setControl(i + fixed_offset, new_ctrls[(size_t)i]);
310 
311  if (fix_first_and_last) setControl(num_ctrls - 1, last_pos);
312 
313  double err = 0;
314  if (llsq.getSquaredError(err))
315  return err;
316 
317  VectorXdConstMap sol_vec(sol, (intx)num_unknowns);
318  return (coeffs * sol_vec - constants).squaredNorm();
319  }

326  template <typename InputIterator>
327  bool refineParameters(InputIterator begin, InputIterator end, Array<double> & u, intx num_newton_iters)
328  {
329  using namespace Algorithms;
330  typedef typename std::iterator_traits<InputIterator>::value_type PointT;
331 
332  if (!this->hasDeriv(1) || !this->hasDeriv(2))
333  {
334  THEA_ERROR << "SplineN: Reparametrization requires first and second derivatives";
335  return false;
336  }
337 
338  size_t i = 0;
339  for (InputIterator pi = begin; pi != end; ++pi, ++i)
340  {
341  // Target point
342  VectorT p = PointTraitsN<PointT, N, T>::getPosition(*pi);
343 
344  for (intx j = 0; j < num_newton_iters; ++j)
345  {
346  // We're trying to minimize (Q(t) - P)^2, i.e. finding the root of (Q(t) - P).Q'(t) = 0. The corresponding
347  // Newton-Raphson step is t <-- t - f(t)/f'(t), where f(t) = (Q(t) - P).Q'(t). Differentiating, we get
348  // f'(t) = Q'(t).Q'(t) + (Q(t) - P).Q''(t)
349 
350  double t = static_cast<T>(u[i]);
351  VectorT q = this->eval(t, 0);
352  VectorT q1 = this->eval(t, 1);
353  VectorT q2 = this->eval(t, 2);
354 
355  double numer = static_cast<double>((q - p).dot(q1));
356  double denom = static_cast<double>(q1.dot(q1) + (q - p).dot(q2));
357  if (std::fabs(denom) >= Math::eps<double>())
358  u[i] = Math::clamp(u[i] - numer / denom, (double)this->minParam(), (double)this->maxParam());
359  }
360  }
361 
362  return true;
363  }
