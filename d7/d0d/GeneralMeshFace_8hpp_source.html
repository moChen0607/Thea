297  void updateNormal()
298  {
299  // Assume the face is planar.
300  VertexConstIterator vi2 = vertices.begin();
301  VertexConstIterator vi0 = vi2++;
302  VertexConstIterator vi1 = vi2++;
303 
304  if (vertices.size() > 3)
305  {
306  // vi1 might be a concave corner -- we need to add up the cross products at all vertices
307  Vector3 sum_cross = Vector3::Zero();
308  for ( ; vi0 != vertices.end(); ++vi0, ++vi1, ++vi2)
309  {
310  if (vi1 == vertices.end()) vi1 = vertices.begin();
311  if (vi2 == vertices.end()) vi2 = vertices.begin();
312 
313  Vector3 e1 = (*vi0)->getPosition() - (*vi1)->getPosition();
314  Vector3 e2 = (*vi2)->getPosition() - (*vi1)->getPosition();
315  sum_cross += e2.cross(e1);
316  }
317 
318  setNormal(sum_cross.normalized());
319  }
320  else
321  {
322  Vector3 e1 = (*vi0)->getPosition() - (*vi1)->getPosition();
323  Vector3 e2 = (*vi2)->getPosition() - (*vi1)->getPosition();
324  setNormal(e2.cross(e1).normalized()); // counter-clockwise
325  }
326  }

347  bool contains(Vector3 const & p) const
348  {
349  if (vertices.empty()) return false;
350 
351  // Generate a ray for the even-odd test, from p to the midpoint of the first halfedge. Ignore degenerate situations for
352  // now.
353  VertexConstIterator vi = verticesBegin();
354  VertexConstIterator last = vi++;
355  Vector3 u = 0.5 * ((*last)->getPosition() + (*vi)->getPosition()) - p;
356 
357  intx count = 1; // first halfedge is obviously intersected, since we generated the ray through its midpoint
358  for ( ; last != verticesEnd(); ++vi)
359  {
360  if (vi == verticesEnd()) vi = verticesBegin();
361 
362  Vector3 v0 = (*last)->getPosition() - p;
363  Vector3 v1 = (*vi)->getPosition() - p;
364 
365  // If winding order is: vector to first vertex, ray, vector to second vertex, then intersects
366  Vector3 c0 = v0.cross(u);
367  Vector3 c1 = u.cross(v1);
368  if (c0.dot(c1) > 0) // intersects, now check forward or reverse
369  {
370  // Forward if the vector to the point nearest to p on the line containing the edge makes an acute angle with u.
371  //
372  // The point p' on line v + t * e closest to point p is v + t0 * e, where t0 = e.dot(p - v) / e.dot(e)
373  // (see www.geometrictools.com/Documentation/DistancePointLine.pdf).
374  //
375  // We translate p to the origin for simpler computations.
376  Vector3 edge = v1 - v0;
377  Real t0 = -edge.dot(v0) / edge.dot(edge);
378  Vector3 u0 = v0 + t0 * edge;
379 
380  if (u0.dot(u) > 0)
381  count++;
382  }
383 
384  last = vi;
385  }
386 
387  return (count % 2 == 1);
388  }
