75 void
76 GLFramebuffer::attach(AttachmentPoint ap, Texture * texture, Texture::Face face, int64 z_offset)
77 {
78  GLTexture * gl_texture = dynamic_cast<GLTexture *>(texture);
79  debugAssertM((texture && gl_texture) || (!texture && !gl_texture),
80  std::string(getName()) + ": Texture is not a valid OpenGL texture");
81 
82  if (gl_texture)
83  {
84  if (gl_texture->getDimension() == Texture::Dimension::DIM_3D && z_offset >= gl_texture->getDepth())
85  throw Error(std::string(getName()) + ": Z-offset lies outside depth range of 3D texture");
86 
87  if (num_attachments > 0 && (gl_texture->getWidth() != width || gl_texture->getHeight() != height))
88  throw Error(std::string(getName()) + ": Texture to attach does not have same size as existing attachments");
89  }
90 
91  GLTexture * current_attachment = attachment_table[ap];
92  if (gl_texture == current_attachment)
93  return;
94 
95  // Get current framebuffer
96  GLuint orig_fb = (GLuint)glGetInteger(GL_FRAMEBUFFER_BINDING_EXT);
97 
98  try
99  {
100  // If we aren't already bound, bind us now
101  if (orig_fb != gl_fbid)
102  {
103  glBindFramebufferEXT(GL_FRAMEBUFFER_EXT, gl_fbid);
104  THEA_CHECK_GL_OK
105  }
106 
107  if (gl_texture)
108  {
109  // Get the GL attachment point
110  GLenum gl_ap = GLFramebuffer__apToGLenum(ap);
111 
112  // Attach the texture
113  switch (gl_texture->getDimension())
114  {
115  case Texture::Dimension::DIM_1D:
116  glFramebufferTexture1DEXT(GL_FRAMEBUFFER_EXT, gl_ap, (GLenum)gl_texture->getGLTarget(), (GLuint)gl_texture->getGLID(),
117  0);
118  break;
119 
120  case Texture::Dimension::DIM_2D:
121  case Texture::Dimension::DIM_RECTANGLE:
122  glFramebufferTexture2DEXT(GL_FRAMEBUFFER_EXT, gl_ap, (GLenum)gl_texture->getGLTarget(), (GLuint)gl_texture->getGLID(),
123  0);
124  break;
125 
126  case Texture::Dimension::DIM_3D:
127  glFramebufferTexture3DEXT(GL_FRAMEBUFFER_EXT, gl_ap, (GLenum)gl_texture->getGLTarget(), (GLuint)gl_texture->getGLID(),
128  0, z_offset);
129  break;
130 
131  case Texture::Dimension::DIM_CUBE_MAP:
132  glFramebufferTexture2DEXT(GL_FRAMEBUFFER_EXT, gl_ap, GLTexture::toGLCubeMapFace(face), (GLuint)gl_texture->getGLID(),
133  0);
134  break;
135 
136  default: throw Error(std::string(getName()) + ": Unsupported texture dimensionality");
137  }
138 
139  THEA_CHECK_GL_OK
140  num_attachments++;
141 
142  if (num_attachments == 1) // this is the first attachment
143  {
144  width = gl_texture->getWidth();
145  height = gl_texture->getHeight();
146  }
147 
148  if (GLFramebuffer__isDrawBuffer(ap))
149  {
150  if (std::find(gl_draw_buffers.begin(), gl_draw_buffers.end(), gl_ap) == gl_draw_buffers.end())
151  gl_draw_buffers.push_back(gl_ap);
152  }
153  }
154  else
155  {
156  // Remove any current attachment at this point
157  glFramebufferTexture2DEXT(GL_FRAMEBUFFER_EXT, GLFramebuffer__apToGLenum(ap), GL_TEXTURE_2D, 0, 0);
158  THEA_CHECK_GL_OK
159  num_attachments--;
160  }
161 
162  attachment_table[ap] = gl_texture;
163  }
164  catch (...)
165  {
166  // Restore the original framebuffer
167  if (orig_fb != gl_fbid) glBindFramebufferEXT(GL_FRAMEBUFFER_EXT, orig_fb);
168  throw;
169  }
170 
171  // Restore the original framebuffer
172  if (orig_fb != gl_fbid)
173  {
174  glBindFramebufferEXT(GL_FRAMEBUFFER_EXT, orig_fb);
175  THEA_CHECK_GL_OK
176  }
177 }

180 GLFramebuffer::detach(AttachmentPoint ap)
181 {
182  // Get current framebuffer
183  GLuint orig_fb = (GLuint)glGetInteger(GL_FRAMEBUFFER_BINDING_EXT);
184 
185  try
186  {
187  // If we aren't already bound, bind us now
188  if (orig_fb != gl_fbid)
189  {
190  glBindFramebufferEXT(GL_FRAMEBUFFER_EXT, gl_fbid);
191  THEA_CHECK_GL_OK
192  }
193 
194  if (attachment_table[ap])
195  {
196  // Get the GL attachment point
197  GLenum gl_ap = GLFramebuffer__apToGLenum(ap);
198 
199  glFramebufferTexture2DEXT(GL_FRAMEBUFFER_EXT, gl_ap, GL_TEXTURE_2D, 0, 0);
200  THEA_CHECK_GL_OK

211  }
212  }
213  catch (...)

215  // Restore the original framebuffer

229 GLFramebuffer::detachAll()
230 {
231  // Get current framebuffer
232  GLuint orig_fb = (GLuint)glGetInteger(GL_FRAMEBUFFER_BINDING_EXT);
233 
234  try
235  {
236  // If we aren't already bound, bind us now
237  if (orig_fb != gl_fbid)
238  {
239  glBindFramebufferEXT(GL_FRAMEBUFFER_EXT, gl_fbid);
240  THEA_CHECK_GL_OK
241  }
242 
243  // Remove all current attachments
244  for (int32 ap = 0; ap < AttachmentPoint::MAX_ATTACHMENTS; ++ap)
245  {
246  if (attachment_table[ap])

249  THEA_CHECK_GL_OK

252  num_attachments--;
253  }
254  }
255 
256  assert(num_attachments == 0);
257 
258  gl_draw_buffers.clear();
259  }
260  catch (...)
261  {
262  // Restore the original framebuffer
263  if (orig_fb != gl_fbid) glBindFramebufferEXT(GL_FRAMEBUFFER_EXT, orig_fb);
264  throw;
265  }
266

275 void
276 GLFramebuffer::use()
277 {
278  glBindFramebufferEXT(GL_FRAMEBUFFER_EXT, gl_fbid);
279  GLenum status = glCheckFramebufferStatusEXT(GL_FRAMEBUFFER_EXT);
280  switch (status)
281  {
282  case GL_FRAMEBUFFER_COMPLETE_EXT: break;
283 
284  case GL_FRAMEBUFFER_INCOMPLETE_ATTACHMENT_EXT:
285  case GL_FRAMEBUFFER_INCOMPLETE_MISSING_ATTACHMENT_EXT:
286  throw Error(std::string(getName()) + ": Framebuffer is not attachment-complete");
287 
288  case GL_FRAMEBUFFER_INCOMPLETE_DIMENSIONS_EXT:
289  throw Error(std::string(getName()) + ": Framebuffer attachments aren't of the same size");
290 
291  case GL_FRAMEBUFFER_INCOMPLETE_FORMATS_EXT:
292  throw Error(std::string(getName()) + ": Color attachments do not all have the same format");
293 
294  case GL_FRAMEBUFFER_INCOMPLETE_DRAW_BUFFER_EXT:
295  throw Error(std::string(getName()) + ": Draw buffer doesn't have a valid bound texture");
296 
297  case GL_FRAMEBUFFER_INCOMPLETE_READ_BUFFER_EXT:
298  throw Error(std::string(getName()) + ": Read buffer doesn't have a valid bound texture");
299 
300  default: throw Error(std::string(getName()) + ": Unknown/implementation-dependent framebuffer error");
301  }
302 
303  glDrawBuffersARB((GLsizei)gl_draw_buffers.size(), &gl_draw_buffers[0]);
304  glViewport(0, 0, (GLsizei)width, (GLsizei)height);
305  THEA_CHECK_GL_OK
306 }
