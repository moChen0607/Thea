72 void
73 Stopwatch::tock()
74 {
75  System::endCycleCount(cycleStart);
76  double now = System::time();
77  lastDuration = now - timeStart;
78 
79  if (std::fabs(emwaDuration - lastDuration) > std::max(emwaDuration, lastDuration) * 0.50)
80  {
81  // Off by more than 50%
82  emwaDuration = lastDuration;
83  }
84  else
85  {
86  emwaDuration = lastDuration * 0.05 + emwaDuration * 0.95;
87  }
88 
89  lastCycleCount = cycleStart - cycleOverhead;
90 
91  if (lastCycleCount < 0)
92  {
93  lastCycleCount = 0;
94  }
95 
96  if (lastTockTime != -1.0)
97  {
98  m_fps = 1.0 / (now - lastTockTime);
99 
100  double const BLEND = 0.01;
101  emwaFPS = m_fps * BLEND + emwaFPS * (1.0 - BLEND);
102 
103  double const MAX_DISCREPANCY_PERCENTAGE = 0.25;
104  if (std::fabs(emwaFPS - m_fps) > std::max(emwaFPS, m_fps) * MAX_DISCREPANCY_PERCENTAGE)
105  {
106  // The difference between emwa and m_fps is way off, so
107  // update emwa directly.
108  emwaFPS = m_fps * 0.20 + emwaFPS * 0.80;
109  }
110 
111  // Update m_smoothFPS only when the value varies significantly.
112  // We round so as to not mislead the user as to the accuracy of
113  // the number.
114  if (m_smoothFPS == 0)
115  {
116  m_smoothFPS = m_fps;
117  }
118  else if (emwaFPS <= 20)
119  {
120  if (std::fabs(m_smoothFPS - emwaFPS) > 0.75)
121  {
122  // Small number and display is off by more than 0.75; round to the nearest 0.1
123  m_smoothFPS = floor(emwaFPS * 10.0 + 0.5) / 10.0;
124  }
125  }
126  else if (std::fabs(m_smoothFPS - emwaFPS) > 1.25)
127  {
128  // Large number and display is off by more than 1.25; round to the nearest 1.0
129  m_smoothFPS = std::floor(emwaFPS + 0.5);
130  }
131  }
132 
133  lastTockTime = now;
134 
135  alwaysAssertM(inBetween, "Stopwatch: tock() called without matching tick()");
136  inBetween = false;
137 }
