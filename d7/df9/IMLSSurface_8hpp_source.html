373 template < typename MeshT, typename std::enable_if< Graphics::IsGeneralMesh<MeshT>::value, int >::type >
374 void
375 IMLSSurface::addMesh(MeshT const & mesh, Array<IndexedTriangle> & tris)
376 {
377  typedef UnorderedMap<typename MeshT::Vertex const *, int> VertexIndexMap;
378  VertexIndexMap vertex_indices;
379 
380  size_t base = verts.size();
381  size_t new_size = verts.size() + static_cast<size_t>(mesh.numVertices());
382  verts.resize(new_size);
383  phi.resize(new_size);
384 
385  size_t i = base;
386  for (typename MeshT::VertexConstIterator vi = mesh.verticesBegin(); vi != mesh.verticesEnd(); ++vi, ++i)
387  {
388  verts[i] = vi->getPosition();
389  phi[i] = 0;
390  vertex_indices[&(*vi)] = i;
391  }
392 
393  size_t i0, i1, i2, i3;
394  Polygon3 poly;
395  Array<intx> tri_indices;
396  for (typename MeshT::FaceConstIterator fi = mesh.facesBegin(); fi != mesh.facesEnd(); ++fi)
397  {
398  if (fi->isTriangle())
399  {
400  typename MeshT::Face::VertexConstIterator vi = fi->verticesBegin();
401  i0 = vertex_indices[&(**vi)]; ++vi;
402  i1 = vertex_indices[&(**vi)]; ++vi;
403  i2 = vertex_indices[&(**vi)];
404 
405  tris.push_back(IndexedTriangle(IndexedVertexTriple(&verts, i0, i1, i2)));
406  }
407  else if (fi->isQuad())
408  {
409  // FIXME: Handle non-convex quads
410 
411  typename MeshT::Face::VertexConstIterator vi = fi->verticesBegin();
412  i0 = vertex_indices[&(**vi)]; ++vi;
413  i1 = vertex_indices[&(**vi)]; ++vi;
414  i2 = vertex_indices[&(**vi)]; ++vi;
415  i3 = vertex_indices[&(**vi)];
416 
417  tris.push_back(IndexedTriangle(IndexedVertexTriple(&verts, i0, i1, i2)));
418  tris.push_back(IndexedTriangle(IndexedVertexTriple(&verts, i0, i2, i3)));
419  }
420  else
421  {
422  poly.clear();
423  for (typename MeshT::Face::VertexConstIterator vi = fi->verticesBegin(); vi != fi->verticesEnd(); ++vi)
424  poly.addVertex((*vi)->getPosition(), static_cast<intx>(vertex_indices[&(**vi)]));
425 
426 #ifdef THEA_DEBUG_BUILD
427  intx num_tris = poly.triangulate(tri_indices);
428  debugAssertM(tri_indices.size() == static_cast<size_t>(3 * num_tris),
429  "IMLSSurface: MeshT face triangulation error");
430 #else
431  poly.triangulate(tri_indices);
432 #endif
433 
434  for (size_t i = 0; i < tri_indices.size(); i += 3)
435  tris.push_back(IndexedTriangle(IndexedVertexTriple(&verts,
436  static_cast<size_t>(tri_indices[i]),
437  static_cast<size_t>(tri_indices[i + 1]),
438  static_cast<size_t>(tri_indices[i + 2]))));
439  }
440  }

443 template < typename MeshT, typename std::enable_if< Graphics::IsDCELMesh<MeshT>::value, int >::type >
444 void
445 IMLSSurface::addMesh(MeshT const & mesh, Array<IndexedTriangle> & tris)
446 {
447  typedef UnorderedMap<typename MeshT::Vertex const *, int> VertexIndexMap;
448  VertexIndexMap vertex_indices;
449 
450  size_t base = verts.size();
451  size_t new_size = verts.size() + static_cast<size_t>(mesh.numVertices());
452  verts.resize(new_size);
453  phi.resize(new_size);
454 
455  size_t i = base;
456  for (typename MeshT::VertexConstIterator vi = mesh.verticesBegin(); vi != mesh.verticesEnd(); ++vi, ++i)
457  {
458  verts[i] = vi->getPosition();
459  phi[i] = 0;
460  vertex_indices[&(*vi)] = i;
461  }
462 
463  size_t i0, i1, i2, i3;
464  Polygon3 poly;
465  Array<intx> tri_indices;
466  for (typename MeshT::FaceConstIterator fi = mesh.facesBegin(); fi != mesh.facesEnd(); ++fi)
467  {
468  if (fi->isTriangle())
469  {
470  typename MeshT::Halfedge const * he = fi->getHalfedge();
471  i0 = vertex_indices[he->getOrigin()]; he = he->next();
472  i1 = vertex_indices[he->getOrigin()]; he = he->next();
473  i2 = vertex_indices[he->getOrigin()];
474 
475  tris.push_back(IndexedTriangle(IndexedVertexTriple(&verts, i0, i1, i2)));
476  }
477  else if (fi->isQuad())
478  {
479  // FIXME: Handle non-convex quads
480 
481  typename MeshT::Halfedge const * he = fi->getHalfedge();
482  i0 = vertex_indices[he->getOrigin()]; he = he->next();
483  i1 = vertex_indices[he->getOrigin()]; he = he->next();
484  i2 = vertex_indices[he->getOrigin()]; he = he->next();
485  i3 = vertex_indices[he->getOrigin()];
486 
487  tris.push_back(IndexedTriangle(IndexedVertexTriple(&verts, i0, i1, i2)));
488  tris.push_back(IndexedTriangle(IndexedVertexTriple(&verts, i0, i2, i3)));
489  }
490  else if (fi->numVertices() > 0)
491  {
492  poly.clear();
493  typename MeshT::Halfedge const * he = fi->getHalfedge();
494  do
495  {
496  poly.addVertex(he->getOrigin()->getPosition(), static_cast<intx>(vertex_indices[he->getOrigin()]));
497  he = he->next();
498 
499  } while (he != fi->getHalfedge());
500 
501 #ifdef THEA_DEBUG_BUILD
502  intx num_tris = poly.triangulate(tri_indices);
503  debugAssertM(tri_indices.size() == static_cast<size_t>(3 * num_tris), "IMLSSurface: Mesh face triangulation error");
504 #else
505  poly.triangulate(tri_indices);
506 #endif
507 
508  for (size_t i = 0; i < tri_indices.size(); i += 3)
509  tris.push_back(IndexedTriangle(IndexedVertexTriple(&verts,
510  static_cast<size_t>(tri_indices[i]),
511  static_cast<size_t>(tri_indices[i + 1]),
512  static_cast<size_t>(tri_indices[i + 2]))));
513  }
514  }
515 }
