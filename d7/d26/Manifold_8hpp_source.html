58  template <typename FaceT>
59  static bool makeManifold(Array<FaceT> & faces, intx num_vertices, Array<intx> & vertex_map)
60  {
61  if (num_vertices <= 0 || faces.size() <= 0)
62  return false;
63 
64  size_t nf = faces.size();
65  size_t nv = (size_t)num_vertices;
66 
67  // Initialize every vertex to map to itself
68  vertex_map.resize(nv);
69  for (size_t i = 0; i < nv; ++i)
70  vertex_map[i] = (intx)i;
71 
72  // Associate each vertex with its incident faces
73  Array< Array<size_t> > v2f(nv);
74  for (size_t i = 0; i < nf; ++i)
75  for (typename FaceT::const_iterator vi = faces[i].begin(); vi != faces[i].end(); ++vi)
76  {
77  debugAssertM(*vi >= 0 && (intx)*vi < num_vertices,
78  format("Manifold: Vertex index %ld out of range [0, %ld)", (intx)*vi, num_vertices));
79  v2f[(size_t)*vi].push_back(i);
80  }
81 
82  // Queue the set of vertices
83  Stack<size_t> vertex_stack;
84  for (intx i = (intx)nv - 1; i >= 0; --i)
85  vertex_stack.push((size_t)i);
86 
87  // Split the faces at each vertex into manifold groups
88  while (!vertex_stack.empty())
89  {
90  size_t i = vertex_stack.top();
91  vertex_stack.pop();
92 
93  // Set of edges (represented by their further vertices) incident at this vertex that have already been observed to be
94  // shared by two faces
95  Set<size_t> shared_edges;
96 
97  // Group the faces into maximal edge-connected components
98  typedef UnionFind<size_t> UnionFind;
99  UnionFind uf(v2f[i].begin(), v2f[i].end());
100 
101  for (size_t j = 0; j < v2f[i].size(); ++j)
102  for (size_t k = j + 1; k < v2f[i].size(); ++k)
103  if (shareEdgeAtVertex(faces[v2f[i][j]], faces[v2f[i][k]], i, shared_edges))
104  uf.merge((intx)j, (intx)k);
105 
106  // Retain only the faces edge-connected to the first one, assigning the rest to a copy of the vertex
107  bool created_new_vertex = false;
108  for (size_t j = 1; j < v2f[i].size(); ++j)
109  {
110  if (!uf.sameSet(0, j))
111  {
112  size_t face = v2f[i][j];
113  size_t copy_index = v2f.size() - 1;
114 
115  if (!created_new_vertex) // create a copy of the vertex
116  {
117  copy_index++;
118  v2f.push_back(Array<size_t>());
119  vertex_map.push_back(vertex_map[i]);
120  vertex_stack.push(copy_index);
121 
122  created_new_vertex = true;
123  }
124 
125  v2f[copy_index].push_back(face);
126 
127  // Update the vertex index of this face to point to the copy
128  for (typename FaceT::iterator vi = faces[face].begin(); vi != faces[face].end(); ++vi)
129  if (*vi == (typename FaceT::value_type)i)
130  {
131  *vi = (typename FaceT::value_type)copy_index;
132  break;
133  }
134  }
135  }
136 
137  // If we made a copy of this vertex, we can get rid of faces reassigned to the copy
138  if (created_new_vertex)
139  {
140  Array<size_t> nbd;
141  for (size_t j = 0; j < v2f[i].size(); ++j)
142  if (uf.sameSet(0, j))
143  nbd.push_back(v2f[i][j]);
144 
145  v2f[i] = nbd;
146  }
147  }
148 
149  if ((intx)vertex_map.size() > num_vertices)
150  {
151  THEA_DEBUG << "Manifold: Non-manifold vertices found and fixed";
152  return true;
153  }
154  else
155  return false;
156  }

167  template <typename FaceT>
168  static bool getNeighboringVertices(FaceT const & face, size_t vertex, size_t & nbr1, size_t & nbr2)
169  {
170  // Find the neighbors of the vertex in the first face
171  size_t last_vertex = getLastIndex(face);
172  size_t next_vertex;
173  bool found = false;
174  for (typename FaceT::const_iterator vi = face.begin(); vi != face.end(); ++vi)
175  {
176  next_vertex = (size_t)(vi + 1 == face.end() ? *face.begin() : *(vi + 1));
177  size_t this_vertex = (size_t)*vi;
178 
179  if (this_vertex == next_vertex || last_vertex == next_vertex)
180  throw Error("Manifold: Face has repeated vertices");
181 
182  if (this_vertex == vertex)
183  {
184  found = true;
185  break;
186  }
187  last_vertex = this_vertex;
188  }
189 
190  if (!found)
191  return false;
192 
193  nbr1 = (size_t)last_vertex;
194  nbr2 = (size_t)next_vertex;
195  return true;
196  }

202  template <typename FaceT>
203  static bool shareEdgeAtVertex(FaceT const & face1, FaceT const & face2, size_t vertex,
204  Set<size_t> & shared_edges)
205  {
206  size_t u1, u2, w1, w2;
207  if (!getNeighboringVertices(face1, vertex, u1, u2))
208  throw Error("Manifold: Vertex does not belong to first face");
209 
210  if (!getNeighboringVertices(face2, vertex, w1, w2))
211  throw Error("Manifold: Vertex does not belong to second face");
212 
213  bool ret = false;
214  if ((u1 == w1 || u1 == w2) && shared_edges.find(u1) == shared_edges.end())
215  {
216  ret = true;
217  shared_edges.insert(u1);
218  }
219 
220  if ((u2 == w1 || u2 == w2) && shared_edges.find(u2) == shared_edges.end())
221  {
222  ret = true;
223  shared_edges.insert(u2);
224  }
225 
226  return ret;
227  }
