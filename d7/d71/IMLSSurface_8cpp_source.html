197 void
198 IMLSSurface::computeEnclosingPhi()
199 {
200  typedef std::pair<size_t, double> PairID;
201  Array<PairID> outside;
202 
203  double mean = 0;
204  size_t const num_verts = bounded ? verts.size() - 8: verts.size();
205 
206  if (num_verts > 0)
207  {
208  for (size_t i = 0; i < num_verts; ++i)
209  {
210  double val = (*this)(verts[i]);
211  mean += val;
212  if (val > 0)
213  outside.push_back(PairID(i, val));
214  }
215 
216  isolevel = mean / num_verts;
217  }
218  else
219  isolevel = 0;
220 
221  THEA_DEBUG << "IMLSSurface: Isolevel = " << isolevel;
222 
223  for (double s = 1; !outside.empty(); s *= 2)
224  {
225  THEA_DEBUG << "IMLSSurface: " << outside.size() << " vertices outside";
226 
227  for (Array<PairID>::const_iterator oi = outside.begin(); oi != outside.end(); ++oi)
228  phi[oi->first] -= s * oi->second;
229 
230  computeUnweightedRec(kdtree.getRoot());
231 
232  Array<PairID> next_outside;
233  for (Array<PairID>::const_iterator oi = outside.begin(); oi != outside.end(); ++oi)
234  {
235  double val = (*this)(verts[oi->first]);
236  if (val > std::fabs(std::numeric_limits<double>::epsilon() * phi[oi->first]))
237  next_outside.push_back(PairID(oi->first, val));
238  }
239 
240  outside.swap(next_outside);
241  }

244 void
245 IMLSSurface::computeCentroidsRec(TriangleKDTree::Node const * start)
246 {
247  using namespace IMLSSurfaceInternal;
248 
249  if (!start) return;
250 
251  NodeAttribute & my_attrib = const_cast<NodeAttribute &>(start->attr()); // yes, yes, it's ugly...
252 
253  if (start->isLeaf())
254  {
255  my_attrib.area = 0;
256  my_attrib.centroid = Vector3d::Zero();
257  my_attrib.normal = Vector3d::Zero();
258 
259  for (TriangleKDTree::Node::ElementIndexConstIterator ei = start->elementIndicesBegin(); ei != start->elementIndicesEnd();
260  ++ei)
261  {
262  IndexedTriangle const & tri = kdtree.getElements()[*ei];
263 
264  my_attrib.area += tri.getArea();
265  my_attrib.centroid += tri.getArea() * tri.getCentroid().cast<double>();
266  my_attrib.normal += tri.getNormal().cast<double>();
267  }
268 
269  if (my_attrib.area > 0) my_attrib.centroid /= my_attrib.area;
270  my_attrib.normal_len = my_attrib.normal.norm();
271  }
272  else
273  {
274  TriangleKDTree::Node const * lo = start->getLowChild();
275  TriangleKDTree::Node const * hi = start->getHighChild();
276 
277  computeCentroidsRec(lo);
278  computeCentroidsRec(hi);
279 
280  NodeAttribute const & lo_attrib = lo->attr();
281  NodeAttribute const & hi_attrib = hi->attr();
282 
283  my_attrib.area = lo_attrib.area + hi_attrib.area;
284  if (my_attrib.area > 0)
285  my_attrib.centroid = (lo_attrib.area * lo_attrib.centroid + hi_attrib.area * hi_attrib.centroid) / my_attrib.area;
286  else
287  my_attrib.centroid = Vector3d::Zero();
288 
289  my_attrib.normal = lo_attrib.normal + hi_attrib.normal;
290  my_attrib.normal_len = my_attrib.normal.norm();
291  }

358 void
359 IMLSSurface::eval(Vector3 const & p, Functor & functor) const
360 {
361  using namespace IMLSSurfaceInternal;
362 
363  // Find the nodes that need to be evaluated
364  Array<NodeValuePair> remain;
365  closestLeaf(kdtree, p, remain);
366 
367  // Compute the (negative) maximum error of integration -- stored negative so that the max error is first when sorted by <
368  for (Array<NodeValuePair>::iterator ri = remain.begin(); ri != remain.end(); ++ri)
369  {
370  double max_w2 = weight2(ri->value);
371  double min_w2 = weight2(ri->node->getBounds().squaredMaxDistance(p));
372  ri->value = (max_w2 - min_w2) * ri->node->attr().area;
373  }
374 
375  // Heapify the set of nodes, by decreasing order of approximation error
376  std::make_heap(remain.begin(), remain.end());
377 
378  // Now add up the contributions from the queued nodes
379  IndexedTriangle const * tris = kdtree.getElements();
380  while (!remain.empty() && !functor.acceptableError(remain.front().value))
381  {
382  TriangleKDTree::Node const * current = remain.front().node;
383  std::pop_heap(remain.begin(), remain.end());
384  remain.pop_back();
385 
386  if (current->isLeaf())
387  {
388  for (TriangleKDTree::Node::ElementIndexConstIterator ii = current->elementIndicesBegin();
389  ii != current->elementIndicesEnd(); ++ii)
390  functor.evalTri(p, tris[*ii]);
391  }
392  else
393  {
394  TriangleKDTree::Node const * lo = current->getLowChild();
395  double max_w2 = weight2(lo->getBounds().squaredDistance(p));
396  double min_w2 = weight2(lo->getBounds().squaredMaxDistance(p));
397  remain.push_back(NodeValuePair(lo, (max_w2 - min_w2) * lo->attr().area));
398  std::push_heap(remain.begin(), remain.end());
399 
400  TriangleKDTree::Node const * hi = current->getHighChild();
401  max_w2 = weight2(hi->getBounds().squaredDistance(p));
402  min_w2 = weight2(hi->getBounds().squaredMaxDistance(p));
403  remain.push_back(NodeValuePair(hi, (max_w2 - min_w2) * hi->attr().area));
404  std::push_heap(remain.begin(), remain.end());
405  }
406  }
407 
408  // Approximate the contribution from the remaining nodes
409  for (Array<NodeValuePair>::iterator ri = remain.begin(); ri != remain.end(); ++ri)
410  functor.evalNode(p, *ri->node);
411 }

413 void
414 IMLSSurface::evalRec(Vector3 const & p, TriangleKDTree::Node const * start, Functor & functor) const
415 {
416  if (!start) return;
417 
418  if (start->isLeaf())
419  {
420  IndexedTriangle const * tris = kdtree.getElements();
421  for (TriangleKDTree::Node::ElementIndexConstIterator ii = start->elementIndicesBegin(); ii != start->elementIndicesEnd();
422  ++ii)
423  functor.evalTri(p, tris[*ii]);
424  }
425  else
426  {
427  TriangleKDTree::Node const * c[2] = { start->getLowChild(), start->getHighChild() };
428 
429  // Make sure c[1] is the further child
430  double d[2] = { c[0]->getBounds().squaredDistance(p), c[1]->getBounds().squaredDistance(p) };
431  if (d[0] > d[1])
432  {
433  std::swap(c[0], c[1]);
434  std::swap(d[0], d[1]);
435  }
436 
437  // Nearer child first, since it's likely to have larger error
438  for (int i = 0; i < 2; ++i)
439  {
440  double max_w2 = weight2(d[i]);
441  double min_w2 = weight2(c[i]->getBounds().squaredMaxDistance(p));
442  double err = (max_w2 - min_w2) * c[i]->attr().area;
443 
444  if (functor.acceptableError(err))
445  functor.evalNode(p, *c[i]);
446  else
447  evalRec(p, c[i], functor);
448  }
449  }
450 }

572  NodeAttribute const & attrib = node.attr();
573  Vector3d v(p.cast<double>() - attrib.centroid);
574  double w = surf.weight(v.squaredNorm());
575  Vector3d dw(-4 * w * v);
576  w *= w;
577  dw *= w;
578 
579  double plane_dist = attrib.normal.dot(v);
580  sums.I += w * attrib.area;
581  sums.I_phi += w * attrib.unweighted + plane_dist * w;
582  sums.dI += dw * attrib.area;
583  sums.dI_phi += attrib.normal * w + dw * (attrib.unweighted + plane_dist);
584 }

594 {

667 TriangleKDTree::Node const *
668 closestLeaf(TriangleKDTree const & kdtree, Vector3 const & query, Array<NodeValuePair> & explored)
669 {
670  // Find the leaves containing the query point, if any
671  Array<NodeValuePair> leaf_queue, internal_queue;
672  traverseKDTree(kdtree, query, leaf_queue, internal_queue);
673 
674  explored.clear();
675 
676  // A comparator to heapify the queues in increasing order of distance from the query
677  std::greater<NodeValuePair> cmp;
678 
679  double closest_dist2 = std::numeric_limits<double>::infinity();
680  TriangleKDTree::Node const * closest_node = nullptr;
681 
682  // Find the closest leaf (at triangle level) in the leaf queue
683  if (!leaf_queue.empty())
684  {
685  // Create a minheap on the leaf queue
686  std::make_heap(leaf_queue.begin(), leaf_queue.end(), cmp);
687 
688  size_t num_remaining = leaf_queue.size();
689  while (num_remaining > 0 && leaf_queue.front().value < closest_dist2)
690  {
691  NodeValuePair const & first = leaf_queue.front();
692  double new_dist2 = triSquaredDistance(kdtree, *first.node, query);
693  if (new_dist2 < closest_dist2)
694  {
695  closest_node = first.node;
696  closest_dist2 = new_dist2;
697  }
698 
699  explored.push_back(NodeValuePair(first.node, new_dist2));
700 
701  std::pop_heap(leaf_queue.begin(), leaf_queue.end(), cmp);
702  num_remaining--;
703  }
704 
705  // Add remaining leaves
706  explored.insert(explored.end(), leaf_queue.begin(), leaf_queue.begin() + num_remaining);
707  }
708 
709  // Now look for closer nodes by searching the unexplored internal nodes
710  std::make_heap(internal_queue.begin(), internal_queue.end(), cmp);
711  while (!internal_queue.empty() && internal_queue.front().value < closest_dist2)
712  {
713  NodeValuePair first = internal_queue.front();
714  std::pop_heap(internal_queue.begin(), internal_queue.end(), cmp);
715  internal_queue.pop_back();
716 
717  if (first.node->isLeaf()) // queued node is a leaf, check if it is the closest by examining triangles
718  {
719  double new_dist2 = triSquaredDistance(kdtree, *first.node, query);
720  if (new_dist2 < closest_dist2)
721  {
722  closest_node = first.node;
723  closest_dist2 = new_dist2;
724  }
725 
726  explored.push_back(NodeValuePair(first.node, new_dist2));
727  }
728  else // queued node is internal, check if its children could be closer than the current minimum
729  {
730  double lo_dist2 = first.node->getLowChild()->getBounds().squaredDistance(query);
731  if (lo_dist2 < closest_dist2)
732  {
733  internal_queue.push_back(NodeValuePair(first.node->getLowChild(), lo_dist2));
734  std::push_heap(internal_queue.begin(), internal_queue.end(), cmp);
735  }
736  else
737  explored.push_back(NodeValuePair(first.node->getLowChild(), lo_dist2));
738 
739  double hi_dist2 = first.node->getHighChild()->getBounds().squaredDistance(query);
740  if (hi_dist2 < closest_dist2)
741  {
742  internal_queue.push_back(NodeValuePair(first.node->getHighChild(), hi_dist2));
743  std::push_heap(internal_queue.begin(), internal_queue.end(), cmp);
744  }
745  else
746  explored.push_back(NodeValuePair(first.node->getHighChild(), hi_dist2));
747  }
748  }
749 
750  // Add remaining nodes
751  explored.insert(explored.end(), internal_queue.begin(), internal_queue.end());
752 
753  return closest_node;
754 }

782 void
783 lineIntegrals(double k1, double k2, double & I1, double & Ix, double & dI1, double & dIx, double & dIx2)
784 {

786  // I1 = -Beta * 1/(2*k2^(3/2)) + (k2 - k1*(k1+1)) / (2*k2*(k1^2+k2)*((k1+1)^2+k2))
787  // Ix = Beta * k1/(2*k2^(3/2)) + (k1 + 1) / (2*k2*((k1+1)^2+k2))
788  // dI1 = 1/8 * ( -Beta*3/k2^(5/2) +
789  // (5*k2^3 - (k1*(k1+1)-3)*k2^2 - k1*(k1+1)*(9*k1*(k1+1)+5)*k2 - 3*k1^3*(k1+1)^3)
790  // / (k2^2*(k1^2+k2)^2*((k1+1)^2+k2)^2) )
791  // dIx = 1/8 * ( Beta*3*k1/k2^(5/2) +
792  // ((k1^2+k2)*((3*(k1+1)+1)*k2^2 + (k1+1)*(6*k1^2+3*k1+2)*k2 + 3*k1^2*(k1+1)^3))
793  // /(k2^2*(k1^2+k2)^2*((k1+1)^2+k2)^2) )
794  // dIx2 = 1/8 * ( -Beta*(3*k1^2+k2)/k2^(5/2) -
795  // ((k1^2+k2)^2*(k2^2 + (k1+1)*(4*k1-1)*k2 + 3*k1*(k1+1)^3))
796  // /(k2^2*(k1^2+k2)^2*((k1+1)^2+k2)^2) )
797 
798  static double const K2_EPSILON = 1.0e-200;
799  if (k2 < K2_EPSILON)
800  {
801  I1 = Ix = dI1 = dIx = dIx2 = 0;
802  return;
803  }
804 
805  double sqrt_k2 = std::sqrt(k2);
806  double k1_p1 = k1 + 1;
807  double k1_2 = k1 * k1;
808  double k2_2 = k2 * k2;
809  double k1_p1_2 = k1_p1 * k1_p1;
810  double t1 = k2 + k1_p1_2;
811  double t2 = k2 + k1_2;
812  double t3 = 3 * k1_p1_2 * k1_p1 * k1;
813 
814  I1 = dI1 = (std::atan(k1_p1 / sqrt_k2) - std::atan(k1 / sqrt_k2)) / (2 * k2 * sqrt_k2);
815  dI1 *= 0.75 / k2;
816  Ix = -k1 * I1;
817  dIx = -k1 * dI1;
818  dIx2 = 0.25 * I1 * (3 * k1_2 + k2) / k2;
819 
820  double denom = 0.5 / (k2 * t1);
821  double ddenom = 0.125 / (k2_2 * t2 * t2 * t1 * t1);
822 
823  I1 += (k2 - k1 * k1_p1) * denom / t2;
824  Ix += k1_p1 * denom;
825 
826  dI1 += (5 * k2 * k2_2 - (k1 * k1_p1 - 3) * k2_2 - k1 * k1_p1 * (9 * k1 * k1_p1 + 5) * k2 - k1_2 * t3) * ddenom;
827  dIx += t2 * ((3 * k1_p1 + 1) * k2_2 + k1_p1 * (6 * k1_2 + 3 * k1 + 2) * k2 + k1 * t3) * ddenom;
828  dIx2 -= t2 * t2 * (k2_2 + k1_p1 * (4 * k1 - 1) * k2 + t3) * ddenom;
829 }

865 Vector2d
866 splitTriangleQuadrature(Vector3 const & x, Vector3 const & pm, Vector3 const & p1, Vector3 const & p2, double vm, double v1,
867  double v2, double eps2_)
868 {
869  Vector3 pp(p1), pn(p2);
870  double vp(v1), vn(v2);
871  if ((pp - x).squaredNorm() > (pn - x).squaredNorm())
872  {
873  // Swap so that pp is closest to x
874  pp = p2; pn = p1;
875  vp = v2; vn = v1;
876  }
877 
878  Vector3d d1 = (pp - pm).cast<double>();
879  Vector3d d2 = (pn - pm).cast<double>();
880  Vector3d d3 = (pm - x).cast<double>();
881 
882  static double const T1_EPSILON = 1.0e-100;
883  double t1 = d1.squaredNorm();
884  if (std::fabs(t1) < T1_EPSILON) return Vector2d::Zero(); // early exit if t1 is 0
885 
886  double t2 = d1.dot(d2);
887  double t3 = d1.dot(d3);
888  double t4 = d2.squaredNorm();
889  double t5 = d2.dot(d3) * 2;
890  double t6 = d3.squaredNorm() + eps2_;
891 
892  // Compute height (divided by 2 * sqrt(t1)); early exit if triangle flat
893  double height = t4 / t1 - t2 * t2 / (t1 * t1);
894  if (height <= 0) return Vector2d::Zero();
895  height = std::sqrt(height) / 2.0;
896 
897  double vt1 = vn - vm;
898  double vt2 = vp - vm;
899 
900  double alpha = 2.0 / 3.0;
901  double sum1 = 0, sum2 = 0;
902  double weight = (1.0 / alpha - alpha);
903  double I1, Ix, u_1, denom, k1, k2;
904  double u = alpha;
905  double last_li1 = 0, last_li2 = 0;
906 
907  // Integrate using the trapezoid rule with non-uniform sampling
908  for (; u > 0.01f; u *= alpha)
909  {
910  sum1 += last_li1;
911  sum2 += last_li2;
912 
913  u_1 = 1.0 - u;
914  denom = 1.0 / (u_1 * u_1 * t1);
915  k1 = (t3 + u * t2) * u_1 * denom;
916  k2 = (t6 + u * t5 + u * u * t4) * denom - k1 * k1;
917 
918  lineIntegrals(k1, k2, I1, Ix);
919 
920  // THEA_CONSOLE << "k1 = " << k1 << ", k2 = " << k2 << ", I1 = " << I1 << ", Ix = " << Ix
921  // << ", vm = " << vm << ", vt1 = " << vt1 << ", u_1 = " << u_1 << ", vt2 = " << vt2;
922 
923  denom *= (u / u_1);
924  last_li1 = ((vm + u * vt1) * I1 + u_1 * vt2 * Ix) * denom;
925  last_li2 = I1 * denom;
926  }
927 
928  sum1 *= weight;
929  sum1 += last_li1 / alpha;
930  sum2 *= weight;
931  sum2 += last_li2 / alpha;
932 
933  // Add the last trapezoid covering the remaining area
934  denom = 1.0 / t1;
935  k1 = t3 * denom;
936  k2 = t6 * denom - k1 * k1;
937 
938  lineIntegrals(k1, k2, I1, Ix);
939 
940  denom *= u / alpha;
941  sum1 += (vm * I1 + vt2 * Ix) * denom;
942  sum2 += I1 * denom;
943 
944  sum1 *= height;
945  sum2 *= height;
946 
947  Vector2d result;
948  result[0] = sum1;
949  result[1] = sum2;
950  return result;
951 }

1035  i_data.dI += (d3 * dI1 + d1 * dIx) * denom;

1037 
1038  i_data *= height;
1039  i_data.dI *= 4;
1040  i_data.dI_phi *= 4;
1041 
1042  return i_data;

1047 triangleQuadrature(F quad_func, Vector3 const & x, Vector3 const & p0, Vector3 const & p1, Vector3 const & p2,

1073  }

1075  case 5:

1112  u = -cp.dot(t) * denom;
1113

1122  {
1123  double p_v = v - u * ab.dot(ca) / ca.dot(ca);

1125  state = 1;
1126  else if (p_v >= 1)
1127  state = 4;
1128  else
1129  {
1130  u = 0; v = p_v;
1131  sqdist = ((1 - v) * ap + v * cp).squaredNorm();
1132  return 5;
1133  }
1134  }
1135 
1136  if (v <= eps)
1137  {
1138  double p_u = u - v * ca.dot(ab) / ab.dot(ab);
1139  if (p_u <= eps)
1140  state = 1;
1141  else if (p_u >= 1)
1142  state = 2;
1143  else
1144  {
1145  u = p_u; v = 0;
1146  sqdist = ((1 - u) * ap + u * bp).squaredNorm();
1147  return 3;
1148  }
1149  }
1150

1153  {

1155  double p_u = u + uv * s;
1156  double p_v = v + uv * (1 - s);
1157  if (p_v <= eps)
1158  state = 2;
1159  else if (p_u <= eps)
1160  state = 4;
1161  else
1162  {
1163  u = p_u; v = p_v;
1164  sqdist = (u * bp + v * cp).squaredNorm();

1172  u = 0; v = 0;

1175

1181  case 4:
