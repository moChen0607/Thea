104  double compute(Vector3 const & position, intx num_rays = -1) const
105  {
106  static intx const DEFAULT_NUM_RAYS = 100;
107 
108  if (num_rays <= 0) // 0 is probably user error, snap it to default as well though this should not be relied upon
109  num_rays = DEFAULT_NUM_RAYS;
110 
111  Vector3 u;
112  intx num_escaped = 0;
113  for (intx i = 0; i < num_rays; ++i)
114  {
115  Random::common().sphere(u[0], u[1], u[2]);
116  Vector3 offset = 0.001f * scale * u;
117  Ray3 ray = Ray3(position + offset, u);
118 
119  bool hit = precomp_kdtree ? precomp_kdtree->template rayIntersects<RayIntersectionTester>(ray)
120  : kdtree->template rayIntersects<RayIntersectionTester>(ray);
121  if (!hit)
122  num_escaped++;
123  }
124 
125  return (double)num_escaped / num_rays;
126  }
