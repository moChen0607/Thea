114  double computeEuclidean(Vector3 const & position, Real max_distance) const
115  {
116  bool process_all = (max_distance < 0);
117  if (process_all)
118  max_distance = this->getNormalizationScale();
119 
120  EuclideanCallback callback(position);
121 
122  if (process_all)
123  {
124  intx num_samples = this->numSamples();
125  for (intx i = 0; i < num_samples; ++i)
126  callback(i, this->getSamplePosition(i));
127  }
128  else
129  {
130  Ball3 ball(position, max_distance);
131 
132  if (this->hasExternalKDTree())
133  this->getMutableExternalKDTree()->template processRangeUntil<IntersectionTester>(ball, std::ref(callback));
134  else
135  this->getMutableInternalKDTree()->template processRangeUntil<IntersectionTester>(ball, std::ref(callback));
136  }
137 
138  return callback.getAverageDistance() / max_distance;
139  }

151  double computeGeodesic(Vector3 const & position, Real max_distance) const
152  {
153  bool process_all = (max_distance < 0);
154  if (process_all)
155  max_distance = this->getNormalizationScale();
156 
157  SampleGraph * graph = const_cast<SampleGraph *>(this->getSampleGraph());
158  alwaysAssertM(graph, "AverageDistance: Non-null sample graph required to compute geodesic distances");
159 
160  // Find the sample closest to the query position and use it as the source for all distance calculations
161  intx seed_index = -1;
162  if (this->hasExternalKDTree())
163  seed_index = this->getMutableExternalKDTree()->template closestElement<MetricL2>(position);
164  else
165  seed_index = this->getMutableInternalKDTree()->template closestElement<MetricL2>(position);
166 
167  alwaysAssertM(seed_index >= 0, "AverageDistance: Seed sample for geodesic distances not found");
168 
169  // Assume the graph and the kd-tree have samples in the same sequence
170  SampleGraph::SurfaceSample * seed_sample = const_cast<SampleGraph::SurfaceSample *>(&graph->getSample(seed_index));
171 
172  ShortestPaths<SampleGraph> shortest_paths;
173  GeodesicCallback callback;
174  shortest_paths.dijkstraWithCallback(*graph, seed_sample, std::ref(callback), (process_all ? -1 : max_distance));
175 
176  return callback.getAverageDistance() / max_distance;
177  }
