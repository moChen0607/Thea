223 // Add a face of a general mesh to a set of triangles.
224 template <typename MeshT, typename TriangleT>
225 typename std::enable_if< Graphics::IsGeneralMesh<MeshT>::value >::type
226 addFace(MeshT & mesh, typename MeshT::Face & face, Array<TriangleT> & tris)
227 {
228  if (face.isTriangle())
229  {
230  typename MeshT::Face::VertexIterator vi = face.verticesBegin();
231  typename MeshT::Vertex * v0 = *(vi++);
232  typename MeshT::Vertex * v1 = *(vi++);
233  typename MeshT::Vertex * v2 = *vi;
234  tris.push_back(TriangleT(typename TriangleT::VertexTriple(v0, v1, v2, &face, &mesh)));
235  }
236  else if (face.isQuad())
237  {
238  typename MeshT::Vertex * v[4];
239  typename MeshT::Face::VertexIterator vi = face.verticesBegin();
240  v[0] = *(vi++);
241  v[1] = *(vi++);
242  v[2] = *(vi++);
243  v[3] = *vi;
244 
245  intx i0, j0, k0;
246  intx i1, j1, k1;
247  int num_tris = Polygon3::triangulateQuad(v[0]->getPosition(), v[1]->getPosition(), v[2]->getPosition(), v[3]->getPosition(),
248  i0, j0, k0, i1, j1, k1);
249 
250  if (num_tris > 0)
251  {
252  tris.push_back(TriangleT(typename TriangleT::VertexTriple(v[i0], v[j0], v[k0], &face, &mesh)));
253 
254  if (num_tris > 1)
255  tris.push_back(TriangleT(typename TriangleT::VertexTriple(v[i1], v[j1], v[k1], &face, &mesh)));
256  }
257  }
258  else
259  {
260  Array<typename MeshT::Vertex *> face_vertices;
261  Polygon3 poly;
262  Array<intx> tri_indices;
263 
264  intx i = 0;
265  for (typename MeshT::Face::VertexIterator vi = face.verticesBegin(); vi != face.verticesEnd(); ++vi, ++i)
266  {
267  face_vertices.push_back(*vi);
268  poly.addVertex((*vi)->getPosition(), i);
269  }
270 
271  poly.triangulate(tri_indices);
272  for (size_t j = 0; j < tri_indices.size(); j += 3)
273  {
274  tris.push_back(TriangleT(typename TriangleT::VertexTriple(face_vertices[(size_t)tri_indices[j]],
275  face_vertices[(size_t)tri_indices[j + 1]],
276  face_vertices[(size_t)tri_indices[j + 2]],
277  &face, &mesh)));
278  }
279  }
280 }

291 // Add a face of a DCEL mesh to a set of triangles.
292 template <typename MeshT, typename TriangleT>
293 typename std::enable_if< Graphics::IsDCELMesh<MeshT>::value >::type
294 addFace(MeshT & mesh, typename MeshT::Face & face, Array<TriangleT> & tris)
295 {
296  if (face.isTriangle())
297  {
298  typename MeshT::Halfedge * he = face.getHalfedge();
299  typename MeshT::Vertex * v0 = he->getOrigin();
300  typename MeshT::Vertex * v1 = he->next()->getOrigin();
301  typename MeshT::Vertex * v2 = he->next()->next()->getOrigin();
302  tris.push_back(TriangleT(typename TriangleT::VertexTriple(v0, v1, v2, &face, &mesh)));
303  }
304  else if (face.isQuad())
305  {
306  typename MeshT::Vertex * v[4];
307  typename MeshT::Halfedge * he = face.getHalfedge();
308  v[0] = he->getOrigin();
309  v[1] = he->next()->getOrigin();
310  v[2] = he->next()->next()->getOrigin();
311  v[3] = he->next()->next()->next()->getOrigin();
312 
313  intx i0, j0, k0;
314  intx i1, j1, k1;
315  int num_tris = Polygon3::triangulateQuad(v[0]->getPosition(), v[1]->getPosition(), v[2]->getPosition(), v[3]->getPosition(),
316  i0, j0, k0, i1, j1, k1);
317 
318  if (num_tris > 0)
319  {
320  tris.push_back(TriangleT(typename TriangleT::VertexTriple(v[i0], v[j0], v[k0], &face, &mesh)));
321 
322  if (num_tris > 1)
323  tris.push_back(TriangleT(typename TriangleT::VertexTriple(v[i1], v[j1], v[k1], &face, &mesh)));
324  }
325  }
326  else
327  {
328  Array<typename MeshT::Vertex *> face_vertices;
329  Polygon3 poly;
330  Array<intx> tri_indices;
331 
332  typename MeshT::Halfedge * he = face.getHalfedge();
333  intx num_verts = face.numVertices();
334  for (intx i = 0; i < num_verts; ++i)
335  {
336  face_vertices.push_back(he->getOrigin());
337  poly.addVertex(he->getOrigin()->getPosition(), i);
338  he = he->next();
339  }
340 
341  poly.triangulate(tri_indices);
342  for (size_t j = 0; j < tri_indices.size(); j += 3)
343  {
344  tris.push_back(TriangleT(typename TriangleT::VertexTriple(face_vertices[(size_t)tri_indices[j]],
345  face_vertices[(size_t)tri_indices[j + 1]],
346  face_vertices[(size_t)tri_indices[j + 2]],
347  &face, &mesh)));
348  }
349  }
350 }

361 // Add a face of a display mesh to a set of triangles.
362 template <typename MeshT, typename TriangleT>
363 typename std::enable_if< Graphics::IsDisplayMesh<MeshT>::value >::type
364 addFace(MeshT & mesh, typename MeshT::Face & face, Array<TriangleT> & tris)
365 {
366  typedef typename TriangleT::VertexTriple VertexTriple;
367 
368  if (face.hasTriangles())
369  {
370  typename MeshT::IndexArray const & tri_indices = mesh.getTriangleIndices();
371  size_t beg = 3 * (size_t)face.getFirstTriangle();
372  size_t end = beg + 3 * (size_t)face.numTriangles();
373  for (size_t i = beg; i < end; i += 3)
374  {
375  tris.push_back(TriangleT(VertexTriple(tri_indices[i], tri_indices[i + 1], tri_indices[i + 2], &mesh, (intx)i / 3,
376  VertexTriple::FaceType::TRIANGLE)));
377  }
378  }
379 
380  if (face.hasQuads())
381  {
382  typename MeshT::VertexArray const & vertices = mesh.getVertices();
383  typename MeshT::IndexArray const & quad_indices = mesh.getQuadIndices();
384  size_t beg = 4 * (size_t)face.getFirstQuad();
385  size_t end = beg + 4 * (size_t)face.numQuads();
386  for (size_t i = beg; i < end; i += 4)
387  {
388  intx i0, j0, k0;
389  intx i1, j1, k1;
390  int num_tris = Polygon3::triangulateQuad(vertices[(size_t)quad_indices[i ]],
391  vertices[(size_t)quad_indices[i + 1]],
392  vertices[(size_t)quad_indices[i + 2]],
393  vertices[(size_t)quad_indices[i + 3]],
394  i0, j0, k0, i1, j1, k1);
395 
396  intx quad = (intx)i / 4;
397  if (num_tris > 0)
398  {
399  tris.push_back(TriangleT(VertexTriple(quad_indices[(size_t)(i + i0)],
400  quad_indices[(size_t)(i + j0)],
401  quad_indices[(size_t)(i + k0)],
402  &mesh, quad, VertexTriple::FaceType::QUAD)));
403 
404  if (num_tris > 1)
405  {
406  tris.push_back(TriangleT(VertexTriple(quad_indices[(size_t)(i + i1)],
407  quad_indices[(size_t)(i + j1)],
408  quad_indices[(size_t)(i + k1)],
409  &mesh, quad, VertexTriple::FaceType::QUAD)));
410  }
411  }
412  }
413  }
414 }

416 // Convert the faces of a display mesh to a set of triangles.
417 template <typename MeshT, typename TriangleT>
418 typename std::enable_if< Graphics::IsDisplayMesh<MeshT>::value >::type
419 buildTriangleList(MeshT & mesh, Array<TriangleT> & tris)
420 {
421  typedef MeshT Mesh;
422  typedef TriangleT Triangle;
423  typedef typename Triangle::VertexTriple VertexTriple;
424 
425  typename Mesh::IndexArray const & tri_indices = mesh.getTriangleIndices();
426  for (size_t i = 0; i < tri_indices.size(); i += 3)
427  tris.push_back(Triangle(VertexTriple(tri_indices[i], tri_indices[i + 1], tri_indices[i + 2], &mesh, (intx)i / 3,
428  VertexTriple::FaceType::TRIANGLE)));
429 
430  // FIXME: Handle non-convex faces
431  typename Mesh::IndexArray const & quad_indices = mesh.getQuadIndices();
432  for (size_t i = 0; i < quad_indices.size(); i += 4)
433  {
434  intx quad = (intx)i / 4;
435  tris.push_back(Triangle(VertexTriple(quad_indices[i], quad_indices[i + 1], quad_indices[i + 2], &mesh, quad,
436  VertexTriple::FaceType::QUAD)));
437  tris.push_back(Triangle(VertexTriple(quad_indices[i], quad_indices[i + 2], quad_indices[i + 3], &mesh, quad,
438  VertexTriple::FaceType::QUAD)));
