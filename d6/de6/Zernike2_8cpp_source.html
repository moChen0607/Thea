31 void
32 Zernike2::generateBasisLUT() const
33 {
34  if (lut_generated) return;
35 
36  int lut_size = 2 * opts.lut_radius + 1;
37  lut.resize(boost::extents[opts.angular_steps][opts.radial_steps][lut_size + 1][lut_size + 1]); // one row/column of padding
38 
39  int max_radius = opts.lut_radius;
40  double angle, temp, radius;
41 
42  for (int y = 0; y < lut_size; ++y)
43  for (int x = 0; x < lut_size; ++x)
44  {
45  radius = std::sqrt((double)(Math::square(x - max_radius) + Math::square(y - max_radius)));
46 
47  if(radius < max_radius)
48  {
49  angle = std::atan2((double)(y - max_radius), (double)(x - max_radius));
50 
51  for (int p = 0; p < opts.angular_steps; ++p)
52  for (int r = 0; r < opts.radial_steps; ++r)
53  {
54  temp = std::cos(radius * Math::pi() * r / max_radius);
55  lut[p][r][x][y] = temp * std::complex<double>(std::cos(angle * p), std::sin(angle * p));
56  }
57  }
58  else
59  {
60  for (int p = 0; p < opts.angular_steps; ++p)
61  for (int r = 0; r < opts.radial_steps; ++r)
62  lut[p][r][x][y] = std::complex<double>(0.0, 0.0);
63  }
64  }
65 
66  // Add some padding in case we index beyond the border
67  for (int p = 0; p < opts.angular_steps; ++p)
68  for (int r = 0; r < opts.radial_steps; ++r)
69  for (int i = 0; i <= lut_size; ++i)
70  lut[p][r][i][lut_size] = lut[p][r][lut_size][i] = std::complex<double>(0.0, 0.0);
71 
72  lut_generated = true;
73 }
