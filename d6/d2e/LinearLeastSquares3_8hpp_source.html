66 // Fitting linear models to sets of objects that map to single points in 3-space.
67 template <typename T>
68 class LinearLeastSquares3<T, typename std::enable_if< IsNonReferencedPointN<T, 3>::value >::type>
69 {
70  public:
71  template <typename InputIterator>
72  static double fitLine(InputIterator begin, InputIterator end, Line3 & line, Vector3 * centroid = nullptr)
73  {
74  Vector3d center;
75  Matrix3d cov = covMatrix(begin, end, center);
76 
77  double sum = 0;
78  for (auto iter = makeRefIterator(begin); iter != makeRefIterator(end); ++iter)
79  {
80  Vector3d diff = PointTraitsN<T, 3>::getPosition(*iter).template cast<double>() - center;
81  sum += (diff[0] * diff[0] + diff[1] * diff[1] + diff[2] * diff[2]);
82  }
83 
84  Matrix3d m = sum * Matrix3d::Identity() - cov;
85  Eigen::SelfAdjointEigenSolver<Matrix3d> eigensolver;
86  if (eigensolver.computeDirect(m).info() != Eigen::Success)
87  throw Error("LinearLeastSquares3: Could not eigensolve matrix");
88 
89  // Eigenvalues are in increasing order
90  line = Line3::fromPointAndDirection(center.cast<Real>(), eigensolver.eigenvectors().col(0).cast<Real>());
91  if (centroid) *centroid = center.cast<Real>();
92  return eigensolver.eigenvalues()[0];
93  }
94 
95  template <typename InputIterator>
96  static double fitPlane(InputIterator begin, InputIterator end, Plane3 & plane, Vector3 * centroid = nullptr)
97  {
98  Vector3d center;
99  Matrix3d cov = covMatrix(begin, end, center);
100 
101  Eigen::SelfAdjointEigenSolver<Matrix3d> eigensolver;
102  if (eigensolver.computeDirect(cov).info() != Eigen::Success)
103  throw Error("LinearLeastSquares3: Could not eigensolve covariance matrix");
104 
105  // Eigenvalues are non-negative (covariance matrix is non-negative definite) and in increasing order
106  plane = Plane3::fromPointAndNormal(center.cast<Real>(), eigensolver.eigenvectors().col(0).cast<Real>());
107  if (centroid) *centroid = center.cast<Real>();
108  return eigensolver.eigenvalues()[0];
109  }
110 
111  private:
113  template <typename InputIterator>
114  static Matrix3d covMatrix(InputIterator begin, InputIterator end, Vector3d & centroid)
115  {
116  centroid = CentroidN<T, 3>::compute(begin, end).template cast<double>();
117 
118  Matrix3d m = Matrix3d::Zero();
119  for (auto iter = makeRefIterator(begin); iter != makeRefIterator(end); ++iter)
120  {
121  Vector3d diff = PointTraitsN<T, 3>::getPosition(*iter).template cast<double>() - centroid;
122  m += diff * diff.transpose(); // outer product
123  }
124 
125  return m;
126  }
