22 int
23 binaryTreeDepth(intx num_elems, int max_elems_in_leaf, Real split_ratio)
24 {
25  alwaysAssertM(num_elems >= 0, "Math: Can't compute binary tree depth for negative number of elements");
26  alwaysAssertM(max_elems_in_leaf > 0, "Math: Can't compute binary tree depth for non-positive number of elements at leaf");
27  alwaysAssertM(split_ratio > 0 && split_ratio < 1, "Math: Binary tree split ratio must be in range (0, 1)");
28 
29  if (num_elems <= 0) return 0;
30 
31  double log_inv_split_ratio = -std::log(split_ratio);
32  int est_depth = (int)std::ceil(std::log(num_elems / (double)max_elems_in_leaf) / log_inv_split_ratio);
33 
34  return est_depth < 0 ? 0 : est_depth; // check shouldn't be necessary but do it just in case
35 }

62 // Distinct real roots x1, x2 of quadratic equation c0 + c1 * x + c2 * x^2 = 0
63 int
64 solveQuadratic(double c0, double c1, double c2, double * roots)
65 {
66  using namespace MathInternal;
67 
68  if (std::fabs(c2) < SOLVER_EPSILON)
69  return solveLinear(c1, c0, roots);
70  else
71  {
72  double d = c1 * c1 - 4 * c2 * c0;
73 
74  if (std::fabs(d) < SOLVER_EPSILON)
75  {
76  *roots = -c1 / (2 * c2);
77  return 1;
78  }
79  else if (d > 0)
80  {
81  double sqrt_d = std::sqrt(d);
82  roots[0] = (-c1 + sqrt_d) / (2 * c2);
83  roots[1] = (-c1 - sqrt_d) / (2 * c2);
84  return 2;
85  }
86  }
87 
88  return 0;
89 }

91 // Real roots of cubic equation c0 + c1 * x + c2 * x^2 + c3 * x^3 = 0
92 int
93 solveCubic(double c0, double c1, double c2, double c3, double * roots)
94 {
95  using namespace MathInternal;
96 
97  static double const RAD_120 = degreesToRadians(120);
98  static double const RAD_240 = degreesToRadians(240);
99 
100  int num_roots = 0;
101 
102  if (std::fabs(c3) < SOLVER_EPSILON)
103  num_roots = solveQuadratic(c0, c1, c2, roots);
104  else if (std::fabs(c2) < SOLVER_EPSILON && std::fabs(c1) < SOLVER_EPSILON)
105  {
106  *roots = std::pow(-c0 / c3, 1.0 / 3.0);
107  num_roots = 1;
108  }
109  else if (std::fabs(c0) < SOLVER_EPSILON)
110  {
111  *roots = 0;
112  num_roots = 1;
113 
114  double quad_roots[2];
115  int num_quad_roots = solveQuadratic(c1, c2, c3, quad_roots);
116  for (int i = 0; i < num_quad_roots; ++i)
117  if (std::fabs(quad_roots[i]) > SOLVER_EPSILON)
118  roots[num_roots++] = quad_roots[i];
119  }
120  else
121  {
122  double k = 0, p = 0, q = 0;
123  if (std::fabs(c2) > SOLVER_EPSILON)
124  {
125  k = (-c2) / (3 * c3);
126  p = (3 * c3 * c1 - c2 * c2) / (-3 * c3 * c3);
127  q = (2 * c2 * c2 * c2 - 9 * c3 * c2 * c1 + 27 * c3 * c3 * c0) / (-27 * c3 * c3 * c3);
128  }
129  else
130  {
131  k = 0;
132  p = -c1 / c3;
133  q = -c0 / c3;
134  }
135 
136  double p_d3_e3 = p * p * p / 27;
137  double w = q * q / 4 - p_d3_e3;
138 
139  if (w < 0)
140  {
141  double cos3a = q / (2 * std::sqrt(p_d3_e3));
142  double a = std::acos(cos3a) / 3;
143  double t = std::sqrt(p * (4.0 / 3.0));
144 
145  roots[0] = t * std::cos(a) + k;
146  roots[1] = t * std::cos(a + RAD_120) + k;
147  roots[2] = t * std::cos(a + RAD_240) + k;
148  num_roots = 3;
149  }
150  else
151  {
152  if (w < SOLVER_EPSILON)
153  *roots = 2 * MathInternal::cbrt(q / 2) + k;
154  else
155  *roots = MathInternal::cbrt(q / 2 + std::sqrt(w)) + MathInternal::cbrt(q / 2 - std::sqrt(w)) + k;
156 
157  num_roots = 1;
158  }
159  }
160 
161  return num_roots;
162 }

164 // Real roots of quartic equation c0 + c1 * x + c2 * x^2 + c3 * x^3 + c4 * x^4 = 0
165 int
166 solveQuartic(double c0, double c1, double c2, double c3, double c4, double * roots)
167 {
168  using namespace MathInternal;
169 
170  int num_roots = 0;
171 
172  if (std::fabs(c4) < SOLVER_EPSILON)
173  num_roots = solveCubic(c0, c1, c2, c3, roots);
174  else if (std::fabs(c3) < SOLVER_EPSILON && std::fabs(c2) < SOLVER_EPSILON && std::fabs(c1) < SOLVER_EPSILON)
175  {
176  double x = -c0 / c4;
177  if (std::fabs(x) < SOLVER_EPSILON)
178  {
179  *roots = 0;
180  num_roots = 1;
181  }
182  else if (x > 0)
183  {
184  roots[0] = std::sqrt(std::sqrt(x));
185  roots[1] = -roots[0];
186  num_roots = 2;
187  }
188  else
189  num_roots = 0;
190  }
191  else if (std::fabs(c0) < SOLVER_EPSILON)
192  {
193  *roots = 0;
194  num_roots = 1;
195 
196  double cubic_roots[3];
197  int num_cubic_roots = solveCubic(c1, c2, c3, c4, cubic_roots);
198  for (int i = 0; i < num_cubic_roots; ++i)
199  if (std::fabs(cubic_roots[i]) > SOLVER_EPSILON)
200  roots[num_roots++] = cubic_roots[i];
201  }
202  else
203  {
204  double a3 = c3 / c4;
205  double a2 = c2 / c4;
206  double a1 = c1 / c4;
207  double a0 = c0 / c4;
208 
209  double cubic_roots[3];
210  int num_cubic_roots = solveCubic(a1 * a1 - a0 * (4 * a2 - a3 * a3), 8 * a0 - 2 * a1 * a3, 4 * a2, -8, cubic_roots);
211  if (num_cubic_roots <= 0)
212  num_roots = 0;
213  else
214  {
215  double hi_root = cubic_roots[0];
216  if (num_cubic_roots > 1) hi_root = std::max(hi_root, cubic_roots[1]);
217  if (num_cubic_roots > 2) hi_root = std::max(hi_root, cubic_roots[2]);
218 
219  num_roots = 0;
220 
221  double h2 = hi_root * hi_root;
222  if (h2 > a0)
223  {
224  double b = std::sqrt(h2 - a0);
225  double m = (a3 * hi_root - a1) / (2 * b);
226 
227  double quad_roots[2][2] = { {0, 0}, {0, 0} };
228  int nqr[2] = {
229  solveQuadratic(hi_root + b, (a3 / 2) + m, 1, (double *)quad_roots[0]),
230  solveQuadratic(hi_root - b, (a3 / 2) - m, 1, (double *)quad_roots[1])
231  };
232 
233  for (int i = 0; i < 2; ++i)
234  for (int j = 0; j < nqr[i]; ++j)
235  roots[num_roots++] = quad_roots[i][j];
236  }
237  }
238  }
239 
240  return num_roots;
241 }
