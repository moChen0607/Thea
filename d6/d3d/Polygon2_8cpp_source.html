81 Polygon2::triangulateInterior(Array<Vector2> & tri_verts, Array<intx> & tri_indices,
82  Array<bool> * tri_vert_is_boundary,TriangulationOptions const & options) const
83 {
84  struct triangulateio in, out;
85 
86  tri_verts.clear();
87  tri_indices.clear();
88 
89  if (impl->vertices.size() < 3)
90  return 0;
91 
92  in.numberofpointattributes = 0;
93  in.pointlist = new REAL[2 * (int)impl->vertices.size()];
94  in.pointmarkerlist = nullptr;
95 
96  in.numberofpoints = 0;
97  for (size_t i = 0, j = 0; i < impl->vertices.size(); ++i)
98  {
99  Vector3 const & p = impl->vertices[i].position;
100 
101  // Explicitly remove successive duplicate vertices from input, Triangle seems to not like them very much
102  if (i <= 0 || (p - impl->vertices[i - 1].position).squaredNorm() > 1.0e-10f)
103  {
104  in.pointlist[j++] = p.x();
105  in.pointlist[j++] = p.y();
106  in.numberofpoints++;
107  }
108  }
109 
110  if (in.numberofpoints < 3)
111  return 0;
112 
113  // THEA_CONSOLE << "Triangulating polygon with " << in.numberofpoints << " vertices";
114  // for (int i = 0; i < in.numberofpoints; ++i)
115  // THEA_CONSOLE << " (" << in.pointlist[2 * i] << ", " << in.pointlist[2 * i + 1] << ")";
116 
117  in.numberofsegments = in.numberofpoints;
118  in.segmentlist = new int[2 * in.numberofsegments];
119  in.segmentmarkerlist = nullptr;
120  in.numberofholes = 0;
121  in.numberofregions = 0;
122 
123  for (int i = 0, j = 1, k = 0; i < in.numberofpoints; ++i, ++j, k += 2)
124  {
125  if (j == in.numberofpoints)
126  j = 0;
127 
128  in.segmentlist[k ] = i;
129  in.segmentlist[k + 1] = j;
130  }
131 
132  out.pointlist = nullptr;
133  out.trianglelist = nullptr;
134  out.pointmarkerlist = nullptr;
135 
136  std::string opt_str = format("p" // triangulate planar straight-line graph (PSLG)
137  "q" // quality mesh generation by Delaunay refinement, adding Steiner points
138  "a%0.32lf" // area bound on output triangles
139  "j" // remove unused vertices from output (e.g. duplicate vertices in input)
140  "P" // don't output segments
141  "z" // index everything from zero
142  "Y" // no new vertices on boundary
143  "Q" // quiet mode
144  , options.area_bound
145  );
146 
147  if (options.max_steiner_points >= 0)
148  opt_str += format("S%ld", options.max_steiner_points);
149 
150  if (!tri_vert_is_boundary)
151  opt_str += "B"; // don't output boundary markers
152 
153  char * opt_c_str = new char[opt_str.size() + 1];
154  std::strcpy(opt_c_str, opt_str.c_str());
155 
156  ::triangulate(opt_c_str, &in, &out, nullptr);
157 
158  // Free input arrays
159  delete [] in.pointlist;
160  delete [] in.segmentlist;
161  delete [] opt_c_str;
162 
163  THEA_DEBUG << "Polygon2: " << in.numberofpoints << " vertices triangulated into " << out.numberofpoints << " vertices and "
164  << out.numberoftriangles << " triangles";
165 
166  if (out.numberofpoints < 0 || out.numberoftriangles < 0) // should never happen
167  return 0;
168 
169  tri_verts.resize((size_t)out.numberofpoints);
170  for (size_t i = 0, j = 0; i < tri_verts.size(); ++i, j += 2)
171  tri_verts[i] = Vector2((Real)out.pointlist[j], (Real)out.pointlist[j + 1]);
172 
173  tri_indices.resize(3 * (size_t)out.numberoftriangles);
174  for (size_t i = 0; i < tri_indices.size(); ++i)
175  tri_indices[i] = (intx)out.trianglelist[i];
176 
177  if (tri_vert_is_boundary)
178  {
179  tri_vert_is_boundary->resize((size_t)out.numberofpoints);
180  for (size_t i = 0; i < tri_vert_is_boundary->size(); ++i)
181  (*tri_vert_is_boundary)[i] = (out.pointmarkerlist[i] != 0);
182  }
183 
184  // Free arrays allocated by Triangle
185  trifree(out.pointlist);
186  trifree(out.pointmarkerlist);
187  trifree(out.trianglelist);
188 
189  return (intx)tri_indices.size() / 3;
190 }
