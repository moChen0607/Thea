108 std::string
109 GLCaps::getDriverVersion()
110 {
111  if (computeVendor() == MESA)
112  {
113  // Mesa includes the driver version in the renderer version e.g., "1.5 Mesa 6.4.2"
114  static std::string _glVersion = (char *)glGetString(GL_VERSION);
115  std::size_t i = _glVersion.rfind(' ');
116  if (i == std::string::npos)
117  return "Unknown (bad MESA driver string)";
118  else
119  return _glVersion.substr(i + 1, _glVersion.length() - i);
120  }
121 
122 #ifdef THEA_WINDOWS
123  // Locate the driver on Windows and get the version this systems expects Windows 2000/XP/Vista
124  std::string videoDriverKey;
125  bool canCheckVideoDevice = GLInternal::registryKeyExists("HKEY_LOCAL_MACHINE\\HARDWARE\\DEVICEMAP\\VIDEO");
126  if (canCheckVideoDevice)
127  {
128  // Find the driver expected to load
129  std::string videoDeviceKey = "HKEY_LOCAL_MACHINE\\HARDWARE\\DEVICEMAP\\VIDEO";
130  std::string videoDeviceValue = "\\Device\\Video";
131  int videoDeviceNum = 0;
132 
133  while (GLInternal::registryValueExists(videoDeviceKey, format("%s%d", videoDeviceValue.c_str(), videoDeviceNum)))
134  ++videoDeviceNum;
135 
136  // Find the key where the installed driver lives
137  std::string installedDriversKey;
138  GLInternal::registryReadString(videoDeviceKey, format("%s%d", videoDeviceValue.c_str(), videoDeviceNum - 1),
139  installedDriversKey);
140 
141  // Find and remove the "\Registry\Machine\" part of the key
142  std::size_t subKeyIndex = installedDriversKey.find('\\', 1);
143  subKeyIndex = installedDriversKey.find('\\', subKeyIndex + 1);
144 
145  installedDriversKey.erase(0, subKeyIndex);
146 
147  // Read the list of driver files this display driver installed/loads. This is a multi-string value, but we only care about
148  // the first entry so reading one string is fine
149  std::string videoDrivers;
150  GLInternal::registryReadString("HKEY_LOCAL_MACHINE" + installedDriversKey, "InstalledDisplayDrivers", videoDrivers);
151 
152  if (videoDrivers.find(',', 0) != std::string::npos)
153  videoDrivers = videoDrivers.substr(0, videoDrivers.find(',', 0));
154 
155  char systemDirectory[512] = "";
156  GetSystemDirectoryA(systemDirectory, sizeof(systemDirectory));
157 
158  std::string driverFileName = format("%s\\%s.dll", systemDirectory, videoDrivers.c_str());
159 
160  DWORD dummy;
161  int size = GetFileVersionInfoSizeA((LPCSTR)driverFileName.c_str(), &dummy);
162  if (size == 0)
163  return "Unknown (Can't find driver)";
164 
165  void * buffer = new uint8[size];
166  if (GetFileVersionInfoA((LPCSTR)driverFileName.c_str(), 0, size, buffer) == 0)
167  {
168  delete[] (uint8*)buffer;
169  return "Unknown (Can't find driver)";
170  }
171 
172  // Interpret the VS_VERSIONINFO header pseudo-struct
173  VS_VERSIONINFO * pVS = (VS_VERSIONINFO *)buffer;
174  debugAssertM(!wcscmp(pVS->szKey, L"VS_VERSION_INFO"), "GLCaps: Version info not found");
175 
176  uint8 * pVt = (uint8 *) &pVS->szKey[wcslen(pVS->szKey) + 1];
177 
178 # define roundoffs(a,b,r) (((uint8 *)(b) - (uint8 *)(a) + ((r) - 1)) & ~((r) - 1))
179 # define roundpos(b, a, r) (((uint8 *)(a)) + roundoffs(a, b, r))
180 
181  VS_FIXEDFILEINFO * pValue = (VS_FIXEDFILEINFO *) roundpos(pVt, pVS, 4);
182 
183 # undef roundoffs
184 # undef roundpos
185 
186  std::string result = "Unknown (Can't find driver)";
187 
188  if (pVS->wValueLength)
189  {
190  result = format("%d.%d.%d.%d",
191  pValue->dwProductVersionMS >> 16,
192  pValue->dwProductVersionMS & 0xFFFF,
193  pValue->dwProductVersionLS >> 16,
194  pValue->dwProductVersionLS & 0xFFFF);
195  }
196 
197  delete[] (uint8 *)buffer;
198 
199  return result;
200  }
201  else
202  return "Unknown (Can't find driver)";
203 #else
204  return "Unknown";
205 #endif
206 }

229 bool
230 GLCaps::createHeadlessContext()
231 {
232  if (has_headless_context)
233  return true;
234 
235 #if defined(THEA_GL_OSMESA)
236 
237 #if OSMESA_MAJOR_VERSION * 100 + OSMESA_MINOR_VERSION >= 305
238  // specify Z, stencil, accum sizes
239  headless_context = OSMesaCreateContextExt(OSMESA_RGBA, 16, 0, 0, nullptr);
240 #else
241  headless_context = OSMesaCreateContext(OSMESA_RGBA, nullptr);
242 #endif
243 
244  if (!headless_context)
245  {
246  THEA_ERROR << "GLCaps: Could not create OSMesa context";
247  return false;
248  }
249 
250  // Bind to a small dummy pixmap instead of binding to a window
251  static int const DUMMY_FB_WIDTH = 32;
252  static int const DUMMY_FB_HEIGHT = 32;
253  headless_buffer = new char[DUMMY_FB_WIDTH * DUMMY_FB_HEIGHT];
254  if (!headless_buffer)
255  {
256  THEA_ERROR << "GLCaps: Could not allocate dummy OSMesa offscreen buffer";
257  return false;
258  }
259 
260  if (!OSMesaMakeCurrent(headless_context, headless_buffer, GL_UNSIGNED_BYTE, DUMMY_FB_WIDTH, DUMMY_FB_HEIGHT))
261  {
262  THEA_ERROR << "GLCaps: Could not make new OSMesa context current";
263  return false;
264  }
265 
266  has_headless_context = true;
267 
268 #elif defined(THEA_WINDOWS)
269 
270  // TODO
271 
272 #elif defined(THEA_LINUX) || defined(THEA_BSD)
273 
274  headless_display = XOpenDisplay(0);
275  if (!headless_display)
276  {
277  THEA_ERROR << "GLCaps: Could not open X display";
278  return false;
279  }
280 
281  static int attribs[] = {
282  GLX_RGBA,
283  GLX_RED_SIZE, 8,
284  GLX_GREEN_SIZE, 8,
285  GLX_BLUE_SIZE, 8,
286  GLX_ALPHA_SIZE, 8,
287  GLX_DEPTH_SIZE, 24,
288  None
289  };
290 
291  XVisualInfo * vi = glXChooseVisual(headless_display, DefaultScreen(headless_display), attribs);
292  if (!vi)
293  {
294  THEA_ERROR << "GLCaps: Could not choose X visual";
295  return false;
296  }
297 
298  headless_context = glXCreateContext(headless_display, vi, 0, GL_TRUE);
299  if (!headless_context)
300  {
301  THEA_ERROR << "GLCaps: Could not create X context";
302  return false;
303  }
304 
305  // Bind to a small dummy pixmap instead of binding to a window
306  static int const DUMMY_FB_WIDTH = 32;
307  static int const DUMMY_FB_HEIGHT = 32;
308  headless_pixmap = XCreatePixmap(headless_display, DefaultRootWindow(headless_display), DUMMY_FB_WIDTH, DUMMY_FB_HEIGHT, 24);
309  headless_glx_pixmap = glXCreateGLXPixmap(headless_display, vi, headless_pixmap);
310  if (!glXMakeCurrent(headless_display, headless_glx_pixmap, headless_context))
311  {
312  THEA_ERROR << "GLCaps: Could not make new X context current";
313  return false;
314  }
315 
316  has_headless_context = true;
317 
318 #elif defined(THEA_MAC)
319 
320  CGLPixelFormatAttribute attribs[] = {
321  kCGLPFAColorSize, (CGLPixelFormatAttribute)24,
322  kCGLPFAAlphaSize, (CGLPixelFormatAttribute) 8,
323  kCGLPFADepthSize, (CGLPixelFormatAttribute)24,
324  kCGLPFAAccelerated,
325  kCGLPFASampleBuffers, (CGLPixelFormatAttribute) 1,
326  kCGLPFASamples, (CGLPixelFormatAttribute) 4,
327  kCGLPFAAllowOfflineRenderers,
328  (CGLPixelFormatAttribute)0
329  };
330 
331  CGLError err;
332  CGLPixelFormatObj pix;
333  GLint npix;
334  if ((err = CGLChoosePixelFormat(attribs, &pix, &npix)) != kCGLNoError)
335  {
336  THEA_ERROR << "GLCaps: Could not choose CGL pixel format (error: " << CGLErrorString(err) << ')';
337  return false;
338  }
339 
340  err = CGLCreateContext(pix, 0, &headless_context);
341  if (err != kCGLNoError || !headless_context)
342  {
343  THEA_ERROR << "GLCaps: Could not create CGL context (error: " << CGLErrorString(err) << ')';
344  return false;
345  }
346 
347  if ((err = CGLSetCurrentContext(headless_context)) != kCGLNoError)
348  {
349  THEA_ERROR << "GLCaps: Could not make new CGL context current (error: " << CGLErrorString(err) << ')';
350  return false;
351  }
352 
353  has_headless_context = true;
354 
355 #endif
356 
357  return has_headless_context;
358 }

396 void
397 GLCaps::loadExtensions()
398 {
399  alwaysAssertM(glGetString(GL_RENDERER) != nullptr, "Error initializing OpenGL, please check your GL installation");
400 
401  if (_loadedExtensions)
402  return;
403 
404  alwaysAssertM(!_initialized, "Attempt to initialize OpenGL twice");
405  alwaysAssertM(glGetCurrentContext(), "Unable to load OpenGL extensions without a current context");
406 
407  GLenum err = glewInit();
408  if (err != GLEW_OK)
409  throw Error(format("Couldn't load extensions via GLEW (%s)", glewGetErrorString(err)));
410 
411  _loadedExtensions = true;
412 
413  // Initialize statically cached strings
414  vendor();
415  renderer();
416  glVersion();
417  driverVersion();
418 
419  // Initialize cached GL major version pulled from glVersion() for extensions made into 2.0 core
420  std::string const glver = glVersion();
421  _hasGLMajorVersion2 = beginsWith(glver, "2.");
422 
423  // GL_ARB_texture_cube_map doesn't work on Radeon Mobility
424  // GL Renderer: MOBILITY RADEON 9000 DDR x86/SSE2
425  // GL Version: 1.3.4204 WinXP Release
426  // Driver version: 6.14.10.6430
427 
428  // GL Vendor: ATI Technologies Inc.
429  // GL Renderer: MOBILITY RADEON 7500 DDR x86/SSE2
430  // GL Version: 1.3.3842 WinXP Release
431  // Driver version: 6.14.10.6371
432 
433  if ((beginsWith(renderer(), "MOBILITY RADEON") || beginsWith(renderer(), "ATI MOBILITY RADEON"))
434  && beginsWith(driverVersion(), "6.14.10.6"))
435  {
436  THEA_WARNING << "This ATI Radeon Mobility card has a known bug with cube maps. Put cube map texture coordinates in the"
437  " normals and use ARB_NORMAL_MAP to work around.";
438  }
439 
440  if (THEA_GL_SUPPORTS(ARB_multitexture))
441  {
442  // Don't use more texture units than allowed at compile time
443  GLint num_texunits;
444  glGetIntegerv(GL_MAX_TEXTURE_UNITS_ARB, &num_texunits);
445  _numTextureUnits = std::min(num_texunits, (int)MAX_TEXTURE_UNITS);
446  }
447  else
448  _numTextureUnits = 1;
449 
450  // NVIDIA cards with GL_NV_fragment_program have different numbers of texture coords, units, and textures
451  if (THEA_GL_SUPPORTS(NV_fragment_program))
452  {
453 // To make this compile in the absence of the extension
454 #ifdef GL_MAX_TEXTURE_COORDS_NV
455  glGetIntegerv(GL_MAX_TEXTURE_COORDS_NV, &_numTextureCoords);
456  _numTextureCoords = Math::clamp(_numTextureCoords, _numTextureUnits, (int)MAX_TEXTURE_UNITS);
457 
458  glGetIntegerv(GL_MAX_TEXTURE_IMAGE_UNITS_NV, &_numTextures);
459  _numTextures = Math::clamp(_numTextures, _numTextureUnits, (int)MAX_TEXTURE_UNITS);
460 #endif
461  }
462  else
463  {
464  _numTextureCoords = _numTextureUnits;
465  _numTextures = _numTextureUnits;
466  }
467 
468  if (!THEA_GL_SUPPORTS(ARB_multitexture))
469  {
470  // No multitexture
471  THEA_DEBUG << "No GL_ARB_multitexture support: forcing number of texture units to no more than 1";
472 
473  _numTextureCoords = std::max(1, _numTextureCoords);
474  _numTextures = std::max(1, _numTextures);
475  _numTextureUnits = std::max(1, _numTextureUnits);
476  }
477 
478  THEA_CHECK_GL_OK
479 
480  _initialized = true;
481 }

483 void
484 GLCaps::checkAllBugs()
485 {
486  if (_checkedForBugs)
487  return;
488 
489  alwaysAssertM(_loadedExtensions, "Cannot check for OpenGL bugs before extensions are loaded");
490 
491  bool is_headless = isHeadless();
492  GLuint fb, color_tex, depth_rb;
493  if (is_headless)
494  {
495  static GLsizei const FB_SIZE = 32;
496 
497  glGenTextures(1, &color_tex);
498  glBindTexture(GL_TEXTURE_2D, color_tex);
499  glTexImage2D(GL_TEXTURE_2D, 0, GL_RGBA8, FB_SIZE, FB_SIZE, 0, GL_BGRA, GL_UNSIGNED_BYTE, nullptr);
500  THEA_CHECK_GL_OK
501 
502  glGenRenderbuffersEXT(1, &depth_rb);
503  glBindRenderbufferEXT(GL_RENDERBUFFER_EXT, depth_rb);
504  glRenderbufferStorageEXT(GL_RENDERBUFFER_EXT, GL_DEPTH_COMPONENT24, FB_SIZE, FB_SIZE);
505  THEA_CHECK_GL_OK
506 
507  glGenFramebuffersEXT(1, &fb);
508  glBindFramebufferEXT(GL_FRAMEBUFFER_EXT, fb);
509  glFramebufferTexture2DEXT(GL_FRAMEBUFFER_EXT, GL_COLOR_ATTACHMENT0_EXT, GL_TEXTURE_2D, color_tex, 0);
510  glFramebufferRenderbufferEXT(GL_FRAMEBUFFER_EXT, GL_DEPTH_ATTACHMENT_EXT, GL_RENDERBUFFER_EXT, depth_rb);
511  THEA_CHECK_GL_OK
512 
513  GLenum status = glCheckFramebufferStatusEXT(GL_FRAMEBUFFER_EXT);
514  if (status != GL_FRAMEBUFFER_COMPLETE_EXT)
515  throw Error(format("GLCaps: Could not create offscreen framebuffer for checking bugs (error code %d)", (int)status));
516 
517  glBindFramebufferEXT(GL_FRAMEBUFFER_EXT, fb);
518  glViewport(0, 0, FB_SIZE, FB_SIZE);
519  THEA_CHECK_GL_OK
520  }
521 
522  checkBug_cubeMapBugs();
523  checkBug_redBlueMipmapSwap();
524  checkBug_mipmapGeneration();
525  checkBug_slowVBO();
526 
527  if (is_headless)
528  {
529  glBindFramebufferEXT(GL_FRAMEBUFFER_EXT, 0);
530  glDeleteRenderbuffersEXT(1, &depth_rb);
531  glDeleteFramebuffersEXT(1, &color_tex);
532  glDeleteTextures(1, &color_tex);
533  }
534 
535  _checkedForBugs = true;
536 }

578 bool
579 GLCaps::supportsTexture(Texture::Format const * fmt)
580 {
581  // First, check if we've already tested this format
582  Map<Texture::Format const *, bool>::iterator existing = _supportedTextureFormat.find(fmt);
583  if (existing == _supportedTextureFormat.end())
584  {
585  bool supportsFormat = false;
586 
587  if (!(fmt->floatingPoint && !THEA_GL_SUPPORTS(ARB_texture_float)))
588  {
589  // Allocate some space for making a dummy texture
590  uint8 bytes[8 * 8 * 4];
591 
592  glPushAttrib(GL_TEXTURE_BIT);
593  {
594  // Clear the error bit
595  glGetError();
596 
597  // See if we can create a texture in this format
598  unsigned int id;
599  glGenTextures(1, &id);
600  glBindTexture(GL_TEXTURE_2D, id);
601 
602  // Clear the old error flag
603  glGetError();
604  // 2D texture, level of detail 0 (normal), internal format, x size from image, y size from image,
605  // border 0 (normal), rgb color data, unsigned byte data, and finally the data itself.
606  glTexImage2D(GL_TEXTURE_2D, 0, fmt->openGLFormat, 8, 8, 0, fmt->openGLBaseFormat, GL_UNSIGNED_BYTE, bytes);
607 
608  supportsFormat = (glGetError() == GL_NO_ERROR);
609 
610  glBindTexture(GL_TEXTURE_2D, 0);
611  glDeleteTextures(1, &id);
612  }
613  glPopAttrib();
614  }
615 
616  _supportedTextureFormat[fmt] = supportsFormat;
617  return supportsFormat;
618  }
619  else
620  return existing->second;
621 }

623 bool
624 GLCaps::supportsRenderBuffer(const Texture::Format* fmt)
625 {
626  // First, check if we've already tested this format
627  Map<Texture::Format const *, bool>::iterator existing = _supportedRenderBufferFormat.find(fmt);
628  if (existing == _supportedRenderBufferFormat.end())
629  {
630  bool supportsFormat = false;
631 
632  if (THEA_GL_SUPPORTS(EXT_framebuffer_object) && !(fmt->floatingPoint && !THEA_GL_SUPPORTS(ARB_texture_float)))
633  {
634  glPushAttrib(GL_COLOR_BUFFER_BIT);
635  {
636  // Clear the error bit
637  glGetError();
638 
639  // See if we can create a render buffer in this format
640  unsigned int id;
641  glGenRenderbuffersEXT (1, &id);
642 
643  // Clear the old error flag
644  glGetError();
645 
646  glBindRenderbufferEXT (GL_RENDERBUFFER_EXT, id);
647  glRenderbufferStorageEXT(GL_RENDERBUFFER_EXT, fmt->openGLFormat, 8, 8);
648 
649  supportsFormat = (glGetError() == GL_NO_ERROR);
650 
651  glBindRenderbufferEXT (GL_RENDERBUFFER_EXT, 0);
652  glDeleteRenderbuffersEXT(1, &id);
653  }
654  glPopAttrib();
655  }
656 
657  _supportedRenderBufferFormat[fmt] = supportsFormat;
658  return supportsFormat;
659  }
660  else
661  return existing->second;
662 }

742  GL_TEXTURE_CUBE_MAP_NEGATIVE_X_ARB,
743  GL_TEXTURE_CUBE_MAP_POSITIVE_Y_ARB,

766  std::memset(image, color[f], N * N * sizeof(unsigned int));

769  // border 0 (normal), rgb color data, unsigned byte data, and finally the data itself.
770  glTexImage2D(target[f], 0, GL_RGBA, N, N, 0, GL_RGBA, GL_UNSIGNED_BYTE, image);

774  glTexParameteri(GL_TEXTURE_CUBE_MAP_ARB, GL_TEXTURE_MIN_FILTER, GL_NEAREST);
775  glTexParameteri(GL_TEXTURE_CUBE_MAP_ARB, GL_TEXTURE_WRAP_S, GL_CLAMP);
776  glTexParameteri(GL_TEXTURE_CUBE_MAP_ARB, GL_TEXTURE_WRAP_T, GL_CLAMP);

801  glDisable(GL_LIGHTING);
802  glDisable(GL_CULL_FACE);
803  glColor3f(1, 1, 1);
804

806  1.000000, -1.000000, 1.000000,
807  1.000000, -1.000000, -1.000000,
808  1.000000, 1.000000, -1.000000,
809  1.000000, 1.000000, 1.000000,
810 
811  -1.000000, 1.000000, 1.000000,
812  -1.000000, 1.000000, -1.000000,
813  -1.000000, -1.000000, -1.000000,
814  -1.000000, -1.000000, 1.000000,
815 
816  1.000000, 1.000000, 1.000000,
817  1.000000, 1.000000, -1.000000,
818  -1.000000, 1.000000, -1.000000,
819  -1.000000, 1.000000, 1.000000,
820 
821  1.000000, -1.000000, 1.000000,
822  -1.000000, -1.000000, 1.000000,
823  -1.000000, -1.000000, -1.000000,
824  1.000000, -1.000000, -1.000000,
825 
826  -1.000000, -1.000000, 1.000000,
827  1.000000, -1.000000, 1.000000,
828  1.000000, 1.000000, 1.000000,
829  -1.000000, 1.000000, 1.000000,
830 
831  -1.000000, 1.000000, -1.000000,
832  1.000000, 1.000000, -1.000000,
833  1.000000, -1.000000, -1.000000,
834  -1.000000, -1.000000, -1.000000
835  };

838  {
839  // First time through, use multitex coord
840  if (i == 1)
841  {
842  // Second time through, use normal map generation
843  glActiveTextureARB(GL_TEXTURE0_ARB + 0);
844  glTexGeni(GL_S, GL_TEXTURE_GEN_MODE, GL_NORMAL_MAP_ARB);
845  glTexGeni(GL_T, GL_TEXTURE_GEN_MODE, GL_NORMAL_MAP_ARB);
846  glTexGeni(GL_R, GL_TEXTURE_GEN_MODE, GL_NORMAL_MAP_ARB);
847  glEnable(GL_TEXTURE_GEN_S);
848  glEnable(GL_TEXTURE_GEN_T);
849  glEnable(GL_TEXTURE_GEN_R);
850  }
851

853  for (int f = 0; f < 6; ++f) {
854  const float s = 10.0f;
855 
856  glMultiTexCoord3fvARB(GL_TEXTURE0_ARB, corner + 12 * f + 0);
857  glNormal3fv(corner + 12 * f + 0);
858  glVertex4f(f * s, 0, -1, 1);
859

861  glNormal3fv(corner + 12 * f + 3);

863

865  glNormal3fv(corner + 12 * f + 6);
866  glVertex4f((f + 1) * s, s, -1, 1);
867

869  glNormal3fv(corner + 12 * f + 9);
870  glVertex4f((f + 1) * s, 0, -1, 1);
871  }

876  glReadPixels(0, (int)(viewport[3] - 5), 60, 1, GL_RGBA, GL_UNSIGNED_BYTE, readback);
877 
878  // Test result for errors
879  bool texbug = false;
880  for (int f = 0; f < 6; ++f)
881  {
882  if ((readback[f * 10 + 5] & 0xFF) != color[f])
883  {
884  texbug = true;
885  break;
886  }
887  }
888 
889  if (i == 0)
890  bug_glMultiTexCoord3fvARB = texbug;
891  else
892  bug_normalMapTexGen = texbug;
893 
894  glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);
895  }
896 
897  glPopAttrib();
898 
899  glDeleteTextures(1, &id);
900 
901  THEA_CHECK_GL_OK

904 void
905 GLCaps::checkBug_redBlueMipmapSwap()
906 {
907  if (!THEA_GL_SUPPORTS(SGIS_generate_mipmap))
908  {
909  // No auto-mipmaps == no bug
910  bug_redBlueMipmapSwap = false;
911  return;
912  }
913 
914  glPushAttrib(GL_ALL_ATTRIB_BITS);
915  GLuint id;
916  glGenTextures(1, &id);
917 
918  glBindTexture(GL_TEXTURE_2D, id);
919 
920  glTexParameteri(GL_TEXTURE_2D, GL_GENERATE_MIPMAP_SGIS, GL_TRUE);
921 
922  int N = 4 * 4 * 3;
923  uint8 * bytes = new uint8[N];
924 
925  std::memset(bytes, 0, 4 *4 *3);
926  for (int i = 0; i < N; i += 3)
927  bytes[i] = 0xFF;
928 
929  // 2D texture, level of detail 0 (normal), internal format, x size from image, y size from image, border 0 (normal), rgb
930  // color data, unsigned byte data, and finally the data itself.
931  glTexImage2D(GL_TEXTURE_2D, 0, GL_RGB8, 4, 4, 0, GL_RGB, GL_UNSIGNED_BYTE, bytes);
932 
933  // Read the data back.
934  glGetTexImage(GL_TEXTURE_2D, 0, GL_RGB, GL_UNSIGNED_BYTE, bytes);
935 
936  // Verify that the data made the round trip correctly
937  bug_redBlueMipmapSwap = (bytes[0] != 0xFF) || (bytes[1] != 0x00) || (bytes[2] != 0x00);
938 
939  delete[] bytes;
940  glDeleteTextures(1, &id);
941  glPopAttrib();
942 
943  THEA_CHECK_GL_OK
944 }
