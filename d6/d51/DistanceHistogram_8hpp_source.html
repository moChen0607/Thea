107  void compute(Histogram & histogram, DistanceType dist_type, Real max_distance = -1, Real pair_reduction_ratio = -1) const
108  {
109  intx num_samples = ldh.numSamples();
110  intx num_distinct_unordered = (num_samples - 1) * num_samples;
111 
112  if (pair_reduction_ratio < 0)
113  pair_reduction_ratio = (Real)std::min(0.5, 1000000.0 / num_distinct_unordered); // don't count (x, x)
114 
115  histogram.setZero(); // don't bother setting the range
116 
117  Real local_reduction_ratio = std::sqrt(pair_reduction_ratio);
118  intx num_queries = Math::clamp((intx)std::ceil(local_reduction_ratio * num_samples), 0, num_samples - 1);
119  if (num_queries <= 0)
120  return;
121 
122  Array<int32> query_indices((size_t)num_queries);
123  Random::common().sortedIntegers(0, (int32)num_samples - 1, (int32)num_queries, &query_indices[0]);
124 
125  Histogram local_histogram(histogram.numBins());
126  for (size_t i = 0; i < query_indices.size(); ++i)
127  {
128  Vector3 p = ldh.getSamplePosition(query_indices[i]);
129  ldh.compute(p, local_histogram, dist_type, max_distance, local_reduction_ratio);
130 
131  // Remove the zero distance from the query point to itself
132  local_histogram.remove(0.0);
133 
134  if (i == 0)
135  histogram.setRange(local_histogram.minValue(), local_histogram.maxValue());
136 
137  histogram.insert(local_histogram);
138  }
139  }
