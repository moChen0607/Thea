35 {
36  private:
37  double mean;
38  double variance;
39  double k; // normalizing constant
40 
41  static double kFromVariance(double var) { return 1.0 / std::sqrt(2 * Math::pi() * var); }
42 
43  public:
44  Gaussian1D() : mean(0), variance(0), k(0) {}
45  Gaussian1D(double mean_, double variance_) : mean(mean_), variance(variance_), k(kFromVariance(variance_)) {}
46 
47  double prob(double x) const
48  {
49  if (variance > 0)
50  return k * (std::exp(-Math::square(x - mean) / (2 * variance)));
51  else

53  }

57  if (variance > 0)

89 class Gaussian
90 {
91  private:
92  VectorX<double> mean;
93  MatrixX<double> inv_cov;
94 
95  void resize(intx ndims)
96  {
97  mean.resize(ndims);
98  inv_cov.resize(ndims, ndims);
99  }

101  public:
102  Gaussian() {}

105  double unnormalizedProb(double const * x) const
106  {
107  double xCx = 0;
108  for (intx i = 0; i < mean.size(); ++i)
109  {
110  double prod = 0;
111  for (intx j = 0; j < mean.size(); ++j)
112  prod += (inv_cov(i, j) * (mean[j] - x[j]));
113 
114  xCx += (mean[i] - x[i]) * prod;
115  }
116 
117  return std::exp(-0.5 * xCx);
118  }

120  double unnormalizedProbFast(double const * x) const
121  {
122  double xCx = 0;
123  for (intx i = 0; i < mean.size(); ++i)
124  {
125  double prod = 0;
126  for (intx j = 0; j < mean.size(); ++j)
127  prod += (inv_cov(i, j) * (mean[j] - x[j]));
128 
129  xCx += (mean[i] - x[i]) * prod;
130  }
131 
132  return Math::fastMinusExp(0.5 * xCx);
133  }

333  // Train the tree.
334  void train(TrainingData const & training_data)
335  {
336  clear();
337 
338  if (training_data.numExamples() <= 0)
339  {
340  THEA_CONSOLE << "HoughForest: - No training examples provided, tree is empty";
341  return;
342  }
343 
344  // Set up root node
345  root = new Node(0);
346  root->elems.resize((size_t)training_data.numExamples());
347  for (intx i = 0; i < training_data.numExamples(); ++i)
348  root->elems[(size_t)i] = i;
349 
350  measureClassCumulativeFrequencies(root->elems, training_data, root->class_cum_freq);
351  root->class_feature_distrib.resize(root->class_cum_freq.size());
352 
353  intx num_nodes = 1;
354  intx max_depth = 0;
355 
356  // Depth-first expansion of the tree
357  std::stack<Node *> nodes;
358  nodes.push(root);
359 
360  while (!nodes.empty())
361  {
362  Node * node = nodes.top();
363  nodes.pop();
364 
365  // Split the node
366  if (node->depth < options.max_depth
367  && (intx)node->elems.size() > options.max_leaf_elements)
368  {
369  intx split_feature;
370  double split_value;
371  MeasureMode measure_mode;
372  if (optimizeTest(node, training_data, split_feature, split_value, measure_mode))
373  {
374  // Set up the node to split the given feature at the given value
375  node->split_feature = split_feature;
376  node->split_value = split_value;
377 
378  if (options.verbose >= 2)
379  THEA_CONSOLE << "HoughForest: - Node test splits feature " << split_feature << " at value " << split_value
380  << " to reduce " << (measure_mode == CLASS_UNCERTAINTY ? "class" : "vote") << " uncertainty";
381 
382  // Create left (< feature_value) and right (>= feature_value) children
383  node->left = new Node(node->depth + 1);
384  node->right = new Node(node->depth + 1);
385 
386  // Update stats
387  num_nodes += 2;
388  if (node->left->depth > max_depth)
389  max_depth = node->left->depth;
390 
391  // Divide the elements into left and right subtrees, depending on their feature values
392  Array<double> features;
393  getNodeFeatures(node, split_feature, training_data, features);
394 
395  Array<double> left_features, right_features;
396  for (size_t i = 0; i < node->elems.size(); ++i)
397  {
398  if (features[i] < split_value)
399  {
400  node->left->elems.push_back(node->elems[i]);
401  left_features.push_back(features[i]);
402  }
403  else
404  {
405  node->right->elems.push_back(node->elems[i]);
406  right_features.push_back(features[i]);
407  }
408  }
409 
410  // Measure frequency of each class
411  measureClassCumulativeFrequencies(node->left->elems, training_data, node->left->class_cum_freq);
412  measureClassCumulativeFrequencies(node->right->elems, training_data, node->right->class_cum_freq);
413 
414  // Estimate distribution of this feature for each class within each child
415  estimateClassFeatureDistributions(left_features, node->left->elems, training_data,
416  node->left->class_feature_distrib);
417  estimateClassFeatureDistributions(right_features, node->right->elems, training_data,
418  node->right->class_feature_distrib);
419 
420  if (options.verbose >= 2)
421  THEA_CONSOLE << "HoughForest: - Left node has " << node->left->numElements() << " elements, right node has "
422  << node->right->numElements() << " elements";
423 
424 #ifdef THEA_HOUGH_SYMMETRIC_VARIANCE
425  // Override the data variances to have equal fuzziness on the left and right sides. This is to compensate for
426  // situations when all the feature values on one or both sides are identical (e.g. all zero), so the variance is
427  // zero.
428  for (size_t i = 0; i < node->left->class_feature_distrib.size(); ++i)
429  {
430  double sep = node->left->class_feature_distrib[i].getMean() - node->right->class_feature_distrib[i].getMean();
431  double var = Math::square(sep / 4);
432  node->left->class_feature_distrib[i].setVariance(var);
433  node->right->class_feature_distrib[i].setVariance(var);
434  }
435 #endif
436 
437  // Free up memory at this node
438  node->elems.clear();
439 
440  // Push children onto stack
441  nodes.push(node->left);
442  nodes.push(node->right);
443  }
444  }
445 
446  // Leaf nodes contain lists of elements which must be sorted by class for efficient sampling from a particular class
447  if (node->isLeaf())
448  sortByClass(node->elems, training_data);
449  }
450 
451  THEA_CONSOLE << "HoughForest: - Trained tree with " << num_nodes << " node(s) and " << max_depth + 1 << " level(s) from "
452  << training_data.numExamples() << " example(s)";
453  }

455  // Cast a single vote for a query point with given features. */
456  bool singleVoteSelf(intx query_class, double const * features, VoteCallback & callback) const
457  {
458  Node * curr = root;
459  while (curr)
460  {
461  if (curr->isLeaf())
462  {
463  // Elements are sorted by class, so we can access the range of elements for the class directly
464  intx start_index = (query_class == 0 ? 0 : curr->class_cum_freq[(size_t)query_class - 1]);
465  intx end_index = curr->class_cum_freq[(size_t)query_class] - 1;
466  if (end_index < start_index)
467  return false;
468 
469  intx index = Random::common().integer(start_index, end_index);
470  double weight = curr->getClassFrequency(query_class) / (double)curr->numElements();
471 
472  if (options.verbose >= 2)
473  {
474  intx actual_index = curr->elems[(size_t)index];
475  THEA_CONSOLE << "HoughForest: Reached leaf with " << end_index - start_index + 1 << " element(s) of class "
476  << query_class << " at depth " << curr->depth << ", casting vote for element " << actual_index
477  << " with weight " << weight;
478 
479  std::ostringstream oss;
480  oss << "HoughForest: - Element features = [";
481  for (intx i = 0; i < num_features; ++i)
482  {
483  if (i > 0) oss << ", ";
484  oss << parent->all_features(actual_index, i);
485  }
486  oss << ']';
487  THEA_CONSOLE << oss.str();
488  }
489 
490  parent->singleSelfVoteByLookup(curr->elems[(size_t)index], weight, callback);
491  break;
492  }
493  else
494  {
495  double feat = features[curr->split_feature];
496  bool done = false;
497 
498  if (options.verbose >= 3)
499  THEA_CONSOLE << "HoughForest: Testing feature " << curr->split_feature << ", value " << feat
500  << " against split value " << curr->split_value << " at depth " << curr->depth;
501 
502  //===================================================================================================================
503  // If either the left or the right subtree has zero samples from the query class, we can just visit the other subtree
504  //===================================================================================================================
505  intx left_freq = curr->left->getClassFrequency(query_class);
506  intx right_freq = curr->right->getClassFrequency(query_class);
507  if (left_freq > 0)
508  {
509  if (right_freq <= 0)
510  {
511  curr = curr->left;
512  done = true;
513  }
514  }
515  else if (right_freq > 0)
516  {
517  curr = curr->right;
518  done = true;
519  }
520  else // this subtree has no samples from the query class
521  return false;
522 
523  //===================================================================================================================
524  // Make a probabilistic choice for which child to step into, for smooth vote distributions. Similar to Gaussian
525  // kd-trees [Adams et al. 2009] but the left and right probabilities are evaluated a little differently.
526  //===================================================================================================================
527  if (!done && options.probabilistic_sampling)
528  {
529  double p_left = curr->left ->class_feature_distrib[(size_t)query_class].probFast(feat);
530  double p_right = curr->right->class_feature_distrib[(size_t)query_class].probFast(feat);
531 
532  p_left *= left_freq;
533  p_right *= right_freq;
534 
535  double p_sum = p_left + p_right;
536  if (p_sum > 0)
537  {
538  double coin_toss = Random::common().uniform01();
539  if (coin_toss * p_sum < p_left)
540  {
541  if (options.verbose >= 3)
542  THEA_CONSOLE << "HoughForest: Traversing left with probabilistic sampling on feature " << curr->split_feature;
543 
544  curr = curr->left;
545  }
546  else
547  {
548  if (options.verbose >= 3)
549  THEA_CONSOLE << "HoughForest: Traversing right with probabilistic sampling on feature "
550  << curr->split_feature;
551 
552  curr = curr->right;
553  }
554 
555  done = true;
556  }
557  }
558 
559  //===================================================================================================================
560  // Make the traditional hard choice
561  //===================================================================================================================
562  if (!done)
563  {
564  if (feat < curr->split_value)
565  {
566  if (options.verbose >= 3)
567  THEA_CONSOLE << "HoughForest: Traversing left with hard decision on feature " << curr->split_feature;
568 
569  curr = curr->left;
570  }
571  else
572  {
573  if (options.verbose >= 3)
574  THEA_CONSOLE << "HoughForest: Traversing right with hard decision on feature " << curr->split_feature;
575 
576  curr = curr->right;
577  }
578  }
579  }
580  }
581 
582  return true;
583  }

613  // Find a suitable splitting decision for a node, based on minimizing classification/regression uncertainty.
614  bool optimizeTest(Node const * node, TrainingData const & training_data, intx & split_feature, double & split_value,
615  MeasureMode & measure_mode) const
616  {
617  if (node->elems.size() < 4) // doesn't make sense splitting smaller sets, and the quadrant index math will fail anyway
618  return false;
619 
620  if (num_classes <= 1)
621  measure_mode = VOTE_UNCERTAINTY;
622  else
623  {
624  measure_mode = (Random::common().coinToss() ? CLASS_UNCERTAINTY : VOTE_UNCERTAINTY);
625  if (measure_mode == CLASS_UNCERTAINTY)
626  {
627  double uncertainty = measureUncertainty(node->elems, training_data, measure_mode);
628  if (uncertainty < options.min_class_uncertainty)
629  measure_mode = VOTE_UNCERTAINTY;
630  }
631  }
632 
633  Array<double> features, reordered_features;
634  double min_uncertainty = -1;
635  split_feature = -1;
636  split_value = 0;
637 
638  intx max_feat_iters = std::min((options.num_feature_expansions + 1) * options.max_candidate_features, num_features);
639  intx max_thresh_iters = std::min(options.max_candidate_thresholds, (intx)node->elems.size());
640 
641  for (intx feat_iter = 0; feat_iter < max_feat_iters; ++feat_iter)
642  {
643  // Generate a random feature
644  // TODO: Rewrite this using randIntegersInRange to avoid repeating features
645  intx test_feature = (max_feat_iters < num_features ? Random::common().integer() % num_features : feat_iter);
646 
647  // Find a suitable split threshold
648  getNodeFeatures(node, test_feature, training_data, features);
649 
650  for (intx thresh_iter = 0; thresh_iter < max_thresh_iters; ++thresh_iter)
651  {
652  double test_threshold = 0;
653  if (max_thresh_iters < (intx)features.size())
654  {
655  // Generate a splitting value in the middle half (second and third quadrants) in the sorted order
656  size_t index = features.size() / 4 + (Random::common().integer() % (features.size() / 2));
657 
658  if (options.verbose >= 3)
659  THEA_CONSOLE << "HoughForest: - Testing split index " << index << " for feature " << test_feature;
660 
661  // Operate on a scratch array so the original features array retains its ordering and can be passed to
662  // measureUncertaintyAfterSplit()
663  if (thresh_iter == 0)
664  reordered_features = features;
665 
666  std::nth_element(reordered_features.begin(), reordered_features.begin() + index, reordered_features.end());
667  test_threshold = reordered_features[index];
668  }
669  else
670  {
671  // We're considering all possible features as thresholds so no need to sort
672  test_threshold = features[(size_t)thresh_iter];
673  }
674 
675  // Measure the uncertainty after the split
676  bool valid_split;
677  double uncertainty = measureUncertaintyAfterSplit(node->elems, features, test_threshold, training_data, measure_mode,
678  valid_split);
679  if (valid_split && (split_feature < 0 || uncertainty < min_uncertainty))
680  {
681  split_feature = test_feature;
682  split_value = refineSplitValue(test_threshold, features);
683  min_uncertainty = uncertainty;
684 
685  if (options.verbose >= 3)
686  THEA_CONSOLE << "HoughForest: - Updated split feature " << split_feature << ", split value " << split_value
687  << ", uncertainty " << min_uncertainty;
688  }
689  }
690 
691  // If we've reached the end of one expansion and we've found a valid splitting feature, we can stop

747  // Measure the classification/regression uncertainty of a collection training examples.
748  double measureUncertainty(Array<intx> const & elems, TrainingData const & training_data, MeasureMode measure_mode) const
749  {
750  if (measure_mode == CLASS_UNCERTAINTY)
751  {
752  // The uncertainty is defined to be the Shannon entropy of the labeling:
753  // Entropy = -\sum_{c \in Classes} p(c) log p(c)
754 
755  Array<intx> class_freq;
756  measureClassFrequencies(elems, training_data, class_freq);
757 
758  double entropy = 0;
759  for (size_t i = 0; i < class_freq.size(); ++i)
760  if (class_freq[i] > 0)
761  {
762  double prob = class_freq[i] / (double)elems.size();
763  entropy -= prob * std::log(prob);
764  }
765 
766  return entropy;
767  }
768  else
769  {
770  // The uncertainty is defined to be the average variance in Hough votes for the same object. Context votes (elements of
771  // one object voting for parameters of another object) are ignored.
772 
773  // First get the class of each training example
774  Array<intx> classes(elems.size());
775  training_data.getClasses((intx)elems.size(), &elems[0], &classes[0]);
776 
777  // Now measure the vote variance per class
778  MatrixX<double> sum_votes(num_classes, max_vote_params); sum_votes.setZero(); // to measure "square of mean"
779  Array<double> sum_vote_sqlen((size_t)num_classes, 0.0); // to measure "mean of squares"
780  Array<intx> class_freq((size_t)num_classes, 0); // only count elements that have valid
781  // Hough votes for their own classes
782  Array<double> vote((size_t)max_vote_params);
783  for (size_t i = 0; i < elems.size(); ++i)
784  {
785  if (classes[i] == BACKGROUND_CLASS) // ignore background class, assumed to have index 0
786  continue;
787 
788  training_data.getSelfVote(elems[i], &vote[0]);
789 
790  // Add up:
791  // - the vote vector
792  // - the square(d length) of the vote
793  for (intx j = 0; j < num_vote_params[(size_t)classes[i]]; ++j)
794  {
795  double v = vote[(size_t)j];
796  sum_votes(classes[i], j) += v;
797  sum_vote_sqlen[(size_t)classes[i]] += (v * v);
798  }
799 
800  class_freq[(size_t)classes[i]]++;
801  }
802 
803  // (Possible) FIXME: The following sum assumes variances for different classes (with possibly different numbers of Hough
804  // parameters) are directly comparable.
805 
806  double tot_var = 0;
807  for (intx i = 0; i < num_classes; ++i)
808  {
809  if (i == BACKGROUND_CLASS)
810  continue;
811 
812  intx num_class_members = class_freq[(size_t)i];
813  if (num_class_members > 0)
814  {
815  double sqlen_of_mean = 0;
816  for (intx j = 0; j < num_vote_params[(size_t)i]; ++j)
817  {
818  double v = sum_votes(i, j);
819  sqlen_of_mean += (v * v);
820  }
821  sqlen_of_mean /= (num_class_members * num_class_members);
822 
823  // Apply: variance = mean of squares - square of mean
824  tot_var += (sum_vote_sqlen[i] / num_class_members - sqlen_of_mean);
825  }
826  }
827 
828  return tot_var / (num_classes - 1); // ignore background class
829  }
830  }

832  // Measure the classification/regression uncertainty after splitting elements along a feature.
833  double measureUncertaintyAfterSplit(Array<intx> const & elems, Array<double> const & features, double split_value,
834  TrainingData const & training_data, MeasureMode measure_mode, bool & valid_split) const
835  {
836  static Real const MAX_ASYMMETRY = 7;
837 
838  valid_split = true;
839 
840  if (elems.empty())
841  {
842  valid_split = false;
843  return 0;
844  }
845

847  for (size_t i = 0; i < elems.size(); ++i)

868  + measureUncertainty(right_elems, training_data, measure_mode);

871  // Adjust the split value to bisect the closest pair from the left and right sides.
872  double refineSplitValue(double split_value, Array<double> const & features) const
873  {
874  double left_max = 0, right_min = 0;
875  bool found_left = false, found_right = false;
876  for (size_t i = 0; i < features.size(); ++i)
877  {
878  double feat = features[i];
879  if (feat < split_value)
880  {
881  if (!found_left)

887  left_max = feat;
888  }
889  else
890  {

894  found_right = true;
895  }
896  else if (feat < right_min)
897  right_min = feat;

899  }

907  // Estimate the distribution of a set of 1D feature values
908  void estimateClassFeatureDistributions(Array<double> const & features, Array<intx> const & elems,
909  TrainingData const & training_data, Array<Gaussian1D> & class_feature_distrib)
910  {
911  alwaysAssertM(!elems.empty(), "HoughForest: Can't model distribution of empty set");
