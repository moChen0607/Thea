86  void compute(Vector3 const & position, intx num_steps, double * features, intx num_walks = -1) const
87  {
88  static intx const DEFAULT_NUM_WALKS = 1000;
89 
90  if (num_walks < 0)
91  num_walks = DEFAULT_NUM_WALKS;
92 
93  // Zero out all features initially
94  for (intx i = 0; i < num_steps; ++i)
95  {
96  features[3 * i ] = 0.0;
97  features[3 * i + 1] = 0.0;
98  features[3 * i + 2] = 0.0;
99  }
100 
101  // Find the sample closest to the query position and use it as the source for all distance calculations
102  intx seed_index = -1;
103  if (this->hasExternalKDTree())
104  seed_index = this->getMutableExternalKDTree()->template closestElement<MetricL2>(position);
105  else
106  seed_index = this->getMutableInternalKDTree()->template closestElement<MetricL2>(position);
107 
108  alwaysAssertM(seed_index >= 0, "RandomWalks: Seed sample for random walks not found");
109 
110  Array<intx> counts((size_t)num_steps, 0);
111  for (intx i = 0; i < num_walks; ++i)
112  {
113  intx walk_steps = walk(seed_index, num_steps, features);
114  for (intx j = 0; j < walk_steps; ++j)
115  counts[(size_t)j]++;
116  }
117 
118  for (intx i = 0; i < num_steps; ++i)
119  {
120  intx count = counts[(size_t)i];
121  features[3 * i ] = (features[3 * i ] / count - position[0]);
122  features[3 * i + 1] = (features[3 * i + 1] / count - position[1]);
123  features[3 * i + 2] = (features[3 * i + 2] / count - position[2]);
124  }
125  }
126 
127  private:
132  intx walk(intx seed_index, intx num_steps, double * features) const
133  {
134  typedef SampleGraph::SurfaceSample::NeighborSet NeighborSet;
135 
136  // Assume the graph and the kd-tree have samples in the same sequence
137  SampleGraph * graph = const_cast<SampleGraph *>(this->getSampleGraph());
138  alwaysAssertM(graph, "RandomWalks: Non-null sample graph required for random walks");
139 
140  SampleGraph::SurfaceSample * sample = const_cast<SampleGraph::SurfaceSample *>(&graph->getSample(seed_index));
141 
142  intx base_index = 0;
143  for (intx i = 0; i < num_steps; ++i, base_index += 3)
144  {
145  NeighborSet const & nbrs = sample->getNeighbors();
146  if (nbrs.isEmpty())
147  return i;
148 
149  intx next_index = Random::common().integer(0, nbrs.size() - 1);
150  sample = nbrs[next_index].getSample();
151 
152  Vector3 const & p = sample->getPosition();
153  features[base_index ] += p[0];
154  features[base_index + 1] += p[1];
155  features[base_index + 2] += p[2];
156  }
157 
158  return num_steps;
159  }
