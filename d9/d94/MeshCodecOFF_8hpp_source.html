129  void readMeshGroup(MeshGroup & mesh_group, BinaryInputStream & input, Codec::BlockHeader const * block_header,
130  ReadCallback * callback) const
131  {
132  mesh_group.clear();
133 
134  BinaryInputStream * in = &input;
135  BinaryInputStream::Ptr tmp_in;
136  if (block_header)
137  {
138  if (block_header->data_size <= 0) { return; }
139  tmp_in = std::make_shared<BinaryInputStream>(input, (int64)block_header->data_size);
140  in = tmp_in.get();
141  }
142 
143  std::string header = trimWhitespace(in->readLine());
144  if (header != "OFF" && !beginsWith(header, "OFF "))
145  throw Error(std::string(getName()) + ": Invalid OFF stream (does not start with 'OFF')");
146 
147  bool binary = (header == "OFF BINARY" || beginsWith(header, "OFF BINARY "));
148  if (binary)
149  readBinary(mesh_group, *in, callback);
150  else
151  readAscii(mesh_group, *in, callback);
152  }

154  void writeMeshGroup(MeshGroup const & mesh_group, BinaryOutputStream & output, bool write_block_header,
155  WriteCallback * callback) const
156  {
157  Codec::BlockHeader bh(this->getMagic());
158  if (write_block_header)
159  bh.markAndSkip(output);
160 
161  { BinaryOutputStream::EndiannessScope scope(output, Endianness::BIG); // binary OFF uses big-endian storage
162 
163  intx num_vertices = 0, num_faces = 0;
164  getStats(mesh_group, num_vertices, num_faces);
165 
166  VertexIndexMap vertex_indices;
167  if (write_opts.binary)
168  {
169  output.printf("OFF BINARY\n");
170  output.writeInt32((int32)num_vertices);
171  output.writeInt32((int32)num_faces);
172  output.writeInt32(0); // num_edges
173  }
174  else
175  output.printf("OFF\n%ld %ld 0\n", num_vertices, num_faces);
176 
177  writeVertices(mesh_group, output, vertex_indices, callback);
178 
179  intx next_index = 0;
180  writeFaces(mesh_group, vertex_indices, output, callback, next_index);
181  }
182 
183  if (write_block_header)
184  bh.calcAndWrite(output);
185  }

189  void readAscii(MeshGroup & mesh_group, BinaryInputStream & in, ReadCallback * callback) const
190  {
191  std::string line = trimWhitespace(in.readLine());
192  while (line.empty() || line[0] == '#')
193  {
194  if (in.hasMore())
195  line = trimWhitespace(in.readLine());
196  else
197  throw Error(std::string(getName()) + ": Unexpected end of input");
198  }
199 
200  std::istringstream counts(line);
201  intx num_vertices, num_faces, num_edges;
202  if (!(counts >> num_vertices >> num_faces >> num_edges))
203  throw Error(std::string(getName()) + ": Could not read mesh statistics on line '" + line + '\'');
204 
205  THEA_CONSOLE << getName() << ": Mesh has " << num_vertices << " vertices, " << num_faces << " faces and " << num_edges
206  << " edges";
207 
208  if (read_opts.skip_empty_meshes && num_vertices <= 0)
209  return;
210 
211  // Create new mesh
212  MeshPtr mesh(new Mesh(std::string(mesh_group.getName()) + "/Mesh"));
213 
214  // Create a builder for the mesh
215  Builder builder(mesh);
216  builder.begin();
217 
218  Array<typename Builder::VertexHandle> vrefs;
219  Array<typename Builder::VertexHandle> face;
220 
221  // Read list of vertices
222  double x, y, z;
223  for (intx v = 0; v < num_vertices; ++v)
224  {
225  std::string line = trimWhitespace(in.readLine());
226  while (line.empty() || line[0] == '#')
227  {
228  if (in.hasMore())
229  line = trimWhitespace(in.readLine());
230  else
231  throw Error(std::string(getName()) + ": Unexpected end of input");
232  }
233 
234  std::istringstream vstr(line);
235  if (!(vstr >> x >> y >> z))
236  throw Error(std::string(getName()) + ": Could not read vertex on line '" + line + '\'');
237 
238  typename Builder::VertexHandle vref = builder.addVertex(Vector3((Real)x, (Real)y, (Real)z),
239  (read_opts.store_vertex_indices ? v : -1));
240  if (callback)
241  callback->vertexRead(mesh.get(), v, vref);
242 
243  vrefs.push_back(vref);
244  }
245 
246  // Read list of faces
247  intx index;
248  intx num_face_vertices;
249  for (intx f = 0; f < num_faces; ++f)
250  {
251  std::string line = trimWhitespace(in.readLine());
252  while (line.empty() || line[0] == '#')
253  {
254  if (in.hasMore())
255  line = trimWhitespace(in.readLine());
256  else
257  throw Error(std::string(getName()) + ": Unexpected end of input");
258  }
259 
260  std::istringstream vstr(line);
261  if (!(vstr >> num_face_vertices))
262  throw Error(std::string(getName()) + ": Could not read number of vertices in face on line '" + line + '\'');
263 
264  if (num_face_vertices > 0)
265  {
266  face.resize(num_face_vertices);
267 
268  bool skip = false;
269  for (intx v = 0; v < num_face_vertices && !skip; ++v)
270  {
271  if (!(vstr >> index))
272  {
273  if (read_opts.strict)
274  throw Error(std::string(getName()) + ": Could not read vertex index on line '" + line + '\'');
275  else
276  {
277  THEA_WARNING << getName() << ": Skipping face, could not read vertex index on line '" << line << '\'';
278  skip = true; break;
279  }
280  }
281 
282  if (index < 0 || index >= (intx)vrefs.size())
283  {
284  if (read_opts.strict)
285  throw Error(getName() + format(": Vertex index %ld out of bounds (#vertices = %ld) on line '%s'",
286  index, (intx)vrefs.size(), line.c_str()));
287  else
288  {
289  THEA_WARNING << getName() << ": Skipping face, vertex index " << index << " out of bounds (#vertices = "
290  << vrefs.size() << ") on line '" << line << '\'';
291  skip = true; break;
292  }
293  }
294 
295  face[(size_t)v] = vrefs[(size_t)index];
296 
297  for (int w = 0; w < v; ++w)
298  if (face[w] == face[v]) // face has repeated vertices
299  {
300  if (read_opts.strict)
301  throw Error(std::string(getName()) + ": Face has repeated vertices on line '" + line + '\'');
302  else
303  {
304  THEA_WARNING << getName() << ": Skipping face with repeated vertices on line '" << line << '\'';
305  skip = true; break;
306  }
307  }
308  }
309 
310  if (!skip)
311  {
312  typename Builder::FaceHandle fref = builder.addFace(face.begin(), face.end(),
313  (read_opts.store_face_indices ? f : -1));
314  if (callback)
315  callback->faceRead(mesh.get(), f, fref);
316  }
317  }
318  }
319 
320  builder.end();
321  mesh_group.addMesh(mesh);
322  }

325  void readBinary(MeshGroup & mesh_group, BinaryInputStream & in, ReadCallback * callback) const
326  {
327  BinaryInputStream::EndiannessScope scope(in, Endianness::BIG); // binary OFF uses big-endian
328 
329  intx num_vertices = (intx)in.readInt32();
330  intx num_faces = (intx)in.readInt32();
331  intx num_edges = (intx)in.readInt32();
332 
333  THEA_CONSOLE << getName() << ": Mesh has " << num_vertices << " vertices, " << num_faces << " faces and " << num_edges
334  << " edges";
335 
336  if (read_opts.skip_empty_meshes && num_vertices <= 0)
337  return;
338 
339  // Create new mesh
340  MeshPtr mesh(new Mesh(std::string(mesh_group.getName()) + "/Mesh"));
341 
342  // Create a builder for the mesh
343  Builder builder(mesh);
344  builder.begin();
345 
346  Array<typename Builder::VertexHandle> vrefs;
347  Array<typename Builder::VertexHandle> face;
348 
349  // Read list of vertices
350  Vector3 vertex;
351  for (intx v = 0; v < num_vertices; ++v)
352  {
353  vertex[0] = (Real)in.readFloat32();
354  vertex[1] = (Real)in.readFloat32();
355  vertex[2] = (Real)in.readFloat32();
356 
357  typename Builder::VertexHandle vref = builder.addVertex(vertex, (read_opts.store_vertex_indices ? v : -1));
358  if (callback)
359  callback->vertexRead(mesh.get(), v, vref);
360 
361  vrefs.push_back(vref);
362  }
363 
364  // Read list of faces
365  for (intx f = 0; f < num_faces; ++f)
366  {
367  int num_face_vertices = (int)in.readInt32();
368  if (num_face_vertices > 0)
369  {
370  face.resize(num_face_vertices);
371 
372  bool skip = false;
373  for (int v = 0; v < num_face_vertices; ++v)
374  {
375  if (skip)
376  {
377  in.skip(4);
378  continue;
379  }
380 
381  intx index = (intx)in.readInt32();
382  if (index < 0 || index >= (intx)vrefs.size())
383  {
384  if (read_opts.strict)
385  throw Error(getName() + format(": Vertex index %ld out of bounds (#vertices = %ld) in face %ld",
386  index, (intx)vrefs.size(), f));
387  else
388  {
389  THEA_WARNING << getName() << ": Skipping face " << f << ", vertex index " << index
390  << " out of bounds (#vertices = " << vrefs.size() << ')';
391  skip = true; break;
392  }
393  }
394 
395  face[v] = vrefs[(size_t)index];
396 
397  for (int w = 0; w < v; ++w)
398  if (face[w] == face[v]) // face has repeated vertices
399  {
400  if (read_opts.strict)
401  throw Error(getName() + format(": Face %ld has repeated vertices", f));
402  else
403  {
404  THEA_WARNING << getName() << ": Skipping face " << f << " with repeated vertices";
405  skip = true; break;
406  }
407  }
408  }
409 
410  int num_color_components = (int)in.readInt32();
411  if (num_color_components > 0)
412  in.skip(4 * num_color_components);
413 
414  if (!skip)
415  {
416  typename Builder::FaceHandle fref = builder.addFace(face.begin(), face.end(),
417  (read_opts.store_face_indices ? f : -1));
418  if (callback)
419  callback->faceRead(mesh.get(), f, fref);
420  }
421  }
422  }
423 
424  builder.end();
425  mesh_group.addMesh(mesh);
426  }

527  template < typename _MeshT, typename std::enable_if< Graphics::IsGeneralMesh<_MeshT>::value
528  || Graphics::IsDCELMesh<_MeshT>::value, int >::type = 0 >
529  void writeFaces(_MeshT const & mesh, VertexIndexMap const & vertex_indices, BinaryOutputStream & output,
530  WriteCallback * callback, intx & next_index) const
531  {
532  for (typename Mesh::FaceConstIterator fi = mesh.facesBegin(); fi != mesh.facesEnd(); ++fi)
533  {
534  typename Mesh::Face const & face = *fi;
535  if (face.numVertices() < 3) continue;
536 
537  if (write_opts.binary)
538  {
539  output.writeInt32((int32)face.numVertices());
540  for (typename Mesh::Face::VertexConstIterator vi = face.verticesBegin(); vi != face.verticesEnd(); ++vi)
541  {
542  typename VertexIndexMap::const_iterator ii = vertex_indices.find(*vi);
543  alwaysAssertM(ii != vertex_indices.end(), std::string(getName()) + ": Vertex index not found");
544 
545  output.writeInt32((int32)ii->second);
546  }
547 
548  output.writeInt32(0); // no color components
549  }
550  else
551  {
552  std::ostringstream os; os << face.numVertices();
553  for (typename Mesh::Face::VertexConstIterator vi = face.verticesBegin(); vi != face.verticesEnd(); ++vi)
554  {
555  typename VertexIndexMap::const_iterator ii = vertex_indices.find(*vi);
556  alwaysAssertM(ii != vertex_indices.end(), std::string(getName()) + ": Vertex index not found");
557 
558  os << ' ' << ii->second;
559  }
560 
561  os << '\n';
562  output.writeBytes((int64)os.str().length(), os.str().data());
563  }
564 
565  if (callback) callback->faceWritten(&mesh, next_index++, &face);
566  }
567  }

570  template < typename _MeshT, typename std::enable_if< Graphics::IsDisplayMesh<_MeshT>::value, int >::type = 0 >
571  void writeFaces(_MeshT const & mesh, VertexIndexMap const & vertex_indices, BinaryOutputStream & output,
572  WriteCallback * callback, intx & next_index) const
573  {
574  typedef std::pair<_MeshT const *, intx> DisplayMeshVRef;
575 
576  for (int type = 0; type < 2; ++type) // 0: triangles, 1: quads
577  {
578  typename Mesh::IndexArray indices = (type == 0 ? mesh.getTriangleIndices() : mesh.getQuadIndices());
579  size_t degree = (type == 0 ? 3 : 4);
580 
581  for (size_t i = 0; i < indices.size(); i += degree)
582  {
583  if (write_opts.binary)
584  {
585  output.writeInt32((int32)degree);
586  for (size_t j = 0; j < degree; ++j)
587  {
588  typename VertexIndexMap::const_iterator ii = vertex_indices.find(DisplayMeshVRef(&mesh, (intx)indices[i + j]));
589  alwaysAssertM(ii != vertex_indices.end(), std::string(getName()) + ": Vertex index not found");
590 
591  output.writeInt32((int32)ii->second);
592  }
593 
594  output.writeInt32(0); // no color components
595  }
596  else
597  {
598  std::ostringstream os; os << degree;
599 
600  for (size_t j = 0; j < degree; ++j)
601  {
602  typename VertexIndexMap::const_iterator ii = vertex_indices.find(DisplayMeshVRef(&mesh, (intx)indices[i + j]));
603  alwaysAssertM(ii != vertex_indices.end(), std::string(getName()) + ": Vertex index not found");
604 
605  os << ' ' << ii->second;
606  }
607 
608  os << '\n';
609  output.writeBytes((int64)os.str().length(), os.str().data());
610  }
611 
612  if (callback)
613  {
614  typename Mesh::Face face(const_cast<Mesh *>(&mesh), degree, (type == 0), (intx)i, 1);
615  callback->faceWritten(&mesh, next_index++, face);
616  }
617  }
618  }
