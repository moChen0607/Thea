19 Quat::Quat(Matrix3 const & rot)
20 {
21  static int const plus1mod3[] = {1, 2, 0};
22 
23  // Find the index of the largest diagonal component. These ? operations hopefully compile to conditional move instructions
24  // instead of branches.
25  int i = (rot(1, 1) > rot(0, 0)) ? 1 : 0;
26  i = (rot(2, 2) > rot(i, i)) ? 2 : i;
27 
28  // Find the indices of the other elements
29  int j = plus1mod3[i];
30  int k = plus1mod3[j];
31 
32  // If we attempted to pre-normalize and trusted the matrix to be perfectly orthonormal, the result would be:
33  //
34  // double c = sqrt((rot(i, i) - (rot(j, j) + rot(k, k))) + 1.0)
35  // v[i] = -c * 0.5
36  // v[j] = -(rot(i, j) + rot(j, i)) * 0.5 / c
37  // v[k] = -(rot(i, k) + rot(k, i)) * 0.5 / c
38  // s = (rot(j, k) - rot(k, j)) * 0.5 / c
39  //
40  // Since we're going to pay the sqrt anyway, we perform a post normalization, which also fixes any poorly normalized input.
41  // Multiply all elements by 2*c in the above, giving nc2 = -c^2
42  double nc2 = ((rot(j, j) + rot(k, k)) - rot(i, i)) - 1.0;
43  v[i] = nc2;
44  s = (rot(j, k) - rot(k, j));
45  v[j] = -(rot(i, j) + rot(j, i));
46  v[k] = -(rot(i, k) + rot(k, i));
47 
48  // We now have the correct result with the wrong magnitude, so normalize it:
49  Real len = std::sqrt(v.squaredNorm() + s * s);
50  if (len > Math::eps<Real>())
51  {
52  v /= len;
53  s /= len;
54  }
55  else
56  {
57  // The quaternion is nearly zero. Make it (0 0 0 1)
58  *this = identity();
59  }
60 }

70 void
71 Quat::toRotationMatrix(Matrix3 & rot) const
72 {
73  // Implementation from Watt and Watt, pg 362
74  // See also http://www.flipcode.com/documents/matrfaq.html#Q54
75 
76  Quat q = normalized();
77 
78  Real xx = 2 * q.x() * q.x();
79  Real xy = 2 * q.x() * q.y();
80  Real xz = 2 * q.x() * q.z();
81  Real xw = 2 * q.x() * q.w();
82 
83  Real yy = 2 * q.y() * q.y();
84  Real yz = 2 * q.y() * q.z();
85  Real yw = 2 * q.y() * q.w();
86 
87  Real zz = 2 * q.z() * q.z();
88  Real zw = 2 * q.z() * q.w();
89 
90  rot << 1 - yy - zz, xy - zw, xz + yw,
91  xy + zw, 1 - xx - zz, yz - xw,
92  xz - yw, yz + xw, 1 - xx - yy;
93 }

105 void
106 Quat::toAxisAngleRotation(Vector3 & axis, double & angle) const
107 {
108  // Decompose the quaternion into an angle and an axis
109  angle = 2 * std::acos(s);
110 
111  axis = v;
112  Real len = std::sqrt(1 - s * s);
113  if (len > Math::eps<Real>())
114  axis /= len;
115 
116  // Reduce the range of the angle
117 
118  if (angle < 0)
119  {
120  angle = -angle;
121  axis = -axis;
122  }
123 
124  while (angle > Math::twoPi())
125  {
126  angle -= Math::twoPi();
127  }
128 
129  if (std::abs(angle) > Math::pi())
130  {
131  angle -= Math::twoPi();
132  }
133 
134  // Make the angle positive.
135  if (angle < 0)
136  {
137  angle = -angle;
138  axis = -axis;
139  }
140 }

142 Quat
143 Quat::slerp(Quat const & target, Real alpha, Real threshold) const
144 {
145  // From: Game Physics -- David Eberly pg 538-540
146  // Modified to include lerp for small angles, which is a common practice.
147  // See also:
148  // http://number-none.com/product/Understanding%20Slerp,%20Then%20Not%20Using%20It/index.html
149 
150  Quat const & quat0 = *this;
151  Quat quat1 = target;
152 
153  debugAssertM(Math::fuzzyEq(quat0.squaredNorm(), (Real)1), "Quat: slerp requires unit quaternions");
154  debugAssertM(Math::fuzzyEq(quat1.squaredNorm(), (Real)1), "Quat: slerp requires unit quaternions");
155 
156  // Angle between quaternion rotations
157  Real cosphi = quat0.dot(quat1);
158  if (cosphi < 0)
159  {
160  // Change the sign and fix the dot product; we need to
161  // loop the other way to get the shortest path
162  quat1 = -quat1;
163  cosphi = -cosphi;
164  }
165 
166  // Returns angle between 0 and pi (assumes both quaternions are unit length)
167  Real phi = std::acos(cosphi);
168  if (phi >= threshold)
169  {
170  // For large angles, slerp
171  Real scale0, scale1;
172  scale0 = std::sin((1 - alpha) * phi);
173  scale1 = std::sin(alpha * phi);
174  return ((quat0 * scale0) + (quat1 * scale1)) / std::sin(phi);
175  }
176  else
177  {
178  // For small angles, linear interpolate
179  return quat0.nlerp(quat1, alpha);
180  }
181 }

191 Quat::operator*(Quat const & other) const
192 {
193  // Following Watt & Watt, page 360
194  Vector3 const & v1 = imag();
195  Vector3 const & v2 = other.imag();
196  Real s1 = s;
197  Real s2 = other.s;
198  return Quat(s1 * v2 + s2 * v1 + v1.cross(v2), s1 * s2 - v1.dot(v2));

201 Quat
202 Quat::unitRandom()
203 {
204  // From "Uniform Random Rotations", Ken Shoemake, Graphics Gems III
205  Real x0 = Random::common().uniform01();
206  Real r1 = std::sqrt(1 - x0),
207  r2 = std::sqrt(x0);
208  Real t1 = (Real)Math::twoPi() * Random::common().uniform01();
209  Real t2 = (Real)Math::twoPi() * Random::common().uniform01();
210  Real c1 = std::cos(t1),
211  s1 = std::sin(t1);
212  Real c2 = std::cos(t2),
213  s2 = std::sin(t2);
214  return Quat(s1 * r1, c1 * r1, s2 * r2, c2 * r2);
215 }
