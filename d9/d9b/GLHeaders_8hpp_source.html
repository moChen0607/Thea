115 inline GLubyte const *
116 theaGLErrorString(GLuint error)
117 {
118  switch (error)
119  {
120 #define THEA_GL_ERROR_STRING_BRANCH(p) case(p): return (GLubyte const *)#p;
121  THEA_GL_ERROR_STRING_BRANCH(GL_NO_ERROR)
122  THEA_GL_ERROR_STRING_BRANCH(GL_INVALID_ENUM)
123  THEA_GL_ERROR_STRING_BRANCH(GL_INVALID_VALUE)
124  THEA_GL_ERROR_STRING_BRANCH(GL_INVALID_OPERATION)
125  THEA_GL_ERROR_STRING_BRANCH(GL_STACK_OVERFLOW)
126  THEA_GL_ERROR_STRING_BRANCH(GL_STACK_UNDERFLOW)
127  THEA_GL_ERROR_STRING_BRANCH(GL_OUT_OF_MEMORY)
128  THEA_GL_ERROR_STRING_BRANCH(GL_TABLE_TOO_LARGE)
129  default: break;
130 #undef THEA_GL_ERROR_STRING_BRANCH
131  }
132 
133  if ((error >= GLU_NURBS_ERROR1) && (error <= GLU_NURBS_ERROR37))
134  return (GLubyte const *)"NURBS error";
135  if ((error >= GLU_TESS_ERROR1) && (error <= GLU_TESS_ERROR8))
136  return (GLubyte const *)"tessellation error";
137 
138  return (GLubyte const *)"unknown error";
139 }
