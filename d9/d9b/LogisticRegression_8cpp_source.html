34 void
35 LogisticRegression::addObservation(double const * x, double y)
36 {
37  // y = 1 / (1 + exp(-a - b.x))
38  // => log(1 / y - 1) = -a - b.x
39  // => a + b.x = log(1 / y - 1)
40 
41  alwaysAssertM(x, "LogisticRegression: Observed vector x cannot be null");
42 
43  llsq_coeffs.push_back(1);
44  llsq_coeffs.insert(llsq_coeffs.end(), x, x + ndims);
45 
46  // c = log(1 / y - 1)
47  // => y = 1 / (1 + exp(c))
48  static double const MIN_Y = 1.0 / std::min(std::numeric_limits<double>::max(), 1.0 + 1.0e+30);
49  static double const MAX_Y = 1.0 / (1.0 + std::max(std::numeric_limits<double>::epsilon(), 1.0e-30));
50 
51  // std::cout << "MIN_Y = " << MIN_Y << ", MAX_Y = " << MAX_Y << std::endl;
52 
53  y = Math::clamp(y, MIN_Y, MAX_Y);
54  llsq_consts.push_back(std::log(1.0 / y - 1.0));
55 
56  // std::cout << "LogisticRegression: Adding LLSQ objective [";
57  // for (size_t i = llsq_coeffs.size() - (size_t)ndims - 1; i < llsq_coeffs.size(); ++i) std::cout << ' ' << llsq_coeffs[i];
58  // std::cout << " ] = " << llsq_consts.back() << std::endl;
59 }

68 bool
69 LogisticRegression::solve(double tolerance)
70 {
71  if (numObservations() <= 0)
72  {
73  THEA_WARNING << "LogisticRegression: Solving empty problem";
74  has_solution = true;
75  solution.resize(0);
76  }
77  else
78  {
79  StdLinearSolver llsq(StdLinearSolver::Method::DEFAULT, StdLinearSolver::Constraint::UNCONSTRAINED);
80  llsq.setTolerance(tolerance);
81 
82  typedef Eigen::Map< MatrixX<double, MatrixLayout::ROW_MAJOR> > M;
83  M a(&llsq_coeffs[0], numObservations(), ndims);
84  has_solution = llsq.solve(MatrixWrapper<M>(&a), &llsq_consts[0]);
85 
86  if (has_solution)
87  solution = VectorXdMap(const_cast<double *>(llsq.getSolution()), ndims);
88  else
89  solution.resize(0);
90  }
91 
92  return has_solution;
93 }
