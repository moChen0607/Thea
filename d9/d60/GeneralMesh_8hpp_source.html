592  void collapseEdge(Edge * edge, int endpoint_to_preserve = 0)
593  {
594  if (!edge)
595  return;
596 
597  if (edge->isSelfLoop())
598  {
599  THEA_WARNING << getName() << ": Edge is self-loop";
600  return;
601  }
602 
603  debugAssertM(endpoint_to_preserve == 0 || endpoint_to_preserve == 1,
604  getNameStr() + ": Endpoint to preserve during edge collapse must be indexed by 0 or 1");
605 
606  Vertex * vertex_to_preserve = edge->getEndpoint(endpoint_to_preserve);
607  Vertex * vertex_to_remove = edge->getEndpoint(1 - endpoint_to_preserve);
608 
609  // Remove all references to this edge, and the vertex to remove, from all adjacent faces
610  bool preserve_edge = false;
611  for (auto fi = edge->facesBegin(); fi != edge->facesEnd(); )
612  {
613  Face * face = *fi;
614  auto fei = face->edgesBegin();
615  auto fvi = face->verticesBegin();
616 
617  bool preserve_edge_ref_to_face = false;
618  while (fei != face->edgesEnd())
619  {
620  if (*fei == edge)
621  {
622  if (*fvi == vertex_to_remove)
623  {
624  fei = face->removeEdge(fei);
625  fvi = face->removeVertex(fvi);
626  }
627  else
628  {
629  // We expect the next vertex to be the one to remove. We'll have to advance the vertex iterator, delete the
630  // vertex, and then advance the edge iterator as well
631  auto next = fvi; ++next;
632  if (next == face->verticesEnd())
633  next = face->verticesBegin();
634 
635  if (*next == vertex_to_remove)
636  {
637  fei = face->removeEdge(fei);
638  fvi = face->removeVertex(next);
639 
640  // Now we must skip over the next edge as well, else edge and vertex iterators will be out of sync
641  if (fei != face->edgesEnd())
642  {
643  if (*fei == edge) // somehow this has happened
644  {
645  THEA_WARNING << getName() << ": Face has repeated edge";
646 
647  preserve_edge_ref_to_face = true;
648  }
649 
650  ++fei;
651  }
652  }
653  else
654  {
655  THEA_WARNING << getName() << ": Edge does not correspond to pair of consecutive vertices";
656 
657  preserve_edge_ref_to_face = true;
658 
659  ++fei;
660  ++fvi;
661  }
662  }
663  }
664  else
665  {
666  ++fei;
667  ++fvi;
668  }
669  }
670 
671  if (!preserve_edge_ref_to_face)
672  fi = edge->removeFace(fi);
673  else
674  {
675  preserve_edge = true;
676  ++fi;
677  }
678  }
679 
680  // Update references of elements adjacent to the vertex that's going to be removed
681  for (auto ei = vertex_to_remove->edgesBegin(); ei != vertex_to_remove->edgesEnd(); ++ei)
682  {
683  // We do the replacement even for the edge that is getting collapsed, since if a face repeats this edge it is going to
684  // be present in the output (as a self-loop)
685 
686  (*ei)->replaceVertex(vertex_to_remove, vertex_to_preserve);
687 
688  if (!(*ei)->hasEndpoint(vertex_to_preserve))
689  vertex_to_preserve->addEdge(*ei);
690  }
691 
692  for (auto fi = vertex_to_remove->facesBegin(); fi != vertex_to_remove->facesEnd(); ++fi)
693  {
694  (*fi)->replaceVertex(vertex_to_remove, vertex_to_preserve);
695 
696  if (!vertex_to_preserve->hasIncidentFace(*fi))
697  vertex_to_preserve->addFace(*fi);
698  }
699 
700  // Remove references to this edge from the vertex that will be preserved (the other one will have its references deleted
701  // wholesale later)
702  if (!preserve_edge)
703  vertex_to_preserve->removeEdge(edge);
704 
705  vertex_to_remove->edges.clear();
706  vertex_to_remove->faces.clear();
707 
708  invalidateGPUBuffers();
709  }

718  Edge * splitEdge(Edge * edge, Vertex * vertex)
719  {
720  if (!edge)
721  return nullptr;
722 
723  if (edge->isSelfLoop())
724  {
725  THEA_DEBUG << getName() << ": Can't split self-loop edge";
726  return nullptr;
727  }
728 
729  if (edge->hasEndpoint(vertex))
730  {
731  THEA_DEBUG << getName() << ": Can't split edge at existing endpoint";
732  return nullptr;
733  }
734 
735  for (auto efi = edge->facesBegin(); efi != edge->facesEnd(); ++efi)
736  if ((*efi)->hasVertex(vertex))
737  {
738  THEA_DEBUG << getName() << ": Can't split edge at vertex on the same face";
739  return nullptr;
740  }
741 
742  Vertex * old_e1 = edge->getEndpoint(1);
743  edge->setEndpoint(1, vertex);
744  edges.push_back(Edge(vertex, old_e1));
745  Edge * new_edge = &edges.back();
746 
747  vertex->addEdge(edge);
748  vertex->addEdge(new_edge);
749 
750  old_e1->removeEdge(edge);
751  old_e1->addEdge(new_edge);
752 
753  for (auto fi = edge->facesBegin(); fi != edge->facesEnd(); ++fi)
754  {
755  new_edge->addFace(*fi);
756 
757  if (!vertex->hasIncidentFace(*fi))
758  vertex->addFace(*fi);
759  }
760 
761  // Now everything's ok except except the references in the faces
762  for (auto fi = edge->facesBegin(); fi != edge->facesEnd(); ++fi)
763  {
764  Face * face = *fi;
765 
766  auto ei = face->edgesBegin();
767  auto vi = face->verticesBegin();
768 
769  while (ei != face->edgesEnd())
770  {
771  if (*ei == edge)
772  {
773  // Insert before or after?
774  if (*vi == old_e1) // sequence is [v1, v0], insert before
775  {
776  face->edges.insert(ei, new_edge);
777  }
778  else // sequence is [v0, v1], insert after
779  {
780  auto next_ei = ei; ++next_ei;
781  face->edges.insert(next_ei, new_edge);
782  }
783 
784  // Vertex goes in the middle, i.e. always after the current vertex
785  auto next_vi = vi; ++next_vi;
786  face->vertices.insert(next_vi, vertex);
787 
788  break;
789  }
790  else
791  {
792  ++ei;
793  ++vi;
794  }
795  }
796  }
797 
798  invalidateGPUBuffers();
799  return new_edge;
800  }

803  Vertex * splitEdge(Edge * edge, Vector3 const & p)
804  {
805  Real s = (p - edge->getEndpoint(1)->getPosition()).norm();
806  Real t = (p - edge->getEndpoint(0)->getPosition()).norm();
807  Real sum = s + t;
808  s /= sum;
809  t /= sum;
810  Vector3 n = s * edge->getEndpoint(0)->getNormal() + t * edge->getEndpoint(1)->getNormal();
811  Vertex * new_vx = addVertex(p, &n);
812 
813  if (!splitEdge(edge, new_vx)) // should generally never happen
814  {
815  vertices.pop_back(); // remove the vertex we just added
816  return nullptr;
817  }
818 
819  invalidateGPUBuffers();
820  return new_vx;
821  }

833  {

836  {

842  {
843  Face * face = *efi;
844 
845  auto fei = face->edgesBegin();
846  auto fvi = face->verticesBegin();
847  while (fei != face->edgesEnd())
848  {

1296  bool isManifoldVertex(Vertex const * vertex) const
1297  {
1298  // Check if the vertex is isolated
1299  if (vertex->numEdges() < 2)
1300  return false;
1301 
1302  // The incident edges must form a continuous sequence around the vertex
1303 
1304  // Find an edge to start from. If the vertex is a boundary vertex, we *must* start from a boundary edge
1305  Edge const * first_edge = *vertex->edgesBegin();
1306  for (auto vei = vertex->edgesBegin(); vei != vertex->edgesEnd(); ++vei)
1307  {
1308  (*vei)->clearAllInternalBits();
1309

1312  }
1313 
1314  // Check if the edge is isolated
1315  if (first_edge->numFaces() <= 0)
1316  return false;
1317

1319  Edge const * edge = first_edge;
1320  Face const * face = edge->isBoundaryEdge() ? nullptr : *edge->facesBegin(); // if starting from a boundary edge, initial
1321  // "face" is empty space before first edge
1322  intx num_visited_edges = 0;
1323  while (true)
1324  {
1325  if (++num_visited_edges >= vertex->numEdges())
1326  break;
1327 
1328  // Step around the vertex
1329  Edge const * next_edge = nullptr;
1330  Face const * next_face = nullptr;
1331  if (!getSuccessor(vertex, face, edge, &next_face, &next_edge))
1332  break;
1333 
1334  // If we've already visited the successor edge, we've looped round too early
1335  if (next_edge->areInternalBitsSet(0xFF))
1336  return false;
1337 
1338  // Mark the current edge as visited
1339  edge->setInternalBits(0xFF, true);
1340 
1341  // Update the pointers
1342  edge = next_edge;
1343  face = next_face;
1344  }

1355  bool replaceEdge(Edge * old_edge, Edge * new_edge)
1356  {
1357  if (old_edge == new_edge)
1358  return false;
1359 
1360  Vertex * old_v[2] = { old_edge->getEndpoint(0), old_edge->getEndpoint(1) };
1361  Vertex * new_v[2] = { new_edge->getEndpoint(0), new_edge->getEndpoint(1) };
1362 
1363  if (old_v[0] == old_v[1]) // can't handle degenerate cases
1364  {
1365  THEA_DEBUG << getName() << ": Can't replace a self-loop";
1366  return false;
1367  }
1368 
1369  if (new_v[0] == new_v[1]) // can't handle degenerate cases
1370  {
1371  THEA_DEBUG << getName() << ": Can't replace with a self-loop";
1372  return false;
1373  }
1374 
1375  Edge const * e00 = old_v[0]->getEdgeTo(new_v[0]);
1376  Edge const * e01 = old_v[0]->getEdgeTo(new_v[1]);
1377  Edge const * e10 = old_v[1]->getEdgeTo(new_v[0]);
1378  Edge const * e11 = old_v[1]->getEdgeTo(new_v[1]);
1379  if ((e00 && e00 != old_edge && e00 != new_edge)
1380  || (e01 && e01 != old_edge && e01 != new_edge)
1381  || (e10 && e10 != old_edge && e10 != new_edge)
1382  || (e11 && e11 != old_edge && e11 != new_edge))
1383  {
1384  THEA_DEBUG << getName() << ": Can't replace an edge with another to which it has a connecting edge";
1385  return false;
1386  }
1387 
1388  for (auto efi = old_edge->facesBegin(); efi != old_edge->facesEnd(); ++efi)
1389  if (new_edge->hasIncidentFace(*efi))
1390  {
1391  THEA_DEBUG << getName() << ": Can't replace an edge with another on the same face";
1392  return false;
1393  }
1394 
1395  bool swap_endpts = false;
1396  if (old_v[0] == new_v[0] || old_v[1] == new_v[1])
1397  swap_endpts = false;
1398  else if (old_v[0] == new_v[1] || old_v[1] == new_v[0])
1399  swap_endpts = true;
1400  else
1401  {
1402  Real err0 = (old_v[0]->getPosition() - new_v[0]->getPosition()).squaredNorm();
1403  Real err1 = (old_v[0]->getPosition() - new_v[1]->getPosition()).squaredNorm();
1404  swap_endpts = (err1 < err0);
1405  }
1406 
1407  if (swap_endpts)
1408  {
1409  Vertex * tmp = new_v[0];
1410  new_v[0] = new_v[1];
1411  new_v[1] = tmp;
1412  }
1413 
1414  // Replace the endpoints
1415  for (int i = 0; i < 2; ++i)
1416  if (old_v[i] != new_v[i])
1417  {
1418  replaceVertex(old_v[i], new_v[i]);
1419  }
1420 
1421  // Now there is a double edge, which we will remove
1422  new_v[0]->removeEdge(old_edge);
1423  new_v[1]->removeEdge(old_edge);
1424  for (auto fi = old_edge->facesBegin(); fi != old_edge->facesEnd(); ++fi)
1425  {
1426  (*fi)->replaceEdge(old_edge, new_edge);
1427  if (!new_edge->hasIncidentFace(*fi))
1428  new_edge->addFace(*fi);
1429  }
1430 
1431  old_edge->faces.clear();
1432 
1433  invalidateGPUBuffers();
1434  return true;
1435  }
