18 // Shared library support. See http://gcc.gnu.org/wiki/Visibility . Quoting loosely from that page, and assuming M is a library-
19 // specific prefix:
20 //
21 // - If M_DLL and M_DLL_EXPORTS are defined, we are building our library as a DLL and symbols should be exported. Something
22 // ending with _EXPORTS is defined by MSVC by default in all projects.
23 //
24 // - If M_DLL_EXPORTS is not defined, we are importing our library and symbols should be imported.
25 //
26 // - If we're building with GCC and __GNUC__ >= 4 , then GCC supports the new features.
27 //
28 // - For every non-templated non-static function definition in your library (both headers and source files), decide if it is
29 // publicly used or internally used:
30 //
31 // - If it is publicly used, mark with M_API like this: extern M_API PublicFunc()
32 //
33 // - If it is only internally used, mark with M_DLL_LOCAL like this: extern M_DLL_LOCAL PublicFunc(). Remember, static
34 // functions need no demarcation, nor does anything which is templated.
35 //
36 // - For every non-templated class definition in your library (both headers and source files), decide if it is publicly used or
37 // internally used:
38 //
39 // - If it is publicly used, mark with M_API like this: class M_API PublicClass
40 //
41 // - If it is only internally used, mark with M_DLL_LOCAL like this: class M_DLL_LOCAL PublicClass
42 //
43 // - Individual member functions of an exported class that are not part of the interface, in particular ones which are private,
44 // and are not used by friend code, should be marked individually with M_DLL_LOCAL.
45 //
46 // - Remember to test your library thoroughly afterwards, including that all exceptions correctly traverse shared object
47 // boundaries.
48 //
49 #ifdef _MSC_VER // should be WIN32?
50 # define THEA_GL_IMPORT __declspec(dllimport)
51 # define THEA_GL_EXPORT __declspec(dllexport)
52 # define THEA_GL_DLL_LOCAL
53 # define THEA_GL_DLL_PUBLIC
54 #else
55 # if (defined __GNUC__ && __GNUC__ >= 4)
56 # define THEA_GL_IMPORT __attribute__ ((visibility("default")))
57 # define THEA_GL_EXPORT __attribute__ ((visibility("default")))
58 # define THEA_GL_DLL_LOCAL __attribute__ ((visibility("hidden")))
59 # define THEA_GL_DLL_PUBLIC __attribute__ ((visibility("default")))
60 # else
61 # define THEA_GL_IMPORT
62 # define THEA_GL_EXPORT
63 # define THEA_GL_DLL_LOCAL
64 # define THEA_GL_DLL_PUBLIC
65 # endif
66 #endif
