30  Array<int> in_triangles;
31  Polygon3 poly;
32  Array<intx> tri_indices;
33  for (size_t i = 0; i < in_faces.size(); ++i)
34  {
35  poly.clear();
36  for (size_t j = 0; j < in_faces[i].size(); ++j)
37  poly.addVertex(in_vertices[in_faces[i][j]]);
38 
39  poly.triangulate(tri_indices);
40  for (size_t j = 0; j < tri_indices.size(); ++j)
41  in_triangles.push_back((int)in_faces[i][tri_indices[j]]);
42  }
43 
44  // Convert the set of input vertices to a packed sequence of floats in groups of 3
45  Array<float> flat_in_vertices(3 * in_vertices.size());
46  for (size_t i = 0, i3 = 0; i < in_vertices.size(); ++i, i3 += 3)

50  flat_in_vertices[i3 + 2] = in_vertices[i].z();
51  }

53  // Call the OpenDX function to convert the input mesh to a collection of oriented manifolds
54  int num_vertices = (int)in_vertices.size(); // number of input vertices
55  int num_triangles = (int)(in_triangles.size() / 3); // number of input vertices
56  float * in_v = const_cast<float *>(&flat_in_vertices[0]); // array of input vertices

64  int new_num_edges = 0; // number of generated edges

86 #ifdef THEA_DEBUG_BUILD
