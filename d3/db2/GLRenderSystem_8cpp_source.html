64 GLRenderSystem::GLRenderSystem(char const * name_)
65 : name(name_), current_framebuffer(nullptr), current_shader(nullptr)

69  if (!THEA_GL_SUPPORTS(ARB_texture_non_power_of_two))

71 
72  if (!THEA_GL_SUPPORTS(ARB_vertex_buffer_object))
73  THEA_WARNING << getName() << ": Vertex/index buffers in GPU memory are not supported";

77 
78  if (!THEA_GL_SUPPORTS(EXT_framebuffer_object))

80 
81  glDisable(GL_LIGHTING);
82  setCullFace(CullFace::NONE);

368 void
369 GLRenderSystem::setTexture(int32 texunit, Texture * texture)
370 {
371  if (THEA_GL_SUPPORTS(ARB_multitexture))
372  glActiveTextureARB(GL_TEXTURE0_ARB + texunit);
373  else if (texunit != 0)
374  throw Error(std::string(getName()) + ": Non-zero texture unit specified but multitexturing is not supported by OpenGL");
375 
376  if (texture)
377  {
378  GLTexture * gltex = dynamic_cast<GLTexture *>(texture);
379  debugAssertM(gltex, std::string(getName()) + ": Attempt to use a non-OpenGL texture with a GL rendersystem");
380 
381  GLenum target = gltex->getGLTarget();
382  GLuint id = gltex->getGLID();
383 
384  glEnable(target);
385  glBindTexture(target, id);
386  }
387  else
388  {
389  // Disable the texture unit
390  glDisable(GL_TEXTURE_1D);
391  glDisable(GL_TEXTURE_2D);
392  glDisable(GL_TEXTURE_3D);
393  glDisable(GL_TEXTURE_CUBE_MAP_ARB);
394  glDisable(GL_TEXTURE_RECTANGLE_ARB);
395  }
396 
397  THEA_CHECK_GL_OK
398 }
