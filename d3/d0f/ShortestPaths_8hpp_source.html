197 template <typename CallbackT>
198 void
199 ShortestPaths<GraphT>::dijkstraWithCallback(Graph & graph, VertexHandle src, CallbackT callback, double limit,
200  UnorderedMap<VertexHandle, double> const * src_region, bool include_unreachable)
201 {
202  if (graph.numVertices() <= 0)
203  return;
204 
205 #ifdef THEA_SHORTEST_PATHS_TIMER
206  Stopwatch timer;
207  timer.tick();
208 #endif
209 
210  // Sanity checks
211  typedef UnorderedMap<VertexHandle, double> DistanceMap;
212  bool has_src_region = (src_region && !src_region->empty());
213  if (has_src_region)
214  {
215  for (typename DistanceMap::const_iterator di = src_region->begin(); di != src_region->end(); ++di)
216  {
217  alwaysAssertM(di->second >= 0, "ShortestPaths: Dijkstra's algorithm requires non-negative distances");
218  }
219  }
220 
221  // Deallocate scratch space only if absolutely necessary. With this policy, the hash table will have at most 2.5 times the
222  // number of vertices (1.5 earlier + 1 on this call) and repeated calls with the same graph will not cause any reallocations.
223  intx num_verts = graph.numVertices();
224  if (scratch.size() > 1.5 * num_verts)
225  scratch = Scratch(); // make sure the memory is actually freed, better than calling scratch.clear()
226 
227  // Create a scratch element representing negative infinity, for the Fibonacci heap
228  static double const INF_DIST = (std::numeric_limits<double>::has_infinity ? std::numeric_limits<double>::infinity()
229  : std::numeric_limits<double>::max());
230  ScratchElement NEG_INF;
231  NEG_INF.dist = -INF_DIST;
232 
233  // Initialize the heap
234  fibheap * dijkstra_queue = fh_makeheap();
235  fh_setcmp(dijkstra_queue, compareDistances);
236  fh_setneginf(dijkstra_queue, &NEG_INF);
237 
238  // Initialize the vertex scratch data
239  for (typename GraphT::VertexIterator vi = graph.verticesBegin(); vi != graph.verticesEnd(); ++vi)
240  {
241  VertexHandle vertex = graph.getVertex(vi);
242  ScratchElement & data = scratch[vertex]; // no new allocs if the element already exists
243 
244  data.vertex = vertex;
245  data.has_pred = false;
246 
247  if (has_src_region)
248  {
249  typename DistanceMap::const_iterator existing = src_region->find(vertex);
250  if (existing != src_region->end())
251  {
252  data.dist = existing->second;
253  data.flag = GREY;
254  data.heap_elem = fh_insert(dijkstra_queue, (void *)&data);
255  continue;
256  }
257  }
258  else
259  {
260  if (vertex == src)
261  {
262  data.dist = 0;
263  data.flag = GREY;
264  data.heap_elem = fh_insert(dijkstra_queue, (void *)&data);
265  continue;
266  }
267  }
268 
269  data.dist = INF_DIST;
270  data.flag = WHITE;
271  }
272 
273 #ifdef THEA_SHORTEST_PATHS_TIMER
274  timer.tock();
275  THEA_CONSOLE << "ShortestPaths: Setting up scratch data and initial heap took " << 1000 * timer.elapsedTime() << "ms";
276  timer.tick();
277 #endif
278 
279 #ifdef THEA_SHORTEST_PATHS_DO_STATS
280  intx num_enqueued = (has_src_region ? (intx)src_region->size() : 1);
281  intx num_iters = 0;
282 #endif
283 
284  ScratchElement tmp_data;
285  while (true)
286  {
287  ScratchElement * data = (ScratchElement *)fh_extractmin(dijkstra_queue);
288  if (!data)
289  break;
290 
291  for (typename GraphT::NeighborIterator ni = graph.neighborsBegin(data->vertex), nbrs_end = graph.neighborsEnd(data->vertex);
292  ni != nbrs_end; ++ni)
293  {
294  VertexHandle nbr = graph.getVertex(ni);
295  double nbr_dist = graph.distance(data->vertex, ni);
296 
297  typename Scratch::iterator nbr_loc = scratch.find(nbr);
298  debugAssertM(nbr_loc != scratch.end(), "ShortestPaths: No scratch entry found for neighboring vertex");
299 
300  ScratchElement * nbr_data = &nbr_loc->second;
301  if (nbr_data->flag == BLACK) // this should not be necessary, but let's do it just to be safe
302  continue;
303 
304  double test_dist = nbr_dist + data->dist;
305  if (test_dist < nbr_data->dist)
306  {
307  // Update the predecessor
308  nbr_data->has_pred = true;
309  nbr_data->pred = data->vertex;
310 
311  // Update the shortest-path distance and reorder the heap
312  if (nbr_data->flag == WHITE)
313  {
314  nbr_data->dist = test_dist;
315  nbr_data->flag = GREY;
316  nbr_data->heap_elem = fh_insert(dijkstra_queue, (void *)nbr_data);
317 
318 #ifdef THEA_SHORTEST_PATHS_DO_STATS
319  num_enqueued++;
320 #endif
321  }
322  else if (nbr_data->flag == GREY)
323  {
324  // Hack the element to a point to a different sample, so old and new data have different pointer values (and hence
325  // aren't seen as identical) but the same distance (so the heap ordering is still valid)
326  tmp_data.dist = nbr_data->dist;
327  fhe_setdata(nbr_data->heap_elem, (void *)&tmp_data);
328 
329  nbr_data->dist = test_dist;
330  fh_replacedata(dijkstra_queue, nbr_data->heap_elem, (void *)nbr_data);
331  }
332  }
333  }
334 
335 #ifdef THEA_SHORTEST_PATHS_DO_STATS
336  num_iters++;
337 #endif
338 
339  data->flag = BLACK;
340 
341  if (limit >= 0 && data->dist > limit) // all remaining distances will be greater than this
342  break;
343 
344  if (callback(data->vertex, data->dist, data->has_pred, data->pred))
345  break;
346  }
347 
348  if (include_unreachable)
349  {
350  for (typename GraphT::VertexIterator vi = graph.verticesBegin(); vi != graph.verticesEnd(); ++vi)
351  {
352  VertexHandle vertex = graph.getVertex(vi);
353  ScratchElement & data = scratch[vertex]; // no new allocs if the element already exists
354  if (data.flag != BLACK)
355  {
356  if (callback(data.vertex, -1, false, VertexHandle()))
357  break;
358  }
359  }
360  }
361 
362  fh_deleteheap(dijkstra_queue);
363 
364 #ifdef THEA_SHORTEST_PATHS_TIMER
365  timer.tock();
366  THEA_CONSOLE << "ShortestPaths: Dijkstra iterations took " << 1000 * timer.elapsedTime() << "ms";
367 #endif
368 
369 #ifdef THEA_SHORTEST_PATHS_DO_STATS
370  THEA_CONSOLE << "ShortestPaths: Enqueued " << num_enqueued << " samples after " << num_iters << " Dijkstra iterations";
371 #endif
372 }
