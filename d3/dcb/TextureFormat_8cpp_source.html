524 TextureFormat const *
525 TextureFormat::fromImageType(AbstractImage::Type type, bool is_depth)
526 {
527  if (is_depth)
528  {
529  switch (type)
530  {
531  case AbstractImage::Type::LUMINANCE_16U : return DEPTH16();
532  case AbstractImage::Type::LUMINANCE_32U : return DEPTH32();
533  case AbstractImage::Type::LUMINANCE_32F : return DEPTH32F();
534  default: throw Error("TextureFormat: No supported depth texture format corresponds to the specified image format");
535  }
536  }
537 
538  enum { COLOR_ORDER_RGB, COLOR_ORDER_BGR } color_order;
539  color_order = (AbstractImage::Channel::RED < AbstractImage::Channel::BLUE ? COLOR_ORDER_RGB : COLOR_ORDER_BGR);
540 
541  switch (type)
542  {
543  case AbstractImage::Type::LUMINANCE_8U : return L8();
544  case AbstractImage::Type::LUMINANCE_16U : return L16();
545  case AbstractImage::Type::LUMINANCE_32F : return L32F();
546  case AbstractImage::Type::RGB_8U : return color_order == COLOR_ORDER_RGB ? RGB8() : BGR8();
547  case AbstractImage::Type::RGBA_8U : return color_order == COLOR_ORDER_RGB ? RGBA8() : BGRA8();
548  case AbstractImage::Type::RGB_16U : return color_order == COLOR_ORDER_RGB ? RGB16() : BGR16();
549  case AbstractImage::Type::RGBA_16U : return color_order == COLOR_ORDER_RGB ? RGBA16() : BGRA16();
550  case AbstractImage::Type::RGB_32F : return color_order == COLOR_ORDER_RGB ? RGB32F() : BGR32F();
551  case AbstractImage::Type::RGBA_32F : return color_order == COLOR_ORDER_RGB ? RGBA32F() : BGRA32F();
552  default: throw Error("TextureFormat: No supported texture format corresponds to the specified image format");
553  }
554 }
