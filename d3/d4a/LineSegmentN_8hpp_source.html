89  VectorT closestPoint(VectorT const & p) const
90  {
91  // Taken from G3D::LineSegment
92 
93  T d2 = direction.squaredNorm();
94  if (Math::fuzzyEq(d2, static_cast<T>(0)))
95  return point;
96 
97  // The vector from the end of the segment to the point in question.
98  VectorT v(p - point);
99 
100  // Projection of v onto the line segment scaled by the length of the segment.
101  T t = direction.dot(v);
102 
103  // Avoid some square roots. Derivation:
104  // t / direction.norm() <= direction.norm()
105  // t <= direction.squaredNorm()
106 
107  if (t >= 0 && t <= d2)
108  {
109  // The point falls within the segment. Normalize direction, divide t by the length of direction.
110  return point + (t / d2) * direction;
111  }
112  else
113  {
114  // The point does not fall within the segment; see which end is closer.
115 
116  // Distance from 0, squared
117  T d0_squared = v.squaredNorm();
118 
119  // Distance from 1, squared
120  T d1_squared = (v - direction).squaredNorm();
121 
122  if (d0_squared < d1_squared) // point 0 is closer
123  return point;
124  else // point 1 is closer
125  return point + direction;
126  }
127  }

227 template <int N, typename T>
228 T
229 closestPtSegmentSegment(Vector<N, T> const & p1, Vector<N, T> const & q1, bool is_line1,
230  Vector<N, T> const & p2, Vector<N, T> const & q2, bool is_line2,
231  T & s, T & t, Vector<N, T> & c1, Vector<N, T> & c2)
232 {
233  typedef Vector<N, T> VectorT;
234 
235  VectorT d1 = q1 - p1; // Direction vector of segment S1
236  VectorT d2 = q2 - p2; // Direction vector of segment S2
237  VectorT r = p1 - p2;
238  T a = d1.squaredNorm(); // Squared length of segment S1, always nonnegative
239  T e = d2.squaredNorm(); // Squared length of segment S2, always nonnegative
240  T f = d2.dot(r);
241 
242  // Check if either or both segments degenerate into points
243  if (Math::fuzzyEq(a, static_cast<T>(0)))
244  {
245  if (Math::fuzzyEq(e, static_cast<T>(0)))
246  {
247  // Both segments degenerate into points
248  s = t = 0;
249  c1 = p1;
250  c2 = p2;
251  return (c1 - c2).dot(c1 - c2);
252  }
253  else
254  {
255  // First segment degenerates into a point
256  s = 0;
257  t = f / e; // s = 0 => t = (b*s + f) / e = f / e
258 
259  if (!is_line2)
260  t = Math::clamp(t, static_cast<T>(0), static_cast<T>(1));
261  }
262  }
263  else
264  {
265  T c = d1.dot(r);
266  if (Math::fuzzyEq(e, static_cast<T>(0)))
267  {
268  // Second segment degenerates into a point
269  t = 0;
270  s = -c / a;
271 
272  if (!is_line1)
273  s = Math::clamp(s, static_cast<T>(0), static_cast<T>(1)); // t = 0 => s = (b*t - c) / a = -c / a
274  }
275  else
276  {
277  // The general nondegenerate case starts here
278  T b = d1.dot(d2);
279  T denom = a * e - b * b; // Always nonnegative
280 
281  // If segments not parallel, compute closest point on L1 to L2, and clamp to segment S1. Else pick arbitrary s (here 0)
282  if (denom != 0)
283  {
284  s = (b * f - c * e) / denom;
285 
286  if (!is_line1)
287  s = Math::clamp(s, static_cast<T>(0), static_cast<T>(1));
288  }
289  else
290  s = 0;
291 
292  // Compute point on L2 closest to S1(s) using t = Dot((P1+D1*s)-P2,D2) / Dot(D2,D2) = (b*s + f) / e
293  t = (b * s + f) / e;
294 
295  if (!is_line2)
296  {
297  // If t in [0,1] done. Else clamp t, recompute s for the new value of t using
298  // s = Dot((P2+D2*t)-P1,D1) / Dot(D1,D1)= (t*b - c) / a and clamp s to [0, 1]
299  if (t < 0)
300  {
301  t = 0;
302  s = -c / a;
303 
304  if (!is_line1)
305  s = Math::clamp(s, static_cast<T>(0), static_cast<T>(1));
306  }
307  else if (t > 1)
308  {
309  t = 1;
310  s = (b - c) / a;
311 
312  if (!is_line1)
313  s = Math::clamp(s, static_cast<T>(0), static_cast<T>(1));
314  }
315  }
316  }
317  }
318 
319  c1 = p1 + s * d1;
320  c2 = p2 + t * d2;
321  return (c1 - c2).dot(c1 - c2);
322 }
