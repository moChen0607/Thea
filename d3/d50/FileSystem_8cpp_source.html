68 bool
69 FileSystem::readWholeFile(std::string const & path, std::string & ret)
70 {
71  if (!fileExists(path))
72  {
73  THEA_ERROR << "FileSystem: File '" << path << "' not found";
74  return false;
75  }
76 
77  int64 length = fileSize(path);
78  if (length <= 0)
79  {
80  ret.clear();
81  return true;
82  }
83 
84  char * buffer = (char *)std::malloc((size_t)length);
85  if (!buffer)
86  {
87  THEA_ERROR << "FileSystem: Could not allocate buffer to hold " << length << " bytes from file '" << path << '\'';
88  return false;
89  }
90 
91  FILE * f = std::fopen(path.c_str(), "rb");
92  if (!f)
93  {
94  THEA_ERROR << "FileSystem: Couldn't open file '" << path << "' for reading";
95  return false;
96  }
97 
98  size_t num_read = std::fread(buffer, 1, length, f);
99  if ((int64)num_read != length)
100  {
101  THEA_ERROR << "FileSystem: Error reading from file '" << path << '\'';
102  return false;
103  }
104 
105  std::fclose(f);
106 
107  ret.assign(buffer, (size_t)length);
108  std::free(buffer);
109 
110  return true;
111 }

115 bool
116 objectSatisfiesConstraints(boost::filesystem::directory_entry const & object, int types,
117  Array<std::string> const & patterns, bool ignore_case)
118 {
119  if (types > 0 && types != FileSystem::ObjectType::ALL)
120  {
121  boost::filesystem::file_status status = object.symlink_status();
122  if (!boost::filesystem::is_symlink(status))
123  status = object.status();
124 
125  bool ok = false;
126 
127  if (!ok && (types & FileSystem::ObjectType::FILE) && boost::filesystem::is_regular_file(status))
128  ok = true;
129 
130  if (!ok && (types & FileSystem::ObjectType::DIRECTORY) && boost::filesystem::is_directory(status))
131  ok = true;
132 
133  if (!ok && (types & FileSystem::ObjectType::SYMLINK) && boost::filesystem::is_symlink(status))
134  ok = true;
135 
136  if (!ok)
137  return false;
138  }
139 
140  if (!patterns.empty())
141  {
142 #if BOOST_VERSION / 100000 > 1 || (BOOST_VERSION / 100000 == 1 && BOOST_VERSION / 100 % 1000 > 45)
143  std::string name = object.path().filename().string();
144 #else
145  std::string name = object.path().filename();
146 #endif
147 
148  if (ignore_case)
149  name = toLower(name);
150 
151  bool ok = false;
152  for (size_t i = 0; !ok && i < patterns.size(); ++i)
153  if (patternMatch(patterns[i], name))
154  ok = true;
155 
156  if (!ok)
157  return false;
158  }
159 
160  return true;
161 }

165 intx
166 FileSystem::getDirectoryContents(std::string const & dir, Array<std::string> & objects, int types, std::string const & patterns,
167  int flags)
168 {
169  if (!directoryExists(dir))
170  return -1;
171 
172  bool ignore_case = (flags & Flags::CASE_INSENSITIVE);
173  bool recursive = (flags & Flags::RECURSIVE);
174  bool sorted = (flags & Flags::SORTED);
175 
176  Array<std::string> patlist;
177  if (!patterns.empty())
178  {
179  stringSplit(patterns, ' ', patlist, true);
180 
181  if (ignore_case)
182  {
183  for (size_t i = 0; i < patlist.size(); ++i)
184  patlist[i] = toLower(patlist[i]);
185  }
186  }
187 
188  objects.clear();
189 
190  if (recursive)
191  {
192  boost::filesystem::recursive_directory_iterator objects_end;
193  for (boost::filesystem::recursive_directory_iterator iter(dir); iter != objects_end; ++iter)
194  if (FileSystemInternal::objectSatisfiesConstraints(*iter, types, patlist, ignore_case))
195  objects.push_back(iter->path().string());
196  }
197  else
198  {
199  boost::filesystem::directory_iterator objects_end;
200  for (boost::filesystem::directory_iterator iter(dir); iter != objects_end; ++iter)
201  if (FileSystemInternal::objectSatisfiesConstraints(*iter, types, patlist, ignore_case))
202  objects.push_back(iter->path().string());
203  }
204 
205  if (sorted)
206  std::sort(objects.begin(), objects.end());
207 
208  return (intx)objects.size();
209 }
