48 FurthestPointSampling::subsample(intx num_orig_points, Vector3 const * orig_points, intx num_desired_points,
49  intx * selected_indices, DistanceType dist_type, bool verbose)
50 {
51  alwaysAssertM(num_desired_points >= 0, "FurthestPointSampling: Can't sample a negative number of points");
52  alwaysAssertM(num_orig_points >= num_desired_points,
53  format("FurthestPointSampling: Can't subsample %ld point(s) from %ld point(s)",
54  num_desired_points, num_orig_points));
55  alwaysAssertM(dist_type == DistanceType::GEODESIC, "FurthestPointSampling: Only geodesic distances currently supported");
56 
57  if (num_desired_points == 0)
58  return 0;
59 
60  // Compute proximity graph
61  SampleGraph graph;
62  graph.setSamples(num_orig_points, orig_points);
63  graph.init();
64 
65  if (verbose)
66  {
67  THEA_CONSOLE << "FurthestPointSampling: Computed proximity graph";
68  std::cout << "FurthestPointSampling: Selecting samples: " << std::flush;
69  }
70 
71  // Repeatedly add the furthest sample from the selected set to the selected set
72  ShortestPaths<SampleGraph> shortest_paths;
73  UnorderedMap<SampleGraph::VertexHandle, double> src_region;
74  int prev_percent = 0;
75  for (intx i = 0; i < num_desired_points; ++i)
76  {
77  SampleGraph::VertexHandle furthest_sample = nullptr;
78  if (src_region.empty())
79  {
80  // Just pick the first sample
81  furthest_sample = const_cast<SampleGraph::VertexHandle>(&graph.getSample(0));
82  }
83  else
84  {
85  FurthestPointSamplingInternal::DijkstraCallback callback;
86  shortest_paths.dijkstraWithCallback(graph, nullptr, std::ref(callback), -1, &src_region,
87  /* include_unreachable = */ true);
88  if (!callback.furthest_sample || src_region.find(callback.furthest_sample) != src_region.end())
89  {
90  THEA_ERROR << "FurthestPointSampling: Could not return enough uniformly separated points";
91  return i;
92  }
93 
94  furthest_sample = callback.furthest_sample;
95  }
96 
97  alwaysAssertM(furthest_sample, "FurthestPointSampling: Furthest point is null");
98 
99  selected_indices[i] = furthest_sample->getIndex();
100  src_region[furthest_sample] = 0;
101 
102  if (verbose)
103  {
104  int curr_percent = (int)std::floor(100 * (i / (float)num_desired_points));
105  if (curr_percent >= prev_percent + 2)
106  {
107  for (prev_percent += 2; prev_percent <= curr_percent; prev_percent += 2)
108  {
109  if (prev_percent % 10 == 0) std::cout << prev_percent << '%' << std::flush;
110  else std::cout << '.' << std::flush;
111  }
112 
113  prev_percent = curr_percent;
114  }
115  }
116  }
117 
118  if (verbose)
119  {
120  std::cout << "done" << std::endl;
121  THEA_CONSOLE << "FurthestPointSampling: Selected " << num_desired_points << " point(s) uniformly by separation";
122  }
123 
124  return num_desired_points;
125 }
