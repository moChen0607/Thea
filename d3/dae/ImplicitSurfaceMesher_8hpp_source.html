180  template <typename FunctorT, typename MeshT>
181  static void meshBloomenthal(FunctorT * surface_functor, Ball3 const & bounding_ball, Vector3 const & pt_near_surface,
182  Options::Bloomenthal const & options, MeshT & result)
183  {
184  BloomenthalEval<FunctorT> func(surface_functor);
185  float size = (float)(options.cell_size < 0 ? bounding_ball.getRadius() / 10.0 : options.cell_size);
186  int bounds = options.max_search_steps < 0 ? 10 : options.max_search_steps;
187  BloomenthalPolygonizer::Polygonizer polygonizer(&func, size, bounds);
188  polygonizer.march(options.tetrahedralize_cubes,
189  (float)pt_near_surface.x(), (float)pt_near_surface.y(), (float)pt_near_surface.z());
190 
191  THEA_LOG << "ImplicitSurfaceMesher: " << polygonizer.no_triangles() << " triangles generated via Bloomenthal";
192 
193  exportMesh(polygonizer, result);
194  }

210  template <typename FunctorT, typename MeshT>
211  static void meshBoissonnatOudot(FunctorT * surface_functor, Ball3 const & bounding_ball,
212  Options::BoissonnatOudot const & options, MeshT & result)
213  {
214  typedef Tr::Geom_traits GT;
215  typedef GT::Sphere_3 Sphere_3;
216  typedef GT::Point_3 Point_3;
217  typedef GT::FT FT;
218  typedef CGAL::Implicit_surface_3< GT, CGALEval<FunctorT> > Surface_3;
219 
220  Tr tr; // 3D Delaunay triangulation
221  C2t3 c2t3(tr); // 2D complex in 3D Delaunay triangulation
222  CGALEval<FunctorT> func(surface_functor); // CGAL-style wrapper to evaluate the function
223 
224  // Define the surface
225  Vector3 const & center = bounding_ball.getCenter();
226  Real radius = bounding_ball.getRadius();
227  Sphere_3 sph(Point_3(static_cast<FT>(center.x()), static_cast<FT>(center.y()), static_cast<FT>(center.z())),
228  static_cast<FT>(radius * radius));
229  Surface_3 surface(func, sph);
230 
231  // Define meshing criteria
232  CGAL::Surface_mesh_default_criteria_3<Tr> criteria(
233  static_cast<FT>(options.min_facet_angle < 0 ? 30 : Math::radiansToDegrees(options.min_facet_angle)),
234  static_cast<FT>(options.min_delaunay_radius < 0 ? 0.1 : options.min_delaunay_radius),
235  static_cast<FT>(options.min_center_separation < 0 ? 0.1 : options.min_center_separation));
236 
237  // Generate mesh
238  CGAL::make_surface_mesh(c2t3, surface, criteria, CGAL::Manifold_with_boundary_tag());
239 
240  THEA_LOG << "ImplicitSurfaceMesher: " << c2t3.number_of_facets() << " faces generated via Boissonnat-Oudot";
241 
242  exportMesh(c2t3, result);
243  }

249  template <typename MeshT> static void exportMesh(BloomenthalPolygonizer::Polygonizer & polygonizer, MeshT & dst)
250  {
251  typedef Graphics::IncrementalMeshBuilder<MeshT> Builder;
252  Builder builder(&dst);
253 
254  Array<typename Builder::VertexHandle> vertices;
255 
256  builder.begin();
257  for (size_t i = 0; i < polygonizer.no_vertices(); ++i)
258  {
259  BloomenthalPolygonizer::VERTEX const & bp_p = polygonizer.get_vertex(i);
260  BloomenthalPolygonizer::NORMAL const & bp_n = polygonizer.get_normal(i);
261  Vector3 p(bp_p.x, bp_p.y, bp_p.z), n(bp_n.x, bp_n.y, bp_n.z);
262 
263  typename Builder::VertexHandle vx = builder.addVertex(p, (intx)i, &n);
264  if (!vx)
265  throw Error("ImplicitSurfaceMesher: Could not add vertex from Bloomenthal polygonizer to mesh");
266 
267  vertices.push_back(vx);
268  }
269 
270  for (size_t i = 0; i < polygonizer.no_triangles(); ++i)
271  {
272  BloomenthalPolygonizer::TRIANGLE const & tri = polygonizer.get_triangle(i);
273  alwaysAssertM(tri.v0 >= 0 && (size_t)tri.v0 < vertices.size()
274  && tri.v1 >= 0 && (size_t)tri.v1 < vertices.size()
275  && tri.v2 >= 0 && (size_t)tri.v2 < vertices.size(),
276  "ImplicitSurfaceMesher: Vertex index in triangle from Bloomenthal polygonizer is out of bounds");
277 
278  typename Builder::VertexHandle vx[3] = { vertices[(size_t)tri.v0],
279  vertices[(size_t)tri.v1],
280  vertices[(size_t)tri.v2] };
281  if (!builder.addFace(vx, vx + 3))
282  throw Error("ImplicitSurfaceMesher: Could not add triangle from Bloomenthal polygonizer to mesh");
283  }
284  builder.end();
285  }

290  template <typename MeshT> static void exportMesh(C2t3 & src, MeshT & dst)
291  {
292  typedef Tr::Geom_traits::Point_3 Point_3;
293 
294  typedef Graphics::IncrementalMeshBuilder<MeshT> Builder;
295  Builder builder(&dst);
296 
297  typedef UnorderedMap<Tr::Vertex const *, typename Builder::VertexHandle> VertexMap;
298  VertexMap vmap;
299 
300  builder.begin();
301  for (C2t3::Vertex_iterator vi = src.vertices_begin(); vi != src.vertices_end(); ++vi)
302  {
303  Tr::Vertex const * vin = &(*vi);
304  Point_3 const & p = vin->point();
305 
306  typename Builder::VertexHandle vout = builder.addVertex(Vector3((Real)p.x(), (Real)p.y(), (Real)p.z()));
307  if (!vout)
308  throw Error("ImplicitSurfaceMesher: Could not add vertex from Boissonnat-Oudot polygonizer to mesh");
309 
310  vmap[vin] = vout;
311  }
312 
313  typename Builder::VertexHandle face_vertices[3];
314  int indices[3];
315  for (C2t3::Facet_iterator fi = src.facets_begin(); fi != src.facets_end(); ++fi)
316  {
317  switch (fi->second)
318  {
319  case 0: indices[0] = 1; indices[1] = 2; indices[2] = 3; break;
320  case 1: indices[0] = 0; indices[1] = 3; indices[2] = 2; break;
321  case 2: indices[0] = 3; indices[1] = 0; indices[2] = 1; break;
322  case 3: indices[0] = 2; indices[1] = 1; indices[2] = 0; break;
323  default:
324  throw Error("ImplicitSurfaceMesher: Mesh created by Boissonnat-Oudot polygonizer has invalid face index in cell");
325  }
326 
327  for (int i = 0; i < 3; ++i)
328  {
329  Tr::Vertex const * vx = &(*fi->first->vertex(indices[i]));
330  typename VertexMap::const_iterator existing = vmap.find(vx);
331  if (existing == vmap.end())
332  throw Error("ImplicitSurfaceMesher: Mesh created by Boissonnat-Oudot polygonizer refers to unmapped vertex");
333 
334  face_vertices[i] = existing->second;
335  }
336 
337  if (!builder.addFace(face_vertices, face_vertices + 3))
338  throw Error("ImplicitSurfaceMesher: Could not add triangle from Boissonnat-Oudot polygonizer to mesh");
339  }
340  builder.end();
341  }
