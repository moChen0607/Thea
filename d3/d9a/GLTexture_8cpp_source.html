202 void
203 GLTexture::glTexImage(void const * bytes, Format const * bytes_format, Face face)
204 {
205  switch (gl_target)
206  {
207  case GL_TEXTURE_1D:
208  glTexImage1D(gl_target, 0, format->openGLFormat, (GLsizei)width, 0, bytes_format->openGLBaseFormat,
209  bytes_format->openGLDataFormat, bytes);
210  break;
211 
212  case GL_TEXTURE_2D:
213  case GL_TEXTURE_RECTANGLE_ARB:
214  glTexImage2D(gl_target, 0, format->openGLFormat, (GLsizei)width, (GLsizei)height, 0, bytes_format->openGLBaseFormat,
215  bytes_format->openGLDataFormat, bytes);
216  break;
217 
218  case GL_TEXTURE_3D:
219 #ifdef GL_VERSION_1_2 // Apple omits the EXT extension after moving 3D textures to core
220  glTexImage3D
221 #else
222  glTexImage3DEXT
223 #endif
224  (gl_target, 0, format->openGLFormat, (GLsizei)width, (GLsizei)height, (GLsizei)depth, 0, bytes_format->openGLBaseFormat,
225  bytes_format->openGLDataFormat, bytes);
226  break;
227 
228  default: // GL_TEXTURE_CUBE_MAP_ARB
229  glTexImage2D(toGLCubeMapFace(face), 0, format->openGLFormat, (GLsizei)width, (GLsizei)height, 0,
230  bytes_format->openGLBaseFormat, bytes_format->openGLDataFormat, bytes);
231  }
232  THEA_CHECK_GL_OK

255 void
256 GLTexture::doSanityChecks()
257 {
258  if (dimension == Dimension::DIM_CUBE_MAP && !THEA_GL_SUPPORTS(ARB_texture_cube_map))
259  throw Error(std::string(getName()) + ": Cube map textures are not supported");
260 
261  if (dimension == Dimension::DIM_RECTANGLE && !THEA_GL_SUPPORTS(ARB_texture_rectangle))
262  throw Error(std::string(getName()) + ": Rectangular textures are not supported");
263 
264  if (width < 1 || height < 1 || depth < 1)
265  throw Error(std::string(getName()) + ": Texture must be at least one pixel wide in each dimension");
266 
267  if (depth > 1 && dimension != Dimension::DIM_3D)
268  throw Error(std::string(getName()) + ": Only a 3D texture can have depth greater than one pixel");
269 
270  if (dimension == Dimension::DIM_1D && (height > 1 || depth > 1))
271  throw Error(std::string(getName()) + ": A 1D texture cannot have height or depth greater than one pixel");
272 
273  if (!(Math::isPowerOf2(width) && Math::isPowerOf2(height) && Math::isPowerOf2(depth)) && dimension != Dimension::DIM_RECTANGLE
274  && !THEA_GL_SUPPORTS(ARB_texture_non_power_of_two)) // rectangular textures can be npot by the spec
275  throw Error(std::string(getName()) + ": Non-power-of-two textures are not supported");
276 }

278 void
279 GLTexture::setOptions(Options const & options)
280 {
281  if (dimension == Dimension::DIM_RECTANGLE && options.wrapMode == WrapMode::TILE)
282  throw Error(std::string(getName())
283  + ": Tiling is not supported for rectangular textures"); // see GL_ARB_texture_rectangle spec
284 
285  GLenum wrap = GL_REPEAT;
286  switch (options.wrapMode)
287  {
288  case WrapMode::CLAMP: wrap = THEA_GL_SUPPORTS(EXT_texture_edge_clamp) ? GL_CLAMP_TO_EDGE : GL_CLAMP; break;
289  case WrapMode::TILE: wrap = GL_REPEAT; break;
290  case WrapMode::ZERO:
291  {
292  wrap = THEA_GL_SUPPORTS(ARB_texture_border_clamp) ? GL_CLAMP_TO_BORDER_ARB : GL_CLAMP;
293  GLfloat border_color[4] = {0, 0, 0, 1};
294  glTexParameterfv(gl_target, GL_TEXTURE_BORDER_COLOR, border_color);
295  break;
296  }
297  default: throw Error(std::string(getName()) + ": Unsupported wrap mode");
298  }
299 
300  glTexParameteri(gl_target, GL_TEXTURE_WRAP_S, wrap);
301  glTexParameteri(gl_target, GL_TEXTURE_WRAP_T, wrap);
302  glTexParameteri(gl_target, GL_TEXTURE_WRAP_R, wrap);
303 
304  THEA_CHECK_GL_OK
305 
306  int8 has_mipmaps = (options.interpolateMode == InterpolateMode::NEAREST_MIPMAP
307  || options.interpolateMode == InterpolateMode::BILINEAR_MIPMAP
308  || options.interpolateMode == InterpolateMode::TRILINEAR);
309 
310  if (dimension == Dimension::DIM_RECTANGLE && has_mipmaps)
311  throw Error(std::string(getName())
312  + ": Mipmapping is not supported for rectangular textures"); // see GL_ARB_texture_rectangle spec
313 
314  switch (options.interpolateMode)
315  {
316  case InterpolateMode::NEAREST_NO_MIPMAP:
317  glTexParameteri(gl_target, GL_TEXTURE_MAG_FILTER, GL_NEAREST);
318  glTexParameteri(gl_target, GL_TEXTURE_MIN_FILTER, GL_NEAREST);
319  break;
320 
321  case InterpolateMode::NEAREST_MIPMAP:
322  glTexParameteri(gl_target, GL_TEXTURE_MAG_FILTER, GL_NEAREST);
323  glTexParameteri(gl_target, GL_TEXTURE_MIN_FILTER, GL_NEAREST_MIPMAP_NEAREST);
324  break;
325 
326  case InterpolateMode::BILINEAR_NO_MIPMAP:
327  glTexParameteri(gl_target, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
328  glTexParameteri(gl_target, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
329  break;
330 
331  case InterpolateMode::BILINEAR_MIPMAP:
332  glTexParameteri(gl_target, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
333  glTexParameteri(gl_target, GL_TEXTURE_MIN_FILTER, GL_LINEAR_MIPMAP_NEAREST);
334  break;
335 
336  case InterpolateMode::TRILINEAR:
337  glTexParameteri(gl_target, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
338  glTexParameteri(gl_target, GL_TEXTURE_MIN_FILTER, GL_LINEAR_MIPMAP_LINEAR);
339  break;
340 
341  default: throw Error(std::string(getName()) + ": Unsupported texture interpolation mode");
342  }
343 
344  THEA_CHECK_GL_OK
345 
346  if (has_mipmaps)
347  {
348  if (THEA_GL_SUPPORTS(SGIS_generate_mipmap))
349  glTexParameteri(gl_target, GL_GENERATE_MIPMAP_SGIS, GL_TRUE);
350  else if (GL_VERSION_1_4) // moved to core in 1.4
351  glTexParameteri(gl_target, GL_GENERATE_MIPMAP, GL_TRUE);
352  else
353  {
354  throw Error(std::string(getName()) + ": Automatic mipmap generation not supported");
355  // TODO: Implement manual fallback (or replace everything with glGenerateMipmap for new versions of GL).
356  }
357  }
358 
359  THEA_CHECK_GL_OK
360 
361  if (THEA_GL_SUPPORTS(ARB_shadow))
362  {
363  if (options.depthReadMode == DepthReadMode::NORMAL)
364  {
365  glTexParameteri(gl_target, GL_DEPTH_TEXTURE_MODE, GL_INTENSITY);
366  glTexParameteri(gl_target, GL_TEXTURE_COMPARE_MODE_ARB, GL_NONE);
367  }
368  else
369  {
370  glTexParameteri(gl_target, GL_DEPTH_TEXTURE_MODE, GL_INTENSITY);
371  glTexParameteri(gl_target, GL_TEXTURE_COMPARE_MODE_ARB, GL_COMPARE_R_TO_TEXTURE_ARB);
372  glTexParameteri(gl_target, GL_TEXTURE_COMPARE_FUNC_ARB,
373  (options.depthReadMode == DepthReadMode::LEQUAL) ? GL_LEQUAL : GL_GEQUAL);
374  }
375  }
376  else if (options.depthReadMode != DepthReadMode::NORMAL)
377  throw Error(std::string(getName()) + ": Comparison-based depth read modes are not supported");
378 
379  THEA_CHECK_GL_OK
380 }

415 void
416 GLTexture::updateSubImage(AbstractImage const & image, int64 src_x, int64 src_y, int64 src_z, int64 src_width, int64 src_height,
417  int64 src_depth, int64 dst_x, int64 dst_y, int64 dst_z, Face face)
418 {
419  if (!image.isValid())
420  throw Error(std::string(getName()) + ": Cannot update texture from invalid image");
421 
422  alwaysAssertM(src_x >= 0 && src_y >= 0 && src_z >= 0
423  && src_x + src_width <= image.getWidth()
424  && src_y + src_height <= image.getHeight()
425  && src_z + src_depth <= image.getDepth(),
426  std::string(getName()) + ": All or part of subimage lies outside source image boundaries");
427  alwaysAssertM(dst_x >= 0 && dst_y >= 0 && dst_z >= 0
428  && dst_x + src_width <= width && dst_y + src_height <= height && dst_z + src_depth <= depth,
429  std::string(getName()) + ": All or part of subimage lies outside texture boundaries");
430 
431  Format const * bytes_format = TextureFormat::fromImageType(AbstractImage::Type(image.getType()));
432 
433  { GLScope scope(GL_TEXTURE_BIT | GL_ENABLE_BIT); // Can we do without ENABLE_BIT? The doc is unclear.
434  GLClientScope client_scope(GL_CLIENT_PIXEL_STORE_BIT);
435 
436  glEnable(gl_target);
437  glBindTexture(gl_target, gl_id);
438  THEA_CHECK_GL_OK
439 
440  GLTexture__setDefaultUnpackingOptions(image.getRowAlignment());
441  glPixelStorei(GL_UNPACK_ROW_LENGTH, image.getWidth());
442  glPixelStorei(GL_UNPACK_SKIP_ROWS, src_y);
443  glPixelStorei(GL_UNPACK_SKIP_PIXELS, src_x);
444 
445  switch (gl_target)
446  {
447  case GL_TEXTURE_1D:
448  glTexSubImage1D(gl_target, 0, dst_x, (GLsizei)src_width, bytes_format->openGLBaseFormat, bytes_format->openGLDataFormat,
449  image.getData());
450  break;
451 
452  case GL_TEXTURE_2D:
453  case GL_TEXTURE_RECTANGLE_ARB:
454  glTexSubImage2D(gl_target, 0, dst_x, dst_y, (GLsizei)src_width, (GLsizei)src_height, bytes_format->openGLBaseFormat,
455  bytes_format->openGLDataFormat, image.getData());
456  break;
457 
458  case GL_TEXTURE_3D:
459 #ifdef GL_VERSION_1_2 // Apple omits the EXT extension after moving 3D textures to core
460  glTexSubImage3D
461 #else
462  glTexSubImage3DEXT
463 #endif
464  (gl_target, 0, dst_x, dst_y, dst_z, (GLsizei)src_width, (GLsizei)src_height, (GLsizei)src_depth,
465  bytes_format->openGLBaseFormat, bytes_format->openGLDataFormat, image.getData());
466  break;
467 
468  default: // GL_TEXTURE_CUBE_MAP_ARB
469  glTexSubImage2D(toGLCubeMapFace(face), 0, dst_x, dst_y, (GLsizei)src_width, (GLsizei)src_height,
470  bytes_format->openGLBaseFormat, bytes_format->openGLDataFormat, image.getData());
471  }
472  THEA_CHECK_GL_OK
473  }
