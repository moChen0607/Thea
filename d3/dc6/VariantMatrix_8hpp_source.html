194  // Functions from Serializable
195  void read(BinaryInputStream & input, Codec const & codec = Codec_AUTO(), bool read_block_header = false)
196  {
197  if (codec == Codec_AUTO())
198  {
199  auto which = input.readMatrix(read_block_header, dense, sparse);
200  type = (which == 0 ? Type::DENSE : Type::SPARSE);
201  }
202  else
203  {
204  // Decide based on the type of codec supplied
205  if (dynamic_cast< MatrixCodec<Dense> const * >(&codec))
206  {
207  input.readMatrix(read_block_header, dense, codec);
208  type = Type::DENSE;
209  }
210  else if (dynamic_cast< MatrixCodec<Sparse> const * >(&codec))
211  {
212  input.readMatrix(read_block_header, sparse, codec);
213  type = Type::SPARSE;
214  }
215  else
216  throw Error("VariantMatrix: Codec is not valid for either the dense or sparse matrix types");
217  }
218  }
