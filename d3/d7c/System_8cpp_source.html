56 void
57 System::initTime()
58 {
59 #ifdef THEA_WINDOWS
60 
61  m_counterFrequency = 0;
62  m_start = 0;
63 
64  LARGE_INTEGER cf, st;
65  if (QueryPerformanceFrequency(&cf))
66  {
67  if (QueryPerformanceCounter(&st))
68  {
69  m_counterFrequency = (int64)cf.QuadPart;
70  m_start = (int64)st.QuadPart;
71  }
72  }
73 
74  struct _timeb t;
75  _ftime(&t);
76 
77  m_realWorldGetTickTime0 = (double)t.time - t.timezone * 60 + (t.dstflag ? 60 * 60 : 0);
78 
79 #else
80 
81  gettimeofday(&m_start, nullptr);
82 
83  // "sse" = "seconds since epoch". The time function returns the seconds since the epoch GMT (perhaps more correctly called
84  // UTC).
85  time_t gmt = ::time(nullptr);
86 
87  // No call to free or delete is needed, but subsequent calls to asctime, ctime, mktime, etc. might overwrite local_time_vals.
88  tm * localTimeVals = localtime(&gmt);
89  time_t local = gmt;
90 
91  if (localTimeVals)
92  {
93  // tm_gmtoff is already corrected for daylight savings.
94  local = local + localTimeVals->tm_gmtoff;
95  }
96 
97  m_realWorldGetTickTime0 = local;
98 
99 #ifdef THEA_MAC
100  SInt32 cpu_speed;
101  Gestalt('pclk', &cpu_speed);
102 
103  m_OSXCPUSpeed = (int32)cpu_speed;
104  m_secondsPerNS = 1.0 / 1.0e9;
105 #endif
106 
107 #endif
108 }

110 double
111 System::time()
112 {
113 #ifdef THEA_WINDOWS
114  LARGE_INTEGER now;
115  QueryPerformanceCounter(&now);
116 
117  return ((double)(now.QuadPart - instance().m_start) / instance().m_counterFrequency) + instance().m_realWorldGetTickTime0;
118 #else
119  // Linux resolution defaults to 100Hz. There is no need to do a separate RDTSC call as gettimeofday actually uses RDTSC
120  // when on systems that support it, otherwise it uses the system clock.
121  struct timeval now;
122  gettimeofday(&now, nullptr);
123 
124  return (now.tv_sec - instance().m_start.tv_sec) +
125  (now.tv_usec - instance().m_start.tv_usec) / 1e6
126  + instance().m_realWorldGetTickTime0;
127 #endif
128 }

144 uint64
145 System::getCycleCount()
146 {
147  uint32 timehi, timelo;
148  // Use the assembly instruction rdtsc, which gets the current cycle count (since the process started) and puts it in edx:eax.
149  __asm
150  {
151  rdtsc;
152  mov timehi, edx;
153  mov timelo, eax;
154  }
155  return ((uint64)timehi << 32) + (uint64)timelo;
156 }

194 System::endCycleCount(uint64 & cycle_count)
195 {

198  Nanoseconds diffNS = AbsoluteDeltaToNanoseconds(end, UInt64ToUnsignedWide(cycle_count));
199  cycle_count = (uint64)((double)instance().m_OSXCPUSpeed
