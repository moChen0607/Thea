72  void \
73  GLVAR::func_name(int64 start_elem, int64 num_elems_to_update, type_ const * array) \
74  { \
75  if (num_elems_to_update <= 0) return; \
76  \
77  alwaysAssertM(isValid(), "GLVAR: Can't update invalid VAR"); \
78  \
79  if (num_elems > 0) \
80  { \
81  alwaysAssertM(gl_type == gl_type_ && num_components == num_components_ && gl_target == gl_target_, \
82  "GLVAR: Can't update non-empty VAR with elements of a different type"); \
83  } \
84  else \
85  { \
86  gl_type = gl_type_; \
87  num_components = num_components_; \
88  elem_size = sizeof(type_); \
89  gl_target = gl_target_; \
90  } \
91  \
92  int64 offset_bytes = start_elem * elem_size; \
93  int64 num_bytes = num_elems_to_update * elem_size; \

95  if (total_size > capacity) \

102  }
