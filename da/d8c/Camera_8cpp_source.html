72 void
73 Camera::updateCachedProjectionTransform() const
74 {
75  if (proj_changed)
76  {
77  cached_proj_transform = (projection_type == ProjectionType::ORTHOGRAPHIC)
78  ? Math::orthogonalProjection(left, right, bottom, top, near_dist, far_dist,
79  proj_y_dir == ProjectedYDirection::UP ? true : false)
80  : Math::perspectiveProjection(left, right, bottom, top, near_dist, far_dist,
81  proj_y_dir == ProjectedYDirection::UP ? true : false);
82 
83  cached_inv_proj_transform = cached_proj_transform.inverse();
84 
85  proj_changed = false;
86  }

112 Camera::read(BinaryInputStream & input, Codec const & codec, bool read_block_header)
113 {
114  if (read_block_header)
115  input.skip(Codec::BlockHeader::SERIALIZED_LENGTH); // header is not needed
116 
117  { BinaryInputStream::EndiannessScope scope(input, Endianness::LITTLE);
118 
119  frame = input.readCoordinateFrame3();
120 
121  uint8 pt = input.readUInt8();
122  projection_type = (pt == 0 ? ProjectionType::ORTHOGRAPHIC : ProjectionType::PERSPECTIVE);
123 
124  left = static_cast<Real>(input.readFloat32());
125  right = static_cast<Real>(input.readFloat32());
126  bottom = static_cast<Real>(input.readFloat32());
127  top = static_cast<Real>(input.readFloat32());
128  near_dist = static_cast<Real>(input.readFloat32());
129  far_dist = static_cast<Real>(input.readFloat32());
130 
131  uint8 pyd = input.readUInt8();
132  proj_y_dir = (pyd == 0 ? ProjectedYDirection::UP : ProjectedYDirection::DOWN);
133  }
134 }

136 void
137 Camera::write(BinaryOutputStream & output, Codec const & codec, bool write_block_header) const
138 {
139  Codec::BlockHeader header("Camera");
140  if (write_block_header)
141  header.markAndSkip(output);
142 
143  { BinaryOutputStream::EndiannessScope scope(output, Endianness::LITTLE);
144 
145  output.writeCoordinateFrame3(frame);
146  output.writeUInt8(projection_type == ProjectionType::ORTHOGRAPHIC ? 0 : 1);
147  output.writeFloat32(static_cast<float32>(left));
148  output.writeFloat32(static_cast<float32>(right));
149  output.writeFloat32(static_cast<float32>(bottom));
150  output.writeFloat32(static_cast<float32>(top));
151  output.writeFloat32(static_cast<float32>(near_dist));
152  output.writeFloat32(static_cast<float32>(far_dist));
153  output.writeUInt8(proj_y_dir == ProjectedYDirection::UP ? 0 : 1);
154  }
155 
156  if (write_block_header)
157  header.calcAndWrite(output);
158 }
