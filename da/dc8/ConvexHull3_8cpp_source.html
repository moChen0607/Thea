46 void
47 ConvexHull3::updateApprox() const
48 {
49  if (approx_updated)
50  return;
51 
52  THEA_DEBUG << "ConvexHull3: Computing convex hull of " << points.size() << " points";
53 
54  // Convert the input points to a packed array of floats
55  Array<float> coords(3 * points.size());
56  size_t base = 0;
57  for (size_t i = 0; i < points.size(); ++i, base += 3)
58  {
59  Vector3 const & p = points[i];
60  coords[base ] = (float)p.x();
61  coords[base + 1] = (float)p.y();
62  coords[base + 2] = (float)p.z();
63  }
64 
65  approx_vertices.clear();
66  approx_indices.clear();
67 
68  if (points.size() == 3)
69  {
70  approx_vertices.resize(3);
71  approx_vertices[0] = points[0];
72  approx_vertices[1] = points[1];
73  approx_vertices[2] = points[2];
74 
75  approx_indices.resize(3);
76  approx_indices[0] = 0;
77  approx_indices[1] = 1;
78  approx_indices[2] = 2;
79  }
80  else if (points.size() > 3)
81  {
82  HullDesc desc(QF_TRIANGLES,
83  (unsigned int)(points.size()),
84  &coords[0],
85  3 * sizeof(float));
86 
87  desc.SetHullFlag(QF_SKIN_WIDTH);
88  desc.mNormalEpsilon = 0.001f;
89  desc.mMaxVertices = (unsigned int)(options.approx.max_vertices_hint > 0 ? options.approx.max_vertices_hint : 100);
90  desc.mSkinWidth = (float)options.approx.skin_width;
91 
92  HullLibrary hull_lib;
93  HullResult result;
94  HullError err = hull_lib.CreateConvexHull(desc, result);
95  THEA_DEBUG << "ConvexHull3: Computed convex hull with " << result.mNumOutputVertices << " vertices, " << result.mNumFaces
96  << " faces, " << result.mNumIndices << " indices and error " << err;
97 
98  approx_vertices.resize((size_t)result.mNumOutputVertices);
99  size_t j = 0;
100  for (size_t i = 0; i < approx_vertices.size(); ++i, j += 3)
101  approx_vertices[i] = Vector3(result.mOutputVertices[j], result.mOutputVertices[j + 1], result.mOutputVertices[j + 2]);
102 
103  if (result.mPolygons) // shouldn't happen since we requested triangles, but still...
104  {
105  THEA_DEBUG << "ConvexHull3: Requested triangles, but higher-degree polygons obtained";
106 
107  Polygon3 poly;
108  Array<intx> tri_indices;
109  size_t num_face_vertices = 0, index;
110 
111  for (size_t i = 0; i < (size_t)result.mNumIndices; i += (1 + num_face_vertices))
112  {
113  num_face_vertices = (size_t)result.mIndices[i];
114 
115  if (num_face_vertices == 3) // again, shouldn't happen
116  {
117  for (int j = 1; j <= 3; ++j)
118  {
119  index = (size_t)result.mIndices[i + j];
120  debugAssertM(index >= 0 && index < approx_vertices.size(), "ConvexHull3: Vertex index out of bounds");
121  approx_indices.push_back(index);
122  }
123  }
124  else if (num_face_vertices > 3) // triangulate
125  {
126  poly.clear();
127  for (size_t j = 1; j <= num_face_vertices; ++j)
128  {
129  index = (size_t)result.mIndices[i + j];
130  debugAssertM(index >= 0 && index < approx_vertices.size(), "ConvexHull3: Vertex index out of bounds");
131  poly.addVertex(approx_vertices[index], (intx)index);
132  }
133 
134  tri_indices.clear();
135  poly.triangulate(tri_indices);
136  for (size_t j = 0; j < tri_indices.size(); ++j)
137  approx_indices.push_back((size_t)tri_indices[j]);
138  }
139  }
140  }
141  else
142  {
143  approx_indices.resize((size_t)result.mNumIndices);
144  size_t index;
145 
146  for (size_t i = 0; i < approx_indices.size(); ++i)
147  {
148  index = (size_t)result.mIndices[i];
149  debugAssertM(index >= 0 && index < approx_vertices.size(), "ConvexHull3: Vertex index out of bounds");
150  approx_indices[i] = index;
151  }
152  }
153 
154  hull_lib.ReleaseResult(result);
155  }
156 
157  approx_updated = true;
158 }
