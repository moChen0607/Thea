236  template <typename T>
237  static int computeFlat(Array<T> const & points, Array<int> & labels, FlatOptions options = FlatOptions::defaults(),
238  int num_clusters_hint = 2)
239  {
240  if (points.size() <= 0)
241  return 0;
242 
243  int num_clusters_found = 0;
244 
245 #ifdef THEA_ENABLE_CLUTO
246  if (options.method == FlatMethod::AUTO)
247  options.method = FlatMethod::CLUTO_GRAPH_CLUSTER_RB;
248 #endif
249 
250  switch (options.method)
251  {
252 #ifdef THEA_ENABLE_CLUTO
253  case FlatMethod::CLUTO_CLUSTER_DIRECT:
254  case FlatMethod::CLUTO_CLUSTER_RB:
255  case FlatMethod::CLUTO_GRAPH_CLUSTER_RB:
256  {
257  int simfun = 0, grmodel = 0, nnbrs = 0, crfun = 0, cstype = 0;
258  toClutoOptions(options, simfun, grmodel, nnbrs, crfun, cstype);
259 
260  ClutoMatrix cm;
261  toClutoMatrix(points, cm);
262 
263  labels.resize(points.size());
264 
265  switch (options.method)
266  {
267  // Refer to the CLUTO manual and to Zhao and Karypis, "Criterion Functions for Document Clustering: Experiments and
268  // Analysis", UMN CS 01-040, 2001, for choice of criterion function in each case
269  case FlatMethod::CLUTO_CLUSTER_DIRECT:
270  {
271  CLUTO_VP_ClusterDirect((int)cm.rows(), // nrows
272  (int)cm.cols(), // ncols
273  nullptr, // rowptr
274  nullptr, // rowind
275  &cm.getValues()[0], // rowval
276  simfun, // simfun
277  crfun, // crfun
278  CLUTO_ROWMODEL_NONE, // rowmodel
279  CLUTO_COLMODEL_NONE, // colmodel
280  1.0f, // colprune
281  10, // ntrials
282  10, // niter
283  Random::common().integer(0, 0xFFFF), // seed
284  0, // dbglvl
285  num_clusters_hint, // nclusters
286  &labels[0]); // part
287 
288  num_clusters_found = num_clusters_hint;
289  break;
290  }
291 
292  case FlatMethod::CLUTO_CLUSTER_RB:
293  {
294  CLUTO_VP_ClusterRB((int)cm.rows(), // nrows
295  (int)cm.cols(), // ncols
296  nullptr, // rowptr
297  nullptr, // rowind
298  &cm.getValues()[0], // rowval
299  simfun, // simfun
300  crfun, // crfun
301  CLUTO_ROWMODEL_NONE, // rowmodel
302  CLUTO_COLMODEL_NONE, // colmodel
303  1.0f, // colprune
304  10, // ntrials
305  10, // niter
306  Random::common().integer(0, 0xFFFF), // seed
307  cstype, // cstype
308  1, // kwayrefine
309  0, // dbglvl
310  num_clusters_hint, // nclusters
311  &labels[0]); // part
312 
313  num_clusters_found = num_clusters_hint;
314  break;
315  }
316 
317  default: // FlatMethod::CLUTO_GRAPH_CLUSTER_RB
318  {
319  float cut_value;
320  num_clusters_found = CLUTO_VP_GraphClusterRB((int)cm.rows(), // nrows
321  (int)cm.cols(), // ncols
322  nullptr, // rowptr
323  nullptr, // rowind
324  &cm.getValues()[0], // rowval
325  simfun, // simfun
326  CLUTO_ROWMODEL_NONE, // rowmodel
327  CLUTO_COLMODEL_NONE, // colmodel
328  1.0f, // colprune
329  grmodel, // grmodel
330  nnbrs, // nnbrs
331  -1, // edgeprune
332  -1, // vtxprune
333  5, // mincmp
334  10, // ntrials
335  Random::common().integer(0, 0xFFFF), // seed
336  cstype, // cstype
337  0, // dbglvl
338  num_clusters_hint, // nclusters
339  &labels[0], // part
340  &cut_value); // crvalue
341  }
342  }
343  break;
344  }
345 #endif
346 
347  default: throw Error("Clustering: Clustering method is not supported");
348  }
349 
350  return num_clusters_found;
351  }

365  template <typename MatrixT>
366  static int computeFlat(MatrixT const & weights, Array<int> & labels, FlatOptions options = FlatOptions::defaults(),
367  int num_clusters_hint = 2)
368  {
369  int num_clusters_found = 0;
370 
371 #ifdef THEA_ENABLE_CLUTO
372  if (options.method == FlatMethod::AUTO)
373  options.method = FlatMethod::CLUTO_GRAPH_CLUSTER_RB;
374 #endif
375 
376  switch (options.method)
377  {
378 #ifdef THEA_ENABLE_CLUTO
379  case FlatMethod::CLUTO_CLUSTER_DIRECT:
380  case FlatMethod::CLUTO_CLUSTER_RB:
381  case FlatMethod::CLUTO_GRAPH_CLUSTER_RB:
382  {
383  alwaysAssertM(weights.rows() == weights.cols(), "Clustering: Adjacency matrix must be square");
384 
385  int simfun = 0, grmodel = 0, nnbrs = 0, crfun = 0, cstype = 0;
386  toClutoOptions(options, simfun, grmodel, nnbrs, crfun, cstype);
387 
388  ClutoMatrix cm;
389  toClutoMatrix(weights, cm);
390  if (cm.getValues().size() <= 0) // more zeroes could have been detected during conversion
391  {
392  THEA_WARNING << "Clustering: Weight matrix is zero, no clusters exist";
393  return 0;
394  }
395 
396  int * rowptr = (cm.getRowIndices().size() > 0 ? &cm.getRowIndices()[0] : nullptr);
397  int * rowind = (cm.getColumnIndices().size() > 0 ? &cm.getColumnIndices()[0] : nullptr);
398  float * rowval = (cm.getValues().size() > 0 ? &cm.getValues()[0] : nullptr);
399 
400  labels.resize((size_t)weights.rows());
401 
402  switch (options.method)
403  {
404  // Refer to the CLUTO manual and to Zhao and Karypis, "Criterion Functions for Document Clustering: Experiments and
405  // Analysis", UMN CS 01-040, 2001, for choice of criterion function in each case
406  case FlatMethod::CLUTO_CLUSTER_DIRECT:
407  {
408  CLUTO_SP_ClusterDirect((int)weights.rows(), // nrows
409  rowptr, // rowptr
410  rowind, // rowind
411  rowval, // rowval
412  crfun, // crfun
413  10, // ntrials
414  10, // niter
415  Random::common().integer(0, 0xFFFF), // seed
416  0, // dbglvl
417  num_clusters_hint, // nclusters
418  &labels[0]); // part
419 
420  num_clusters_found = num_clusters_hint;
421  break;
422  }
423 
424  case FlatMethod::CLUTO_CLUSTER_RB:
425  {
426  CLUTO_SP_ClusterRB((int)weights.rows(), // nrows
427  rowptr, // rowptr
428  rowind, // rowind
429  rowval, // rowval
430  crfun, // crfun
431  10, // ntrials
432  10, // niter
433  Random::common().integer(0, 0xFFFF), // seed
434  cstype, // cstype
435  1, // kwayrefine
436  0, // dbglvl
437  num_clusters_hint, // nclusters
438  &labels[0]); // part
439 
440  num_clusters_found = num_clusters_hint;
441  break;
442  }
443 
444  default: // FlatMethod::CLUTO_GRAPH_CLUSTER_RB
445  {
446  float cut_value;
447  num_clusters_found = CLUTO_SP_GraphClusterRB((int)weights.rows(), // nrows
448  rowptr, // rowptr
449  rowind, // rowind
450  rowval, // rowval
451  nnbrs, // nnbrs
452  -1, // edgeprune
453  -1, // vtxprune
454  5, // mincmp
455  10, // ntrials
456  Random::common().integer(0, 0xFFFF), // seed
457  cstype, // cstype
458  0, // dbglvl
459  num_clusters_hint, // nclusters
460  &labels[0], // part
461  &cut_value); // crvalue
462  }
463  }
464  break;
465  }
466 #endif
467 
468  default: throw Error("Clustering method is not supported");
469  }
470 
471  return num_clusters_found;
472  }

602 template <typename T>
603 void
604 Clustering::toClutoMatrix(Array<T> const & points, ClutoMatrix & cm)
605 {
606  size_t num_points = points.size();
607  size_t num_elems = numElems(points[0]);
608 
609  cm.resize((int)num_points, (int)num_elems);
610  cm.getRowIndices().clear();
611  cm.getValues().resize(num_elems * num_points);
612 
613  float * fp = &cm.getValues()[0];
614  for (size_t i = 0; i < num_points; ++i)
615  fp = copyToFloatArray(points[i], fp);
616 }

626  cm.getRowIndices().clear();
627  cm.getValues().resize(num_elems * num_points);

654 Clustering::toClutoMatrix(MatrixX<T, MatrixLayout::COLUMN_MAJOR> const & m, ClutoMatrix & cm)
655 {
656  cm.getRowIndices().clear();
657  cm.getColumnIndices().clear();
658 
659  if (!m.isEmpty())
660  {
661  cm.getValues().resize(m.rows() * m.cols());
662 
663  Array<float>::iterator cp = cm.getValues().begin();
664  for (int i = 0; i < m.rows(); ++i)
665  for (int j = 0; j < m.cols(); ++j, ++cp)
666  *cp = m(j, i);
667  }
668  else
669  cm.getValues().clear();
670 }
