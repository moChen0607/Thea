27 static bool
28 indicesToInt(int32 type, int64 n, void const * begin, Array<int32> & out)
29 {
30 #define THEA_CSPARSE_CONVERT_AND_PUSH(numtype) \
31  { \
32  out.resize((size_t)n); \
33  for (intx i = 0; i < n; ++i) out[(size_t)i] = ((numtype const *)begin)[i]; \
34  }
35 
36  switch (type)
37  {
38  case NumericType::INT8 : THEA_CSPARSE_CONVERT_AND_PUSH(int8); break;
39  case NumericType::INT16 : THEA_CSPARSE_CONVERT_AND_PUSH(int16); break;
40  case NumericType::INT32 : THEA_CSPARSE_CONVERT_AND_PUSH(int32); break;
41  case NumericType::INT64 : THEA_CSPARSE_CONVERT_AND_PUSH(int64); break;
42  case NumericType::UINT8 : THEA_CSPARSE_CONVERT_AND_PUSH(uint8); break;
43  case NumericType::UINT16 : THEA_CSPARSE_CONVERT_AND_PUSH(uint16); break;
44  case NumericType::UINT32 : THEA_CSPARSE_CONVERT_AND_PUSH(uint32); break;
45  case NumericType::UINT64 : THEA_CSPARSE_CONVERT_AND_PUSH(uint64); break;
46 
47  default:
48  THEA_ERROR << "CSPARSELinearSolver: Unsupported compressed sparse matrix index type";
49  return false;
50  }
51 
52  return true;
53 
54 #undef THEA_CSPARSE_CONVERT_AND_PUSH
55 }

57 // Return value: 1: symmetric, 2: upper-triangular, 3: lower-triangular, 0: other
58 int32
59 isSymmetric(cs * A)
60 {
61  if (A->m != A->n)
62  return 0;
63 
64  if (A->n <= 0 || A->nzmax <= 0 || A->p[A->n] <= 0) // empty matrix is symmetric by default
65  return 1;
66 
67  // Checking for upper or lower triangular is fast
68  bool is_upper = true;
69  bool is_lower = true;
70  for (int32 j = 0 ; j < A->n; ++j)
71  {
72  for (int32 p = A->p[j]; p < A->p[j + 1]; ++p)
73  {
74  if (A->i[p] > j) is_upper = false;
75  if (A->i[p] < j) is_lower = false;
76  }
77  }
78 
79  if (is_upper) return 2;
80  if (is_lower) return 3;
81 
82  // Checking for symmetric is relatively slower since we compute the transpose first
83  static float64 const EPSILON = 1.0e-200;
84  cs * At = cs_transpose(A, 1);
85  bool is_sym = true;
86 
87  // First check that all the column pointers are the same
88  for (int32 j = 0 ; j < A->n; ++j)
89  if (A->p[j] != At->p[j])
90  {
91  is_sym = false;
92  break;
93  }
94 
95  // Now check that all the row indices are the same
96  if (is_sym)
97  {
98  int32 nnz = A->p[A->n];
99  for (int32 p = 0; p < nnz; ++p)
100  if (A->i[p] != At->i[p])
101  {
102  is_sym = false;
103  break;
104  }
105  }
106 
107  // Finally check that all the values are the same (floating point comparisons so most expensive)
108  if (is_sym)
109  {
110  int32 nnz = A->p[A->n];
111  for (int32 p = 0; p < nnz; ++p)
112  if (std::fabs(A->x[p] - At->x[p]) > EPSILON)
113  {
114  is_sym = false;
115  break;
116  }
117  }
118 
119  // Cleanup
120  cs_spfree(At);
121 
122  return is_sym ? 1 : 0;
123 }

132 cs *
133 makeSymmetric(cs * A)
134 {
135  cs * At, * C;
136  At = cs_transpose(A, 1);
137  cs_fkeep(At, &dropDiag, nullptr); // drop diagonal entries from At
138  C = cs_add(A, At, 1, 1); // C = A + At
139  cs_spfree(At) ;
140  return C;
141 }

152 int8
153 CSPARSELinearSolver::solve(AbstractMatrix<float64> const & a, float64 const * b, AbstractOptions const * options)
154 {
155  using namespace CSPARSELinearSolverInternal;
156 
157  has_solution = false;
158  solution.resize(0);
159 
160  try
161  {
162  if (a.asSparse() && a.asSparse()->asCompressed())
163  {
164  if (a.rows() <= 0 || a.cols() <= 0)
165  {
166  THEA_WARNING << getName() << ": Attempting to solve empty system";
167  has_solution = true;
168  return has_solution;
169  }
170 
171  std::string method = toUpper(options ? options->getString("method", "LU") : "LU");
172 
173  // Only QR factorization allows rectangular matrices, returning least-squares solutions
174  alwaysAssertM(method == "QR" || Math::isSquare(a),
175  std::string(getName()) + ": Coefficient matrix for method '" + method + "' must be square");
176 
177  AbstractCompressedSparseMatrix<float64> const & sm = *a.asSparse()->asCompressed();
178 
179  // Convert the coefficient matrix to CSPARSE format
180  alwaysAssertM(sm.isColumnMajor(),
181  std::string(getName()) + ": Coefficient matrix is not in compressed column (CSC) format");
182  alwaysAssertM(sm.isFullyCompressed(), std::string(getName()) + ": Operator matrix is not fully compressed");
183 
184  // TODO: avoid the copy when the input indices are actually ints
185  Array<int32> irow;
186  indicesToInt(sm.getInnerIndexType(), sm.numStoredElements(), sm.getInnerIndices(), irow);
187 
188  Array<int32> pcol;
189  indicesToInt(sm.getOuterIndexType(), sm.outerSize(), sm.getOuterIndices(), pcol);
190 
191  int32 nnz = (int32)sm.numStoredElements();
192  alwaysAssertM(nnz == pcol[pcol.size() - 1],
193  std::string(getName()) + ": (n + 1)th entry of pcol array should be number of non-zeros");
194 
195  // Create the coefficient matrix
196  cs C;
197  C.nzmax = nnz;
198  C.m = sm.rows();
199  C.n = sm.cols();
200  C.p = &pcol[0];
201  C.i = irow.empty() ? nullptr : &irow[0];
202  C.x = const_cast<float64 *>(sm.getValues());
203  C.nz = -1;
204 
205  // Check if the matrix is symmetric, and if necessary symmetrize it if is triangular
206  cs * Cp = &C;
207  int32 sym = isSymmetric(&C);
208  if (sym == 2 || sym == 3)
209  {
210  if (options && options->getInteger("symmetrize-triangular", 0) != 0)
211  Cp = makeSymmetric(&C);
212  else
213  sym = 0;
214  }
215 
216  // Initialize the solutions vector with the constants vector. This will be overwritten by the solver.
217  solution.resize(std::max(C.m, C.n));
218  Algorithms::fastCopy(b, b + C.m, solution.data());
219  if (solution.size() > C.m) // underdetermined system
220  solution.tail(solution.size() - C.m).setZero();
221 
222  int32 ok = 0;
223  try
224  {
225  if (method == "CHOLESKY")
226  {
227  THEA_DEBUG << getName() << ": Trying to solve linear system by Cholesky factorization";
228 
229  if (!sym)
230  throw Error("Cholesky factorization requires a symmetric positive-definite coefficient matrix");
231 
232  ok = cs_cholsol(Cp, solution.data(), 1);
233 
234  THEA_DEBUG << getName() << ": Solution by Cholesky factorization failed:"
235  " please check that the coefficient matrix is symmetric positive-definite";
236  }
237  else if (method == "QR")
238  {
239  THEA_DEBUG << getName() << "Trying to solve linear system by QR factorization";
240 
241  ok = cs_qrsol(Cp, solution.data(), 3);
242  }
243  else if (method == "LU")
244  {
245  THEA_DEBUG << getName() << ": Trying to solve linear system by LU factorization";
246 
247  static float64 const DEFAULT_TOLERANCE = 1e-10;
248  float64 tol = options ? options->getFloat("tol", DEFAULT_TOLERANCE) : DEFAULT_TOLERANCE;
249  ok = cs_lusol(Cp, solution.data(), 1, tol);
250  }
251  else
252  throw Error("Unknown solution method '" + method + '\'');
253  }
254  catch (...)
255  {
256  // Cleanup
257  if (Cp != &C)
258  cs_spfree(Cp);
259 
260  throw;
261  }
262 
263  // Cleanup
264  if (Cp != &C)
265  cs_spfree(Cp);
266 
267  has_solution = (ok != 0);
268  }
269  else
270  throw Error("Unsupported coefficient matrix format: only compressed column (CSC) allowed)");
271  }
272  THEA_STANDARD_CATCH_BLOCKS(return false;, ERROR, "%s: Error solving linear system", getName())
273 
274  return has_solution;
275 }
