83  VectorT closestPoint(VectorT const & p) const
84  {
85  T t = (p - origin).dot(direction); // we'll normalize direction later
86  if (t < 0)
87  return origin;
88  else
89  {
90  T dir_sqlen = direction.squaredNorm();
91  if (Math::fuzzyEq(dir_sqlen, static_cast<T>(0)))
92  return origin;
93 
94  // Direction has to be normalized twice -- once during the dot product above and once in the scaling below. We can
95  // combine the two divisions and avoid the square roots.
96  return origin + (t / dir_sqlen) * direction;
97  }
