206  void readMeshGroup(MeshGroup & mesh_group, BinaryInputStream & input, Codec::BlockHeader const * block_header,
207  ReadCallback * callback) const
208  {
209  mesh_group.clear();
210 
211  BinaryInputStream * in = &input;
212  BinaryInputStream::Ptr tmp_in;
213  if (block_header)
214  {
215  if (block_header->data_size <= 0) { return; }
216  tmp_in = std::make_shared<BinaryInputStream>(input, (int64)block_header->data_size);
217  in = tmp_in.get();
218  }
219 
220  using CodecOBJInternal::VTN;
221  typedef UnorderedMap<VTN, typename Builder::VertexHandle> VTNVertexMap;
222  typedef UnorderedMap<intx, typename Builder::VertexHandle> IndexVertexMap;
223  VTNVertexMap vtn_refs;
224  IndexVertexMap vrefs;
225  Array<typename Builder::VertexHandle> face;
226 
227  // OBJ is not neatly divided into separate meshes (e.g. *all* the vertices can be put at the beginning), so we need to
228  // cache the vertices and add them to meshes on-demand.
229  Array<Vector3> vertices;
230  Array<Vector2> texcoords;
231  Array<Vector3> normals;
232 
233  std::string line;
234  double x, y, z;
235  intx index;
236 
237  std::string group_name = std::string(mesh_group.getName()) + (read_opts.flatten ? "/FlattenedMesh" : "/AnonymousMesh0");
238  int anon_index = 0;
239 
240  MeshPtr mesh;
241  std::shared_ptr<Builder> bp;
242  Builder * builder = nullptr;
243 
244  intx num_faces = 0;
245 
246  while (in->hasMore())
247  {
248  std::string line = trimWhitespace(in->readLine());
249  bool done = false;
250  while (line.empty() || !(line[0] == 'v' || line[0] == 'f' || line[0] == 'p' || line[0] == 'g' || line[0] == 'o'))
251  {
252  if (in->hasMore())
253  line = trimWhitespace(in->readLine());
254  else
255  {
256  done = true;
257  break;
258  }
259  }
260 
261  if (done) break;
262 
263  if (line[0] == 'v' && line.length() >= 2)
264  {
265  std::istringstream vstr(line); vstr.get();
266 
267  if (line[1] == 't')
268  {
269  if (!read_opts.ignore_texcoords) // texcoord
270  {
271  vstr.get();
272  if (!(vstr >> x >> y))
273  throw Error(std::string(getName()) + ": Could not read texture coordinate on line '" + line + '\'');
274 
275  texcoords.push_back(Vector2((Real)x, (Real)y));
276  }
277  }
278  else if (line[1] == 'n')
279  {
280  if (!read_opts.ignore_normals) // normal
281  {
282  vstr.get();
283  if (!(vstr >> x >> y >> z))
284  throw Error(std::string(getName()) + ": Could not read normal on line '" + line + '\'');
285 
286  normals.push_back(Vector3((Real)x, (Real)y, (Real)z));
287  }
288  }
289  else if (line[1] == ' ' || line[1] == '\t') // vertex
290  {
291  if (!(vstr >> x >> y >> z))
292  throw Error(std::string(getName()) + ": Could not read vertex position on line '" + line + '\'');
293 
294  vertices.push_back(Vector3((Real)x, (Real)y, (Real)z));
295  }
296  }
297  else if ((line[0] == 'f' || line[0] == 'p') && line.length() >= 2 && (line[1] == ' ' || line[1] == '\t')) // face
298  {
299  // If no mesh+builder have been created yet, create them
300  if (!builder)
301  {
302  mesh = MeshPtr(new Mesh(group_name));
303  bp = std::shared_ptr<Builder>(new Builder(mesh));
304  builder = bp.get();
305  builder->begin();
306  }
307 
308  face.clear();
309  size_t field_begin = line.find_first_not_of("fp \t"), field_end = 0;
310  bool bad_face = false;
311  while (field_end != std::string::npos)
312  {
313  field_end = line.find_first_of(" \t", field_begin);
314 
315  std::istringstream fstr(line.substr(field_begin, (field_end == std::string::npos ? std::string::npos
316  : field_end - field_begin)));
317  fstr.setf(std::ios::skipws);
318 
319  if (!read_opts.ignore_texcoords || !read_opts.ignore_normals) // use the VTN map
320  {
321  // OBJ stores a vertex reference as VertexIndex[/[TexCoordIndex][/NormalIndex]]
322  VTN vtn; vtn[0] = 0, vtn[1] = 0; vtn[2] = 0;
323 
324  bool bad_index = false;
325  if (!(fstr >> vtn[0]))
326  bad_index = true;
327 
328  if (!bad_index && fstr.get() == '/')
329  {
330  if (fstr.peek() == '/')
331  {
332  if (!read_opts.ignore_normals)
333  {
334  fstr.get();
335  if (!(fstr >> vtn[2]))
336  bad_index = true;
337  }
338  }
339  else
340  {
341  if (!(fstr >> vtn[1]))
342  bad_index = true;
343 
344  if (!bad_index)
345  {
346  if (read_opts.ignore_texcoords) // reset field
347  vtn[1] = 0;
348 
349  if (!read_opts.ignore_normals && fstr.get() == '/')
350  {
351  if (!(fstr >> vtn[2]))
352  bad_index = true;
353  }
354  }
355  }
356  }
357 
358  if (bad_index)
359  {
360  THEA_WARNING << getName() << ": Could not read index";
361  bad_face = true; break;
362  }
363 
364  if (std::abs(vtn[0]) < 1 || std::abs(vtn[0]) > (intx)vertices.size())
365  {
366  THEA_WARNING << getName() << ": Vertex index " << vtn[0] << " out of bounds (#vertices = "
367  << vertices.size() << ')';
368  bad_face = true; break;
369  }
370 
371  if (!read_opts.ignore_texcoords && std::abs(vtn[1]) > (intx)texcoords.size())
372  {
373  THEA_WARNING << getName() << ": Texture coordinate index " << vtn[1] << " out of bounds (#texcoords = "
374  << texcoords.size() << ')';
375  vtn[1] = 0;
376  }
377 
378  if (!read_opts.ignore_normals && std::abs(vtn[2]) > (intx)normals.size())
379  {
380  THEA_WARNING << getName() << ": Normal index " << vtn[2] << " out of bounds (#normals = "
381  << normals.size() << ')';
382  vtn[2] = 0;
383  }
384 
385  // Negative indices indicate counting from the last element. We'll subtract one eventually, so -1 should currently
386  // map to vertices.size().
387  if (vtn[0] < 0) vtn[0] = (intx)vertices.size() + 1 + vtn[0];
388  if (vtn[1] < 0) vtn[1] = (intx)texcoords.size() + 1 + vtn[1];
389  if (vtn[2] < 0) vtn[2] = (intx)normals.size() + 1 + vtn[2];
390 
391  // Add the vertex referenced by the triple to the mesh builder if it has not already been added
392  typename VTNVertexMap::const_iterator existing = vtn_refs.find(vtn);
393  if (existing == vtn_refs.end())
394  {
395  typename Builder::VertexHandle vref = builder->addVertex(vertices[(size_t)vtn[0] - 1],
396  read_opts.store_vertex_indices ? vtn[0] - 1 : -1,
397  vtn[2] > 0 ? &normals[(size_t)vtn[2] - 1] : nullptr,
398  nullptr, // color
399  vtn[1] > 0 ? &texcoords[(size_t)vtn[1] - 1] : nullptr);
400  if (callback)
401  callback->vertexRead(mesh.get(), vtn[0] - 1, vref);
402 
403  vtn_refs[vtn] = vref;
404  face.push_back(vref);
405  }
406  else
407  face.push_back(existing->second);
408  }
409  else
410  {
411  if (!(fstr >> index))
412  {
413  THEA_WARNING << getName() << ": Could not read index";
414  bad_face = true; break;
415  }
416 
417  if (std::abs(index) < 1 || std::abs(index) > (intx)vertices.size())
418  {
419  THEA_WARNING << getName() << ": Vertex index " << index << " out of bounds (#vertices = "
420  << vertices.size() << ')';
421  bad_face = true; break;
422  }
423 
424  // OBJ indices start from 1. Negative indices indicate counting from the last element.
425  index = (index < 0 ? (intx)vertices.size() + index : index - 1);
426 
427  // Add the referenced vertex to the mesh builder if it has not already been added
428  typename IndexVertexMap::const_iterator existing = vrefs.find(index);
429  if (existing == vrefs.end())
430  {
431  typename Builder::VertexHandle vref = builder->addVertex(vertices[(size_t)index],
432  (read_opts.store_vertex_indices ? index : -1));
433  if (callback)
434  callback->vertexRead(mesh.get(), index, vref);
435 
436  vrefs[index] = vref;
437  face.push_back(vref);
438  }
439  else
440  face.push_back(existing->second);
441  }
442 
443  if (field_end != std::string::npos)
444  field_begin = line.find_first_not_of(" \t", field_end);
445  }
446 
447  if (!bad_face)
448  {
449  typename Builder::FaceHandle fref = builder->addFace(face.begin(), face.end(),
450  (read_opts.store_face_indices ? num_faces : -1));
451  if (callback)
452  callback->faceRead(mesh.get(), num_faces, fref);
453 
454  num_faces++;
455  }
456  else
457  {
458  if (read_opts.strict)
459  throw Error(std::string(getName()) + ": Malformed face: '" + line + '\'');
460  else
461  THEA_WARNING << getName() << ": Skipping malformed face: '" << line << '\'';
462  }
463  }
464  else if (!read_opts.flatten
465  && ((line[0] == 'g' || line[0] == 'o') && (line.length() < 2 || line[1] == ' ' || line[1] == '\t'))) // group
466  {
467  // Add the previous mesh to the mesh group
468  if (builder)
469  {
470  builder->end();
471  if (builder->numFaces() > 0 || !read_opts.skip_empty_meshes)
472  {
473  if (read_opts.verbose)
474  {
475  THEA_CONSOLE << getName() << ": Mesh " << mesh->getName() << " has " << builder->numVertices()
476  << " vertices and " << builder->numFaces() << " faces";
477  }
478 
479  mesh_group.addMesh(mesh);
480  }
481  }
482 
483  // Read the new group name
484  group_name = trimWhitespace(line.substr(1));
485  if (group_name.empty())
486  group_name = format("%s/AnonymousMesh%d", mesh_group.getName(), ++anon_index);
487 
488  // Create a new mesh and a builder for it
489  mesh = MeshPtr(new Mesh(group_name));
490  vrefs.clear(); // start a new set of vertex handles for the new mesh
491  vtn_refs.clear();
492  bp = std::shared_ptr<Builder>(new Builder(mesh)); // old builder gets destroyed here
493  builder = bp.get();
494  builder->begin();
495  }
496  // Else ignore the line
497  }
498 
499  // Add the final mesh to the mesh group
500  if (builder)
501  {
502  builder->end();
503  if (builder->numVertices() > 0 || !read_opts.skip_empty_meshes)
504  {
505  if (read_opts.verbose)
506  {
507  THEA_CONSOLE << getName() << ": Mesh " << mesh->getName() << " has " << builder->numVertices() << " vertices and "
508  << builder->numFaces() << " faces";
509  }
510 
511  mesh_group.addMesh(mesh);
512  }
513  }
514 
515  THEA_CONSOLE << getName() << ": Read " << mesh_group.numMeshes() << " submesh(es) with a total of " << vertices.size()
516  << " vertices and " << num_faces << " faces";
517  }

520  WriteCallback * callback) const
521  {
522  Codec::BlockHeader bh(this->getMagic());
523  if (write_block_header)

555  template < typename _MeshT, typename std::enable_if< Graphics::IsGeneralMesh<_MeshT>::value
556  || Graphics::IsDCELMesh<_MeshT>::value, int >::type = 0 >
557  void writeVertices(_MeshT const & mesh, BinaryOutputStream & output, VertexIndexMap & vertex_indices,
558  WriteCallback * callback) const
559  {
560  intx vertex_index = (intx)vertex_indices.size() + 1; // OBJ numbers vertices starting from 1
561  for (typename Mesh::VertexConstIterator vi = mesh.verticesBegin(); vi != mesh.verticesEnd(); ++vi, ++vertex_index)
562  {
563  output.printf("v %f %f %f\n", vi->getPosition().x(), vi->getPosition().y(), vi->getPosition().z());
564  vertex_indices[&(*vi)] = vertex_index;
565  if (callback) callback->vertexWritten(&mesh, vertex_index - 1, &(*vi));
566  }
567 
568  if (!write_opts.ignore_normals)
569  {
570  for (typename Mesh::VertexConstIterator vi = mesh.verticesBegin(); vi != mesh.verticesEnd(); ++vi)
571  output.printf("vn %f %f %f\n", vi->getNormal().x(), vi->getNormal().y(), vi->getNormal().z());
572  }
573  }

576  template < typename _MeshT, typename std::enable_if< Graphics::IsDisplayMesh<_MeshT>::value, int >::type = 0 >
577  void writeVertices(_MeshT const & mesh, BinaryOutputStream & output, VertexIndexMap & vertex_indices,
578  WriteCallback * callback) const
579  {
580  typedef std::pair<_MeshT const *, intx> DisplayMeshVRef;
581  typename Mesh::VertexArray const & vertices = mesh.getVertices();
582  intx vertex_index = (intx)vertex_indices.size() + 1; // OBJ numbers vertices starting from 1
583 
584  for (size_t i = 0; i < vertices.size(); ++i, ++vertex_index)
585  {
586  Vector3 const & v = vertices[i];
587  output.printf("v %f %f %f\n", v.x(), v.y(), v.z());
588  vertex_indices[DisplayMeshVRef(&mesh, (intx)i)] = vertex_index;
589  if (callback) callback->vertexWritten(&mesh, vertex_index - 1, (intx)i);
590  }
591 
592  if (!write_opts.ignore_texcoords && mesh.hasTexCoords())
593  {
594  typename Mesh::TexCoordArray const & texcoords = mesh.getTexCoords();
595  alwaysAssertM(texcoords.size() == vertices.size(),
596  std::string(getName()) + ": Mesh has unequal numbers of vertices and texture coordinates");
597 
598  for (size_t i = 0; i < texcoords.size(); ++i)
599  {
600  Vector2 const & t = texcoords[i];
601  output.printf("vt %f %f\n", t.x(), t.y());
602  }
603  }
604 
605  if (!write_opts.ignore_normals && mesh.hasNormals())
606  {
607  typename Mesh::NormalArray const & normals = mesh.getNormals();
608  alwaysAssertM(normals.size() == vertices.size(),
609  std::string(getName()) + ": Mesh has unequal numbers of vertices and normals");
610 
611  for (size_t i = 0; i < normals.size(); ++i)
612  {
613  Vector3 const & n = normals[i];
614  output.printf("vn %f %f %f\n", n.x(), n.y(), n.z());
615  }
616  }
617  }

635  template < typename _MeshT, typename std::enable_if< Graphics::IsGeneralMesh<_MeshT>::value
636  || Graphics::IsDCELMesh<_MeshT>::value, int >::type = 0 >
637  void writeFaces(_MeshT const & mesh, VertexIndexMap const & vertex_indices, BinaryOutputStream & output,
638  WriteCallback * callback, intx & next_index) const
639  {
640  if (write_opts.skip_empty_meshes && mesh.numFaces() <= 0)
641  return;
642 
643  if (!write_opts.flatten)
644  output.printf("\ng %s\n", mesh.getName());
645 
646  for (typename Mesh::FaceConstIterator fi = mesh.facesBegin(); fi != mesh.facesEnd(); ++fi)
647  {
648  typename Mesh::Face const & face = *fi;
649  if (face.numVertices() < 3) continue;
650 
651  std::ostringstream os; os << 'f';
652  for (typename Mesh::Face::VertexConstIterator vi = face.verticesBegin(); vi != face.verticesEnd(); ++vi)
653  {
654  typename VertexIndexMap::const_iterator ii = vertex_indices.find(*vi);
655  alwaysAssertM(ii != vertex_indices.end(), std::string(getName()) + ": Vertex index not found");
656 
657  if (!write_opts.ignore_normals)
658  os << ' ' << ii->second << "//" << ii->second;
659  else
660  os << ' ' << ii->second;
661  }
662 
663  os << '\n';
664  output.writeBytes((int64)os.str().length(), os.str().data());
665 
666  if (callback) callback->faceWritten(&mesh, next_index++, &face);
667  }
668  }

671  template < typename _MeshT, typename std::enable_if< Graphics::IsDisplayMesh<_MeshT>::value, int >::type = 0 >
672  void writeFaces(_MeshT const & mesh, VertexIndexMap const & vertex_indices, BinaryOutputStream & output,
673  WriteCallback * callback, intx & next_index) const
674  {
675  if (write_opts.skip_empty_meshes && mesh.numFaces() <= 0)
676  return;
