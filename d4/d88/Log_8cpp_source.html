34 std::string
35 currentDateTimeToString()
36 {
37  std::time_t raw_time;
38  std::tm const * time_info;
39 
40  std::time(&raw_time); // get raw system time
41  time_info = std::localtime(&raw_time); // break it down into components w.r.t. the current locale
42  std::ostringstream os;
43  os << 1900 + time_info->tm_year << '-'
44  << std::setw(2) << std::right << std::setfill('0') << 1 + time_info->tm_mon << '-'
45  << std::setw(2) << std::right << std::setfill('0') << time_info->tm_mday << ' '
46  << std::setw(2) << std::right << std::setfill('0') << time_info->tm_hour << ':'
47  << std::setw(2) << std::right << std::setfill('0') << time_info->tm_min << ':'
48  << std::setw(2) << std::right << std::setfill('0') << time_info->tm_sec;
49  return os.str();
50 }
