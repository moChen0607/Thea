68 uint32
69 Random::bits()
70 {
71  // See http://en.wikipedia.org/wiki/Mersenne_twister
72 
73  // Make a local copy of the index variable to ensure that it is not out of bounds
74  int32 localIndex = index;
75 
76  // Automatically checks for index < 0 if corrupted by unsynchronized threads.
77  if ((uint32)localIndex >= (uint32)N)
78  {
79  generate();
80  localIndex = 0;
81  }
82 
83  // Increment the global index. It may go out of bounds on multiple threads, but the above check ensures that the array index
84  // actually used never goes out of bounds. It doesn't matter if we grab the same array index twice on two threads, since the
85  // distribution of random numbers will still be uniform.
86  ++index;
87 
88  // Return the next random in the sequence
89  uint32 r = state[localIndex];
90 
91  // Temper the result
92  r ^= r >> U;
93  r ^= (r << S) & B;
94  r ^= (r << T) & C;
95  r ^= r >> L;
96  return r;
97 }

99 void
100 Random::generate()
101 {
102  // Lower R bits
103  static uint32 const LOWER_MASK = (1LU << R) - 1;
104 
105  // Upper (32 - R) bits
106  static uint32 const UPPER_MASK = 0xFFFFFFFF << R;
107  static uint32 const mag01[2] = { 0UL, (uint32)A };
108 
109  if (m_threadsafe)
110  lock.lock();
111 
112  // First N - M bits
113  for (unsigned int i = 0; i < N - M; ++i)
114  {
115  uint32 x = (state[i] & UPPER_MASK) | (state[i + 1] & LOWER_MASK);
116  state[i] = state[i + M] ^ (x >> 1) ^ mag01[x & 1];
117  }
118 
119  // Remaining bits
120  for (unsigned int i = N - M + 1; i < N - 1; ++i)
121  {
122  uint32 x = (state[i] & UPPER_MASK) | (state[i + 1] & LOWER_MASK);
123  state[i] = state[i + (M - N)] ^ (x >> 1) ^ mag01[x & 1];
124  }
125 
126  uint32 y = (state[N - 1] & UPPER_MASK) | (state[0] & LOWER_MASK);
127  state[N - 1] = state[M - 1] ^ (y >> 1) ^ mag01[y & 1];
128  index = 0;
129 
130  if (m_threadsafe)
131  lock.unlock();
132 }

159 Real
160 Random::gaussian(Real mean, Real stddev)
161 {
162  // Using Box-Mueller method from http://www.taygeta.com/random/gaussian.html
163  // Modified to specify standard deviation and mean of distribution
164  Real w, x1, x2;
165 
166  // Loop until w is less than 1 so that log(w) is negative
167  do
168  {
169  x1 = uniform(-1.0f, 1.0f);
170  x2 = uniform(-1.0f, 1.0f);
171  w = (Real)(Math::square(x1) + Math::square(x2));
172  }
173  while (w > 1.0f);
174 
175  // Transform to gassian distribution. Multiply by the variance \a sigma (stddev^2) and add the \a mean.
176  return x2 * (Real)Math::square(stddev) * std::sqrt((-2.0f * std::log(w)) / w) + mean;
177 }

179 void
180 Random::cosHemi(Real & x, Real & y, Real & z)
181 {
182  Real const e1 = uniform01();
183  Real const e2 = uniform01();
184 
185  // Jensen's method
186  Real const sin_theta = std::sqrt(1.0f - e1);
187  Real const cos_theta = std::sqrt(e1);
188  Real const phi = 6.28318531f * e2;
189 
190  x = std::cos(phi) * sin_theta;
191  y = std::sin(phi) * sin_theta;
192  z = cos_theta;
193 
194  // We could also use Malley's method (pbrt p.657), since they are the same cost:
195  //
196  // r = sqrt(e1);
197  // t = 2*pi*e2;
198  // x = cos(t)*r;
199  // y = sin(t)*r;
200  // z = sqrt(1.0 - x*x + y*y);
201 }

204 Random::cosPowHemi(Real const k, Real & x, Real & y, Real & z)
205 {
206  Real const e1 = uniform01();
207  Real const e2 = uniform01();
208 
209  Real const cos_theta = std::pow(e1, 1.0f / (k + 1.0f));
210  Real const sin_theta = std::sqrt(1.0f - Math::square(cos_theta));
211  Real const phi = 6.28318531f * e2;
212 
213  x = std::cos(phi) * sin_theta;
214  y = std::sin(phi) * sin_theta;
215  z = cos_theta;

225 void
226 Random::sphere(Real & x, Real & y, Real & z)
227 {
228  // Adapted from: Robert E. Knop, "Algorithm 381: Random Vectors Uniform in Solid Angle", CACM, 13, p326, 1970.
229  // It uses the fact that the projection of the uniform 2-sphere distribution onto any axis (in this case the Z axis) is
230  // uniform.
231 
232  Real m2; // squared magnitude
233  do
234  {
235  x = 2 * uniform01() - 1;
236  y = 2 * uniform01() - 1;
237  m2 = x * x + y * y;
238 
239  } while (m2 > 1 || m2 < 1e-10);
240 
241  // The squared length happens to be uniformly distributed in [0, 1]. Since the projection of the 2-sphere distribution onto
242  // the Z axis is uniform (can be derived from the observation that the area of a spherical cap is linear in its height), we
243  // can use the squared length, rescaled to [-1, 1], as the Z value (latitude).
244  z = 2 * m2 - 1;
245 
246  // x and y now locate the longitude of the point after scaling to lie on the sphere
247  Real s = 2 * std::sqrt(1 - m2); // this factor ensures x^2 + y^2 + z^2 = 1
248  x *= s;
249  y *= s;
250 }

252 void
253 Random::unitVector(intx n, Real * v)
254 {
255  // The implementation follows https://mc-stan.org/docs/2_18/reference-manual/unit-vector-section.html and uses the following
256  // result of Marsaglia [1972]: if n coordinates are independently sampled from n 1-D standard normal distributions, and the
257  // resulting vector normalized by dividing by its length, then the unit vector so generated is distributed uniformly on the
258  // (n - 1)-sphere.
259 
260  alwaysAssertM(n >= 1, "Random: Invalid dimensions for sampling unit vectors");
261 
262  while (true) // can theoretically go into a long loop but the chances of even two iterations are vanishingly small
263  {
264  Real sqlen = 0;
265  for (intx i = 0; i < n; ++i)
266  {
267  Real c = gaussian(0, 1);
268  v[i] = c;
269  sqlen += (c * c);
270  }
271 
272  if (sqlen > Math::square(Math::eps<Real>()))
273  {
274  Real len = std::sqrt(sqlen);
275  for (intx i = 0; i < n; ++i)
276  v[i] /= len;
277 
278  break;
279  }
280  }
281 }

302 void
303 Random::sortedIntegers(int32 lo, int32 hi, int32 m, int32 * selected)
304 {
305  int32 remaining = m;
306  for (int32 i = lo; i <= hi; ++i)
307  {
308  // Select m of remaining n - i
309  int32 r = RandomInternal::bigRand(*this);
310  if ((r % (hi - i + 1)) < remaining)
311  {
312  selected[m - remaining] = i;
313  remaining--;
314 
315  if (remaining <= 0)
316  return;
317  }
318  }
