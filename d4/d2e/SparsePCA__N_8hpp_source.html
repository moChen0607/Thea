61 // Principal component analysis of objects that map to single points in N-space.
62 template <int N, typename T, typename ScalarT>
63 class SparsePCA_N<T, N, ScalarT, typename std::enable_if< IsNonReferencedPointN<T, N>::value >::type>
64 {
65  public:
66  typedef Vector<N, ScalarT> VectorT;
67 
68  template <typename InputIterator> static bool compute(InputIterator begin, InputIterator end, ScalarT variances[N],
69  VectorT axes[N], VectorT * centroid = nullptr, ScalarT lambda = -1,
70  ScalarT eps = -1, intx max_iters = -1)
71  {
72  if (lambda < 0)
73  lambda = (ScalarT)1.0 / ((ScalarT)N * (ScalarT)std::log((double)N));
74 
75  if (eps < 0)
76  eps = Math::eps<ScalarT>();
77 
78  if (max_iters < 0)
79  max_iters = 100 * N * N;
80 
81  // Compute initial PCA basis
82  VectorT ctr;
83  if (!centroid)
84  centroid = &ctr;
85 
86  PCA_N<T, N>::compute(begin, end, variances, axes, centroid);
87 
88  // Normalize initial variances
89  ScalarT normalized_variances[N];
90  normalizeVariances<N>(variances, normalized_variances);
91 
92  // Compute initial cost
93  ScalarT old_cost = cost<N>(normalized_variances, axes, lambda);
94 
95  // Compute the lambda for the 2-D sweep
96  ScalarT lambda2 = lambda * (ScalarT)(std::log((double)N) / std::log(2.0));
97 
98  // Iterate to minimize the cost function
99  bool updated = false;
100  for (intx iter = 0; iter < max_iters; ++iter)
101  {
102  bool updated_in_iter = false;
103 
104  for (intx i = 0; i < N; ++i)
105  for (intx j = i + 1; j < N; ++j)
106  {
107  VectorT selected_axes[2] = { axes[i], axes[j] };
108 
109  ScalarT vars[2] = { variances[i], variances[j] };
110  ScalarT nvars[2]; normalizeVariances<2>(vars, nvars);
111  ScalarT best_cost = cost<2>(nvars, selected_axes, lambda2);
112 
113  static intx const ANG_ITERS = 180;
114  for (intx k = 1; k < ANG_ITERS; ++k)
115  {
116  double ang = Math::pi() * (k / (double)ANG_ITERS);
117  double s = std::sin(ang);
118  double c = std::cos(ang);
119  VectorT rot_axes[2] = { c * selected_axes[0] - s * selected_axes[1],
120  s * selected_axes[0] + c * selected_axes[1] };
121 
122  vars[0] = computeVariance(begin, end, rot_axes[0], *centroid);
123  vars[1] = computeVariance(begin, end, rot_axes[1], *centroid);
124  normalizeVariances<2>(vars, nvars);
125 
126  ScalarT rot_cost = cost<2>(nvars, rot_axes, lambda2);
127  if (rot_cost < best_cost)
128  {
129  axes[i] = rot_axes[0]; axes[j] = rot_axes[1];
130  variances[i] = vars[0]; variances[j] = vars[1];
131 
132  best_cost = rot_cost;
133  updated = updated_in_iter = true;
134  }
135  }
136  }
137 
138  if (updated_in_iter)
139  {
140  normalizeVariances<N>(variances, normalized_variances);
141  ScalarT new_cost = cost<N>(normalized_variances, axes, lambda);
142 
143  // THEA_CONSOLE << "Cost after iteration " << iter << " = " << new_cost;
144 
145  if (std::abs(old_cost - new_cost) < eps)
146  {
147  sortAxes(variances, axes);
148  return true;
149  }
150 
151  old_cost = new_cost;
152  }
153  else
154  {
155  if (updated)
156  sortAxes(variances, axes);
157 
158  return true;
159  }
160  }
161 
162  if (updated)
163  sortAxes(variances, axes);
164 
165  THEA_WARNING << "SparsePCA" << N << ": No convergence after " << max_iters
166  << " iterations, returning best result found so far (cost = " << old_cost << ')';
167  return false;
168  }

170  private:
171  // Compute variance along a basis direction.
172  template <typename InputIterator>
173  static ScalarT computeVariance(InputIterator begin, InputIterator end, VectorT const & axis, VectorT const & centroid)
174  {
175  ScalarT var = 0;
176  for (auto pi = makeRefIterator(begin); pi != makeRefIterator(end); ++pi)
177  var += Math::square(axis.dot(PointTraitsN<T, N, ScalarT>::getPosition(*pi) - centroid));
178 
179  return var;
180  }
181 
182  // Compute normalized variances.
183  template <int M> static void normalizeVariances(ScalarT const * variances, ScalarT * normalized_variances)
184  {
185  ScalarT sum = 0;
186  for (intx i = 0; i < M; ++i)
187  sum += variances[i];
188 
189  if (sum > Math::eps<ScalarT>())
190  {
191  for (intx i = 0; i < M; ++i)
192  normalized_variances[i] = variances[i] / sum;
193  }
194  else
195  {
196  for (intx i = 0; i < M; ++i)
197  normalized_variances[i] = 0;
198  }
199  }
200 
201  // Cost function.
202  template <int M> static ScalarT cost(ScalarT const * normalized_variances, VectorT axes[M], ScalarT lambda)
203  {
204  return cost1<M>(normalized_variances) + lambda * cost2<M>(axes);
205  }
206 
207  // First term of cost function.
208  template <int M> static ScalarT cost1(ScalarT const * normalized_variances)
209  {
210  // \sum_{i = 0}^{N - 1} -d_m \log d_m
211 
212  ScalarT c = 0;
213  for (intx i = 0; i < M; ++i)
214  if (normalized_variances[i] > Math::eps<ScalarT>())
215  c += -normalized_variances[i] * std::log(normalized_variances[i]);
216 
217  return c;
218  }
219 
220  // Second term of cost function.
221  template <int M> static ScalarT cost2(VectorT axes[M])
222  {
223  // \sum_{i = 0}^{N - 1} \sum_{j = 0}^{N - 1} -axes_{i, j}^2 \log -axes_{i, j}^2
224 
225  ScalarT c = 0;
226  for (intx i = 0; i < M; ++i)
227  for (intx j = 0; j < N; ++j)
228  {
229  ScalarT sq = axes[i][j] * axes[i][j];
230  if (sq > Math::eps<ScalarT>())
231  c += -sq * std::log(sq);
232  }
233 
234  return c;
235  }
236 
237  // Sort axes from largest to smallest variance.
238  static void sortAxes(ScalarT variances[N], VectorT axes[N])
239  {
240  for (intx i = 0; i < N; ++i)
241  for (intx j = i + 1; j < N; ++j)
242  {
243  if (variances[i] < variances[j])
244  {
245  std::swap(variances[i], variances[j]);
246  std::swap(axes[i], axes[j]);
247  }
248  }
249  }
