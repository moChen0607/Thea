89  typedef UnorderedMap<intx, size_t> ComponentIndexMap;
90  ComponentIndexMap component_indices;
91 
92  // Loop over faces, adding each to the appropriate result subarray
93  size_t component_index;
94  for (size_t i = 0; i < faces.size(); ++i)
95  {
96  FaceT * face = faces[i];
97  intx rep = uf.find(uf.getObjectID(face));
98  typename ComponentIndexMap::iterator existing = component_indices.find(rep);
99  if (existing == component_indices.end())
100  {
101  component_index = component_indices.size();
102  component_indices[rep] = component_index;
103  components[component_index].clear();
104  components[component_index].reserve(uf.sizeOfSet(rep));
105  }
106  else
107  component_index = existing->second;
108 
109  components[component_index].push_back(face);
110  }
111 
112  return (intx)components.size();

121  typename std::enable_if< Graphics::IsGeneralMesh<MeshT>::value, int >::type = 0 >
122  static void unifyAdjacentFaces(MeshT & mesh, FaceUnionFind & uf)
123  {
124  for (typename MeshT::EdgeIterator ei = mesh.edgesBegin(); ei != mesh.edgesEnd(); ++ei)
125  {
126  for (typename MeshT::Edge::FaceIterator fi = ei->facesBegin(); fi != ei->facesEnd(); ++fi)
127  for (typename MeshT::Edge::FaceIterator fj = ei->facesBegin(); fj != fi; ++fj)
128  {
129  intx handle1 = uf.getObjectID(*fi);
130  intx handle2 = uf.getObjectID(*fj);
131  uf.merge(handle1, handle2);
132  }
133  }

143  static void unifyAdjacentFaces(MeshT & mesh, FaceUnionFind & uf)
