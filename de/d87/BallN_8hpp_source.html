145  bool rayIntersects(RayN<N, T> const & ray, T max_time = -1) const
146  {
147  if (max_time >= 0)
148  return rayIntersectionTime(ray, max_time) >= 0;
149 
150  VectorT co = ray.getOrigin() - center;
151 
152  T c = co.squaredNorm() - radius * radius;
153  if (c <= 0) // origin is inside ball
154  return true;
155 
156  T a = ray.getDirection().squaredNorm();
157  T b = 2 * co.dot(ray.getDirection());
158 
159  // Solve quadratic a * t^2 + b * t + c = 0
160  T b2 = b * b;
161  T det = b2 - 4 * a * c;
162  if (det < 0) return false;
163 
164  if (a > 0)
165  return b <= 0 || det >= b2;
166  else if (a < 0)
167  return b >= 0 || det >= b2;
168  else
169  return false;
170  }

172  T rayIntersectionTime(RayN<N, T> const & ray, T max_time = -1) const
173  {
174  VectorT co = ray.getOrigin() - center;
175 
176  T c = co.squaredNorm() - radius * radius;
177  if (c <= 0) // origin is inside ball
178  return 0;
179 
180  // We could do an early test to see if the distance from the ray origin to the ball is less than
181  // max_time * ray.getDirection().norm(), but it would involve a square root so might as well solve the quadratic.
182 
183  T a = ray.getDirection().squaredNorm();
184  T b = 2 * co.dot(ray.getDirection());
185 
186  // Solve quadratic a * t^2 + b * t + c = 0
187  T det = b * b - 4 * a * c;
188  if (det < 0) return -1;
189 
190  T d = std::sqrt(det);
191  T t = -1;
192  if (a > 0)
193  {
194  T s0 = -b - d;
195  if (s0 >= 0)
196  t = s0 / (2 * a);
197  else
198  {
199  T s1 = -b + d;
200  if (s1 >= 0)
201  t = s1 / (2 * a);
202  }
203  }
204  else if (a < 0)
205  {
206  T s0 = -b + d;
207  if (s0 <= 0)
208  t = s0 / (2 * a);
209  else
210  {
211  T s1 = -b - d;
212  if (s1 <= 0)
213  t = s1 / (2 * a);
214  }
215  }
216 
217  if (max_time >= 0 && t > max_time)
218  return -1;
219  else
220  return t;
221  }
