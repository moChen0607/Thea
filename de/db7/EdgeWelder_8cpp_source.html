92  void * getDirectedEdge(Vector3 const & e0, Vector3 const & e1) const
93  {
94  typedef std::pair<EdgeMap::const_iterator, EdgeMap::const_iterator> IteratorRange;
95  Long3Pair base_key(toGrid(e0), toGrid(e1));
96 
97  // Every edge within welding distance of the given edge must be inside a neighbor of the grid cell containing the edge
98  for (int dx0 = -1; dx0 <= 1; ++dx0)
99  for (int dy0 = -1; dy0 <= 1; ++dy0)
100  for (int dz0 = -1; dz0 <= 1; ++dz0)
101  for (int dx1 = -1; dx1 <= 1; ++dx1)
102  for (int dy1 = -1; dy1 <= 1; ++dy1)
103  for (int dz1 = -1; dz1 <= 1; ++dz1)
104  {
105  Long3Pair key(Long3(base_key.first.x + dx0, base_key.first.y + dy0, base_key.first.z + dz0),
106  Long3(base_key.second.x + dx1, base_key.second.y + dy1, base_key.second.z + dz1));
107  IteratorRange nbrs = edge_map.equal_range(key);
108  for (EdgeMap::const_iterator ni = nbrs.first; ni != nbrs.second; ++ni)
109  {
110  if ((ni->second.e0 - e0).squaredNorm() <= squared_weld_radius
111  && (ni->second.e1 - e1).squaredNorm() <= squared_weld_radius)
112  return ni->second.edge;
113  }
114  }
115 
116  return nullptr;
117  }
