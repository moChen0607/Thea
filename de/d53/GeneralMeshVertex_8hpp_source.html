189  void updateNormal()
190  {
191  if (!faces.empty())
192  {
193  Vector3 sum_normals = Vector3::Zero();
194  for (FaceConstIterator fi = faces.begin(); fi != faces.end(); ++fi)
195  sum_normals += (*fi)->getNormal(); // weight by face area?
196 
197  normal_normalization_factor = sum_normals.norm();
198  if (normal_normalization_factor < 1e-20f)
199  setNormal(Vector3::Zero());
200  else
201  setNormal(sum_normals / normal_normalization_factor);
202  }
203  else
204  {
205  setNormal(Vector3::Zero());
206  normal_normalization_factor = 0;
207  }
208 
209  has_precomputed_normal = false;
210  }
