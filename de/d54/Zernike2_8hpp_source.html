156 template <int N, typename T, typename ScalarT>
157 intx
158 Zernike2::compute(AbstractAddressableMatrix<T> const & distrib, double center_x, double center_y, double radius,
159  Zernike2::MomentMatrix<N, ScalarT> & moments) const
160 {
161  alwaysAssertM(radius > 0, "Zernike2: Radius must be greater than zero");
162 
163  this->generateBasisLUT();
164 
165  moments.resize(Eigen::NoChange, numMoments());
166  moments.setZero();
167 
168  intx ncols = distrib.cols();
169  intx nrows = distrib.rows();
170 
171  // Don't go outside the specified radius
172  intx min_x = std::max(0L, (intx)std::ceil (center_x - radius));
173  intx max_x = std::min(ncols - 1, (intx)std::floor(center_x + radius));
174 
175  intx min_y = std::max(0L, (intx)std::ceil (center_y - radius));
176  intx max_y = std::min(nrows - 1, (intx)std::floor(center_y + radius));
177 
178  double r_radius = opts.lut_radius / radius;
179 
180  std::complex<double> x1, x2, x3;
181  double dx, dy, tx, ty;
182  intx ix, iy;
183  intx count = 0;
184  for (intx y = min_y; y <= max_y; ++y)
185  {
186  for (intx x = min_x; x <= max_x; ++x)
187  {
188  T const & density = distrib.at(y, x);
189  if (!IsZero<N>::check(density))
190  {
191  dx = x - center_x;
192  dy = y - center_y;
193  tx = dx * r_radius + opts.lut_radius;
194  ty = dy * r_radius + opts.lut_radius;
195  ix = (intx)tx;
196  iy = (intx)ty;
197  dx = tx - ix;
198  dy = ty - iy;
199 
200  // Summation of basis function
201  for (intx p = 0; p < opts.angular_steps; ++p)
202  {
203  for (intx r = 0; r < opts.radial_steps; ++r)
204  {
205  x1 = lut[p][r][ix][iy ] + (lut[p][r][ix + 1][iy ] - lut[p][r][ix][iy ]) * dx;
206  x2 = lut[p][r][ix][iy + 1] + (lut[p][r][ix + 1][iy + 1] - lut[p][r][ix][iy + 1]) * dx;
207  x3 = x1 + (x2 - x1) * dy;
208 
209  Accum<N, ScalarT>::add(density, x3, moments.col(p * opts.radial_steps + r));
210  }
211  }
212 
213  count++;
214  }
215  }
216  }
217 
218  if (count > 0)
219  moments /= (ScalarT)count;
220 
221  return count;
222 }
