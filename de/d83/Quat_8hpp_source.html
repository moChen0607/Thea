282  Quat log() const
283  {
284  if ((v[0] == 0) && (v[1] == 0) && (v[2] == 0))
285  {
286  if (s > 0)
287  {
288  return Quat(0, 0, 0, std::log(s));
289  }
290  else if (s < 0)
291  {
292  // Log of a negative number. Multivalued, any number of the form (PI * v, ln(-q.s))
293  return Quat((Real)Math::pi(), 0, 0, std::log(-s));
294  }
295  else
296  {
297  // Log of zero!
298  static Real const n = Math::nan<Real>();
299  return Quat(n, n, n, n);
300  }
301  }
302  else
303  {
304  // Partly imaginary.
305  Real imag_len = v.norm();
306  Real theta = std::atan2(imag_len, s);
307  Real t = theta / imag_len;
308  return Quat(v * t, std::log(norm()));
309  }
310  }
