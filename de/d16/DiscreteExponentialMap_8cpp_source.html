51  void parametrize(SampleGraph const & sample_graph, intx origin_index_, Vector3 const & u_axis_, Vector3 const & v_axis_,
52  Real radius_)
53  {
54  clear();
55 
56  SurfaceSample * origin_sample = const_cast<SurfaceSample *>(&sample_graph.getSamples()[(size_t)origin_index_]);
57  origin = origin_sample->getPosition();
58  u_axis = u_axis_.normalized(); // renormalize to be safe
59  v_axis = v_axis_.normalized();
60  tangent_plane = Plane3::fromPointAndNormal(origin, u_axis.cross(v_axis));
61  radius = radius_;
62  blend_bandwidth_squared = (options.blendUpwind() ? Math::square(3 * sample_graph.getAverageSeparation()) : 0);
63 
64  geodesics.dijkstraWithCallback(const_cast<SampleGraph &>(sample_graph), origin_sample, std::ref(*this), radius);
65  }

103  // This class also acts as the callback during Dijkstra search. VertexHandle is a pointer to a sample.
104  bool operator()(Geodesics::VertexHandle vertex, double distance, bool has_pred, Geodesics::VertexHandle pred)
105  {
106  ParamData curr_data;
107 
108  if (has_pred)
109  {
110  ParamDataMap::const_iterator existing_pred = param_data.find(pred);
111  alwaysAssertM(existing_pred != param_data.end(), "SampleGraph: No parametrization data associated with predecessor");
112  ParamData const & pred_data = existing_pred->second;
113 
114  Vector3 p = vertex->getPosition();
115  Vector3 sum_p = pred_data.uv_transform * p;
116  Real sum_weights = 1;
117 
118  if (options.blendUpwind())
119  {
120  // Blend in the positions of nearby upwind points (the predecessor's visited neighbors) to make the estimate a bit
121  // more robust
122  SurfaceSample::NeighborSet const & pred_nbrs = pred->getNeighbors();
123  for (int i = 0; i < pred_nbrs.size(); ++i)
124  {
125  SurfaceSample::Neighbor const & pred_nbr = pred_nbrs[i];
126  ParamDataMap::const_iterator existing_nbr = param_data.find(pred_nbr.getSample());
127  if (existing_nbr != param_data.end()) // already assigned parameters
128  {
129  ParamData const & pred_nbr_data = existing_nbr->second;
130 
131  Vector3 offset = pred_nbr.getSample()->getPosition() - p;
132  Real weight = kernelFastGaussianSqDistUnscaled(offset.squaredNorm(), blend_bandwidth_squared);
133 
134  sum_p += weight * (pred_nbr_data.uv_transform * p);
135  sum_weights += weight;
136  }
137  }
138  }
139 
140  // Now do the DEM unwinding based on the averaged position
141  unwind(tangent_plane, sum_p / sum_weights, &pred_data, curr_data);
142  Vector3 offset = curr_data.proj_p - origin;
143  curr_data.uv = Vector2(offset.dot(u_axis), offset.dot(v_axis));
144  curr_data.pred_data = &pred_data; // hopefully the map class guarantees this will be preserved
145 
146  params[vertex->getIndex()] = (options.normalize() ? curr_data.uv / radius : curr_data.uv);
147 
148  // THEA_CONSOLE << "Assigned params " << curr_data.uv;
149  }
150  else
151  {
152  unwind(tangent_plane, vertex->getPosition(), nullptr, curr_data);
153  Vector3 offset = curr_data.proj_p - origin;
154  curr_data.uv = Vector2(offset.dot(u_axis), offset.dot(v_axis));
155  curr_data.pred_data = nullptr;
156 
157  params[vertex->getIndex()] = (options.normalize() ? curr_data.uv / radius : curr_data.uv);
158 
159  // THEA_CONSOLE << "Assigned params " << curr_data.uv << " (no pred)";
160  }
161 
162  param_data[vertex] = curr_data;
163 
164  return false;
165  }

173  // Parametrize the point as an increment from its predecessor. Sets curr.proj_p and curr.uv_transform.
174  void
175  unwind(Plane3 const & tangent_plane, Vector3 pos_in_pred_frame, ParamData const * pred, ParamData & curr)
176  {
177  if (pred)
178  {
179  ParamData const * pred_pred = pred->pred_data;
180  if (pred_pred)
181  {
182  Vector3 edge = pos_in_pred_frame - pred->proj_p;
183  Vector3 prev_edge = pred->proj_p - pred_pred->proj_p;
184 
185  Vector3 right = prev_edge.cross(tangent_plane.getNormal()).normalized();
186  Vector3 v = edge - (edge.dot(right) * right);
187  if (v.squaredNorm() > 1.0e-10f) // TODO: Is this a generally applicable epsilon?
188  {
189  AffineTransform3 pred_inc = AffineTransform3::translation(pred->proj_p)
190  * AffineTransform3::rotationArc(v.normalized(), prev_edge.normalized(), false)
191  * AffineTransform3::translation(-pred->proj_p);
192  curr.proj_p = pred_inc * pos_in_pred_frame;
193  curr.uv_transform = pred_inc * pred->uv_transform;
194  }
195  else
196  {
197  curr.proj_p = tangent_plane.closestPoint(pos_in_pred_frame);
198  curr.uv_transform = AffineTransform3::translation(curr.proj_p - pos_in_pred_frame) * pred->uv_transform;
199  }
200  }
201  else // the predecessor is the source
202  {
203  Vector3 edge = pos_in_pred_frame - pred->proj_p;
204  Vector3 tp = tangent_plane.closestPoint(pos_in_pred_frame);
205  Vector3 v = tp - pred->proj_p;
206  if (v.squaredNorm() > 1.0e-10f) // TODO: Is this a generally applicable epsilon?
207  {
208  AffineTransform3 pred_inc = AffineTransform3::translation(pred->proj_p)
209  * AffineTransform3::rotationArc(edge.normalized(), v.normalized(), false)
210  * AffineTransform3::translation(-pred->proj_p);
211  curr.proj_p = pred_inc * pos_in_pred_frame;
212  curr.uv_transform = pred_inc * pred->uv_transform;
213  }
214  else // hope this never happens, because we're going to squash the edge to zero length
215  {
216  curr.proj_p = tp;
217  curr.uv_transform = AffineTransform3::translation(curr.proj_p - pos_in_pred_frame) * pred->uv_transform;
218  }
219  }
220  }
221  else // this is the source
222  {
223  // Map to origin
224  Vector3 tp = tangent_plane.closestPoint(pos_in_pred_frame); // just in case, but it should be on the plane anyway
225  curr.proj_p = tp;
226  curr.uv_transform = AffineTransform3::translation(tp - pos_in_pred_frame);
227  }
228  }
