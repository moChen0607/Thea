470  void read_AUTO(BinaryInputStream & input, bool read_block_header, ReadCallback * callback)
471  {
472  if (read_block_header)
473  {
474  // Try to identify the codec by the magic string
475  Codec::BlockHeader header; header.read(input);
476  MeshCodec<Mesh> const * codec = nullptr;
477  intx codec_index = 0;
478  while ((codec = getDefaultCodec(codec_index++)))
479  if (codec->getMagic() == header.magic)
480  {
481  codec->readMeshGroup(*this, input, &header, callback);
482  return;
483  }
484  }
485  else
486  {
487  // Try to identify by filename extension
488  MeshCodec<Mesh> const * codec = codecFromPath(input.getPath());
489  if (codec)
490  {
491  codec->readMeshGroup(*this, input, nullptr, callback);
492  return;
493  }
494  }
495 
496  throw Error(getNameStr() + ": Could not detect mesh encoding from input. Please specify the encoding explicitly.");
497  }

500  static MeshCodec<Mesh> const * codecFromPath(std::string path)
501  {
502  path = toLower(path);
503  MeshCodec<Mesh> const * codec = nullptr;
504  intx codec_index = 0;
505  while ((codec = getDefaultCodec(codec_index++)))
506  {
507  for (size_t j = 0; j < codec->getExtensions().size(); ++j)
508  if (endsWith(path, '.' + codec->getExtensions()[j]))
509  return codec;
510  }
511 
512  return nullptr;
513  }

516  static MeshCodec<Mesh> const * getDefaultCodec(intx index)
517  {
518  // A set of default codecs that should be implemented for each mesh type
519  static Codec3DS<Mesh> const codec_3DS;
520  static CodecOBJ<Mesh> const codec_OBJ;
521  static CodecOFF<Mesh> const codec_OFF;
522  static CodecPLY<Mesh> const codec_PLY;
523  static MeshCodec<Mesh> const * codecs[] = { &codec_3DS, &codec_OBJ, &codec_OFF, &codec_PLY };
524  static int NUM_CODECS = (intx)(sizeof(codecs) / sizeof(MeshCodec<Mesh> const *));
525 
526  if (index >= 0 && index < NUM_CODECS)
527  return codecs[index];
528  else
529  return nullptr;
530  }
