52 void
53 GLShader::attachModuleFromString(ModuleType type, char const * source)
54 {
55  GLenum gltype;
56  switch (type)
57  {
58  case ModuleType::VERTEX: gltype = GL_VERTEX_SHADER_ARB; break;
59  case ModuleType::FRAGMENT: gltype = GL_FRAGMENT_SHADER_ARB; break;
60  case ModuleType::GEOMETRY:
61  {
62  if (THEA_GL_SUPPORTS(EXT_geometry_shader4))
63  gltype = GL_GEOMETRY_SHADER_EXT;
64  else if (THEA_GL_SUPPORTS(ARB_geometry_shader4))
65  gltype = GL_GEOMETRY_SHADER_ARB; // this should have exactly the same value as the EXT version, but let's play safe
66  else
67  throw Error(std::string(getName()) + ": This OpenGL installation does not support geometry shaders");
68 
69  break;
70  }
71  default:
72  throw Error(std::string(getName()) + ": Unknown module type");
73  }
74 
75  GLhandleARB module_id = glCreateShaderObjectARB(gltype);
76  THEA_CHECK_GL_OK
77 
78  int32 src_length = (int32)std::strlen(source);
79  GLcharARB const * src_ptr = (GLcharARB const *)source;
80  glShaderSourceARB(module_id, 1, &src_ptr, &src_length);
81  THEA_CHECK_GL_OK
82 
83  glCompileShaderARB(module_id);
84  checkBuildStatus(module_id, GL_OBJECT_COMPILE_STATUS_ARB, "Failed to compile shader module");
85 
86  glAttachObjectARB(program_id, module_id);
87  THEA_CHECK_GL_OK
88 
89  if (type == ModuleType::VERTEX) has_vertex_module = true;
90  if (type == ModuleType::FRAGMENT) has_fragment_module = true;
91  if (has_vertex_module && has_fragment_module) complete = true;
92 
93  // Update the list of active shader uniforms
94  if (complete) readActiveUniforms();
95 
96  linked = false;
97 }

127 void
128 GLShader::readActiveUniforms()
129 {
130  alwaysAssertM(complete, std::string(getName()) + ": GL will throw an error unless the shader is complete");
131 
132  uniforms.clear();
133  int32 last_texture_unit = 0;
134 
135  GLint max_uniform_name_length = 0, num_active_uniforms = 0;
136 
137  // On ATI cards, we are required to call glUseProgramObjectARB before glGetUniformLocationARB,
138  // so we always bind the program first.
139 
140  // First, store the old program.
141  GLhandleARB old_program = glGetHandleARB(GL_PROGRAM_OBJECT_ARB);
142 
143  // The shader must be linked before we can use it
144  link();
145  glUseProgramObjectARB(program_id);
146  THEA_CHECK_GL_OK
147 
148  try
149  {
150  // Get the number of uniforms, and the length of the longest name.
151  glGetObjectParameterivARB(program_id, GL_OBJECT_ACTIVE_UNIFORM_MAX_LENGTH_ARB, &max_uniform_name_length);
152  glGetObjectParameterivARB(program_id, GL_OBJECT_ACTIVE_UNIFORMS_ARB, &num_active_uniforms);
153 
154  Array<GLcharARB> name_chars(max_uniform_name_length);
155 
156  // Loop over glGetActiveUniformARB and store the results away.
157  for (int32 i = 0; i < num_active_uniforms; ++i)
158  {
159  UniformData data;
160 
161  glGetActiveUniformARB(program_id, i, max_uniform_name_length, nullptr, &data.size, &data.type, &name_chars[0]);
162  std::string name(&name_chars[0]);
163 
164  data.location = glGetUniformLocationARB(program_id, &name_chars[0]);
165 
166  bool is_gl_builtin = (data.location == -1 || beginsWith(name, "gl_"));
167  if (!is_gl_builtin)
168  {
169  if (GLShader__isSamplerType(data.type))
170  {
171  if (data.size > 1)
172  throw Error(std::string(getName()) + ": Arrays of samplers are not supported");
173 
174  data.texunit = last_texture_unit++;
175  }
176 
177  uniforms[name] = data;
178  }
179  }
180  }
181  catch (...)
182  {
183  glUseProgramObjectARB(old_program);
184  throw;
185  }
186 
187  glUseProgramObjectARB(old_program);
188  THEA_CHECK_GL_OK
189 }

191 void
192 GLShader::bindUniforms()
193 {
194  for (Uniforms::iterator ui = uniforms.begin(); ui != uniforms.end(); ++ui)
195  {
196  if (!ui->second.has_value)
197  throw Error(std::string(getName()) + ": Uniform '" + ui->first + "' has not been set");
198 
199  if (ui->second.requires_rebind)
200  {
201  ui->second.requires_rebind = false;
202 
203  switch (ui->second.type)
204  {
205  case GL_FLOAT: glUniform1fARB(ui->second.location, ui->second.value.f_val); break;
206  case GL_INT: glUniform1iARB(ui->second.location, ui->second.value.i_val); break;
207  case GL_BOOL: glUniform1iARB(ui->second.location, ui->second.value.i_val); break;
208 
209  case GL_FLOAT_VEC2_ARB: glUniform2fvARB(ui->second.location, ui->second.size, &ui->second.value.f_array[0]); break;
210  case GL_INT_VEC2_ARB: glUniform2ivARB(ui->second.location, ui->second.size, &ui->second.value.i_array[0]); break;
211  case GL_BOOL_VEC2_ARB: glUniform2ivARB(ui->second.location, ui->second.size, &ui->second.value.i_array[0]); break;
212 
213  case GL_FLOAT_VEC3_ARB: glUniform3fvARB(ui->second.location, ui->second.size, &ui->second.value.f_array[0]); break;
214  case GL_INT_VEC3_ARB: glUniform3ivARB(ui->second.location, ui->second.size, &ui->second.value.i_array[0]); break;
215  case GL_BOOL_VEC3_ARB: glUniform3ivARB(ui->second.location, ui->second.size, &ui->second.value.i_array[0]); break;
216 
217  case GL_FLOAT_VEC4_ARB: glUniform4fvARB(ui->second.location, ui->second.size, &ui->second.value.f_array[0]); break;
218  case GL_INT_VEC4_ARB: glUniform4ivARB(ui->second.location, ui->second.size, &ui->second.value.i_array[0]); break;
219  case GL_BOOL_VEC4_ARB: glUniform4ivARB(ui->second.location, ui->second.size, &ui->second.value.i_array[0]); break;
220 
221  case GL_FLOAT_MAT2_ARB: glUniformMatrix2fvARB(ui->second.location, ui->second.size, GL_TRUE,
222  &ui->second.value.f_array[0]); break;
223  case GL_FLOAT_MAT3_ARB: glUniformMatrix3fvARB(ui->second.location, ui->second.size, GL_TRUE,
224  &ui->second.value.f_array[0]); break;
225  case GL_FLOAT_MAT4_ARB: glUniformMatrix4fvARB(ui->second.location, ui->second.size, GL_TRUE,
226  &ui->second.value.f_array[0]); break;
227 
228  case GL_SAMPLER_1D_ARB:
229  case GL_SAMPLER_2D_ARB:
230  case GL_SAMPLER_3D_ARB:
231  case GL_SAMPLER_CUBE_ARB:
232  case GL_SAMPLER_2D_RECT_ARB:
233  case GL_SAMPLER_2D_SHADOW_ARB:
234  case GL_SAMPLER_2D_RECT_SHADOW_ARB:
235  {
236  GLenum tex_type = GLShader__toCanonicalType(ui->second.type);
237  glEnable(tex_type);
238 
239  if (THEA_GL_SUPPORTS(ARB_multitexture))
240  glActiveTextureARB(GL_TEXTURE0_ARB + ui->second.texunit);
241  else
242  {
243  debugAssertM(ui->second.texunit == 0,
244  std::string(getName()) + ": Multitexturing not supported, texture unit must be zero");
245  }
246 
247  glBindTexture(tex_type, ui->second.value.texture->getGLID());
248  glUniform1iARB(ui->second.location, ui->second.texunit);
249  ui->second.requires_rebind = true; // samplers must always be rebound
250  break;
251  }
252 
253  default:
254  throw Error(std::string(getName()) + ": Type of uniform '" + ui->first + "' not handled");
255  }
256  }
257  }

285 {

298  }
299  else
300  throw Error(std::string(getName()) + ": " + error_msg + " (unknown error)");
301  }
