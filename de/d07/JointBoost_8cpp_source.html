89 intx
90 JointBoost::train(TrainingData const & training_data_, TrainingData const * validation_data_)
91 {
92  alwaysAssertM(training_data_.numExamples() > 0, "JointBoost: No training examples provided");
93  alwaysAssertM(num_features == training_data_.numFeatures(), "JointBoost: Unexpected number of features in training data");
94 
95  // Create a temporary reference to the training data so we don't have to pass it to every function
96  training_data = &training_data_;
97 
98  // Set values of parameters
99  intx min_rounds = options.min_boosting_rounds;
100  intx max_rounds = options.max_boosting_rounds;
101 
102  if (min_rounds <= 0)
103  min_rounds = 1;
104 
105  if (max_rounds < 0)
106  max_rounds = num_classes;
107 
108  if (max_rounds < min_rounds)
109  max_rounds = min_rounds;
110 
111  double min_fractional_error_reduction = options.min_fractional_error_reduction;
112  if (min_fractional_error_reduction < 0)
113  min_fractional_error_reduction = 0.001;
114 
115  feature_sampling_fraction = options.feature_sampling_fraction;
116  if (feature_sampling_fraction <= 0)
117  feature_sampling_fraction = 1.0 / max_rounds;
118 
119  max_thresholds_fraction = options.max_thresholds_fraction;
120  if (max_thresholds_fraction <= 0)
121  max_thresholds_fraction = 1.0; // test all features as thresholds by default
122 
123  // Get class names from training data, if available
124  training_data->getClassNames(class_names);
125  alwaysAssertM(class_names.empty() || (intx)class_names.size() == num_classes,
126  "JointBoost: Incorrect number of class names specified");
127 
128  // Get classes for training data
129  Array<intx> classes;
130  training_data->getClasses(classes);
131 
132  // Initialize the weights
133  Array<double> training_weights;
134  training_data->getWeights(training_weights);
135  weights.resize(num_classes, (intx)classes.size());
136  if (!training_weights.empty())
137  {
138  for (size_t i = 0; i < training_weights.size(); ++i)
139  for (intx c = 0; c < num_classes; ++c)
140  weights(c, (intx)i) = training_weights[i];
141  }
142  else
143  weights.fill(1);
144 
145  if (options.verbose)
146  {
147  THEA_CONSOLE << "JointBoost: Starting training (" << num_classes << " classes, " << num_features << " features, "
148  << training_data_.numExamples() << " examples)";
149  THEA_CONSOLE << "JointBoost: -- min_rounds = " << min_rounds;
150  THEA_CONSOLE << "JointBoost: -- max_rounds = " << max_rounds;
151  THEA_CONSOLE << "JointBoost: -- min_fractional_error_reduction = " << min_fractional_error_reduction;
152  THEA_CONSOLE << "JointBoost: -- feature_sampling_fraction = " << feature_sampling_fraction;
153  THEA_CONSOLE << "JointBoost: -- max_thresholds_fraction = " << max_thresholds_fraction;
154  THEA_CONSOLE << "JointBoost: -- force_exhaustive = " << options.force_exhaustive;
155  THEA_CONSOLE << "JointBoost: -- force_greedy = " << options.force_greedy;
156  THEA_CONSOLE << "JointBoost: -- verbose = " << options.verbose;
157  }
158 
159  // Do several rounds of boosting, adding a new stump (weak learner) in each round
160  stumps.clear();
161  double error = -1;
162  double validation_error = -1;
163  for (intx round = 0; round < max_rounds; ++round)
164  {
165  if (options.verbose) THEA_CONSOLE << "JointBoost: Training round " << round;
166 
167  // Create the stump for this round
168  SharedStump::Ptr stump(new SharedStump);
169 
170  // Compute the k values for the stump
171  stump->k.resize((size_t)num_classes, 0);
172  for (intx c = 0; c < num_classes; ++c)
173  {
174  double k_numer = 0, k_denom = 0;
175  for (size_t i = 0; i < classes.size(); ++i)
176  {
177  double w = weights(c, (intx)i);
178  int z = (classes[i] == c ? +1 : -1);
179 
180  k_numer += (w * z);
181  k_denom += w;
182  }
183 
184  stump->k[(size_t)c] = (k_denom != 0 ? k_numer / k_denom : 0);
185  }
186 
187  // Optimize this stump
188  double round_err = optimizeStump(*stump, classes);
189  double round_validation_err = -1;
190 
191  if (options.verbose)
192  THEA_CONSOLE << "JointBoost: Error in round " << round << " = " << round_err << " with " << stump->toString();
193 
194  if (round_err <= 0)
195  break;
196 
197  // Check if we should stop at this point
198  if (round >= min_rounds)
199  {
200  // Check if error reduction is below threshold
201  if (error >= 0)
202  {
203  double err_reduction = error - round_err;
204  if (err_reduction < min_fractional_error_reduction * error)
205  {
206  if (options.verbose)
207  THEA_CONSOLE << "JointBoost: Stopping training, fractional error reduction less than threshold";
208 
209  break;
210  }
211  }
212 
213  // Check if error has increased on validation set
214  if (validation_data_)
215  {
216  round_validation_err = computeValidationError(*validation_data_, stump);
217 
218  if (validation_error >= 0 && round_validation_err > validation_error)
219  {
220  if (options.verbose)
221  THEA_CONSOLE << "JointBoost: Stopping training, validation error increased from " << validation_error << " to "
222  << round_validation_err;
223 
224  break;
225  }
226  }
227  }
228 
229  // We have new error values
230  error = round_err;
231  validation_error = round_validation_err;
232 
233  // Add the stump to the strong classifier
234  stumps.push_back(stump);
235 
236  // Update weights
237  Array<double> stump_features;
238  training_data->getFeature(stump->f, stump_features);
239 
240  for (intx c = 0; c < num_classes; ++c)
241  for (size_t i = 0; i < classes.size(); ++i)
242  {
243  // THEA_CONSOLE << " weights[c = " << c << ", i = " << i << "] = " << weights(c, (intx)i);
244 
245  int z = (classes[i] == c ? +1 : -1);
246  double h = (*stump)(stump_features[i], c);
247 
248  // THEA_CONSOLE << " h[i = " << i << ", c = " << c << "] = " << h << ", z = " << z;
249 
250  weights(c, (intx)i) *= std::exp(-z * h);
251 
252  // THEA_CONSOLE << " weights[c = " << c << ", i = " << i << "] = " << weights(c, (intx)i);
253  }
254  }
255 
256  training_data = nullptr;
257 
258  if (options.verbose)
259  {
260  if (validation_data_)
261  {
262  if (validation_error < 0)
263  validation_error = computeValidationError(*validation_data_);
264 
265  THEA_CONSOLE << "JointBoost: Completed training, added " << stumps.size() << " stump(s) with final error " << error
266  << ", validation error " << validation_error;
267  }
268  else
269  THEA_CONSOLE << "JointBoost: Completed training, added " << stumps.size() << " stump(s) with final error " << error;
270  }
271 
272  return (intx)stumps.size();
273 }

275 double
276 JointBoost::optimizeStump(SharedStump & stump, Array<intx> const & stump_classes)
277 {
278  if (num_features <= 0)
279  {
280  THEA_WARNING << "JointBoost: Can't optimize stump with " << num_features << " features";
281  return -1;
282  }
283 
284  // Select a random fraction of the features
285  Array<int32> candidate_features;
286  if (feature_sampling_fraction >= 1)
287  {
288  for (intx f = 0; f < num_features; ++f)
289  candidate_features.push_back(f);
290  }
291  else
292  {
293  int32 num_candidate_features = std::max((int32)Math::round(feature_sampling_fraction * num_features), (int32)1);
294  candidate_features.resize((size_t)num_candidate_features);
295  Random::common().integers(0, (int32)num_features - 1, num_candidate_features, &candidate_features[0]);
296  }
297 
298  if (options.verbose && (intx)candidate_features.size() < num_features)
299  THEA_CONSOLE << "JointBoost: Optimizing over " << candidate_features.size() << " randomly selected feature(s)";
300 
301  SharedStump test = stump;
302  double min_err = -1;
303  for (size_t f = 0; f < candidate_features.size(); ++f)
304  {
305  test.f = candidate_features[f];
306 
307  // Get feature values
308  Array<double> stump_features;
309  training_data->getFeature(test.f, stump_features);
310 
311  // Optimize the stump via an exhaustive or a greedy search over subsets of classes
312  double err = 0;
313  if ((num_classes <= 5 && !options.force_greedy) || options.force_exhaustive)
314  err = optimizeStumpExhaustive(test, stump_features, stump_classes);
315  else
316  err = optimizeStumpGreedy(test, stump_features, stump_classes);
317 
318  // THEA_CONSOLE << "JointBoost: Error when splitting on feature " << test.f << " = " << err;
319 
320  // Did we improve the classification?
321  if (err >= 0 && (min_err < 0 || err < min_err))
322  {
323  min_err = err;
324  stump = test;
325  }
326  }
327 
328  return min_err;
329 }

331 double
332 JointBoost::optimizeStumpExhaustive(SharedStump & stump, Array<double> const & stump_features,
333  Array<intx> const & stump_classes)
334 {
335  // Loop over all possible subsets of classes
336  SharedStump test = stump;
337  uintx num_subsets = ((uintx)1 << num_classes); // assume no more classes than an uintx can hold
338  double min_err = -1;
339  double cum_num_thresholds = 0;
340  for (uintx subset = 1; subset + 1 < num_subsets; ++subset)
341  {
342  test.n = SharingSet((SharingSet::size_type)num_classes, subset);
343 
344  // THEA_CONSOLE << "JointBoost: Testing stump " << test.toString();
345 
346  intx num_thresholds = 0;
347  double err = fitStump(test, stump_features, stump_classes, &num_thresholds);
348  if (err >= 0 && (min_err < 0 || err < min_err))
349  {
350  stump = test;
351  min_err = err;
352  }
353 
354  cum_num_thresholds += num_thresholds;
355  }
356 
357  if (options.verbose)
358  {
359  THEA_CONSOLE << "JointBoost: Generated " << cum_num_thresholds / (num_subsets - 2)
360  << " candidate feature threshold(s) on average";
361  }
362 
363  return min_err;
364 }

366 double
367 JointBoost::optimizeStumpGreedy(SharedStump & stump, Array<double> const & stump_features,
368  Array<intx> const & stump_classes)
369 {
370  Array<SharedStump> candidate_stumps;
371  Array<double> candidate_errors;
372 
373  SharedStump test = stump;
374  SharingSet current_n((SharingSet::size_type)num_classes, 0); // initially empty
375  double cum_num_thresholds = 0;
376  intx num_fitted_stumps = 0;
377  for (intx c = 0; c < num_classes - 1; ++c)
378  {
379  // Add the single new class to the current set that jointly gives the minimum error
380  SharedStump best_stump;
381  double min_err = -1;
382  bool min_found = false;
383  for (intx c = 0; c < num_classes - 1; ++c)
384  {
385  if (current_n[(SharingSet::size_type)c])
386  continue;
387 
388  test.n = current_n;
389  test.n[(SharingSet::size_type)c] = true;
390 
391  // THEA_CONSOLE << "JointBoost: Testing stump " << test.toString();
392 
393  intx num_thresholds = 0;
394  double err = fitStump(test, stump_features, stump_classes, &num_thresholds);
395  if (err >= 0 && (min_err < 0 || err < min_err))
396  {
397  best_stump = test;
398  min_err = err;
399  min_found = true;
400  }
401 
402  cum_num_thresholds += num_thresholds;
403  num_fitted_stumps++;
404  }
405 
406  if (!min_found) // this shouldn't normally happen, it means every attempt to fit gave an error
407  break;
408 
409  candidate_stumps.push_back(best_stump);
410  candidate_errors.push_back(min_err);
411 
412  // THEA_CONSOLE << "JointBoost: Greedy search candidate " << best_stump.toString() << " with error " << min_err;
413 
414  current_n = best_stump.n;
415  }
416 
417  double min_err = -1;
418  for (size_t i = 0; i < candidate_stumps.size(); ++i)
419  {
420  if (min_err < 0 || candidate_errors[i] < min_err)
421  {
422  stump = candidate_stumps[i];
423  min_err = candidate_errors[i];
424  }
425  }
426 
427  if (options.verbose)
428  {
429  THEA_CONSOLE << "JointBoost: Generated " << cum_num_thresholds / num_fitted_stumps
430  << " candidate feature threshold(s) on average";
431  }
432 
433  return min_err;
434 }

456 void
457 getClassificationAccuracy(double split_value, SharingSet const & pos_classes, Array<double> const & features,
458  Array<intx> const & classes, Array<size_t> const & sorted_indices, Array<int> & accuracy)
459 {
460  accuracy.resize(features.size());
461  for (size_t i = 0; i < features.size(); ++i)
462  {
463  size_t index = sorted_indices[i];
464 
465  intx c = classes[index];
466  double f = features[index];
467  bool is_positive = pos_classes[(SharingSet::size_type)c];
468  bool correct = ((is_positive && f > split_value) || (!is_positive && f <= split_value));
469  accuracy[index] = (correct ? +1 : -1);
470  }
471 }

473 double
474 splitQuality(MatrixX<double> const & weights, Array<intx> const & classes, Array<int> const & accuracy)
475 {
476  // We'll define the split quality as
477  //
478  // sum { weights[classes[i], i] * accuracy[i] }
479  //
480  // where accuracy[i] is +1 if example i has a positive class and its feature above the split index, or has a negative class
481  // and is below the split index, and -1 otherwise.
482  //
483  // As mentioned in getCandidateThresholds(), the absolute value of the split quality is relevant, not its sign.
484 
485  double quality = 0;
486  for (size_t i = 0; i < accuracy.size(); ++i)
487  quality += (accuracy[i] * weights(classes[i], (intx)i));
488 
489  return quality;
490 }

492 void
493 getCandidateThresholds(SharingSet const & pos_classes, MatrixX<double> const & weights, Array<double> const & features,
494  Array<intx> const & classes, intx max_thresholds, Array<double> & thresholds)
495 {
496  // A good threshold separates the positive classes from the negative classes. In other words, we want either many positive
497  // examples > theta and many negative examples <= theta, or vice versa. Note that the positive examples need *not* be mostly
498  // on the "positive side of" (greater than) theta -- we just require theta to separate the positive and negative examples
499  // as well as possible. (Hence, the absolute value of splitQuality() is relevant, not its sign.)
500 
501  // Sort the examples by feature value
502  Array<size_t> sorted_indices;
503  sortIndexed(features, sorted_indices);
504 
505  // Get the minimum and maximum feature values
506  double lo = features[0], hi = features[0];
507  for (size_t i = 1; i < features.size(); ++i)
508  {
509  if (features[i] < lo) lo = features[i];
510  if (features[i] > hi) hi = features[i];
511  }
512 
513  // Choose a tolerance in comparing features
514  double tolerance = 1.0e-10 * (hi - lo);
515 
516  // Generate samples and adjust their position to be better cuts
517  typedef UnorderedSet<size_t> IndexSet;
518  IndexSet threshold_set;
519  for (intx t = 0; t < max_thresholds; ++t)
520  {
521  // Start with a random seed
522  size_t index = (size_t)Random::common().integer(0, (int32)sorted_indices.size() - 1);
523  size_t mapped_index = sorted_indices[index];
524  double threshold = features[mapped_index];
525 
526  // Measure the classification accuracy (+/-1) per example, and the overall quality of the split
527  Array<int> accuracy;
528  getClassificationAccuracy(threshold, pos_classes, features, classes, sorted_indices, accuracy);
529  double quality = splitQuality(weights, classes, accuracy);
530 
531  // Iteratively try to improve it by moving one step to the right or left
532  for (size_t round = 0; round < sorted_indices.size(); ++round)
533  {
534  // Get the quality change in each direction
535  double offset_qualities[2] = { 0, 0 };
536  bool equal_quality[2] = { false, false };
537  for (int offset = -1; offset <= 1; offset += 2)
538  {
539  if ((offset < 0 && index == 0) || (offset > 0 && index == sorted_indices.size() - 1))
540  continue;
541 
542  size_t offset_index = (size_t)(index + offset);
543  size_t mapped_offset_index = sorted_indices[offset_index];
544 
545  // THEA_CONSOLE << "index = " << index << ", mapped_offset_index = " << mapped_offset_index;
546 
547  double offset_quality = quality;
548 
549  if (Math::fuzzyEq(features[mapped_offset_index], features[mapped_index], tolerance))
550  {
551  equal_quality[(offset + 1) >> 1] = true;
552  }
553  else
554  {
555  if (offset < 0) // move left
556  {
557  // The current example's classification accuracy is inverted. The offset example remains unchanged.
558  intx c = classes[mapped_index];
559  offset_quality -= 2 * (accuracy[mapped_index] * weights(c, (intx)mapped_index));
560  }
561  else // move right
562  {
563  // The offset example's classification accuracy is inverted. The current example remains unchanged.
564  intx c = classes[mapped_offset_index];
565  offset_quality -= 2 * (accuracy[mapped_offset_index] * weights(c, (intx)mapped_offset_index));
566  }
567  }
568 
569  offset_qualities[(offset + 1) >> 1] = offset_quality;
570  }
571 
572  // Which is the better direction to move?
573  int best_offset = (std::fabs(offset_qualities[0]) > std::fabs(offset_qualities[1]) ? 0 : 1);
574  if (std::fabs(offset_qualities[best_offset]) > std::fabs(quality)
575  || (equal_quality[best_offset] && Random::common().uniform01() < 0.5)) // if equal, move with 50% probability
576  {
577  // If move left, current example's accuracy is inverted
578  if (best_offset == 0)
579  accuracy[mapped_index] = -accuracy[mapped_index];
580 
581  // Move to index + offset
582  index = (size_t)(index + 2 * best_offset - 1);
583  mapped_index = sorted_indices[index];
584  quality = offset_qualities[best_offset];
585 
586  // If move right, offset example's accuracy is inverted
587  if (best_offset == 1)
588  accuracy[mapped_index] = -accuracy[mapped_index];
589  }
590  else if (!equal_quality[best_offset])
591  break;
592  }
593 
594  threshold_set.insert(mapped_index);
595  }
596 
597  thresholds.clear();
598  for (IndexSet::const_iterator ti = threshold_set.begin(); ti != threshold_set.end(); ++ti)
599  thresholds.push_back(features[*ti]);
600 }

604 double
605 JointBoost::fitStump(SharedStump & stump, Array<double> const & stump_features, Array<intx> const & stump_classes,
606  intx * num_generated_thresholds)
607 {
608  using namespace JointBoostInternal;
609 
610  // Find good places to cut the features (candidate values of the threshold theta)
611  Array<double> thresholds;
612 
613 #ifdef JOINT_BOOST_TEST_ALL_THRESHOLDS
614  thresholds = stump_features;
615 #else
616  intx max_thresholds = (intx)std::ceil(max_thresholds_fraction * stump_features.size());
617  if (max_thresholds < (intx)stump_features.size())
618  getCandidateThresholds(stump.n, weights, stump_features, stump_classes, max_thresholds, thresholds);
619  else
620  thresholds = stump_features;
621 #endif
622 
623  if (num_generated_thresholds)
624  *num_generated_thresholds = (intx)thresholds.size();
625 
626  // THEA_CONSOLE << "JointBoost: Generated " << thresholds.size() << " candidate threshold(s) for " << stump.toString();
627  // for (size_t t = 0; t < thresholds.size(); ++t)
628  // THEA_CONSOLE << "JointBoost: -- threshold[" << t << "] = " << thresholds[t];
629 
630  // Precalculate the error term depending on stump.k -- it does not depend on threshold selection
631  double err_k = 0;
632  for (intx c = 0; c < num_classes; ++c)
633  {
634  if (!stump.n[(SharingSet::size_type)c])
635  {
636  for (size_t i = 0; i < stump_classes.size(); ++i)
637  {
638  double w = weights(c, (intx)i);
639  int z = (stump_classes[i] == c ? +1 : -1);
640 
641  err_k += w * Math::square(z - stump.k[(size_t)c]); // k is precalculated outside this function
642  }
643  }
644  }
645 
646  // THEA_CONSOLE << "err_k = " << err_k;
647 
648  // Fit the stump parameters to the observed data
649  intx best_threshold = -1;
650  double min_err = -1;
651  for (size_t t = 0; t < thresholds.size(); ++t)
652  {
653  double theta = thresholds[t];
654  double a_numer = 0, a_denom = 0, b_numer = 0, b_denom = 0;
655  for (intx c = 0; c < num_classes; ++c)
656  {
657  if (stump.n[(SharingSet::size_type)c])
658  {
659  for (size_t i = 0; i < stump_classes.size(); ++i)
660  {
661  double w = weights(c, (intx)i);
662  int z = (stump_classes[i] == c ? +1 : -1);
663 
664  if (stump_features[i] > theta)
665  {
666  a_numer += w * z;
667  a_denom += w;
668  }
669  else
670  {
671  b_numer += w * z;
672  b_denom += w;
673  }
674  }
675  }
676  }
677 
678  double a = (a_denom != 0 ? a_numer / a_denom : 0);
679  double b = (b_denom != 0 ? b_numer / b_denom : 0);
680 
681  double err = (1 - Math::square(a)) * a_denom + (1 - Math::square(b)) * b_denom + err_k;
682 
683  if (best_threshold < 0 || err < min_err)
684  {
685  best_threshold = (intx)t;
686  min_err = err;
687 
688  stump.a = a;
689  stump.b = b;
690  stump.theta = theta;
691  }
692  }
693 
694  return min_err;
695 }

697 intx
698 JointBoost::predict(double const * features, double * class_probabilities) const
699 {
700  intx best_class = -1;
701  double best_H = -1;
702  double sum_probs = 0;
703 
704  for (intx c = 0; c < num_classes; ++c)
705  {
706  double H = 0;
707  for (size_t m = 0; m < stumps.size(); ++m)
708  {
709  SharedStump const & stump = *stumps[m];
710  double h = stump(features[stump.f], c);
711  H += h;
712  }
713 
714  if (class_probabilities)
715  {
716  double p = std::exp(H); // softmax transform, see TextonBoost paper and Friedman/Hastie/Tibshirani
717  class_probabilities[c] = p;
718  sum_probs += p;
719  }
720 
721  if (best_class < 0 || H > best_H)
722  {
723  best_class = c;
724  best_H = H;
725  }
726  }
727 
728  // Normalize probabilities
729  if (class_probabilities && sum_probs > 0)
730  {
731  for (intx c = 0; c < num_classes; ++c)
732  class_probabilities[c] /= sum_probs;
733  }
734 
735  return best_class;
736 }

739 JointBoost::computeValidationError(TrainingData const & validation_data_, SharedStump::Ptr new_stump)
740 {
741  alwaysAssertM(validation_data_.numFeatures() == num_features, "JointBoost: Validation set has different number of features");
742 
743  if (new_stump)
744  stumps.push_back(new_stump);
745 
746  MatrixX<double, MatrixLayout::ROW_MAJOR> validation_features(validation_data_.numExamples(), num_features);
747  Array<double> feat;
748  for (intx i = 0; i < num_features; ++i)
749  {
750  validation_data_.getFeature(i, feat);
751  validation_features.col(i) = VectorXdMap(&feat[0], (intx)feat.size());
752  }
753 
754  Array<intx> validation_classes;
755  Array<double> validation_weights;
756 
757  validation_data_.getClasses(validation_classes);
758  validation_data_.getWeights(validation_weights);
759 
760  alwaysAssertM((intx)validation_classes.size() == validation_features.rows(),
761  "JointBoost: Ground truth classes for validation data do not match number of examples");
762 
763  double err = 0;
764  for (intx i = 0; i < validation_features.rows(); ++i)
765  {
766  intx predicted = predict(validation_features.row(i).data());
767  if (predicted != validation_classes[i])
768  err += (validation_weights.empty() ? 1.0 : validation_weights[i]);
769  }
770 
771  if (new_stump)
772  stumps.pop_back();
773
