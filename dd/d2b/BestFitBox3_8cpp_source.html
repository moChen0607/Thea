74 static Quat
75 rotationArc(Vector3 const & v0, Vector3 const & v1)
76 {
77  Vector3 cross = v0.cross(v1);
78  Real d = v0.dot(v1);
79  Real s = std::sqrt((1 + d) * 2);
80  Real recip = 1.0f / s;
81 
82  Vector3 rc = recip * cross;
83  return Quat(rc, 0.5f * s);
84 }

86 // Convert a plane equation to a rigid transform.
87 static void
88 planeToCFrame(Plane3 const & plane, Vector3 const & centroid, CoordinateFrame3 & cframe)
89 {
90  Vector3 n;
91  Real d;
92  plane.getEquation(n.x(), n.y(), n.z(), d);
93 
94  Matrix3 rot;
95  if (n.y() < -0.999)
96  {
97  rot << 1, 0, 0,
98  0, -1, 0,
99  0, 0, -1;
100  }
101  else
102  {
103  Quat quat = rotationArc(Vector3::UnitY(), n);
104  rot = quat.toRotationMatrix();
105  }
106 
107  cframe = CoordinateFrame3::_fromAffine(AffineTransform3(rot, centroid));
108 }

122 // Compute the OBB for a set of points relative to a transform matrix and see if it is smaller than the current best value. If
123 // so, return it.
124 static void
125 computeOBB(Array<Vector3> const & points, CoordinateFrame3 const & cframe, OBB & current_best_result, bool overwrite)
126 {
127  if (points.empty())
128  return;
129 
130  Vector3 lo, hi;
131  lo = hi = cframe.pointToObjectSpace(points[0]);
132 
133  Vector3 p;
134  for (size_t i = 1; i < points.size(); ++i)
135  {
136  p = cframe.pointToObjectSpace(points[i]);
137  lo = lo.cwiseMin(p);
138  hi = hi.cwiseMax(p);
139  }
140 
141  Vector3 e = hi - lo;
142  double volume = e.x() * e.y() * e.z();
143  if (overwrite || volume < current_best_result.volume)
144  {
145  Vector3 c = 0.5f * (lo + hi);
146  Vector3 he = 0.5f * e;
147  current_best_result = OBB(c - he, c + he, cframe, volume);
148  }
149 }

151 static void
152 computeBestFitOBB(Array<Vector3> const & points, Box3 & result, bool has_up, Vector3 const & up)
153 {
154  if (points.empty())
155  {
156  result = Box3();
157  return;
158  }
159  else if (points.size() == 1)
160  {
161  result = Box3(AxisAlignedBox3(points[0]));
162  return;
163  }
164 
165  Vector3 centroid;
166  CoordinateFrame3 cframe;
167 
168  if (has_up)
169  {
170  centroid = CentroidN<Vector3, 3>::compute(points.begin(), points.end());
171  cframe = CoordinateFrame3::_fromAffine(AffineTransform3(Math::orthonormalBasis(up), centroid));
172  }
173  else
174  {
175  Plane3 plane;
176  LinearLeastSquares3<Vector3>::fitPlane(points.begin(), points.end(), plane, &centroid);
177  planeToCFrame(plane, centroid, cframe);
178  }
179 
180  OBB best_obb;
181  computeOBB(points, cframe, best_obb, true);
182 
183  Matrix3 rot = Math::rotationAxisAngle((has_up ? up : Vector3::UnitY()), Math::degreesToRadians(10));
184  for (int a = 10; a < 180; a += 10)
185  {
186  cframe._setRotation(cframe.getRotation() * rot);
187  computeOBB(points, cframe, best_obb, false);
188  }
189 
190  result = Box3(AxisAlignedBox3(best_obb.lo, best_obb.hi), best_obb.cframe);
191 }
