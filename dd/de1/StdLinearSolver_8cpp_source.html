41  // Solve the linear system Ax = b for a dense double-precision matrix A.
42  template < typename MatrixT, typename ScalarT,
43  typename std::enable_if< std::is_same<typename MatrixT::value_type, ScalarT>::value, int >::type = 0 >
44  bool solve(Eigen::MatrixBase<MatrixT> const & a, ScalarT const * b, AbstractOptions const * options = nullptr)
45  {
46  if (a.rows() < a.cols())
47  THEA_WARNING << "StdLinearSolver: Fewer objectives than dimensions -- the solution will not be unique";
48 
49  has_solution = false;
50  try
51  {
52  if (a.rows() <= 0 || a.cols() <= 0)
53  throw Error("Empty coefficient matrix");
54 
55  intx num_objectives = a.rows();
56  ndims = a.cols();
57 
58  switch (constraint)
59  {
60  case StdLinearSolver::Constraint::NON_NEGATIVE:
61  {
62  if (method != StdLinearSolver::Method::DEFAULT && method != StdLinearSolver::Method::NNLS)
63  throw Error("Unsupported method for non-negative least squares problems");
64 
65  // Values will be overwritten anyway, so make copies
66  MatrixX<double, MatrixLayout::COLUMN_MAJOR> nnls_a = a.template cast<double>(); // NNLS needs Fortran COLUMN-MAJOR
67  VectorX<double> nnls_b = Eigen::Map< VectorX<ScalarT> const >(b, num_objectives).template cast<double>();
68 
69  solution.resize(ndims);
70 
71  double rnorm;
72  Array<double> w((size_t)ndims);
73  Array<double> zz((size_t)num_objectives);
74  Array<int> index((size_t)ndims);
75  int mode;
76 
77  int mda = (int)num_objectives, im = (int)num_objectives, in = (int)ndims;
78 
79  // NOTE: Assume float64 == double, for passing solution vector to NNLS
80  nnls_c(nnls_a.data(), &mda, &im, &in, nnls_b.data(), solution.data(), &rnorm, &w[0], &zz[0], &index[0], &mode);
81 
82  if (mode == 1)
83  has_solution = true;
84  else
85  {
86  switch (mode)
87  {
88  // Should never be 2 since we've checked for this above, but recheck all the same
89  case 2: THEA_DEBUG << "StdLinearSolver: NNLS error (bad problem dimensions)"; break;
90  case 3: THEA_DEBUG << "StdLinearSolver: NNLS error (iteration count exceeded)"; break;
91  default: THEA_DEBUG << "StdLinearSolver: Unknown NNLS error";
92  }
93  }
94 
95  break;
96  }
97 
98  case StdLinearSolver::Constraint::UNCONSTRAINED:
99  {
100  switch (method)
101  {
102  case StdLinearSolver::Method::HOUSEHOLDER_QR:
103  {
104  Eigen::HouseholderQR<typename MatrixT::PlainObject> solver(a);
105  solution = solver.solve(Eigen::Map< VectorX<ScalarT> const >(b, num_objectives));
106  has_solution = true;
107  break;
108  }
109 
110  case StdLinearSolver::Method::DEFAULT: // slower than plain Householder, but more accurate
111  case StdLinearSolver::Method::COL_PIV_HOUSEHOLDER_QR:
112  {
113  Eigen::ColPivHouseholderQR<typename MatrixT::PlainObject> solver(a);
114  if (tolerance >= 0) solver.setThreshold(tolerance);
115  solution = solver.solve(Eigen::Map< VectorX<ScalarT> const >(b, num_objectives));
116  has_solution = true;
117  break;
118  }
119 
120  case StdLinearSolver::Method::FULL_PIV_HOUSEHOLDER_QR:
121  {
122  Eigen::FullPivHouseholderQR<typename MatrixT::PlainObject> solver(a);
123  if (tolerance >= 0) solver.setThreshold(tolerance);
124  solution = solver.solve(Eigen::Map< VectorX<ScalarT> const >(b, num_objectives));
125  has_solution = true;
126  break;
127  }
128 
129  case StdLinearSolver::Method::COMPLETE_ORTHOGONAL_DECOMPOSITION:
130  {
131  Eigen::CompleteOrthogonalDecomposition<typename MatrixT::PlainObject> solver(a);
132  if (tolerance >= 0) solver.setThreshold(tolerance);
133  solution = solver.solve(Eigen::Map< VectorX<ScalarT> const >(b, num_objectives));
134  has_solution = true;
135  break;
136  }
137 
138  case StdLinearSolver::Method::BDCSVD:
139  {
140  Eigen::BDCSVD<typename MatrixT::PlainObject> solver(a);
141  if (tolerance >= 0) solver.setThreshold(tolerance);
142  solution = solver.solve(Eigen::Map< VectorX<ScalarT> const >(b, num_objectives));
143  has_solution = true;
144  break;
145  }
146 
147  default:
148  throw Error("Unsupported method for unconstrained dense least squares problems");
149  }
150 
151  break;
152  }
153 
154  default:
155  throw Error("Unsupported constraint");
156  }
157  }
158  THEA_STANDARD_CATCH_BLOCKS(return false;, ERROR, "%s",
159  "StdLinearSolver: Error solving dense linear least-squares system")
160 
161  return has_solution;
162  }

165  template < typename MatrixT, typename ScalarT,
166  typename std::enable_if< std::is_same<typename MatrixT::value_type, ScalarT>::value, int >::type = 0 >
167  bool solve(Eigen::SparseMatrixBase<MatrixT> const & a, ScalarT const * b, AbstractOptions const * options = nullptr)
168  {

172  has_solution = false;
173  try

202  // Use one of the iterative solvers to solve the dense or sparse problem. Returns true if a suitable solver was found, NOT
203  // if the problem was successfully solved.
204  template <typename MatrixT, typename ScalarT> bool solveIterative(MatrixT const & a, ScalarT const * b)
205  {
206  switch (method)
207  {
208  case StdLinearSolver::Method::CONJUGATE_GRADIENT:
209  {
210  Eigen::ConjugateGradient<typename MatrixT::PlainObject> solver;
211  if (tolerance >= 0) solver.setTolerance((ScalarT)tolerance);
212  if (max_iters > 0) solver.setMaxIterations(max_iters);
213  solver.compute(a);
214  if (solver.info() == Eigen::Success)
215  {
216  solution = solver.solve(Eigen::Map< VectorX<ScalarT> >(const_cast<ScalarT *>(b), a.rows()));
217  has_solution = (solver.info() == Eigen::Success);
218  }
219  break;
220  }
221 
222  case StdLinearSolver::Method::DEFAULT:
223  case StdLinearSolver::Method::LEAST_SQUARES_CONJUGATE_GRADIENT:
224  {
225  Eigen::LeastSquaresConjugateGradient<typename MatrixT::PlainObject> solver;
226  if (tolerance >= 0) solver.setTolerance((ScalarT)tolerance);
227  if (max_iters > 0) solver.setMaxIterations(max_iters);
228  solver.compute(a);
229  if (solver.info() == Eigen::Success)
230  {
231  solution = solver.solve(Eigen::Map< VectorX<ScalarT> >(const_cast<ScalarT *>(b), a.rows()));
232  has_solution = (solver.info() == Eigen::Success);
233  }
234  break;
235  }
236 
237  case StdLinearSolver::Method::BICGSTAB:
238  {
239  Eigen::BiCGSTAB<typename MatrixT::PlainObject> solver;
240  if (tolerance >= 0) solver.setTolerance((ScalarT)tolerance);
241  if (max_iters > 0) solver.setMaxIterations(max_iters);
242  solver.compute(a);
243  if (solver.info() == Eigen::Success)
244  {
245  solution = solver.solve(Eigen::Map< VectorX<ScalarT> >(const_cast<ScalarT *>(b), a.rows()));
246  has_solution = (solver.info() == Eigen::Success);
247  }
248  break;
249  }
250 
251  default: return false;
252  }
253 
254  return true;
255  }

257  // Use a solver based on sparse factorization, for column-major matrices. Returns true if a suitable solver was found, NOT
258  // if the problem was successfully solved.
259  template < typename MatrixT, typename ScalarT,
260  typename std::enable_if< !(MatrixT::Flags & Eigen::RowMajorBit), int >::type = 0 >
261  bool solveSparseFactorize(MatrixT const & a, ScalarT const * b)
262  {
263  switch (method)
264  {
265  case StdLinearSolver::Method::SIMPLICIALT_LLT:
266  {
267  Eigen::SimplicialLLT<typename MatrixT::PlainObject> solver;
268  solver.compute(a);
269  if (solver.info() == Eigen::Success)
270  {
271  solution = solver.solve(Eigen::Map< VectorX<ScalarT> >(const_cast<ScalarT *>(b), a.rows()));
272  has_solution = (solver.info() == Eigen::Success);
273  }
274  break;
275  }
276 
277  case StdLinearSolver::Method::SIMPLICIALT_LDLT:
278  {
279  Eigen::SimplicialLDLT<typename MatrixT::PlainObject> solver;
280  solver.compute(a);
281  if (solver.info() == Eigen::Success)
282  {
283  solution = solver.solve(Eigen::Map< VectorX<ScalarT> >(const_cast<ScalarT *>(b), a.rows()));
284  has_solution = (solver.info() == Eigen::Success);
285  }
286  break;
287  }
288 
289  case StdLinearSolver::Method::SPARSE_LU:
290  {
291  Eigen::SparseLU<typename MatrixT::PlainObject> solver;
292  if (tolerance >= 0) solver.setPivotThreshold((ScalarT)tolerance);
293  solver.compute(a);
294  if (solver.info() == Eigen::Success)
295  {
296  solution = solver.solve(Eigen::Map< VectorX<ScalarT> >(const_cast<ScalarT *>(b), a.rows()));
297  has_solution = (solver.info() == Eigen::Success);
298  }
299  break;
300  }
301 
302  case StdLinearSolver::Method::DEFAULT:
303  case StdLinearSolver::Method::SPARSE_QR:
304  {
305  Eigen::SparseQR<typename MatrixT::PlainObject, Eigen::AMDOrdering<typename MatrixT::StorageIndex> > solver;
306  if (tolerance >= 0) solver.setPivotThreshold((ScalarT)tolerance);
307  solver.compute(a);
308  if (solver.info() == Eigen::Success)
309  {
310  solution = solver.solve(Eigen::Map< VectorX<ScalarT> >(const_cast<ScalarT *>(b), a.rows()));
311  has_solution = (solver.info() == Eigen::Success);
312  }
313  break;
314  }
315 
316  default: return false;
317  }
318 
319  return true;
320  }

344 // Given a storage index type, dispatch a call to StdLinearSolverImpl::solve() with the correct pointer conversions
345 template <MatrixLayout::Value L, typename ScalarT>
346 bool
347 implSolve(StdLinearSolverImpl * impl, int storage_type, intx nr, intx nc, intx nnz, void const * in, void const * out,
348  ScalarT const * val, void const * nzc, ScalarT const * b, AbstractOptions const * opt)
349 {
350  void * in2 = const_cast<void *>(in);
351  void * out2 = const_cast<void *>(out);
352  ScalarT * val2 = const_cast<ScalarT *>(val);
353  void * nzc2 = const_cast<void *>(nzc);
354 
355  switch (storage_type)
356  {
357 // A casting bug in the current Eigen prevents us using StorageIndex shorter than an int.
358 // Specifically, OrderingMethods/Amd.h, L104: dense = (std::min)(n-2, dense);
359 // Here n and dense are of type StorageIndex, but n - 2 is of type int for StorageIndex shorter than an int, so std::min fails
360 // because the arguments are not of the same type.
361 //
362 // case NumericType::INT8:
363 // return impl->solve(SM<L, int8 >(nr, nc, nnz, (int8 *)in2, (int8 *)out2, val2, (int8 *)nzc2), b, opt);
364 // case NumericType::INT16:
365 // return impl->solve(SM<L, int16 >(nr, nc, nnz, (int16 *)in2, (int16 *)out2, val2, (int16 *)nzc2), b, opt);
366  case NumericType::INT32:
367  return impl->solve(SM<L, int32 >(nr, nc, nnz, (int32 *)in2, (int32 *)out2, val2, (int32 *)nzc2), b, opt);
368  case NumericType::INT64:
369  return impl->solve(SM<L, int64 >(nr, nc, nnz, (int64 *)in2, (int64 *)out2, val2, (int64 *)nzc2), b, opt);
370  default: THEA_ERROR << "StdLinearSolver: Unsupported index type";
371  }
372 
373  return false;
374 }

448 int8
449 StdLinearSolver::solve(AbstractMatrix<float64> const & a, float64 const * b, AbstractOptions const * options)
450 {
451  if (a.asAddressable() && a.asAddressable()->asDense())
452  {
453  AbstractDenseMatrix<float64> const & dm = *a.asAddressable()->asDense();
454  if (dm.isRowMajor())
455  {
456  Eigen::Map< MatrixX<float64, MatrixLayout::ROW_MAJOR> const > wrapped(dm.data(), dm.rows(), dm.cols());
457  return impl->solve(wrapped, b, options);
458  }
459  else // col-major
460  {
461  Eigen::Map< MatrixX<float64, MatrixLayout::COLUMN_MAJOR> const > wrapped(dm.data(), dm.rows(), dm.cols());
462  return impl->solve(wrapped, b, options);
463  }
464  }
465  else if (a.asSparse() && a.asSparse()->asCompressed())
466  {
467  AbstractCompressedSparseMatrix<float64> const & sm = *a.asSparse()->asCompressed();
468  int storage_type = sm.getInnerIndexType();
469  if (storage_type != sm.getOuterIndexType() || storage_type != sm.getNonZeroCountType())
470  {
471  // TODO: Convert to integer arrays of consistent type to work around this problem
472 
473  THEA_ERROR << "StdLinearSolver: Different indices have different storage types -- cannot convert to SparseMatrix";
474  return false;
475  }
476 
477  if (sm.isRowMajor())
478  return StdLinearSolverInternal::implSolve<MatrixLayout::ROW_MAJOR>(impl, storage_type, sm.rows(), sm.cols(),
479  sm.numStoredElements(), sm.getOuterIndices(),
480  sm.getInnerIndices(), sm.getValues(),
481  sm.getNonZeroCounts(), b, options);
482  else
483  return StdLinearSolverInternal::implSolve<MatrixLayout::COLUMN_MAJOR>(impl, storage_type, sm.rows(), sm.cols(),
484  sm.numStoredElements(), sm.getOuterIndices(),
485  sm.getInnerIndices(), sm.getValues(),
486  sm.getNonZeroCounts(), b, options);
487  }
488  else
489  {
490  THEA_ERROR << "StdLinearSolver: Unsupported matrix type";
491  return false;
492  }
