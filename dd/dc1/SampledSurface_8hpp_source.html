92  template <typename TriangleT>
93  Vector3 smoothNormal(TriangleT const & tri, Vector3 const & p)
94  {
95  Vector3 b = tri.barycentricCoordinates(p);
96 
97  Vector3 n0 = tri.getVertices().getVertexNormal(0);
98  Vector3 n1 = tri.getVertices().getVertexNormal(1);
99  Vector3 n2 = tri.getVertices().getVertexNormal(2);
100 
101  return b[0] * n0 + b[1] * n1 + b[2] * n2;
102  }

105  template <typename MeshT>
106  void computeSamples(MeshSampler<MeshT> & sampler, intx num_samples, Array<SurfaceSample> & samples)
107  {
108  typedef typename MeshSampler<MeshT>::Triangle Triangle;
109 
110  static intx const DEFAULT_NUM_SAMPLES = 5000;
111  if (num_samples < 0) num_samples = DEFAULT_NUM_SAMPLES;
112 
113  Array<Vector3> positions;
114  Array<Triangle const *> tris;
115  sampler.sampleEvenlyByArea(num_samples, positions, nullptr, &tris);
116 
117  samples.clear();
118  for (size_t i = 0; i < positions.size(); ++i)
119  samples.push_back(SurfaceSample(positions[i], smoothNormal(*tris[i], positions[i])));
120  }
121 
123  void computeScaleFromSamples()
124  {
125  BestFitSphere3 bsphere;
126  PointCollectorN<BestFitSphere3, 3>(&bsphere).addPoints(samples.begin(), samples.end());
127  scale = bsphere.getDiameter();
128  }

140  SampledSurface(intx num_samples, Vector3 const * positions, Vector3 const * normals, Real normalization_scale = -1)
141  : sample_kdtree(nullptr), precomp_kdtree(nullptr), owns_sample_graph(true), sample_graph(nullptr),
142  scale(normalization_scale)
143  {
144  alwaysAssertM(num_samples >= 0, "SampledSurface: Number of precomputed samples must be non-negative");
145  alwaysAssertM(positions != nullptr, "SampledSurface: Null array of sample positions");
146  alwaysAssertM(normals != nullptr, "SampledSurface: Null array of sample normals");
147 
148  samples.resize((size_t)num_samples);
149  for (size_t i = 0; i < samples.size(); ++i)
150  samples[i] = SurfaceSample(positions[i], normals[i]);

212  SampledSurface(ExternalSampleKDTree const * sample_kdtree_, Real normalization_scale = -1)
213  : sample_kdtree(nullptr), precomp_kdtree(sample_kdtree_), owns_sample_graph(true), sample_graph(nullptr),
