273 DisplayMesh::Face
274 DisplayMesh::addFace(int num_vertices, intx const * face_vertex_indices_, intx source_face_index)
275 {
276  if (num_vertices < 3)
277  {
278  THEA_DEBUG << getName() << ": Skipping face -- too few vertices (" << num_vertices << ')';
279  return Face();
280  }
281 
282  if (num_vertices == 3)
283  return Face(this, 3, true,
284  addTriangle(face_vertex_indices_[0], face_vertex_indices_[1], face_vertex_indices_[2], source_face_index), 1);
285 
286  if (num_vertices == 4)
287  return Face(this, 4, false,
288  addQuad(face_vertex_indices_[0], face_vertex_indices_[1], face_vertex_indices_[2], face_vertex_indices_[3],
289  source_face_index),
290  1);
291 
292  Polygon3 poly;
293  for (int i = 0; i < num_vertices; ++i)
294  {
295  intx vi = face_vertex_indices_[i];
296  debugAssertM(vi >= 0 && vi < (intx)vertices.size(), getName() + format(": Vertex index %ld out of bounds", vi));
297 
298  poly.addVertex(vertices[vi], vi);
299  }
300 
301  intx num_tris = poly.triangulate(triangulated_indices);
302  if (num_tris <= 0)
303  return Face();
304 
305  // debugAssertM(num_tris == 3 * (num_vertices - 2),
306  // getName() + format(": Triangulation of polygonal face yielded %l triangles, whereas %l were expected",
307  // num_tris, num_vertices - 2));
308 
309  alwaysAssertM((source_face_index >= 0 && 3 * tri_source_face_indices.size() == tris.size())
310  || (source_face_index < 0 && tri_source_face_indices.empty()),
311  getNameStr() + ": Mesh must have all or no triangle face source indices");
312 
313  intx starting_index = numTriangles();
314  size_t num_tri_verts = (size_t)(3 * num_tris);
315  for (size_t i = 0; i < num_tri_verts; i += 3)
316  {
317  tris.push_back((uint32)triangulated_indices[i ]);
318  tris.push_back((uint32)triangulated_indices[i + 1]);
319  tris.push_back((uint32)triangulated_indices[i + 2]);
320  if (source_face_index >= 0) tri_source_face_indices.push_back(source_face_index);
321  }
322 
323  invalidateGPUBuffers();
324 
325  return Face(this, num_vertices, true, starting_index, num_tris);
326 }

387 void
388 DisplayMesh::computeAveragedVertexNormals()
389 {
390  bool topo_change = (normals.size() != vertices.size());
391 
392  normals.resize(vertices.size());
393  for (size_t i = 0; i < normals.size(); ++i)
394  normals[i] = Vector3::Zero();
395 
396  // TODO: weight normals by face area?
397  Vector3 n;
398  uint32 i0, i1, i2, i3;
399  for (size_t i = 0; i < tris.size(); i += 3)
400  {
401  i0 = tris[i], i1 = tris[i + 1], i2 = tris[i + 2];
402  Vector3 const & v0 = vertices[i0];
403  Vector3 const & v1 = vertices[i1];
404  Vector3 const & v2 = vertices[i2];
405 
406  n = (v2 - v1).cross(v0 - v1).normalized();
407  normals[i0] += n;
408  normals[i1] += n;
409  normals[i2] += n;
410  }
411 
412  for (size_t i = 0; i < quads.size(); i += 4)
413  {
414  i0 = quads[i], i1 = quads[i + 1], i2 = quads[i + 2], i3 = quads[i + 3];
415  Vector3 const & v0 = vertices[i0];
416  Vector3 const & v1 = vertices[i1];
417  Vector3 const & v2 = vertices[i2];
418 
419  n = (v2 - v1).cross(v0 - v1).normalized();
420  normals[i0] += n;
421  normals[i1] += n;
422  normals[i2] += n;
423  normals[i3] += n;
424  }
425 
426  for (size_t i = 0; i < normals.size(); ++i)
427  normals[i] = normals[i].normalized();
428 
429  invalidateGPUBuffers(topo_change ? BufferID::ALL : BufferID::NORMAL);
430 }

441 void
442 DisplayMesh::updateEdges()
443 {
444  edges.clear();
445 
446  if (wireframe_enabled)
447  {
448  typedef std::pair<uint32, uint32> Edge;
449  typedef UnorderedSet<Edge> EdgeSet;
450 
451  EdgeSet added_edges;
452  Edge edge;
453 
454  for (size_t i = 0; i < tris.size(); i += 3)
455  for (size_t j = 0; j < 3; ++j)
456  {
457  uint32 ei0 = tris[i + j];
458  uint32 ei1 = tris[j == 2 ? i : (i + j + 1)];
459 
460  // Order so lower index is first, since we're considering edges to be undirected
461  if (ei0 < ei1)
462  edge = Edge(ei0, ei1);
463  else
464  edge = Edge(ei1, ei0);
465 
466  EdgeSet::iterator existing = added_edges.find(edge);
467  if (existing == added_edges.end())
468  {
469  edges.push_back(ei0);
470  edges.push_back(ei1);
471  added_edges.insert(edge);
472  }
473  }
474 
475  for (size_t i = 0; i < quads.size(); i += 4)
476  for (size_t j = 0; j < 4; ++j)
477  {
478  uint32 ei0 = quads[i + j];
479  uint32 ei1 = quads[j == 3 ? i : (i + j + 1)];
480 
481  // Order so lower index is first, since we're considering edges to be undirected
482  if (ei0 < ei1)
483  edge = Edge(ei0, ei1);
484  else
485  edge = Edge(ei1, ei0);
486 
487  EdgeSet::iterator existing = added_edges.find(edge);
488  if (existing == added_edges.end())
489  {
490  edges.push_back(ei0);
491  edges.push_back(ei1);
492  added_edges.insert(edge);
493  }
494  }
495  }
496 
497  THEA_DEBUG << getName() << ": Mesh has " << edges.size() / 2 << " edges";
498 }

516  new_vertices.push_back(vertices[i0]);
517  new_vertices.push_back(vertices[i1]);
518  new_vertices.push_back(vertices[i2]);
519 
520  if (!normals.empty())
521  {
522  new_normals.push_back(normals[i0]);

524  new_normals.push_back(normals[i2]);

526 
527  if (!colors.empty())
528  {
529  new_colors.push_back(colors[i0]);
530  new_colors.push_back(colors[i1]);
531  new_colors.push_back(colors[i2]);

533 
534  if (!texcoords.empty())
535  {

537  new_texcoords.push_back(texcoords[i1]);
538  new_texcoords.push_back(texcoords[i2]);
539  }
540 
541  tris[i ] = new_vindex;
542  tris[i + 1] = new_vindex + 1;
543  tris[i + 2] = new_vindex + 2;

610 void
611 DisplayMesh::uploadToGraphicsSystem(RenderSystem & render_system)
612 {
613  if (changed_buffers == 0) return;
614 
615  if (changed_buffers == BufferID::ALL)
616  {
617  if (var_area) var_area->reset();
618  vertices_var = normals_var = colors_var = texcoords_var = tris_var = quads_var = edges_var = nullptr;
619 
620  if (vertices.empty() || (tris.empty() && quads.empty()))
621  {
622  if (var_area)
623  {
624  render_system.destroyVARArea(var_area);
625  var_area = nullptr;
626  }
627 
628  allGPUBuffersAreValid();
629  return;
630  }
631 
632  static int const PADDING = 32;
633  intx vertex_bytes = !vertices.empty() ? 3 * 4 * (intx)vertices.size() + PADDING : 0; // 3 * float
634  intx normal_bytes = hasNormals() ? 3 * 4 * (intx)normals.size() + PADDING : 0; // 3 * float
635  intx color_bytes = hasColors() ? 4 * 4 * (intx)colors.size() + PADDING : 0; // 4 * float
636  intx texcoord_bytes = hasTexCoords() ? 2 * 4 * (intx)texcoords.size() + PADDING : 0; // 2 * float
637 
638  updateEdges();
639 
640 #ifdef THEA_DISPLAY_MESH_NO_INDEX_ARRAY
641  intx num_bytes = vertex_bytes + normal_bytes + color_bytes + texcoord_bytes + PADDING;
642 #else
643  intx tri_bytes = !tris.empty() ? 4 * (intx)tris.size() + PADDING : 0; // uint32
644  intx quad_bytes = !quads.empty() ? 4 * (intx)quads.size() + PADDING : 0; // uint32
645  intx edge_bytes = !edges.empty() ? 4 * (intx)edges.size() + PADDING : 0; // uint32
646 
647  intx num_bytes = vertex_bytes + normal_bytes + color_bytes + texcoord_bytes + tri_bytes + quad_bytes + edge_bytes + PADDING;
648 #endif
649

655 
656  std::string vararea_name = getNameStr() + " VAR area";
657  var_area = render_system.createVARArea(vararea_name.c_str(), num_bytes, VARArea::Usage::WRITE_OCCASIONALLY, true);
658  if (!var_area) throw Error(getNameStr() + ": Couldn't create VAR area");
659  }
660  // Else no need to reset var_area, we've done it above
661  }
662  else
663  {
664  std::string vararea_name = getNameStr() + " VAR area";
665  var_area = render_system.createVARArea(vararea_name.c_str(), num_bytes, VARArea::Usage::WRITE_OCCASIONALLY, true);
666  if (!var_area) throw Error(getNameStr() + ": Couldn't create VAR area");
667  }
668 
669  if (!vertices.empty())
670  {
671  vertices_var = var_area->createArray(vertex_bytes);
672  if (!vertices_var) throw Error(getNameStr() + ": Couldn't create vertices VAR");
673  }
674 
675  if (hasNormals())
676  {
677  normals_var = var_area->createArray(normal_bytes);
678  if (!normals_var) throw Error(getNameStr() + ": Couldn't create normals VAR");
679  }
680 
681  if (hasColors())
682  {
683  colors_var = var_area->createArray(color_bytes);
684  if (!colors_var) throw Error(getNameStr() + ": Couldn't create colors VAR");
685  }
686 
687  if (hasTexCoords())
688  {
689  texcoords_var = var_area->createArray(texcoord_bytes);
690  if (!texcoords_var) throw Error(getNameStr() + ": Couldn't create texcoords VAR");
691  }
692 
693 #ifndef THEA_DISPLAY_MESH_NO_INDEX_ARRAY
694  if (!tris.empty())
695  {
696  tris_var = var_area->createArray(tri_bytes);
697  if (!tris_var) throw Error(getNameStr() + ": Couldn't create triangle indices VAR");
698  }
699 
700  if (!quads.empty())
701  {
702  quads_var = var_area->createArray(quad_bytes);
703  if (!quads_var) throw Error(getNameStr() + ": Couldn't create quad indices VAR");
704  }
705 
706  if (!edges.empty())
707  {
708  edges_var = var_area->createArray(edge_bytes);
709  if (!edges_var) throw Error(getNameStr() + ": Couldn't create edge indices VAR");
710  }
711 
712  if (!tris.empty()) tris_var->updateIndices (0, (intx)tris.size(), &tris[0]);
713  if (!quads.empty()) quads_var->updateIndices(0, (intx)quads.size(), &quads[0]);
714  if (!edges.empty()) edges_var->updateIndices(0, (intx)edges.size(), &edges[0]);
715 #endif
716 
717  if (!vertices.empty()) vertices_var->updateVectors (0, (intx)vertices.size(), &vertices[0]);
718  if (hasNormals()) normals_var->updateVectors (0, (intx)normals.size(), &normals[0]);
719  if (hasColors()) colors_var->updateColors (0, (intx)colors.size(), &colors[0]);
720  if (hasTexCoords()) texcoords_var->updateVectors(0, (intx)texcoords.size(), &texcoords[0]);
721  }
722  else
723  {
724  if (!gpuBufferIsValid(BufferID::VERTEX) && !vertices.empty())
725  vertices_var->updateVectors(0, (intx)vertices.size(), &vertices[0]);
726 
727  if (!gpuBufferIsValid(BufferID::NORMAL) && hasNormals())
728  normals_var->updateVectors (0, (intx)normals.size(), &normals[0]);
729 
730  if (!gpuBufferIsValid(BufferID::COLOR) && hasColors())
731  colors_var->updateColors(0, (intx)colors.size(), &colors[0]);
732 
733  if (!gpuBufferIsValid(BufferID::TEXCOORD) && hasTexCoords())
734  texcoords_var->updateVectors(0, (intx)texcoords.size(), &texcoords[0]);
735  }
736

740 void
741 DisplayMesh::draw(RenderSystem & render_system, AbstractRenderOptions const & options) const
742 {
743  if (options.drawEdges() && !wireframe_enabled)
744  throw Error(getNameStr() + ": Can't draw mesh edges when wireframe mode is disabled");
745 
746  const_cast<DisplayMesh *>(this)->uploadToGraphicsSystem(render_system);
747 
748  if (!vertices_var) return;
749  if (!options.drawFaces() && !options.drawEdges()) return;
750  if (!options.drawFaces() && !edges_var) return;
751  if (!options.drawEdges() && !tris_var && !quads_var) return;
752 
753  render_system.beginIndexedPrimitives();
754 
755  render_system.setVertexArray(vertices_var);
756  if (options.sendNormals() && normals_var) render_system.setNormalArray(normals_var);
757  if (options.sendColors() && colors_var) render_system.setColorArray(colors_var);
758  if (options.sendTexCoords() && texcoords_var) render_system.setTexCoordArray(0, texcoords_var);
759 
760  if (options.drawFaces())
761  {
762  if (options.drawEdges())
763  {
764  render_system.pushShapeFlags();
765  render_system.setPolygonOffset(true, 2);
766  }
767 
768 #ifdef THEA_DISPLAY_MESH_NO_INDEX_ARRAY
769  if (!tris.empty()) render_system.sendIndices(RenderSystem::Primitive::TRIANGLES, (intx)tris.size(), &tris[0]);
770  if (!quads.empty()) render_system.sendIndices(RenderSystem::Primitive::QUADS, (intx)quads.size(), &quads[0]);
771 #else
772  if (!tris.empty())
773  {
774  render_system.setIndexArray(tris_var);
775  render_system.sendIndicesFromArray(RenderSystem::Primitive::TRIANGLES, 0, (intx)tris.size());
776  }
777 
778  if (!quads.empty())
779  {
780  render_system.setIndexArray(quads_var);
781  render_system.sendIndicesFromArray(RenderSystem::Primitive::QUADS, 0, (intx)quads.size());
782  }
783 #endif
784 
785  if (options.drawEdges())
786  render_system.popShapeFlags();
787  }
788 
789  if (options.drawEdges())
790  {
791  render_system.pushShader();
792  render_system.pushColorFlags();
793  render_system.pushTextures();
794 
795  render_system.setShader(nullptr);
796  render_system.setColorArray(nullptr);
797  render_system.setTexCoordArray(0, nullptr);
798  render_system.setNormalArray(nullptr);
799  render_system.setColor(ColorRGBA(options.edgeColor())); // set default edge color (TODO: handle per-edge colors)
800  render_system.setTexture(0, nullptr);
801 
802 #ifdef THEA_DISPLAY_MESH_NO_INDEX_ARRAY
803  if (!edges.empty()) render_system.sendIndices(RenderSystem::Primitive::LINES, (intx)edges.size(), &edges[0]);
804 #else
805  if (!edges.empty())
806  {
807  render_system.setIndexArray(edges_var);
808  render_system.sendIndicesFromArray(RenderSystem::Primitive::LINES, 0, (intx)edges.size());
809  }
810 #endif
811 
812  render_system.popTextures();
813  render_system.popColorFlags();
814  render_system.popShader();
815  }
816 
817  render_system.endIndexedPrimitives();
818 }
