368  template <typename RayQueryStructureT> void init(RayQueryStructureT const * surface)
369  {
370  if (initialized) // don't initialize twice
371  return;
372 
373  // Aggregate samples
374  Array<SurfaceSample *> sample_ptrs(samples.size() + dense_samples.size());
375 
376  for (size_t i = 0; i < samples.size(); ++i)
377  sample_ptrs[i] = &samples[i];
378 
379  for (size_t i = 0; i < dense_samples.size(); ++i)
380  {
381  dense_samples[i].setIndex((intx)(samples.size() + i));
382  sample_ptrs[samples.size() + i] = &dense_samples[i];
383  }
384 
385  // Clear any prior adjacency data
386  for (size_t i = 0; i < sample_ptrs.size(); ++i)
387  sample_ptrs[i]->getNeighbors().clear();
388 
389  if (sample_ptrs.size() <= 1) // nothing to do
390  return;
391 
392  // Create kd-tree on samples
393  SampleKDTree sample_kdtree(sample_ptrs.begin(), sample_ptrs.end());
394 
395  // Get a measure of the average pairwise separation of samples
396  avg_separation = 0;
397  intx num_trials = std::min(100L, (intx)sample_ptrs.size());
398  for (intx i = 0; i < num_trials; ++i)
399  {
400  size_t index = (size_t)Random::common().integer(0, (int32)sample_ptrs.size() - 1);
401  FilterSelf filter(sample_ptrs[index]);
402  sample_kdtree.pushFilter(&filter);
403  intx nn_index = sample_kdtree.closestElement<MetricL2>(sample_ptrs[index]->getPosition());
404  alwaysAssertM(nn_index >= 0, "SampleGraph: Nearest neighbor of sample not found");
405  avg_separation += (sample_ptrs[(size_t)nn_index]->getPosition()
406  - sample_ptrs[index]->getPosition()).squaredNorm();
407  sample_kdtree.popFilter();
408  }
409 
410  avg_separation = std::sqrt(avg_separation / num_trials); // RMS average
411 
412  // Find the neighbors of each sample
413  Real sep_scale = std::sqrt((Real)options.max_degree); // assume uniform distribution on 2D surface
414  for (size_t i = 0; i < sample_ptrs.size(); ++i)
415  findSampleNeighbors(sample_ptrs[i], sample_kdtree, sep_scale * avg_separation, surface);
416 
417  // Extract adjacencies between original set of samples
418  if (!dense_samples.empty())
419  extractOriginalAdjacencies(sample_ptrs);
420 
421  // Update the average separation to the correct value
422  updateAverageSeparation();
423 
424  initialized = true;
425  }

462  template <typename RayQueryStructureT> class NeighborFunctor
463  {
464  public:
466  NeighborFunctor(SurfaceSample * sample_, bool has_normals_, RayQueryStructureT const * surface_)
467  : sample(sample_), has_normals(has_normals_), surface(surface_)
468  {
469  debugAssertM(sample, "SampleGraph: Can't create neighbor functor without valid source sample");
470  }
471 
473  bool operator()(intx index, SurfaceSample * nbr)
474  {
475  Real sep = 0;
476  if (isValidNeighbor(nbr, sep))
477  sample->getNeighbors().insert(SurfaceSample::Neighbor(nbr, sep));
478 
479  return false;
480  }
481 
482  private:
487  bool isValidNeighbor(SurfaceSample const * nbr, Real & sep)
488  {
489  // Tests are (hopefully) in decreasing order of speed
490 
491  // Identity test
492  if (sample == nbr)
493  return false;
494 
495  // Angle test
496  if (has_normals)
497  {
498  static Real const MIN_DOT = -0.5f;
499  if (sample->getNormal().dot(nbr->getNormal()) < MIN_DOT)
500  return false;
501  }
502 
503  // Duplication test which compares only the samples themselves (since numerical error can cause multiple attempts to
504  // insert the same sample with slightly different separations)
505  for (int i = 0; i < sample->numNeighbors(); ++i)
506  if (nbr == sample->getNeighbors()[i].getSample())
507  return false;
508 
509  // Compute the separation of the samples
510  Vector3 diff = nbr->getPosition() - sample->getPosition();
511  Real sqsep = diff.squaredNorm();
512  sep = Math::fastSqrt(sqsep);
513 
514  // Reachability test: lift points off the surface by an amount proportional to their separation, and see if the line
515  // connecting them is blocked by the surface
516  if (surface && has_normals)
517  {
518  static Real const LIFT_FACTOR = 5;
519  Vector3 lift_dir = (sample->getNormal() + nbr->getNormal()).normalized();
520  typename RayQueryStructureT::RayT ray(sample->getPosition() + LIFT_FACTOR * sep * lift_dir, diff);
521  if (surface->template rayIntersects<RayIntersectionTester>(ray, 1))
522  return false;
523  }
524 
525  return true;
526  }
527 
528  SurfaceSample * sample;
529  bool has_normals;
530  RayQueryStructureT const * surface;
531 
532  }; // struct NeighborFunctor

535  template <typename RayQueryStructureT>
536  void findSampleNeighbors(SurfaceSample * sample, SampleKDTree & sample_kdtree, Real init_radius,
537  RayQueryStructureT const * surface)
538  {
539  static int const MAX_ITERS = 3;
540  static Real const RADIUS_EXPANSION_FACTOR = 2.0f;
541  int min_degree = Math::clamp((int)(0.25 * options.max_degree), 4, options.max_degree);
542 
543  Real radius = init_radius;
544  for (int i = 0; i < MAX_ITERS; ++i)
545  {
546  sample->getNeighbors().clear();
547  NeighborFunctor<RayQueryStructureT> func(sample, has_normals, surface);
548 
549  Ball3 nbd(sample->getPosition(), radius);
550  sample_kdtree.processRangeUntil<IntersectionTester>(nbd, func);
551 
552  if (sample->getNeighbors().size() >= min_degree)
553  break;
554  else
555  radius *= RADIUS_EXPANSION_FACTOR;
556  }
557  }

565  void updateAverageSeparation()
566  {
567  avg_separation = 0;
568  intx num_edges = 0; // double-counts, but we'll ignore that for now
569  for (size_t i = 0; i < samples.size(); ++i)
570  {
571  SurfaceSample::NeighborSet const & nbrs = samples[i].getNeighbors();
572  for (int j = 0; j < nbrs.size(); ++j)
573  avg_separation += nbrs[j].getSeparation();
574 
575  num_edges += (intx)nbrs.size();
576  }
577 
578  if (num_edges > 0)
579  avg_separation /= num_edges;
580  }
