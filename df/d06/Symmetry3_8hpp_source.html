54 // Symmetries of point clouds.
55 template <typename T>
56 class /* THEA_API */ Symmetry3<T, typename std::enable_if< IsNonReferencedPointN<T, 3>::value >::type>
57 {
58  public:
59  template <typename InputIterator>
60  static double findPlane(InputIterator begin, InputIterator end, Plane3 & plane,
61  Vector3 const * precomputed_centroid = nullptr, intx num_rounds = -1)
62  {
63  if (begin == end) // no points, early exit
64  return false;
65 
66  if (num_rounds <= 0)
67  num_rounds = 5;
68 
69  Vector3 centroid = (precomputed_centroid ? *precomputed_centroid : CentroidN<T, 3>::compute(begin, end));
70  Real radius = 0;
71  for (auto pi = makeRefIterator(begin); pi != makeRefIterator(end); ++pi)
72  radius = std::max(radius, (PointTraitsN<T, 3>::getPosition(*pi) - centroid).squaredNorm());
73 
74  radius = std::sqrt(radius);
75 
76  Array<Vector3> vertices;
77  Array<intx> triangles;
78  GeodesicSphere3::compute(2, vertices, &triangles);
79 
80  intx best_dir = -1;
81  double best_error = -1;
82  size_t first_vertex_of_round = 0;
83  for (intx round = 0; round < num_rounds; ++round)
84  {
85  for (size_t i = first_vertex_of_round; i < vertices.size(); ++i)
86  {
87  Plane3 candidate = Plane3::fromPointAndNormal(centroid, vertices[i]);
88  double err = symmetryError(begin, end, candidate, centroid, radius);
89  if (best_dir < 0 || err < best_error)
90  {
91  best_dir = (intx)i;
92  best_error = err;
93  plane = candidate;
94  }
95  }
96 
97  THEA_DEBUG << "GeodesicSphere3: Round " << round << ", best plane = " << plane.toString() << ", error = " << best_error;
98 
99  // Further localize the search to a smaller set of directions more tightly clustered around the best one
100  if (round < num_rounds - 1)
101  {
102  first_vertex_of_round = vertices.size();
103  Array<intx> tris_to_subdivide;
104 
105  // Collect triangles incident on the best direction vertex
106  UnorderedSet<intx> nbr_verts;
107  for (size_t i = 0; i < triangles.size(); i += 3)
108  {
109  if ((intx)triangles[i] == best_dir || (intx)triangles[i + 1] == best_dir || (intx)triangles[i + 2] == best_dir)
110  {
111  tris_to_subdivide.push_back(triangles[i ]);
112  tris_to_subdivide.push_back(triangles[i + 1]);
113  tris_to_subdivide.push_back(triangles[i + 2]);
114 
115  if (triangles[i ] != best_dir) nbr_verts.insert(triangles[i ]);
116  if (triangles[i + 1] != best_dir) nbr_verts.insert(triangles[i + 1]);
117  if (triangles[i + 2] != best_dir) nbr_verts.insert(triangles[i + 2]);
118  }
119  }
120 
121  // Collect triangles incident on the one-hop neighbor vertices
122  for (size_t i = 0; i < triangles.size(); i += 3)
123  {
124  if ((intx)triangles[i] == best_dir || (intx)triangles[i + 1] == best_dir || (intx)triangles[i + 2] == best_dir)
125  continue; // already added
126 
127  if (nbr_verts.find((intx)triangles[i ]) != nbr_verts.end()
128  || nbr_verts.find((intx)triangles[i + 1]) != nbr_verts.end()
129  || nbr_verts.find((intx)triangles[i + 2]) != nbr_verts.end())
130  {
131  tris_to_subdivide.push_back(triangles[i ]);
132  tris_to_subdivide.push_back(triangles[i + 1]);
133  tris_to_subdivide.push_back(triangles[i + 2]);
134  }
135  }
136 
137  triangles.clear();
138  GeodesicSphere3::compute(1, vertices, tris_to_subdivide, &triangles);
139  }
140  }
141 
142  return best_error;
143  }

145  private:
147  template <typename InputIterator>
148  static double symmetryError(InputIterator begin, InputIterator end, Plane3 const & plane, Vector3 const & centroid,
149  Real radius)
150  {
151  static size_t const NUM_BINS = 10;
152 
153  double bins[2][NUM_BINS][NUM_BINS]; // 0: negative side, 1: positive side
154  intx count[2][NUM_BINS][NUM_BINS];
155 
156  for (size_t i = 0; i < NUM_BINS; ++i)
157  for (size_t j = 0; j < NUM_BINS; ++j)
158  {
159  bins[0][i][j] = bins[1][i][j] = 0.0;
160  count[0][i][j] = count[1][i][j] = 0;
161  }
162 
163  Vector3 nrm = plane.getNormal();
164  Matrix3 basis = Math::orthonormalBasis(nrm);
165  Vector3 u = basis.col(0);
166  Vector3 v = basis.col(1);
167 
168  intx total_count[2] = { 0, 0 };
169  for (auto pi = makeRefIterator(begin); pi != makeRefIterator(end); ++pi)
170  {
171  Vector3 p = PointTraitsN<T, 3>::getPosition(*pi) - centroid;
172  Real pu = 0.5f * (p.dot(u) / radius + 1);
173  Real pv = 0.5f * (p.dot(v) / radius + 1);
174 
175  int bin_i = Math::clamp((int)(NUM_BINS * pu), 0, NUM_BINS - 1);
176  int bin_j = Math::clamp((int)(NUM_BINS * pv), 0, NUM_BINS - 1);
177 
178  Real pw = p.dot(nrm);
179  if (pw >= 0)
180  {
181  bins[1][bin_i][bin_j] += pw;
182  count[1][bin_i][bin_j]++;
183  total_count[1]++;
184  }
185 
186  if (pw <= 0) // double-count points exactly on the plane, if any
187  {
188  bins[0][bin_i][bin_j] += std::fabs(pw);
189  count[0][bin_i][bin_j]++;
190  total_count[0]++;
191  }
192  }
193 
194  double dist_error = 0, count_error = 0;
195  for (size_t i = 0; i < NUM_BINS; ++i)
196  for (size_t j = 0; j < NUM_BINS; ++j)
197  {
198  intx c0 = count[0][i][j];
199  intx c1 = count[1][i][j];
200  if (c0 == 0 && c1 == 0)
201  continue;
202 
203  double avg0 = c0 > 0 ? bins[0][i][j] / c0 : 0.0;
204  double avg1 = c1 > 0 ? bins[1][i][j] / c1 : 0.0;
205 
206  double weight = c0 + c1;
207  dist_error += (std::fabs(avg0 - avg1) / radius) * weight;
208  count_error += (c0 < c1 ? 1.0 - c0 / (double)c1 : 1.0 - c1 / (double)c0) * weight;
209  }
210 
211  double sum_weights = total_count[0] + total_count[1];
212  dist_error /= sum_weights;
213  count_error /= sum_weights;
214 
215  return 0.5f * (dist_error + count_error);
216  }
