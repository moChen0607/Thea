72 void
73 BestFitEllipsoid3::update() const
74 {
75  if (updated)
76  return;
77 
78  if (points.empty())
79  center = axis[0] = axis[1] = axis[2] = Vector3::Zero();
80  else if (points.size() == 1)
81  {
82  center = points[0];
83  axis[0] = axis[1] = axis[2] = Vector3::Zero();
84  }
85  else
86  {
87 #ifdef THEA_ENABLE_CGAL
88 
89  typedef CGAL::Cartesian<double> Kernel;
90  typedef CGAL::MP_Float ET;
91  typedef CGAL::Approximate_min_ellipsoid_d_traits_3<Kernel, ET> Traits;
92  typedef Traits::Point Point;
93  typedef CGAL::Approximate_min_ellipsoid_d<Traits> AME;
94 
95  Array<Point> p(points.size());
96  for (size_t i = 0; i < points.size(); ++i)
97  {
98  Vector3 const & point = points[i];
99  p[i] = Point(point.x(), point.y(), point.z());
100  }
101 
102  Traits traits;
103  AME ame(eps, p.begin(), p.end(), traits);
104 
105  if (!ame.is_full_dimensional())
106  axis[0] = axis[1] = axis[2] = Vector3::Zero(); // FIXME
107  else
108  {
109  int coord = 0;
110  for (AME::Center_coordinate_iterator c_it = ame.center_cartesian_begin(); coord < 3; ++c_it, ++coord)
111  center[coord] = *c_it;
112 
113  AME::Axes_lengths_iterator l_it = ame.axes_lengths_begin();
114  Real lengths[3];
115  for (int i = 0; i < 3; ++i, ++l_it)
116  {
117  int coord = 0;
118  for (AME::Axes_direction_coordinate_iterator d_it = ame.axis_direction_cartesian_begin(i); coord < 3; ++d_it, ++coord)
119  axis[i][coord] = (Real)(*d_it);
120 
121  lengths[i] = (Real)(*l_it);
122  }
123 
124  if (lengths[0] < lengths[2]) // CGAL bug? Lengths returned in ascending order although spec says otherwise
125  {
126  std::swap(lengths[0], lengths[2]);
127  std::swap(axis[0], axis[2]);
128  }
129 
130  // Compute oriented bounding box before we scale the unit axes by the dimensions of the ellipsoid
131  CoordinateFrame3 cframe(RigidTransform3::_fromAffine(
132  AffineTransform3((Matrix3() << axis[0], axis[1], axis[2]).finished())));
133  Vector3 min, max;
134  for (size_t i = 0; i < points.size(); ++i)
135  {
136  Vector3 op = cframe.pointToObjectSpace(points[i]);
137 
138  if (i == 0)
139  min = max = op;
140  else
141  {
142  min = op.cwiseMin(min);
143  max = op.cwiseMax(max);
144  }
145  }
146 
147  Vector3 obb_center = cframe.pointToWorldSpace((Real)0.5 * (min + max));
148  cframe.setTranslation(obb_center);
149 
150  Vector3 half_extent = (Real)0.5 * (max - min);
151  obb = Box3(AxisAlignedBox3(-half_extent, half_extent), cframe);
152 
153  // Now scale the axes to the dimensions of the ellipsoid
154  axis[0] *= lengths[0];
155  axis[1] *= lengths[1];
156  axis[2] *= lengths[2];
157 
158  // Make sure the axes, in order, form a right-handed system
159  if ((axis[0].cross(axis[1])).dot(axis[2]) < 0)
160  axis[2] = -axis[2];
161  }
162 
163 #else
164 
165  throw FatalError("BestFitEllipsoid3: Requires CGAL for 2 or more points");
166 
167 #endif
168  }
169 
170  updated = true;
171 }
