142  template <typename VertexInputIterator> static Real computeArea(VertexInputIterator vbegin, VertexInputIterator vend)
143  {
144  typedef typename std::iterator_traits<VertexInputIterator>::value_type VertexT;
145 
146  static Real const EPSILON = 1e-10f;
147 
148  Vector3 normal = computeNormal(vbegin, vend);
149  if (normal.squaredNorm() < EPSILON)
150  return 0;
151 
152  VertexInputIterator v0 = vbegin;
153  VertexInputIterator v1 = incrementIterator(v0, vbegin, vend);
154 
155  if (v1 == v0) // polygon is a point
156  return 0;
157 
158  Vector3 p0 = Algorithms::PointTraitsN<VertexT, 3>::getPosition(*v0);
159  Vector3 c = Vector3::Zero();
160  for (VertexInputIterator vi = vbegin; vi != vend; ++vi)
161  {
162  Vector3 p1 = Algorithms::PointTraitsN<VertexT, 3>::getPosition(*v1);
163  c += p0.cross(p1);
164 
165  p0 = p1;
166  v1 = incrementIterator(v1, vbegin, vend);
167  }
168 
169  return std::fabs(0.5f * c.dot(normal));
170  }

180  template <typename VertexInputIterator> static Vector3 computeNormal(VertexInputIterator vbegin, VertexInputIterator vend)
181  {
182  typedef typename std::iterator_traits<VertexInputIterator>::value_type VertexT;
183 
184  static Real const EPSILON = 1e-10f;
185 
186  VertexInputIterator v0 = vbegin;
187  VertexInputIterator v1 = incrementIterator(v0, vbegin, vend);
188  VertexInputIterator v2 = incrementIterator(v1, vbegin, vend);
189 
190  if (v1 == v0 || v2 == v0) // too few vertices
191  return Vector3::Zero();
192 
193  Vector3 p0 = Algorithms::PointTraitsN<VertexT, 3>::getPosition(*v0);
194  Vector3 p1 = Algorithms::PointTraitsN<VertexT, 3>::getPosition(*v1);
195 
196  for (VertexInputIterator vi = vbegin; vi != vend; ++vi)
197  {
198  Vector3 p2 = Algorithms::PointTraitsN<VertexT, 3>::getPosition(*v2);
199  Vector3 normal = (p2 - p1).cross(p0 - p1);
200  if (normal.squaredNorm() >= EPSILON)
201  return normal.normalized();
202 
203  p0 = p1;
204  p1 = p2;
205  v2 = incrementIterator(v2, vbegin, vend);
206  }
207 
208  return Vector3::Zero(); // degenerate or too small
209  }

269  template <typename T>
270  static int triangulateQuad(Vector<3, T> const & p0, Vector<3, T> const & p1,
271  Vector<3, T> const & p2, Vector<3, T> const & p3,
272  intx & i0, intx & j0, intx & k0,
273  intx & i1, intx & j1, intx & k1,
274  T const & epsilon)
275  {
276  typedef Vector<3, T> VectorT;
277 
278  // We have two diagonals to split along. Try one and if it produces a triangle outside the polygon, pick the other one
279 
280  // First diagonal is p0-p2
281  VectorT n0 = (p1 - p0).cross(p2 - p0);
282  VectorT n1 = (p2 - p0).cross(p3 - p0);
283 
284  if (n0.dot(n1) < 0)
285  {
286  // Flip to diagonal p1-p3
287  n0 = (p2 - p1).cross(p3 - p1);
288  n1 = (p3 - p1).cross(p0 - p1);
289 
290  i0 = 1; j0 = 2; k0 = 3;
291  i1 = 1; j1 = 3; k1 = 0;
292  }
293  else
294  {
295  i0 = 0; j0 = 1; k0 = 2;
296  i1 = 0; j1 = 2; k1 = 3;
297  }
298 
299  // Check for degenerate triangles
300  T e2 = (epsilon < 0 ? Math::eps<T>() * Math::eps<T>() : epsilon * epsilon);
301  if (n0.squaredNorm() < e2)
302  {
303  if (n1.squaredNorm() < e2)
304  return 0;
305  else
306  {
307  i0 = i1; j0 = j1; k0 = k1; // the second triangle is the only non-degenerate one
308  return 1;
309  }
310  }
311  else if (n1.squaredNorm() < e2)
312  return 1;
313  else
314  return 2;
315  }
