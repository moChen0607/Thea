97  T squaredDistance(LineT const & other, VectorT * this_pt = nullptr, VectorT * other_pt = nullptr) const
98  {
99  // Adapted from Christer Ericson, "Real-Time Collision Detection", Morgan-Kaufman, 2005.
100 
101  VectorT r = point - other.point;
102  T b = direction.dot(other.direction);
103  T f = other.direction.dot(r);
104  T denom = 1 - b * b; // always nonnegative
105 
106  // If segments not parallel, compute closest point on L1 to L2. Else pick arbitrary s (here 0).
107  T s = 0;
108  if (Math::fuzzyGt(denom, static_cast<T>(0)))
109  {
110  T c = direction.dot(r);
111  s = (b * f - c) / denom;
112  }
113 
114  VectorT c1 = point + s * direction;
115  if (this_pt)
116  *this_pt = c1;
117 
118  // Compute point on L2 closest to S1(s) using t = Dot((P1+D1*s)-P2,D2) / Dot(D2,D2) = (b*s + f)
119  T t = b * s + f;
120  VectorT c2 = other.point + t * direction;
121  if (other_pt)
122  *other_pt = c2;
123 
124  return (c1 - c2).squaredNorm();
125  }
