64  (intx)points.size(), N));

169  T rayIntersectionTime(RayN<N, T> const & ray, T max_time = -1) const
170  {
171  T numer = dist - normal.dot(ray.getOrigin());
172  T denom = normal.dot(ray.getDirection());
173  if (std::abs(denom) < Math::eps(numer, denom))
174  return -1;
175  else
176  {
177  T t = numer / denom;
178  return (max_time < 0 || t <= max_time) ? t : -1;
179  }
