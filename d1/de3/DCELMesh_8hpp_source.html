513  bool stitchBoundaryLoops(Halfedge * loop0_start, Halfedge * loop1_start)
514  {
515  // In the first pass, find the length of the loops. This makes the stopping condition easier to evaluate since we don't
516  // need to ensure that we're not examining a deleted halfedge, and also allows us to check for some errors before we make
517  // any changes.
518  Halfedge * e0 = loop0_start, * e1 = loop1_start;
519  intx n = 0;
520  do
521  {
522  if (!e0->isBoundaryHalfedge() || !e1->isBoundaryHalfedge())
523  {
524  THEA_ERROR << getName() << ": Can't stitch loops with non-boundary edges";
525  return false;
526  }
527 
528  if (++n > numEdges())
529  {
530  THEA_ERROR << getName() << ": Loop is not closed";
531  return false;
532  }
533 
534  e0 = e0->next();
535  e1 = e1->next();
536 
537  } while (e0 != loop0_start);
538 
539  if (e1 != loop1_start)
540  {
541  THEA_ERROR << getName() << ": Loops are not of same length";
542  return false;
543  }
544 
545  // Now do the actual stitching
546  e0 = loop0_start; // we should be here already, but let's play safe
547  e1 = loop1_start;
548  Halfedge * next0 = nullptr, * next1 = nullptr;
549  for (intx i = 0; i < n; ++i, e0 = next0, e1 = next1)
550  {
551  if (i < n - 1)
552  {
553  // Store in advance where we're going to go next, since e0 and e1 are going to be deleted in this iteration
554  next0 = e0->next();
555  next1 = e1->origin->leaving; // we're stepping backwards around the second loop
556  do
557  {
558  if (next1->twin_he->next() == e1)
559  break;
560 
561  next1 = next1->nextAroundOrigin();
562 
563  } while (next1 != e1->origin->leaving);
564 
565  next1 = next1->twin_he;
566  if (next1->next() != e1)
567  {
568  THEA_ERROR << getName() << ": Predecessor of halfedge in second loop not found";
569  return false;
570  }
571  }
572 
573  // Shift all halfedges originating at e1's origin to the new origin at the tip of e0
574  Vertex * old_origin = e1->origin, * new_origin = e0->twin_he->origin;
575  Halfedge * he = old_origin->leaving; // this is guaranteed to not be a deleted loop edge, since we're stepping
576  do // backwards in the second loop
577  {
578  he->origin = new_origin;
579  he = he->nextAroundOrigin(); // this is also guaranteed to be a valid operation, the relevant next and twin pointers
580  // are ok
581  } while (he != old_origin->leaving);
582 
583  // Delete the old origin
584  removeVertex(old_origin);
585 
586  // Seal the edge by deleting the loop-side halfedges, and making the face-side halfedges twins
587  Halfedge * t0 = e0->twin_he;
588  Halfedge * t1 = e1->twin_he;
589  intx index = e0->index;
590 
591  removeHalfedge(e0);
592  removeHalfedge(e1);
593 
594  // Update the index of the face-side halfedge in the second loop, so it is stored right after its new twin
595  halfedges.erase(t1);
596  t1->index = index;
597  halfedges.insert(t1);
598 
599  // Update the twin pointers to complete the pairing
600  t0->twin_he = t1;
601  t1->twin_he = t0;
602  }
603 
604  invalidateGPUBuffers();
605  return true;
606  }

817  Face * addFace(size_t num_verts, Vertex ** verts, Halfedge * first, size_t origin, bool reverse,
818  Vector3 const & normal)
819  {
820  // For each vertex, check that the halfedge emanating from it exists as a boundary halfedge, or can be successfully added.
821  // Store either the existing edge to the next vertex, or the preceding boundary halfedge _into_ the vertex.
822  Array<Halfedge *> edges(num_verts);
823  Halfedge * prev = first, * e;
824  size_t start_index = reverse ? (origin > 0 ? origin - 1 : num_verts - 1)
825  : (origin < num_verts - 1 ? origin + 1 : 0);
826  size_t i = start_index, next;
827  do
828  {
829  // See if there is an existing edge
830  next = reverse ? (i > 0 ? i - 1 : num_verts - 1) : (i < num_verts - 1 ? i + 1 : 0);
831  e = verts[i]->getEdgeTo(verts[next]);
832 
833  if (e) // edge exists, needs to be boundary edge
834  {
835  if (!e->isBoundaryHalfedge())
836  {
837  THEA_WARNING << getName() << ": Can't stitch a face to a halfedge that already adjoins a face";
838  return nullptr;
839  }
840 
841  if (prev && prev->next() != e)
842  {
843  THEA_WARNING << getName() << ": Face breaks halfedge order at vertex";
844  return nullptr;
845  }
846 
847  edges[i] = e;
848  prev = e;
849  }
850  else // edge does not exist
851  {
852  if (prev)
853  {
854  // Just to be sure, check that prev->next() is a boundary edge
855  debugAssertM(prev->isBoundaryHalfedge(),
856  getNameStr() + ": Previous halfedge of new face is not currently on the mesh boundary");
857  debugAssertM(prev->next() && prev->next()->isBoundaryHalfedge(),
858  getNameStr()
859  + ": Mesh has internal inconsistency (boundary halfedge not followed by boundary halfedge)");
860 
861  // Twin of prev is predecessor of (i, next) around vertex i.
862  edges[i] = prev;
863  }
864  else
865  {
866  // Check that the edge can be added, i.e. the vertex is on the boundary, and find its predecessor
867  Halfedge * e = findPrevAroundVertex(verts[i], verts[next], normal);
868  if (e)
869  {
870  if (e->twin()->isBoundaryHalfedge())
871  edges[i] = e->twin();
872  else
873  {
874  THEA_WARNING << getName() << ": Can't stitch a face to a halfedge that already adjoins a face";
875  return nullptr;
876  }
877  }
878  else
879  edges[i] = nullptr;
880  }
881 
882  prev = nullptr; // no current edge between i and next
883  }
884 
885  i = next;
886 
887  } while (i != start_index);
888 
889  // Now create the new face
890  Face * face = new Face;
891  face->num_edges = (int)num_verts;
892  face->setNormal(normal);
893 
894  // Now actually create and add the edges. In this loop we directly access private members (ha ha) of the edges to bypass
895  // any consistency checks until we've finished adding the face.
896  Halfedge * new_e, * new_twin, * next_e, * last = nullptr; (void)last; // Squash -Wunused-but-set-variable
897  Vertex * vi, * vnext;
898  intx index0, index1;
899  bool added_canonical_edge_to_face = false;
900 
901 #ifdef THEA_DCELMESH_VERBOSE
902  bool last_existed = false, this_exists = false;
903 #endif
904 
905  i = start_index;
906  vi = verts[i];
907 
908  do
909  {
910  next = reverse ? (i > 0 ? i - 1 : num_verts - 1) : (i < num_verts - 1 ? i + 1 : 0);
911  vnext = verts[next];
912 
913  e = edges[i];
914  if (e && e->origin == vi) // edge exists, no problem
915  {
916 #ifdef THEA_DCELMESH_VERBOSE
917  this_exists = true;
918 #endif
919 
920  debugAssertM(e->getOrigin() == vi && e->getEnd() == vnext, getNameStr() + ": Edge has wrong endpoints");
921  debugAssertM(e->isBoundaryHalfedge(),
922  getNameStr() + ": Can't stitch a face to a halfedge that already adjoins a face");
923  new_e = e;
924 
925  // This edge is necessarily the predecessor of the edge we will add at the next vertex
926  next_e = edges[next];
927 
928  debugAssertM(next_e, getNameStr()
929  + ": Vertex has an existing edge, yet it was not found in search for predecessor");
930 
931  if (next_e->origin != vnext) // no existing next edge
932  edges[next] = e;
933  }
934  else // edge doesn't exist, predecessor is stored
935  {
936 #ifdef THEA_DCELMESH_VERBOSE
937  this_exists = false;
938 #endif
939 
940  // Create a new pair of halfedges
941  nextHalfedgeIndices(index0, index1);
942 
943  new_e = new Halfedge(index0);
944  halfedges.insert(new_e);
945 
946  new_twin = new Halfedge(index1);
947  halfedges.insert(new_twin);
948 
949  // Each is the twin of the other
950  new_e->twin_he = new_twin;
951  new_twin->twin_he = new_e;
952 
953  // Set the halfedge origins
954  new_e->origin = vi;
955  new_twin->origin = vnext;
956 
957  // The new edge is (duh) the new edge connecting this vertex to the next
958  edges[i] = new_e;
959 
960  // Insert the edge at this vertex
961  if (e)
962  {
963  new_twin->next_he = e->next_he;
964  e->next_he = new_e;
965  }
966  else
967  {
968  new_twin->next_he = new_e; // vertex has degree 1 for now (will be fixed when the boundary loops round)
969  vi->leaving = new_e;
970  }
971 
972  // Insert the edge at the next vertex (the new edge is the new predecessor at the next vertex)
973  next_e = edges[next];
974  if (next_e)
975  {
976  if (next_e->origin == vnext) // next edge exists
977  {
978  new_e->next_he = next_e;
979  prev = next_e->prevAroundOrigin(); // this search can be avoided if the first edge was newly created, and we
980  // have looped back to it (since we can reuse the predecessor that was
981  // initially stored); however, in the interest of clarity we'll omit this
982  // optimization
983  prev->twin_he->next_he = new_twin;
984  }
985  else // predecessor is directly stored (note that it points _into_ the next vertex, so it corresponds to prev->twin
986  // of the if block)
987  {
988  new_e->next_he = next_e->next_he;
989  next_e->next_he = new_twin;
990 
991  edges[next] = new_e;
992  }
993  }
994  else // isolated next vertex
995  {
996  new_e->next_he = new_twin; // next vertex has degree 1 for now (will be fixed by next edge)
997  vnext->leaving = new_twin;
998 
999  edges[next] = new_e;
1000  }
1001  }
1002 
1003  // Update the face pointer of the halfedge (the twin, if newly created, becomes a boundary halfedge)
1004  new_e->face = face;
1005  if (!added_canonical_edge_to_face)
1006  {
1007  face->halfedge = new_e;
1008  added_canonical_edge_to_face = true;
1009  }
1010 
1011  debugAssertM(!last || last->next_he == new_e, getNameStr() + ": Next pointers on face boundary not consistent");
1012  last = new_e;
1013 
1014 #ifdef THEA_DCELMESH_VERBOSE
1015  THEA_CONSOLE << "Last existed = " << last_existed << ", this exists = " << this_exists;
1016  last_existed = this_exists;
1017 #endif
1018 
1019  // Update the normal at the vertex
1020  vi->addFaceNormal(normal); // weight by face area?
1021 
1022  i = next;
1023  vi = vnext;
1024 
1025  } while (i != start_index);
1026 
1027  faces.insert(face);
1028 
1029  invalidateGPUBuffers();
1030  return face;
1031  }

1088  static Real ccwAngle(Vector3 const & u, Vector3 const & v, Vector3 const & unit_up)
1089  {
1090  Vector3 v_proj = v - (v.dot(unit_up) * unit_up);
1091  Real s = u.cross(v_proj).norm();
1092  Real c = u.dot(v_proj);
1093  Real ang = std::atan2(s, c);
1094 
1095  return ang < 0 ? 2 * Math::pi() - ang : ang;

1108  Halfedge * findPrevAroundVertex(Vertex * u, Vertex * v, Vector3 const & normal) const
1109  {
1110  Halfedge * first = u->getHalfedge();
1111  if (!first)
1112  return nullptr;
1113 
1114  // Compute the new normal, which defines the plane of projection
1115  Vector3 new_normal = u->estimateUpdatedNormal(normal);
1116  Halfedge * e = first, * best = first;
1117  Real ang, best_ang = 100; // anything > 2 pi
1118  Vector3 edge = v->getPosition() - u->getPosition();
1119  edge = (edge - (edge.dot(new_normal) * new_normal));
1120  do
1121  {
1122  if (e->twin()->isBoundaryHalfedge())
1123  {
1124  Vector3 test_edge = (e->getEnd()->getPosition() - u->getPosition());
1125  ang = ccwAngle(edge, test_edge, new_normal);
1126  if (ang < best_ang)
1127  {
1128  best_ang = ang;
1129  best = e;
1130  }
1131  }
1132 
1133  e = e->nextAroundOrigin(); // the next halfedge around u, counter-clockwise when looking from the "outside"
1134 
1135  } while (e != first);
1136 
1137  return best;
1138  }
