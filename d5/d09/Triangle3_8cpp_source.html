99 /* this edge to edge test is based on Franlin Antonio's gem:
100  "Faster Line Segment Intersection", in Graphics Gems III,
101  pp. 199-202 */
102 #define EDGE_EDGE_TEST(V0,U0,U1) \
103  Bx=U0[i0]-U1[i0]; \
104  By=U0[i1]-U1[i1]; \
105  Cx=V0[i0]-U0[i0]; \
106  Cy=V0[i1]-U0[i1]; \
107  f=Ay*Bx-Ax*By; \
108  d=By*Cx-Bx*Cy; \
109  if((f>0 && d>=0 && d<=f) || (f<0 && d<=0 && d>=f)) \
110  { \
111  e=Ax*Cy-Ay*Cx; \
112  if(f>0) \
113  { \
114  if(e>=0 && e<=f) return 1; \
115  } \
116  else \
117  { \
118  if(e<=0 && e>=f) return 1; \
119  } \
120  }

122 #define EDGE_AGAINST_TRI_EDGES(V0,V1,U0,U1,U2) \
123  { \
124  Real Ax,Ay,Bx,By,Cx,Cy,e,d,f; \
125  Ax=V1[i0]-V0[i0]; \
126  Ay=V1[i1]-V0[i1]; \
127  /* test edge U0,U1 against V0,V1 */ \
128  EDGE_EDGE_TEST(V0,U0,U1); \
129  /* test edge U1,U2 against V0,V1 */ \
130  EDGE_EDGE_TEST(V0,U1,U2); \
131  /* test edge U2,U1 against V0,V1 */ \
132  EDGE_EDGE_TEST(V0,U2,U0); \
133  }

135 #define POINT_IN_TRI(V0,U0,U1,U2) \
136  { \
137  Real a,b,c,d0,d1,d2; \
138  /* is T1 completly inside T2? */ \
139  /* check if V0 is inside tri(U0,U1,U2) */ \
140  a=U1[i1]-U0[i1]; \
141  b=-(U1[i0]-U0[i0]); \
142  c=-a*U0[i0]-b*U0[i1]; \
143  d0=a*V0[i0]+b*V0[i1]+c; \

153  d2=a*V0[i0]+b*V0[i1]+c; \
154  if(d0*d1>0) \

156  if(d0*d2>0) return 1; \

160 int coplanar_tri_tri(Real const N[3], Real const V0[3], Real const V1[3], Real const V2[3],
161  Real const U0[3], Real const U1[3], Real const U2[3])
162 {
163  Real A[3];
164  short i0, i1;
165  /* first project onto an axis-aligned plane, that maximizes the area */
166  /* of the triangles, compute indices: i0,i1. */
167  A[0] = fabs(N[0]);
168  A[1] = fabs(N[1]);
169  A[2] = fabs(N[2]);
170 
171  if (A[0] > A[1])
172  {
173  if (A[0] > A[2])
174  {
175  i0 = 1; /* A[0] is greatest */
176  i1 = 2;
177  }
178  else
179  {
180  i0 = 0; /* A[2] is greatest */
181  i1 = 1;
182  }
183  }
184  else /* A[0]<=A[1] */
185  {
186  if (A[2] > A[1])
187  {
188  i0 = 0; /* A[2] is greatest */
189  i1 = 1;
190  }
191  else
192  {
193  i0 = 0; /* A[1] is greatest */
194  i1 = 2;
195  }
196  }
197 
198  /* test all edges of triangle 1 against the edges of triangle 2 */
199  EDGE_AGAINST_TRI_EDGES(V0, V1, U0, U1, U2);
200  EDGE_AGAINST_TRI_EDGES(V1, V2, U0, U1, U2);
201  EDGE_AGAINST_TRI_EDGES(V2, V0, U0, U1, U2);
202  /* finally, test if tri1 is totally contained in tri2 or vice versa */
203  POINT_IN_TRI(V0, U0, U1, U2);
204  POINT_IN_TRI(U0, V0, V1, V2);
205  return 0;
206 }

209 int tri_tri_intersect(Real const V0[3], Real const V1[3], Real const V2[3],
210  Real const U0[3], Real const U1[3], Real const U2[3])
211 {
212  Real E1[3], E2[3];
213  Real N1[3], N2[3], d1, d2;
214  Real du0, du1, du2, dv0, dv1, dv2;
215  Real D[3];
216  Real isect1[2], isect2[2];
217  Real du0du1, du0du2, dv0dv1, dv0dv2;
218  short index;
219  Real vp0, vp1, vp2;
220  Real up0, up1, up2;
221  Real b, c, max;
222  /* compute plane equation of triangle(V0,V1,V2) */
223  SUB(E1, V1, V0);
224  SUB(E2, V2, V0);
225  CROSS(N1, E1, E2);
226  d1 = -DOT(N1, V0);
227  /* plane equation 1: N1.X+d1=0 */
228  /* put U0,U1,U2 into plane equation 1 to compute signed distances to the plane*/
229  du0 = DOT(N1, U0) + d1;
230  du1 = DOT(N1, U1) + d1;
231  du2 = DOT(N1, U2) + d1;
232  /* coplanarity robustness check */
233 #if USE_EPSILON_TEST==TRUE
234 
235  if (fabs(du0) < EPSILON) du0 = 0;
236 
237  if (fabs(du1) < EPSILON) du1 = 0;
238 
239  if (fabs(du2) < EPSILON) du2 = 0;
240 
241 #endif
242  du0du1 = du0 * du1;
243  du0du2 = du0 * du2;
244 
245  if (du0du1 > 0 && du0du2 > 0) /* same sign on all of them + not equal 0 ? */
246  return 0; /* no intersection occurs */
247 
248  /* compute plane of triangle (U0,U1,U2) */
249  SUB(E1, U1, U0);
250  SUB(E2, U2, U0);
251  CROSS(N2, E1, E2);
252  d2 = -DOT(N2, U0);
253  /* plane equation 2: N2.X+d2=0 */
254  /* put V0,V1,V2 into plane equation 2 */
255  dv0 = DOT(N2, V0) + d2;
256  dv1 = DOT(N2, V1) + d2;
257  dv2 = DOT(N2, V2) + d2;
258 #if USE_EPSILON_TEST==TRUE
259 
260  if (fabs(dv0) < EPSILON) dv0 = 0;
261 
262  if (fabs(dv1) < EPSILON) dv1 = 0;
263 
264  if (fabs(dv2) < EPSILON) dv2 = 0;
265 
266 #endif
267  dv0dv1 = dv0 * dv1;
268  dv0dv2 = dv0 * dv2;
269 
270  if (dv0dv1 > 0 && dv0dv2 > 0) /* same sign on all of them + not equal 0 ? */
271  return 0; /* no intersection occurs */
272 
273  /* compute direction of intersection line */
274  CROSS(D, N1, N2);
275  /* compute and index to the largest component of D */
276  max = fabs(D[0]);
277  index = 0;
278  b = fabs(D[1]);
279  c = fabs(D[2]);
280 
281  if (b > max) max = b, index = 1;
282 
283  if (c > max) max = c, index = 2;
284 
285  /* this is the simplified projection onto L*/
286  vp0 = V0[index];
287  vp1 = V1[index];
288  vp2 = V2[index];
289  up0 = U0[index];
290  up1 = U1[index];
291  up2 = U2[index];
292  /* compute interval for triangle 1 */
293  COMPUTE_INTERVALS(vp0, vp1, vp2, dv0, dv1, dv2, dv0dv1, dv0dv2, isect1[0], isect1[1]);
294  /* compute interval for triangle 2 */
295  COMPUTE_INTERVALS(up0, up1, up2, du0, du1, du2, du0du1, du0du2, isect2[0], isect2[1]);
296  SORT(isect1[0], isect1[1]);
297  SORT(isect2[0], isect2[1]);
298 
299  if (isect1[1] < isect2[0] || isect2[1] < isect1[0]) return 0;
300 
301  return 1;
302 }

569  Real du0, du1, du2, dv0, dv1, dv2;

573  Real isectpointB1[3] = {0}, isectpointB2[3] = {0};

577  Real up0, up1, up2;
578  Real b, c, max;
579  // Real tmp,diff[3];
580  int smallest1, smallest2;

585  d1 = -DOT(N1, V0);

587  /* put U0,U1,U2 into plane equation 1 to compute signed distances to the plane*/
588  du0 = DOT(N1, U0) + d1;
589  du1 = DOT(N1, U1) + d1;
590  du2 = DOT(N1, U2) + d1;
591  /* coplanarity robustness check */
592 #if USE_EPSILON_TEST==TRUE
593

602  du0du2 = du0 * du2;
603

606

609  SUB(E2, U2, U0);
610  CROSS(N2, E1, E2);

627  dv0dv2 = dv0 * dv2;
628

667  if (isect2[0] < isect1[0])

710  }

740  Vector3 const & v0,
741  Vector3 const & v1,
742  Vector3 const & edgeDirection,
743  float const edgeLength,

745 {

768 Vector3 closestPointOnTrianglePerimeter(
769  Vector3 const v[3],
770  Vector3 const edgeDirection[3],
771  float const edgeLength[3],
772  Vector3 const & point,
773  int & edgeIndex)
774 {
775  // Closest point on segment from v[i] to v[i + 1]
776  Vector3 r[3];
777  // Distance squared from r[i] to point
778  float d[3];
779  // Index of the next point
780  static const int next[] = {1, 2, 0};
781 
782  for (int i = 0; i < 3; ++i)
783  {
784  r[i] = closestPointOnLineSegment(v[i], v[next[i]], edgeDirection[i], edgeLength[i], point);
785  d[i] = (r[i] - point).squaredNorm();
786  }
787 
788  if (d[0] < d[1])
789  {
790  if (d[0] < d[2])
791  {
792  // Between v0 and v1
793  edgeIndex = 0;
794  }
795  else
796  {
797  // Between v2 and v0
798  edgeIndex = 2;
799  }
800  }
801  else
802  {
803  if (d[1] < d[2])
804  {
805  // Between v1 and v2
806  edgeIndex = 1;
807  }
808  else
809  {
810  // Between v2 and v0
811  edgeIndex = 2;
812  }
813  }
814 
815  return r[edgeIndex];

839 bool
840 isPointInsideTriangle(
841  Vector3 const & v0,
842  Vector3 const & v1,
843  Vector3 const & v2,
844  int primary_axis,
845  Vector3 const & p)
846 {
847  // Check that the point is within the triangle using a Barycentric coordinate test on a two dimensional plane.
848  int i, j;
849  switch (primary_axis)
850  {
851  case 1 : i = 2; j = 0; break;
852  case 2 : i = 0; j = 1; break;
853  default /* 0 */ : i = 1; j = 2;
854  }
855 
856  // See if all barycentric coordinates are non-negative
857 
858  // 2D area via cross product
859 #define THEA_TRIANGLE_AREA2(d, e, f) (((e)[i] - (d)[i]) * ((f)[j] - (d)[j]) - ((f)[i] - (d)[i]) * ((e)[j] - (d)[j]))
860 
861  // Area of the polygon
862  Real area = THEA_TRIANGLE_AREA2(v0, v1, v2);
863  if (area == 0)
864  {
865  // This triangle has zero area, so the point must not be in it unless the triangle point is the test point.
866  return (v0 == p);
867  }
868 
869  Real inv_area = 1.0f / area;
870  Real b[3];
871 
872  // (Avoid normalization until absolutely necessary)
873  b[0] = THEA_TRIANGLE_AREA2(p, v1, v2) * inv_area;
874  if ((b[0] < 0) || (b[0] > 1))
875  return false;
876 
877  b[1] = THEA_TRIANGLE_AREA2(v0, p, v2) * inv_area;
878  if ((b[1] < 0) || (b[1] > 1))
879  return false;
880 
881  b[2] = 1.0f - b[0] - b[1];
882 
883 #undef THEA_TRIANGLE_AREA2
884 
885  return ((b[2] >= 0) && (b[2] <= 1));
886 }

889 rayTriangleIntersectionTime(Ray3 const & ray, Vector3 const & v0, Vector3 const & edge01, Vector3 const & edge02)
890 {
891  // The code is taken from Dave Eberly's Wild Magic library, v5.3, released under the Boost license:
892  // http://www.boost.org/LICENSE_1_0.txt .
893  static Real const EPS = 1e-30f;
894  Vector3 diff = ray.getOrigin() - v0;
895  Vector3 normal = edge01.cross(edge02);
896  // Solve Q + t*D = b1*E1 + b2*E2 (Q = diff, D = ray direction, E1 = edge01, E2 = edge02, N = Cross(E1,E2)) by
897  // |Dot(D,N)|*b1 = sign(Dot(D,N))*Dot(D,Cross(Q,E2))
898  // |Dot(D,N)|*b2 = sign(Dot(D,N))*Dot(D,Cross(E1,Q))
899  // |Dot(D,N)|*t = -sign(Dot(D,N))*Dot(Q,N)
900  Real DdN = ray.getDirection().dot(normal);
901  int sign;
902 
903  if (DdN > EPS)
904  sign = 1;
905  else if (DdN < -EPS)
906  {
907  sign = -1;
908  DdN = -DdN;
909  }
910  else
911  {
912  // Ray and triangle are parallel, call it a "no intersection" even if the ray does intersect
913  return -1;
914  }
915 
916  Real DdQxE2 = sign * ray.getDirection().dot(diff.cross(edge02));
917 
918  if (DdQxE2 >= 0)
919  {
920  Real DdE1xQ = sign * ray.getDirection().dot(edge01.cross(diff));
921 
922  if (DdE1xQ >= 0)
923  {
924  if (DdQxE2 + DdE1xQ <= DdN)
925  {
926  // Line intersects triangle, check if ray does
927  Real QdN = -sign * diff.dot(normal);
928 
929  if (QdN >= 0)
930  {
931  // Ray intersects triangle.
932  return QdN / DdN;
933  }
934 
935  // else: t < 0, no intersection
936  }
937 
938  // else: b1 + b2 > 1, no intersection
939  }
940 
941  // else: b2 < 0, no intersection
942  }
943 
944  // else: b1 < 0, no intersection
945  return -1;
946 }
