51 // Fitting lines to sets of objects that map to single points in 2-space.
52 template <typename T>
53 class /* THEA_API */ LinearLeastSquares2<T, typename std::enable_if< IsNonReferencedPointN<T, 2>::value >::type>
54 {
55  public:
56  template <typename InputIterator>
57  static double fitLine(InputIterator begin, InputIterator end, Line2 & line, Vector2 * centroid = nullptr)
58  {
59  Vector2d center = CentroidN<T, 2>::compute(begin, end);
60  Matrix2d m = Matrix2d::Zero();
61  for (auto iter = makeRefIterator(begin); iter != makeRefIterator(end); ++iter)
62  {
63  Vector2d diff = Vector2d(PointTraitsN<T, 2>::getPosition(*iter)) - center;
64 
65  m(0, 0) += (diff.y() * diff.y());
66  m(0, 1) -= (diff.x() * diff.y());
67  m(1, 1) += (diff.x() * diff.x());
68  }
69 
70  m(1, 0) = m(0, 1);
71 
72  Eigen::SelfAdjointEigenSolver eigensolver;
73  if (eigensolver.computeDirect(m).info() != Eigen::Success)
74    throw Error("LinearLeastSquares2: Could not eigensolve covariance matrix");
75 
76  // Eigenvalues are non-negative (covariance matrix is non-negative definite) and in increasing order
77  line = Line2::fromPointAndDirection(Vector2(center), Vector2(eigensolver.eigenvectors().col(0)));
78  if (centroid) *centroid = center;
79  return eigensolver.eigenvalues()[0];
80  }
81 
82 }; // class LinearLeastSquares2<Point2>
