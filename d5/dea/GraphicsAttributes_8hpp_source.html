309 // Dirty SFINAE template trick to check if the type T has get...() and set...() functions
310 #define THEA_HAS_GRAPHICS_ATTRIB(name, get_func_name, set_func_name) \
311 template <typename T> \
312 class name \
313 { \
314  public: \
315  typedef char Yes; \
316  typedef struct { char a[2]; } No; \
317 \
318  struct GetBaseMixin { void get_func_name() {} }; \
319  struct SetBaseMixin { void set_func_name() {} }; \
320 \
321  struct GetBase : public T, public GetBaseMixin {}; \
322  struct SetBase : public T, public SetBaseMixin {}; \
323  template <typename S, S> class Helper {}; \
324 \
325  template <typename U> static No testGet(U *, Helper<void (GetBaseMixin::*)(), &U::get_func_name> * = 0); \
326  static Yes testGet(...); \
327 \
328  template <typename U> static No testSet(U *, Helper<void (SetBaseMixin::*)(), &U::set_func_name> * = 0); \
329  static Yes testSet(...); \
330 \
331  public: \
332  static bool const has_get = (sizeof(testGet((GetBase *)(0))) == sizeof(Yes)); \
333  static bool const has_set = (sizeof(testSet((SetBase *)(0))) == sizeof(Yes)); \
334  static bool const value = has_get && has_set; \
335 };
