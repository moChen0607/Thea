32 struct Sector
33 {
34  Vector2 center;
35  Real min_radius, max_radius, min_angle, max_angle;
36  Vector2 corner[4]; // small arc min angle to max angle, large arc min angle to max angle
37  Halfspace min_space, max_space; // positive halfspace contains sector
38 
39  // min_angle, max_ang must be in range [0, 2 * pi]
40  Sector(Vector2 const & center_, Real min_radius_, Real max_radius_, Real min_angle_, Real max_angle_)
41  : center(center_), min_radius(min_radius_), max_radius(max_radius_), min_angle(min_angle_), max_angle(max_angle_)
42  {
43  Real cos_min = Math::fastCos(min_angle);
44  Real sin_min = Math::fastSin(min_angle);
45 
46  Real cos_max = Math::fastCos(max_angle);
47  Real sin_max = Math::fastSin(max_angle);
48 
49  corner[0] = center + Vector2(min_radius * cos_min, min_radius * sin_min);
50  corner[1] = center + Vector2(min_radius * cos_max, min_radius * sin_max);
51 
52  corner[2] = center + Vector2(max_radius * cos_min, max_radius * sin_min);
53  corner[3] = center + Vector2(max_radius * cos_max, max_radius * sin_max);
54 
55  Vector2 delta_min = corner[2] - center;
56  min_space = Halfspace::fromPointAndNormal(center, Vector2(-delta_min.y(), delta_min.x()));
57 
58  Vector2 delta_max = corner[3] - center;
59  max_space = Halfspace::fromPointAndNormal(center, Vector2(delta_max.y(), -delta_max.x()));
60  }
61 
62  bool contains(Vector2 const & p) const
63  {
64  Vector2 delta = p - center;
65  Real sqdist = delta.squaredNorm();
66  if (sqdist < min_radius * min_radius || sqdist > max_radius * max_radius)
67  return false;
68 
69  Real ang = Math::fastArcTan2(delta.y(), delta.x());
70  if (ang < 0) ang += Math::twoPi();
71 
72  return (ang >= min_angle && ang <= max_angle);
73  }
74 
75  bool contains(Rect const & rect) const
76  {
77  if (rect.squaredDistance(center) < min_radius * min_radius)
78  return false;
79 
80  return (contains(rect.getCorner(0))
81  && contains(rect.getCorner(1))
82  && contains(rect.getCorner(2))
83  && contains(rect.getCorner(3)));
84  }
85 
86  // A conservative intersection test -- can return true even when the shapes don't intersect
87  bool maybeIntersects(Rect const & rect) const
88  {
89  // Any corner inside rectangle ==> intersect
90  if (rect.contains(corner[0])
91  || rect.contains(corner[1])
92  || rect.contains(corner[2])
93  || rect.contains(corner[3]))
94  return true;
95 
96  // Rectangle outside ring ==> no intersect
97  if (rect.squaredDistance(center) > max_radius * max_radius
98  || rect.squaredMaxDistance(center) < min_radius * min_radius)
99  return false;
100 
101  // Any rectangle corner inside sector ==> intersect
102  if (contains(rect.getCorner(0))
103  || contains(rect.getCorner(1))
104  || contains(rect.getCorner(2))
105  || contains(rect.getCorner(3)))
106  return true;
107 
108  // Test against min_angle separating ray
109  if (min_space.negativeHalfSpaceContains(rect.getCorner(0))
110  && min_space.negativeHalfSpaceContains(rect.getCorner(1))
111  && min_space.negativeHalfSpaceContains(rect.getCorner(2))
112  && min_space.negativeHalfSpaceContains(rect.getCorner(3)))
113  return false;
114 
115  // Test against max_angle separating ray
116  if (max_space.negativeHalfSpaceContains(rect.getCorner(0))
117  && max_space.negativeHalfSpaceContains(rect.getCorner(1))
118  && max_space.negativeHalfSpaceContains(rect.getCorner(2))
119  && max_space.negativeHalfSpaceContains(rect.getCorner(3)))
120  return false;
121 
122  return true;
123  }

144 struct QuadTree
145 {
146  static int const MIN_NODE_WIDTH = 10;
147 
148  Array<QuadTreeNode> nodes;
149  Image image;
150 
151  QuadTree(Image const & image_)
152  {
153  if (!image_.convert(Image::Type::LUMINANCE_8U, image))
154  throw Error("ShapeContext: Could not initialize quadtree for image");
155 
156  int w = image.getWidth();
157  int h = image.getHeight();
158 
159  nodes.push_back(QuadTreeNode());
160  nodes[0].rect.set(Vector2(0, 0), Vector2(w - 1, h - 1));
161  initWeight(&nodes[0]);
162  buildTree(0);
163  }
164 
165  void clear()
166  {
167  nodes.clear();
168  }

170  void buildTree(size_t node_index)
171  {
172  int xmin, ymin, xmax, ymax;
173  nodes[node_index].getBounds(xmin, ymin, xmax, ymax);
174 
175  int num_xsteps = (xmax - xmin >= MIN_NODE_WIDTH ? 2 : 1);
176  int num_ysteps = (ymax - ymin >= MIN_NODE_WIDTH ? 2 : 1);
177 
178  if (num_xsteps <= 1 && num_ysteps <= 1)
179  return;
180 
181  int xmid = (num_xsteps > 1 ? (xmin + xmax) / 2 : xmax);
182  int ymid = (num_ysteps > 1 ? (ymin + ymax) / 2 : ymax);
183 
184  for (int i = 0; i < num_ysteps; ++i)
185  {
186  int cell_ymin = (i == 0 ? ymin : ymid + 1);
187  int cell_ymax = (i == 0 ? ymid : ymax);
188 
189  for (int j = 0; j < num_xsteps; ++j)
190  {
191  int cell_xmin = (j == 0 ? xmin : xmid + 1);
192  int cell_xmax = (j == 0 ? xmid : xmax);
193 
194  QuadTreeNode child;
195  child.rect.set(Vector2(cell_xmin, cell_ymin), Vector2(cell_xmax, cell_ymax));
196  initWeight(&child); // do this brute force here so we can aggressively prune recursion when weight --> 0
197 
198  if (child.weight > 1e-10)
199  {
200  nodes.push_back(child);
201  nodes[node_index].children.push_back(nodes.size() - 1);
202 
203  // THEA_CONSOLE << "Weight of node " << nodes.size() - 1 << " = " << nodes.back().weight;
204  }
205  }
206  }
207 
208  for (size_t i = 0; i < nodes[node_index].children.size(); ++i)
209  buildTree(nodes[node_index].children[i]);
210  }
211 
212  void initWeight(QuadTreeNode * node)
213  {
214  node->weight = 0;
215 
216  int xmin, ymin, xmax, ymax;
217  node->getBounds(xmin, ymin, xmax, ymax);
218 
219  for (int i = ymin; i <= ymax; ++i)
220  {
221  uint8 const * scanline = (uint8 const *)image.getScanLine(i);
222 
223  for (int j = xmin; j <= xmax; ++j)
224  node->weight += scanline[j];
225  }
226 
227  node->weight /= 255.0; // normalize each pixel's weight to [0, 1]
228  }
229 
230  Real rangeWeight(Sector const & range) const
231  {
232  return rangeWeight(&nodes[0], range);
233  }
234 
235  Real rangeWeight(QuadTreeNode const * node, Sector const & range) const
236  {
237  if (range.contains(node->rect))
238  return node->weight;
239  else if (!range.maybeIntersects(node->rect))
240  return 0;
241 
242  Real weight = 0;
243 
244  if (node->children.empty())
245  {
246  int xmin, ymin, xmax, ymax;
247  node->getBounds(xmin, ymin, xmax, ymax);
248 
249  for (int i = ymin; i <= ymax; ++i)
250  {
251  uint8 const * scanline = (uint8 const *)image.getScanLine(i);
252 
253  for (int j = xmin; j <= xmax; ++j)
254  {
255  if (range.contains(Vector2(j, i)))
256  weight += scanline[j];
257  }
258  }
259 
260  weight /= 255.0; // normalize each pixel's weight to [0, 1]
261  }
262  else
263  {
264  for (size_t i = 0; i < node->children.size(); ++i)
265  weight += rangeWeight(&nodes[node->children[i]], range);
266  }
267 
268  return weight;
269  }

285 void
286 ShapeContext::compute(intx num_radial_bins, intx num_polar_bins, Array<Real> & values, bool ignore_empty_pixels,
287  Real max_radius) const
288 {
289  using namespace ShapeContextInternal;
290 
291  alwaysAssertM(num_radial_bins > 0 && num_polar_bins > 0, "ShapeContext: Number of bins must be positive");
292 
293  int w = qtree->image.getWidth();
294  int h = qtree->image.getHeight();
295 
296  Real rad_limit = (max_radius <= 0 ? std::sqrt((Real)(w * w + h * h)) : max_radius);
297  Real rad_init = rad_limit / (1 << (num_radial_bins - 1));
298  Real ang_step = Math::twoPi() / num_polar_bins;
299 
300  values.resize((size_t)(w * h * num_radial_bins * num_polar_bins));
301  Real * entry = &values[0];
302  intx entry_size = num_radial_bins * num_polar_bins;
303 
304  Real progress_step = h / 5.0f;
305  Real next_progress_milestone = progress_step;
306 
307  static Real const SQRT_3 = std::sqrt(3.0f);
308 
309  for (int i = 0; i < h; ++i)
310  {
311  uint8 const * scanline = (uint8 const *)qtree->image.getScanLine(i);
312 
313  if (i >= next_progress_milestone)
314  {
315  THEA_CONSOLE << "... " << (int)(100 * next_progress_milestone / h) << '%';
316  next_progress_milestone += progress_step;
317  }
318 
319  for (int j = 0; j < w; ++j, entry += entry_size)
320  {
321  if (ignore_empty_pixels && scanline[j] == 0)
322  std::fill(entry, entry + entry_size, 0);
323  else
324  {
325  Real radius = rad_init;
326  Real old_radius = 0;
327 
328  for (int r = 0; r < num_radial_bins; ++r, radius *= 2)
329  {
330  Real ang = ang_step;
331  Real old_ang = 0;
332 
333  for (int p = 0; p < num_polar_bins; ++p, ang += ang_step)
334  {
335  Sector sector(Vector2(j, i), old_radius, radius, old_ang, ang);
336  Real weight = qtree->rangeWeight(sector);
337 
338  // Compensate for increasing size of bins. The formula is derived by computing the increase in area from bin to bin,
339  // but then taking the square root since the image content is assumed to be a sketched curve whose length increases
340  // as the square root of the area increase.
341  if (r > 0)
342  weight /= (SQRT_3 * (1 << (r - 1)));
343 
344  entry[r * num_polar_bins + p] = weight;
345 
346  old_ang = ang;
347  }
348 
349  old_radius = radius;
350  }
351 
352  THEA_CONSOLE << '\n';
353  }
354  }
355  }
356 
357  THEA_CONSOLE << "... 100%";
358 }
