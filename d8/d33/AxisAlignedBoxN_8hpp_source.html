118  template <typename TransformT> AxisAlignedBoxT transformAndBound(TransformT const & tr) const
119  {
120  AxisAlignedBoxT ret; // null box
121  std::bitset<N> counter; // all-zero bitset
122  VectorT corner;
123  do
124  {
125  // Get the current corner
126  for (intx j = 0; j < N; ++j)
127  corner[j] = counter.test((size_t)j) ? hi[j] : lo[j];
128 
129  // Transform and merge it
130  ret.merge(tr * corner);
131 
132  // Get the index of the next corner by adding 1 to the current index
133  for (size_t j = 0; j < N; j++)
134  {
135  if (counter.test(j))
136  counter.set(j, false);
137  else
138  {
139  counter.set(j, true);
140  break;
141  }
142  }
143 
144  } while (counter.any());
145 
146  return ret;
147  }

313  T squaredDistance(AxisAlignedBoxT const & other) const
314  {
315  if (is_null || other.is_null) return 0;
316 
317  VectorT vmax = hi.cwiseMin(other.hi);
318  VectorT vmin = lo.cwiseMax(other.lo);
319 
320  // Each coord of vmax is greater than the corresponding coord of vmin iff the ranges intersect
321 
322  VectorT vdiff = vmin - vmax; // any axes with overlap have negative values here
323  vdiff = vdiff.cwiseMax(VectorT::Zero()); // overlap axes have zero separation

414  RayIntersectionN<N, T> rayIntersection(RayN<N, T> const & ray, T max_time = -1) const
415  {
416  // Early exit
417  if (is_null)
418  return RayIntersectionN<N, T>(-1);
419  else if (max_time >= 0)
420  {
421  T ray_sqlen = ray.getDirection().squaredNorm();
422  T origin_sqdist = squaredDistance(ray.getOrigin()); // fast operation
423  if (origin_sqdist > max_time * max_time * ray_sqlen)
424  return RayIntersectionN<N, T>(-1);
425  }
426 
427  VectorT max_t(-1), location;
428  VectorT const & origin = ray.getOrigin();
429  VectorT const & dir = ray.getDirection();
430  bool inside = true;
431 
432  // Find candidate planes.
433  for (intx i = 0; i < N; ++i)
434  {
435  if (origin[i] < lo[i])
436  {
437  location[i] = lo[i];
438  inside = false;
439 
440  // Calculate distance to candidate plane along this coordinate
441  if (Math::fuzzyNe(dir[i], static_cast<T>(0)))
442  max_t[i] = (lo[i] - origin[i]) / dir[i];
443  }
444  else if (origin[i] > hi[i])
445  {
446  location[i] = hi[i];
447  inside = false;
448 
449  // Calculate distance to candidate plane along this coordinate
450  if (Math::fuzzyNe(dir[i], static_cast<T>(0)))
451  max_t[i] = (hi[i] - origin[i]) / dir[i];
452  }
453  }
454 
455  if (inside) // ray origin inside bounding box
456  return RayIntersectionN<N, T>(0); // undefined normal inside box
457 
458  // Get largest of the max_t's for final choice of intersection
459  int which_plane = 0;
460  for (intx i = 1; i < N; ++i)
461  if (max_t[i] > max_t[which_plane])
462  which_plane = i;
463 
464  // Check final candidate actually inside box
465  if (max_t[which_plane] < static_cast<T>(-0.5)) // miss the box
466  return RayIntersectionN<N, T>(-1);
467 
468  for (intx i = 0; i < N; ++i)
469  {
470  if (i != which_plane)
471  {
472  location[i] = origin[i] + max_t[which_plane] * dir[i];
473  if (location[i] < lo[i] || location[i] > hi[i]) // on this plane we're outside the box extents, so we miss the box
474  return RayIntersectionN<N, T>(-1);
475  }
476  }
477 
478  // Choose the normal to be the plane normal facing into the ray
479  VectorT normal = VectorT::Zero();
480  normal[which_plane] = (dir[which_plane] > 0) ? -1 : 1;
481 
482  return RayIntersectionN<N, T>(max_t[which_plane], &normal);
483  }
