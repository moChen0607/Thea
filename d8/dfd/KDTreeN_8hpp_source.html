469  template <typename InputIterator>
470  void init(InputIterator begin, InputIterator end, intx max_depth_ = -1, intx max_elems_per_leaf_ = -1,
471  bool save_memory = false, bool deallocate_previous_memory = true)
472  {
473  clear(deallocate_previous_memory);
474 
475  if (deallocate_previous_memory)
476  {
477  for (InputIterator ii = begin; ii != end; ++ii, ++num_elems)
478  if (elementPassesFilters(*ii))
479  elems.push_back(*ii);
480  }
481  else
482  {
483  size_t max_new_elems = (size_t)std::distance(begin, end);
484  bool resized = false;
485  if (max_new_elems > elems.size())
486  {
487  if (filters.empty())
488  elems.resize((size_t)std::ceil(1.2 * max_new_elems)); // add a little more space to avoid future reallocs
489  else
490  elems.clear(); // we don't know how many elements will pass the filter
491 
492  resized = true;
493  }
494 
495  if (filters.empty())
496  {
497  std::copy(begin, end, elems.begin());
498  num_elems = (intx)max_new_elems;
499  }
500  else
501  {
502  if (resized)
503  {
504  for (InputIterator ii = begin; ii != end; ++ii)
505  if (elementPassesFilters(*ii))
506  {
507  elems.push_back(*ii);
508  ++num_elems;
509  }
510  }
511  else
512  {
513  typename ElementArray::iterator ei = elems.begin();
514  for (InputIterator ii = begin; ii != end; ++ii)
515  if (elementPassesFilters(*ii))
516  {
517  *(ei++) = *ii;
518  ++num_elems;
519  }
520  }
521  }
522  }
523 
524  if (num_elems <= 0)
525  return;
526 
527  static intx const DEFAULT_MAX_ELEMS_IN_LEAF = 10;
528  max_elems_per_leaf = max_elems_per_leaf_ < 0 ? DEFAULT_MAX_ELEMS_IN_LEAF : max_elems_per_leaf_;
529 
530  // The fraction of elements held by the larger node at each split is 0.5
531  static double const SPLIT_FRACTION = 0.5;
532  intx est_depth = Math::binaryTreeDepth(num_elems, max_elems_per_leaf, SPLIT_FRACTION);
533  max_depth = max_depth_;
534  if (max_depth < 0)
535  max_depth = est_depth;
536  else if (max_depth < est_depth)
537  est_depth = max_depth;
538 
539  // THEA_CONSOLE << "KDTreeN: max_depth = " << max_depth << ", est_depth = " << est_depth;
540 
541  // Each index is stored at most once at each level
542  size_t BUFFER_SAFETY_MARGIN = 10;
543  size_t index_buffer_capacity = num_elems + BUFFER_SAFETY_MARGIN;
544  if (!save_memory)
545  index_buffer_capacity *= (size_t)(1 + est_depth); // reserve space for all levels at once
546 
547  if (deallocate_previous_memory || index_buffer_capacity > 1.3 * index_pool.getBufferCapacity())
548  {
549  // THEA_CONSOLE << "KDTreeN: Resizing index pool: old buffer capacity = " << index_pool.getBufferCapacity()
550  // << ", new buffer capacity = " << index_buffer_capacity;
551  index_pool.init(index_buffer_capacity);
552  }
553 
554  // Assume a complete, balanced binary tree upto the estimated depth to guess the number of leaf nodes
555  size_t node_buffer_capacity = (size_t)(1 << est_depth) + BUFFER_SAFETY_MARGIN;
556  if (deallocate_previous_memory || node_buffer_capacity > 1.3 * node_pool.getBufferCapacity())
557  {
558  // THEA_CONSOLE << "KDTreeN: Resizing node pool: old buffer capacity = " << node_pool.getBufferCapacity()
559  // << ", new buffer capacity = " << node_buffer_capacity;
560  node_pool.init(node_buffer_capacity);
561  }
562 
563  // Create the root node
564  root = node_pool.alloc(1);
565  root->init(0);
566  num_nodes = 1;
567 
568  // THEA_CONSOLE << "Allocated root " << root << " from mempool " << &node_pool;
569 
570  root->num_elems = num_elems;
571  root->elems = index_pool.alloc(root->num_elems);
572 
573  AxisAlignedBoxT elem_bounds;
574  for (size_t i = 0; i < (size_t)num_elems; ++i)
575  {
576  root->elems[i] = i;
577 
578  BoundedTraitsT::getBounds(elems[i], elem_bounds);
579  root->bounds.merge(elem_bounds);
580  }
581 
582  // Expand the bounding box slightly to handle numerical error
583  root->bounds.scaleCentered(BOUNDS_EXPANSION_FACTOR);
584 
585  if (save_memory)
586  {
587  // Estimate the maximum number of indices that will need to be held in the scratch pool at any time during depth-first
588  // traversal with earliest-possible deallocation. This is
589  //
590  // #elements * (sum of series 1 + 1 + SPLIT_FRACTION + SPLIT_FRACTION^2 + ... + SPLIT_FRACTION^(est_depth - 1))
591  // <= #elements * (1 + 1 / (1 - SPLIT_FRACTION))
592  //
593  size_t est_max_path_indices = (size_t)(num_elems * (1 + 1 / (1 - SPLIT_FRACTION)));
594  // THEA_CONSOLE << "KDTreeN: Estimated maximum number of indices on a single path = " << est_max_path_indices;
595 
596  // Create a temporary pool for scratch storage
597  IndexPool tmp_index_pool;
598  tmp_index_pool.init(est_max_path_indices + BUFFER_SAFETY_MARGIN);
599 
600  createTree(root, true, &tmp_index_pool, &index_pool);
601  }
602  else
603  createTree(root, false, &index_pool, nullptr);
604 
605  invalidateBounds();
606  }

832  template <typename MetricT, typename QueryT>
833  NeighborPair closestPair(QueryT const & query, double dist_bound = -1, bool get_closest_points = false) const
834  {
835  if (!root) return NeighborPair(-1);
836 
837  // Early pruning if the entire structure is too far away from the query
838  AxisAlignedBoxT query_bounds;
839  getObjectBounds(query, query_bounds);
840  double mon_approx_dist_bound = (dist_bound >= 0 ? MetricT::computeMonotoneApprox(dist_bound) : -1);
841  if (mon_approx_dist_bound >= 0)
842  {
843  double lower_bound = monotonePruningDistance<MetricT>(root, query, query_bounds);
844  if (lower_bound >= 0 && lower_bound > mon_approx_dist_bound)
845  return NeighborPair(-1);
846  }
847 
848  // If acceleration is enabled, set an upper limit to the distance to the nearest object
849  double accel_bound = accelerationBound<MetricT>(query, dist_bound);
850  if (accel_bound >= 0)
851  {
852  double fudge = 0.001 * getBoundsWorldSpace(*root).getExtent().norm();
853  mon_approx_dist_bound = MetricT::computeMonotoneApprox(accel_bound + fudge);
854  }
855 
856  NeighborPair pair(-1, -1, mon_approx_dist_bound);
857  closestPair<MetricT>(root, query, query_bounds, pair, get_closest_points);
858 
859  return pair;

1549  // have not saved memory by flushing references at internal nodes), then process all these elems.
1550  if (start->num_elems > 0 && range.contains(tr_start_bounds))
