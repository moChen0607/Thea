301  void readHeader(Header & header, BinaryInputStream & in) const
302  {
303  std::string magic = trimWhitespace(in.readLine());
304  if (magic != "ply")
305  throw Error(std::string(getName()) + ": Invalid PLY stream (does not start with 'ply')");
306 
307  header = Header(); // reset
308 
309  bool first = true;
310  std::string line, field;
311  while (in.hasMore())
312  {
313  line = trimWhitespace(in.readLine());
314 
315  if (line.empty())
316  continue;
317 
318  if (line == "end_header")
319  return;
320 
321  std::istringstream line_in(line);
322  if (!(line_in >> field))
323  throw Error(std::string(getName()) + ": Could not read first word of line");
324 
325  if (field == "format")
326  {
327  if (first)
328  {
329  if (!(line_in >> field))
330  throw Error(std::string(getName()) + ": Could not read ascii/binary specifier");
331 
332  // Don't bother looking for the version number at the end of the line
333  if (field == "ascii")
334  header.binary = false;
335  else if (field == "binary_little_endian")
336  {
337  header.binary = true;
338  header.endianness = Endianness::LITTLE;
339  }
340  else if (field == "binary_big_endian")
341  {
342  header.binary = true;
343  header.endianness = Endianness::BIG;
344  }
345  else
346  throw Error(std::string(getName()) + ": Invalid format specifier");
347 
348  first = false;
349  }
350  else
351  throw Error(std::string(getName()) + ": Format specifier must be first line after magic string");
352  }
353  else if (field == "element")
354  {
355  if (!(line_in >> field))
356  throw Error(std::string(getName()) + ": Could not read element type");
357 
358  ElementBlock block;
359  if (!(line_in >> block.num_elems))
360  throw Error(std::string(getName()) + ": Could not read number of elements of type '" + field + '\'');
361 
362  if (field == "vertex")
363  block.type = ElementType::VERTEX;
364  else if (field == "face")
365  block.type = ElementType::FACE;
366  else
367  block.type = ElementType::UNKNOWN;
368 
369  header.elem_blocks.push_back(block);
370  }
371  else if (field == "property")
372  {
373  if (header.elem_blocks.empty())
374  throw Error(std::string(getName()) + ": Property specification outside an element block");
375 
376  Property prop;
377  readPropertyHeader(prop, line_in);
378  if (prop.type == PropertyType::INVALID)
379  throw Error(std::string(getName()) + ": Invalid property type");
380 
381  header.elem_blocks.back().props.push_back(prop);
382  }
383  }
384 
385  throw Error(std::string(getName()) + ": No end_header token, header not closed");
386  }

389  void checkBlock(ElementBlock const & block) const
390  {
391  if (block.type == ElementType::VERTEX)
392  {
393  if (block.props.size() < 3
394  || block.props[0].type == PropertyType::LIST
395  || block.props[1].type == PropertyType::LIST
396  || block.props[2].type == PropertyType::LIST)
397  throw Error(std::string(getName()) + ": Vertex element must start with three numerical coordinates");
398  }
399  else if (block.type == ElementType::FACE)
400  {
401  if (block.props.empty()
402  || block.props[0].type != PropertyType::LIST
403  || (block.props[0].count_type & 0xF000)
404  || (block.props[0].item_type & 0xF000))
405  throw Error(std::string(getName()) + ": Face element must start with a list of integer indices");
406  }
407  }

410  void readAscii(MeshGroup & mesh_group, BinaryInputStream & in, Header const & header, ReadCallback * callback) const
411  {
412  // Create new mesh
413  MeshPtr mesh(new Mesh(std::string(mesh_group.getName()) + "/Mesh"));
414 
415  // Create a builder for the mesh
416  Builder builder(mesh);
417  builder.begin();
418 
419  Array<typename Builder::VertexHandle> vrefs;
420  Array<typename Builder::VertexHandle> face;
421 
422  std::string line;
423  intx num_vertices = 0, num_faces = 0;
424  for (size_t i = 0; i < header.elem_blocks.size(); ++i)
425  {
426  ElementBlock const & block = header.elem_blocks[i];
427  checkBlock(block);
428 
429  for (intx j = 0; j < block.num_elems; ++j)
430  {
431  do
432  {
433  if (!in.hasMore())
434  throw Error(std::string(getName()) + ": Unexpected end of input");
435 
436  line = trimWhitespace(in.readLine());
437 
438  } while (line.empty() || beginsWith(line, "comment"));
439 
440  switch (block.type)
441  {
442  case ElementType::VERTEX:
443  {
444  double x, y, z;
445  std::istringstream vstr(line);
446  if (!(vstr >> x >> y >> z))
447  throw Error(std::string(getName()) + ": Could not read vertex on line '" + line + '\'');
448 
449  typename Builder::VertexHandle vref = builder.addVertex(Vector3((Real)x, (Real)y, (Real)z),
450  (read_opts.store_vertex_indices ? num_vertices : -1));
451  if (callback)
452  callback->vertexRead(mesh.get(), num_vertices, vref);
453 
454  vrefs.push_back(vref);
455  num_vertices++;
456 
457  break;
458  }
459 
460  case ElementType::FACE:
461  {
462  intx index, num_face_vertices;
463  std::istringstream vstr(line);
464  if (!(vstr >> num_face_vertices))
465  throw Error(std::string(getName()) + ": Could not read number of vertices in face on line '" + line + '\'');
466 
467  if (num_face_vertices > 0)
468  {
469  face.resize(num_face_vertices);
470 
471  bool skip = false;
472  for (intx v = 0; v < num_face_vertices && !skip; ++v)
473  {
474  if (!(vstr >> index))
475  {
476  if (read_opts.strict)
477  throw Error(std::string(getName()) + ": Could not read vertex index on line '" + line + '\'');
478  else
479  {
480  THEA_WARNING << getName() << ": Skipping face, could not read vertex index on line '" << line << '\'';
481  skip = true; break;
482  }
483  }
484 
485  if (index < 0 || index >= (intx)vrefs.size())
486  {
487  if (read_opts.strict)
488  throw Error(getName() + format(": Vertex index %ld out of bounds (#vertices = %ld) on line '%s'",
489  index, (intx)vrefs.size(), line.c_str()));
490  else
491  {
492  THEA_WARNING << getName() << ": Vertex index " << index << " out of bounds (#vertices = "
493  << vrefs.size() << ") on line '" << line << '\'';
494  skip = true; break;
495  }
496  }
497 
498  face[(size_t)v] = vrefs[(size_t)index];
499 
500  for (int w = 0; w < v; ++w)
501  if (face[w] == face[v]) // face has repeated vertices
502  {
503  if (read_opts.strict)
504  throw Error(std::string(getName()) + ": Face has repeated vertices on line '" + line + '\'');
505  else
506  {
507  THEA_WARNING << getName() << ": Skipping face with repeated vertices on line '" << line << '\'';
508  skip = true; break;
509  }
510  }
511  }
512 
513  if (!skip)
514  {
515  typename Builder::FaceHandle fref = builder.addFace(face.begin(), face.end(),
516  (read_opts.store_face_indices ? num_faces : -1));
517  if (callback)
518  callback->faceRead(mesh.get(), num_faces, fref);
519 
520  num_faces++;
521  }
522  }
523 
524  break;
525  }
526 
527  default: {}
528  }
529  }
530  }
531 
532  builder.end();
533 
534  if (builder.numVertices() > 0 || builder.numFaces() > 0)
535  mesh_group.addMesh(mesh);
536 
537  THEA_CONSOLE << getName() << ": Read mesh with " << num_vertices << " vertices and " << num_faces << " faces";
538  }

541  template <typename T> T readBinaryNumber(BinaryInputStream & in, PropertyType const & type) const
542  {
543  switch (type)
544  {
545  case PropertyType::INT8: return static_cast<T>(in.readInt8());
546  case PropertyType::INT16: return static_cast<T>(in.readInt16());
547  case PropertyType::INT32: return static_cast<T>(in.readInt32());
548  case PropertyType::UINT8: return static_cast<T>(in.readUInt8());
549  case PropertyType::UINT16: return static_cast<T>(in.readUInt16());
550  case PropertyType::UINT32: return static_cast<T>(in.readUInt32());
551  case PropertyType::FLOAT32: return static_cast<T>(in.readFloat32());
552  case PropertyType::FLOAT64: return static_cast<T>(in.readFloat64());
553  default: throw Error(std::string(getName()) + ": Unknown property type");

558  template <typename T> void readBinaryList(BinaryInputStream & in, Property const & prop, Array<T> & items) const
559  {
560  debugAssertM(prop.type == PropertyType::LIST, std::string(getName()) + ": Can't read non-list property as list");

585  void readBinary(MeshGroup & mesh_group, BinaryInputStream & in, Header const & header, ReadCallback * callback) const
586  {
587  // Create new mesh
588  MeshPtr mesh(new Mesh(std::string(mesh_group.getName()) + "/Mesh"));
589 
590  // Create a builder for the mesh
591  Builder builder(mesh);
592  builder.begin();
593 
594  Array<typename Builder::VertexHandle> vrefs;
595  Array<typename Builder::VertexHandle> face;
596 
597  BinaryInputStream::EndiannessScope scope(in, header.endianness);
598 
599  intx num_vertices = 0, num_faces = 0;
600  for (size_t i = 0; i < header.elem_blocks.size(); ++i)
601  {
602  ElementBlock const & block = header.elem_blocks[i];
603  checkBlock(block);
604 
605  for (intx j = 0; j < block.num_elems; ++j)
606  {
607  switch (block.type)
608  {
609  case ElementType::VERTEX:
610  {
611  Vector3 vertex;
612  vertex[0] = readBinaryNumber<Real>(in, block.props[0].type);
613  vertex[1] = readBinaryNumber<Real>(in, block.props[1].type);
614  vertex[2] = readBinaryNumber<Real>(in, block.props[2].type);
615 
616  for (size_t k = 3; k < block.props.size(); ++i)
617  skipBinaryProperty(in, block.props[k]);
618 
619  typename Builder::VertexHandle vref = builder.addVertex(vertex,
620  (read_opts.store_vertex_indices ? num_vertices : -1));
621  if (callback)
622  callback->vertexRead(mesh.get(), num_vertices, vref);
623 
624  vrefs.push_back(vref);
625  num_vertices++;
626 
627  break;
628  }
629 
630  case ElementType::FACE:
631  {
632  Array<intx> face_vertices;
633  readBinaryList(in, block.props[0], face_vertices);
634 
635  if (!face_vertices.empty())
636  {
637  face.resize(face_vertices.size());
638 
639  bool skip = false;
640  for (size_t v = 0; v < face_vertices.size() && !skip; ++v)
641  {
642  intx index = face_vertices[v];
643  if (index < 0 || index >= (intx)vrefs.size())
644  {
645  if (read_opts.strict)
646  throw Error(getName() + format(": Vertex index %ld out of bounds (#vertices = %ld) in face %ld",
647  index, (intx)vrefs.size(), num_faces));
648  else
649  {
650  THEA_WARNING << getName() << ": Skipping face, vertex index " << index << " out of bounds (#vertices = "
651  << vrefs.size() << ')';
652  skip = true; break;
653  }
654  }
655 
656  face[v] = vrefs[(size_t)index];
657 
658  for (size_t w = 0; w < v; ++w)
659  if (face[w] == face[v]) // face has repeated vertices
660  {
661  if (read_opts.strict)
662  throw Error(getName() + format(": Face %ld has repeated vertices", num_faces));
663  else
664  {
665  THEA_WARNING << getName() << ": Skipping face with repeated vertices";
666  skip = true; break;
667  }
668  }
669  }
670 
671  if (!skip)
672  {
673  typename Builder::FaceHandle fref = builder.addFace(face.begin(), face.end(),
674  (read_opts.store_face_indices ? num_faces : -1));
675  if (callback)
676  callback->faceRead(mesh.get(), num_faces, fref);
677 
678  num_faces++;
679  }
680  }
681 
682  break;
683  }
684 
685  default: {}
686  }
687  }
688  }
689 
690  builder.end();
691 
692  if (builder.numVertices() > 0 || builder.numFaces() > 0)
693  mesh_group.addMesh(mesh);
694 
695  THEA_CONSOLE << getName() << ": Read mesh with " << num_vertices << " vertices and " << num_faces << " faces";
696  }

718  {
