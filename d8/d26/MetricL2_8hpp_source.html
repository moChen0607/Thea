197 template <typename A, typename B, int N, typename T>
198 struct MetricL2Impl<A, B, N, T, typename std::enable_if< !IsPointN<A, N>::value
199  && !MetricL2Internal::TransformedObjectCheck<A>::value
200  && !std::is_pointer<A>::value
201  && IsNonReferencedPointN<B, N>::value >::type>

204

206  {
207  Vector<N, T> p = PointTraitsN<B, N, T>::getPosition(b);
208  return (a.closestPoint(p) - p).squaredNorm();
209  }

212  {
