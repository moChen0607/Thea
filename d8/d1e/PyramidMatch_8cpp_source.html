22 // Integrals of the one-dimensional gaussian with standard deviation 0.741 (chosen to give an integral of 0.5 within distance
23 // 0.5 of the mean). GAUSSIAN_INTEGRALS[i][0], [i][1] and [i][2] give the integrals over the ranges (-infty, 0], [0, 1] and
24 // [1, infty) respectively of the gaussian with mean (i + 0.5) / 10.
25 int const NUM_GAUSSIAN_INTEGRAL_BINS = 10;
26 Real const GAUSSIAN_INTEGRALS[NUM_GAUSSIAN_INTEGRAL_BINS][3] = { {0.473112f, 0.426885f, 0.100003f},
27  {0.419823f, 0.454411f, 0.125766f},
28  {0.367966f, 0.476202f, 0.155832f},
29  {0.318412f, 0.491301f, 0.190287f},
30  {0.271912f, 0.499026f, 0.229062f},
31  {0.229062f, 0.499026f, 0.271912f},
32  {0.190287f, 0.491301f, 0.318412f},
33  {0.155832f, 0.476202f, 0.367966f},
34  {0.125766f, 0.454411f, 0.419823f},
35  {0.100003f, 0.426885f, 0.473112f} };

37 void
38 smoothIncrement1D(Real * dst, int dst_size, int index, Real mean, Real value)
39 {
40  int sub_bin = Math::clamp((int)std::floor(NUM_GAUSSIAN_INTEGRAL_BINS * mean), 0, NUM_GAUSSIAN_INTEGRAL_BINS - 1);
41  dst[std::max(index - 1, 0) ] += value * GAUSSIAN_INTEGRALS[sub_bin][0]; // (-infty, 0]
42  dst[index ] += value * GAUSSIAN_INTEGRALS[sub_bin][1]; // [0, 1]
43  dst[std::min(index + 1, dst_size - 1)] += value * GAUSSIAN_INTEGRALS[sub_bin][2]; // [1, infty)
44 }

46 void
47 smoothIncrement2D(Real * dst, int dst_sx, int dst_sy, int x, int y, Real mean_x, Real mean_y, Real value)
48 {
49  int sub_bin_x = Math::clamp((int)std::floor(NUM_GAUSSIAN_INTEGRAL_BINS * mean_x), 0, NUM_GAUSSIAN_INTEGRAL_BINS - 1);
50  int sub_bin_y = Math::clamp((int)std::floor(NUM_GAUSSIAN_INTEGRAL_BINS * mean_y), 0, NUM_GAUSSIAN_INTEGRAL_BINS - 1);
51 
52  int ix[3] = { std::max(x - 1, 0), x, std::min(x + 1, dst_sx - 1) };
53  int iy[3] = { std::max(y - 1, 0), y, std::min(y + 1, dst_sy - 1) };
54 
55  for (int i = 0; i < 3; ++i)
56  for (int j = 0; j < 3; ++j)
57  dst[iy[i] * dst_sx + ix[j]] += value * GAUSSIAN_INTEGRALS[sub_bin_x][j] * GAUSSIAN_INTEGRALS[sub_bin_y][i];
58 }

60 void
61 smoothIncrement3D(Real * dst, int dst_sx, int dst_sy, int dst_sz, int x, int y, int z, Vector3 const & mean, Real value)
62 {
63  int sub_bin_x = Math::clamp((int)std::floor(NUM_GAUSSIAN_INTEGRAL_BINS * mean.x()), 0, NUM_GAUSSIAN_INTEGRAL_BINS - 1);
64  int sub_bin_y = Math::clamp((int)std::floor(NUM_GAUSSIAN_INTEGRAL_BINS * mean.y()), 0, NUM_GAUSSIAN_INTEGRAL_BINS - 1);
65  int sub_bin_z = Math::clamp((int)std::floor(NUM_GAUSSIAN_INTEGRAL_BINS * mean.z()), 0, NUM_GAUSSIAN_INTEGRAL_BINS - 1);
66 
67  int ix[3] = { std::max(x - 1, 0), x, std::min(x + 1, dst_sx - 1) };
68  int iy[3] = { std::max(y - 1, 0), y, std::min(y + 1, dst_sy - 1) };
69  int iz[3] = { std::max(z - 1, 0), z, std::min(z + 1, dst_sz - 1) };
70 
71  for (int i = 0; i < 3; ++i)
72  for (int j = 0; j < 3; ++j)
73  for (int k = 0; k < 3; ++k)
74  dst[(iz[i] * dst_sy + iy[j]) * dst_sx + ix[k]] += value * GAUSSIAN_INTEGRALS[sub_bin_x][k]
75  * GAUSSIAN_INTEGRALS[sub_bin_y][j]
76  * GAUSSIAN_INTEGRALS[sub_bin_z][i];
77 }

97 template <typename MeanT>
98 void
99 downsample1D(Real const * src, MeanT const * src_means, Real * dst, MeanT * dst_means, int dst_size)
100 {
101  if (src_means)
102  {
103  Real w0, w1, sum_weights;
104 
105  for (int i = 0, j = 0; i < dst_size; ++i, j += 2)
106  {
107  w0 = src[j];
108  w1 = src[j + 1];
109  sum_weights = w0 + w1;
110 
111  if (Math::fuzzyEq(sum_weights, (Real)0))
112  dst_means[i] = 0.5f * halve1D(MeanT(src_means[j]
113  + add1D(src_means[j + 1], 1)));
114  else
115  dst_means[i] = halve1D(MeanT(w0 * src_means[j]
116  + w1 * add1D(src_means[j + 1], 1))) / sum_weights;
117 
118  smoothIncrement1D(dst, dst_size, (int)i, toReal(dst_means[i]), sum_weights);
119  }
120  }
121  else
122  {
123  for (int i = 0, j = 0; i < dst_size; ++i, j += 2)
124  dst[i] = src[j] + src[j + 1];
125  }
126 }

128 template <typename MeanT>
129 void
130 downsample2D(Real const * src, MeanT const * src_means, Real * dst, MeanT * dst_means, int dst_sx, int dst_sy)
131 {
132  static Vector2 const offset01(0, 1);
133  static Vector2 const offset10(1, 0);
134  static Vector2 const offset11(1, 1);
135 
136  int src_sx = 2 * dst_sx;
137  intx i = 0;
138  intx j00 = 0, j10 = 1, j01 = src_sx, j11 = src_sx + 1;
139 
140  if (src_means)
141  {
142  Real w00, w10, w01, w11, sum_weights;
143 
144  for(int y = 0; y < dst_sy; ++y, j00 += src_sx, j10 += src_sx, j01 += src_sx, j11 += src_sx)
145  for(int x = 0; x < dst_sx; ++x, ++i, j00 += 2, j10 += 2, j01 += 2, j11 += 2)
146  {
147  w00 = src[j00];
148  w10 = src[j10];
149  w01 = src[j01];
150  w11 = src[j11];
151  sum_weights = w00 + w10 + w01 + w11;
152 
153  if (Math::fuzzyEq(sum_weights, (Real)0))
154  dst_means[i] = 0.25f * halve2D(MeanT(src_means[j00]
155  + add2D(src_means[j10], offset10)
156  + add2D(src_means[j01], offset01)
157  + add2D(src_means[j11], offset11)));
158  else
159  dst_means[i] = halve2D(MeanT(w00 * src_means[j00]
160  + w10 * add2D(src_means[j10], offset10)
161  + w01 * add2D(src_means[j01], offset01)
162  + w11 * add2D(src_means[j11], offset11))) / sum_weights;
163 
164  smoothIncrement2D(dst, dst_sx, dst_sy, x, y, dst_means[i].x(), dst_means[i].y(), sum_weights);
165  }
166  }
167  else
168  {
169  for(int y = 0; y < dst_sy; ++y, j00 += src_sx, j10 += src_sx, j01 += src_sx, j11 += src_sx)
170  for(int x = 0; x < dst_sx; ++x, ++i, j00 += 2, j10 += 2, j01 += 2, j11 += 2)
171  dst[i] = src[j00] + src[j10] + src[j01] + src[j11];
172  }
173 }

175 void
176 downsample3D(Real const * src, Vector3 const * src_means, Real * dst, Vector3 * dst_means, int dst_sx, int dst_sy, int dst_sz)
177 {
178  static Vector3 const offset001(0, 0, 1);
179  static Vector3 const offset010(0, 1, 0);
180  static Vector3 const offset011(0, 1, 1);
181  static Vector3 const offset100(1, 0, 0);
182  static Vector3 const offset101(1, 0, 1);
183  static Vector3 const offset110(1, 1, 0);
184  static Vector3 const offset111(1, 1, 1);
185 
186  int src_sx = 2 * dst_sx;
187  int src_sxy = src_sx * (2 * dst_sy);
188  intx i = 0;
189  intx j000 = 0, j100 = 1, j010 = src_sx, j110 = src_sx + 1;
190  intx j001 = src_sxy, j101 = src_sxy + 1, j011 = src_sxy + src_sx, j111 = src_sxy + src_sx + 1;
191 
192  if (src_means)
193  {
194  Real w000, w100, w010, w110, w001, w101, w011, w111, sum_weights;
195 
196  for(intx z = 0; z < dst_sz; ++z, j000 += src_sxy, j100 += src_sxy, j010 += src_sxy, j110 += src_sxy,
197  j001 += src_sxy, j101 += src_sxy, j011 += src_sxy, j111 += src_sxy)
198  for(intx y = 0; y < dst_sy; ++y, j000 += src_sx, j100 += src_sx, j010 += src_sx, j110 += src_sx,
199  j001 += src_sx, j101 += src_sx, j011 += src_sx, j111 += src_sx)
200  for(intx x = 0; x < dst_sx; ++x, ++i, j000 += 2, j100 += 2, j010 += 2, j110 += 2,
201  j001 += 2, j101 += 2, j011 += 2, j111 += 2)
202  {
203  w000 = src[j000];
204  w100 = src[j100];
205  w010 = src[j010];
206  w110 = src[j110];
207  w001 = src[j001];
208  w101 = src[j101];
209  w011 = src[j011];
210  w111 = src[j111];
211  sum_weights = w000 + w100 + w010 + w110 + w001 + w101 + w011 + w111;
212 
213  if (Math::fuzzyEq(sum_weights, (Real)0))
214  dst_means[i] = 0.125f * 0.5f * (src_means[j000]
215  + src_means[j100] + offset100
216  + src_means[j010] + offset010
217  + src_means[j110] + offset110
218  + src_means[j001] + offset001
219  + src_means[j101] + offset101
220  + src_means[j011] + offset011
221  + src_means[j111] + offset111);
222  else
223  dst_means[i] = (0.5f / sum_weights) * (w000 * src_means[j000]
224  + w100 * src_means[j100] + offset100
225  + w010 * src_means[j010] + offset010
226  + w110 * src_means[j110] + offset110
227  + w001 * src_means[j001] + offset001
228  + w101 * src_means[j101] + offset101
229  + w011 * src_means[j011] + offset011
230  + w111 * src_means[j111] + offset111);
231 
232  smoothIncrement3D(dst, dst_sx, dst_sy, dst_sz, x, y, z, dst_means[i], sum_weights);
233  }
234  }
235  else
236  {
237  for(intx z = 0; z < dst_sz; ++z, j000 += src_sxy, j100 += src_sxy, j010 += src_sxy, j110 += src_sxy,
238  j001 += src_sxy, j101 += src_sxy, j011 += src_sxy, j111 += src_sxy)
239  for(intx y = 0; y < dst_sy; ++y, j000 += src_sx, j100 += src_sx, j010 += src_sx, j110 += src_sx,
240  j001 += src_sx, j101 += src_sx, j011 += src_sx, j111 += src_sx)
241  for(intx x = 0; x < dst_sx; ++x, ++i, j000 += 2, j100 += 2, j010 += 2, j110 += 2,
242  j001 += 2, j101 += 2, j011 += 2, j111 += 2)
243  dst[i] = src[j000] + src[j100] + src[j010] + src[j110]
244  + src[j001] + src[j101] + src[j011] + src[j111];
245  }
246 }

310  PyramidInternal::downsample1D(src_ptr, src_means_ptr, dst_ptr, dst_means_ptr, dst_sx);

321 Pyramid3D::downsample(Array3D const & src, Array3D & dst, int num_dims_to_downsample, Vector3 * means) const
322 {
323  alwaysAssertM(num_dims_to_downsample > 0, "Pyramid3D: At least one dimension must reduce in size during downsampling");
324 
325  int dst_sx = src.sx / 2;
326  int dst_sy = num_dims_to_downsample > 1 ? src.sy / 2 : src.sy;
327  int dst_sz = num_dims_to_downsample > 2 ? src.sz / 2 : src.sz;
328 
329  if (means)
330  dst = Array3D(dst_sx, dst_sy, dst_sz, 0);
331  else
332  dst = Array3D(dst_sx, dst_sy, dst_sz);
333 
334  if (num_dims_to_downsample >= 3)
335  PyramidInternal::downsample3D(&src.data[0], means, &dst.data[0], means, dst_sx, dst_sy, dst_sz);
336  else if (num_dims_to_downsample == 2)
337  {
338  Real const * src_ptr = &src.data[0];
339  Real * dst_ptr = &dst.data[0];
340  intx src_step = src.sx * src.sy;
341  intx dst_step = dst_sx * dst_sy;
342 
343  if (means)
344  {
345  Vector3 * src_means_ptr = means;
346  Vector3 * dst_means_ptr = means;
347 
348  for (int z = 0; z < dst_sz; ++z, src_ptr += src_step, src_means_ptr += src_step, dst_ptr += dst_step,
349  dst_means_ptr += dst_step)
350  PyramidInternal::downsample2D(src_ptr, src_means_ptr, dst_ptr, dst_means_ptr, dst_sx, dst_sy);
351  }
352  else
353  {
354  for (int z = 0; z < dst_sz; ++z, src_ptr += src_step, dst_ptr += dst_step)
355  PyramidInternal::downsample2D(src_ptr, means, dst_ptr, means, dst_sx, dst_sy);
356  }
357  }
358  else
359  {
360  Real const * src_ptr = &src.data[0];
361  Real * dst_ptr = &dst.data[0];
362 
363  if (means)
364  {
365  Vector3 * src_means_ptr = means;
366  Vector3 * dst_means_ptr = means;
367 
368  for (int z = 0; z < dst_sz; ++z)
369  for (int y = 0; y < dst_sy; ++y, src_ptr += src.sx, src_means_ptr += src.sx, dst_ptr += dst_sx, dst_means_ptr += dst_sx)
370  PyramidInternal::downsample1D(src_ptr, src_means_ptr, dst_ptr, dst_means_ptr, dst_sx);
371  }
372  else
373  {
374  for (int z = 0; z < dst_sz; ++z)
375  for (int y = 0; y < dst_sy; ++y, src_ptr += src.sx, dst_ptr += dst_sx)
376  PyramidInternal::downsample1D(src_ptr, means, dst_ptr, means, dst_sx);

474 void
475 Pyramid2D::read(BinaryInputStream & input, Codec const & codec, bool read_block_header)
476 {
477  if (read_block_header)
478  input.skip(Codec::BlockHeader::SERIALIZED_LENGTH); // not used
479 
480  { BinaryInputStream::EndiannessScope scope(input, Endianness::LITTLE);
481 
482  uint32 dims = input.readUInt32();
483  if (dims != 2) throw Error("Pyramid2D: Number of dimensions must be 2");
484 
485  num_levels = input.readUInt32();
486  levels.clear();
487  if (num_levels > 0) levels.resize(num_levels);
488 
489  nx = ny = 0;
490 
491  for (int i = 0; i < num_levels; ++i)
492  {
493  Array2D & level = levels[i];
494  level.sx = (size_t)input.readUInt32();
495  level.sy = (size_t)input.readUInt32();
496

712 inline Real
713 similarity(PyramidMatch::Kernel kernel, Real x, Real y)
714 {
715  switch (kernel)
716  {
717  case PyramidMatch::Kernel::L1: return -std::fabs(x - y);
718  case PyramidMatch::Kernel::L2: return -Math::square(x - y);
719  case PyramidMatch::Kernel::CHI_SQUARED: return -Math::square(x - y) / (x < 1 ? 1 : x);
720  default: /* HIK */ return std::min(x, y);
721  }
722 }

726 Real
727 PyramidMatch::similarity(Pyramid1D const & pyramid1, Pyramid1D const & pyramid2, Kernel kernel, int levels,
728  Real attenuation_factor)
729 {
730  alwaysAssertM(pyramid1.num_levels == pyramid2.num_levels, "PyramidMatch: Can't compare pyramids of different dimensions");
731 
732  int num_levels = (levels > 0 ? levels : pyramid1.num_levels);
733  if (attenuation_factor <= 0) attenuation_factor = 0.5;
734 
735  Real ans = 0;
736  Real scale = 1;
737  for (int i = 0; i < num_levels; ++i)
738  {
739  Array<Real> const & level_data1 = pyramid1.levels[i];
740  Array<Real> const & level_data2 = pyramid2.levels[i];
741 
742  debugAssertM(level_data1.size() == level_data2.size(), "PyramidMatch: Levels of comparable pyramids should have same size");
743 
744  Real level_ans = 0;
745  for(size_t j = 0; j < level_data1.size(); ++ j)
746  level_ans += PyramidInternal::similarity(kernel, level_data1[j], level_data2[j]);
747 
748  ans += (level_ans * scale);
749  scale *= attenuation_factor;
750  }
751 
752  return ans;
753 }
