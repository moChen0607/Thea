98 // If a block header is present, use it to create a new input stream wrapping the matrix data block, and also get the parameters
99 // stored in the custom field. Else, leave the parameters unchanged, and return a null pointer.

113 inline char
114 detectSeparatorFromLine(std::string const & line, char default_sep)
115 {
116  // Look for separators in the order comma, tab, space
117  if (line.find_first_of(',') != std::string::npos) { return ','; }
118  if (line.find_first_of('\t') != std::string::npos) { return '\t'; }
119  if (line.find_first_of(' ') != std::string::npos) { return ' '; }
120  return default_sep;
121 }

123 // Check from the content of a CSV stream (not the block header) whether it is likely to represent a sparse matrix or a dense
124 // matrix
125 inline bool
126 isSparse(BinaryInputStream & input, char sep)
127 {
128  std::string line1, line2;
129  auto curr_pos = input.getPosition();
130  if (input.hasMore()) line1 = input.readLine();
131  if (input.hasMore()) line2 = input.readLine();
132  input.setPosition(curr_pos);
133 
134  if (line1.empty() || line2.empty())
135  return false; // can't really guess, so assume it is dense
136 
137  if (sep < 0) sep = detectSeparatorFromLine(line1, ',');
138  if (sep < 0) return false; // couldn't detect separator, again default to dense
139 
140  auto n1 = std::count(line1.begin(), line1.end(), sep);
141  auto n2 = std::count(line2.begin(), line2.end(), sep);
142 
143  return (n1 == 1 && n2 == 2); // a non-empty sparse matrix has "#rows,#cols" on the first line, and a triplet on the second
144 }

172  void readMatrix(MatrixT & m, BinaryInputStream & input, Codec::BlockHeader const * block_header) const
173  {
174  BinaryInputStream * in = &input;
175  CodecCSVInternal::Params params{ /* is_sparse = */ false, /* separator = */ separator };
176  auto tmp_in = CodecCSVInternal::parseHeader(input, block_header, params);
177  if (tmp_in) { in = tmp_in.get(); }
178 
179  if (params.is_sparse)
180  throw Error(format("%s: Matrix in stream '%s' is sparse, but a dense target matrix was given",
181  getName(), input.getName()));
182 
183  Array<std::string> lines;
184  while (in->hasMore())
185  {
186  lines.push_back(in->readLine());
187 
188  // If the separator is still not set, try to autodetect it from the first line
189  if (lines.size() == 1 && params.separator < 0)
190  {
191  params.separator = CodecCSVInternal::detectSeparatorFromLine(lines[0], '\n'); // assume entire line is one field
192  if (params.separator < 0)
193  throw Error(format("%s: Could not autodetect the field separator from the stream '%s'",
194  getName(), input.getName()));
195  }
196  }
197 
198  if (lines.empty())
199  {
200  m.resize(0, 0);
201  return;
202  }
203 
204  intx num_rows = (intx)lines.size();
205  intx num_cols = (intx)std::count(lines[0].begin(), lines[0].end(), params.separator) + 1;
206  m.resize(num_rows, num_cols);
207 
208  std::string field;
209  for (intx i = 0; i < num_rows; ++i)
210  {
211  std::istringstream line_in(lines[(size_t)i]);
212  for (intx j = 0; j < num_cols; ++j)
213  {
214  if (!std::getline(line_in, field, params.separator))
215  throw Error(format("%s: Could not read matrix element (%ld, %ld) from stream '%s'",
216  getName(), (long)i, (long)j, input.getName()));
217 
218  m(i, j) = fromString<typename MatrixT::value_type>(field);
219  }
220  }
221  }

223  void writeMatrix(MatrixT const & m, BinaryOutputStream & output, bool write_block_header) const
224  {
225  Codec::BlockHeader bh(this->getMagic());
226  if (write_block_header)
227  bh.markAndSkip(output);
228 
229  // No need to set endianness, CSV is a purely text-based format
230 
231  char sep = (separator < 0 ? ',' : separator);
232  for (intx i = 0; i < m.rows(); ++i)
233  {
234  std::ostringstream row;
235  for (intx j = 0; j < m.cols(); ++j)
236  {
237  if (j > 0) row << sep;
238  row << m(i, j);
239  }
240  row << '\n';
241  output.writeBytes((int64)row.str().length(), row.str().data());
242  }
243 
244  if (write_block_header)
245  bh.calcAndWrite(output);
246  }

277  void readMatrix(MatrixT & m, BinaryInputStream & input, Codec::BlockHeader const * block_header) const
278  {
279  BinaryInputStream * in = &input;
280  CodecCSVInternal::Params params{ /* is_sparse = */ true, /* separator = */ separator };
281  auto tmp_in = CodecCSVInternal::parseHeader(input, block_header, params);
282  if (tmp_in) { in = tmp_in.get(); }
283 
284  if (!params.is_sparse)
285  throw Error(format("%s: Matrix in stream '%s' is dense, but a sparse target matrix was given",
286  getName(), input.getName()));
287 
288  if (!in->hasMore())
289  {
290  m.resize(0, 0);
291  m.data().squeeze();
292  return;
293  }
294 
295  // Read dimensions of matrix
296  std::string rstr, cstr;
297  {
298  auto line = in->readLine();
299 
300  // If the separator is still not set, try to autodetect it from the first line
301  if (params.separator < 0)
302  {
303  params.separator = CodecCSVInternal::detectSeparatorFromLine(line, -1);
304  if (params.separator < 0)
305  throw Error(format("%s: Could not autodetect the field separator from the stream '%s'",
306  getName(), input.getName()));
307  }
308 
309  std::istringstream line_in(line);
310  if (!std::getline(line_in, rstr, params.separator) || !std::getline(line_in, cstr, params.separator))
311  throw Error(format("%s: Could not read matrix dimensions from stream '%s'", getName(), input.getName()));
312 
313  intx nrows = (intx)std::stol(rstr), ncols = (intx)std::stol(cstr);
314  if (nrows < 0 || ncols < 0)
315  throw Error(format("%s: Negative matrix dimension in stream '%s'", getName(), input.getName()));
316 
317  m.resize(nrows, ncols);
318  m.data().squeeze();
319  }
320 
321  // Read triplets and initialize matrix
322  {
323  typedef typename MatrixT::value_type Value;
324  Array< Eigen::Triplet<Value> > triplets;
325  std::string vstr;
326  while (in->hasMore())
327  {
328  std::istringstream line_in(in->readLine());
329  if (!std::getline(line_in, rstr, params.separator)
330  || !std::getline(line_in, cstr, params.separator)
331  || !std::getline(line_in, vstr, params.separator))
332  throw Error(format("%s: Could not read triplet on line %ld of stream '%s'",
333  getName(), (long)triplets.size() + 2, input.getName()));
334 
335  intx row = (intx)std::stol(rstr), col = (intx)std::stol(cstr);
336  auto value = fromString<typename MatrixT::value_type>(vstr);
337  if (row < 0 || row >= m.rows() || col < 0 || col >= m.cols())
338  throw Error(format("%s: Out-of-bounds matrix position (%ld, %ld) in stream '%s'",
339  getName(), (long)row, (long)col, input.getName()));
340 
341  triplets.push_back(Eigen::Triplet<Value>(row, col, value));
342  }
343 
344  m.setFromTriplets(triplets.begin(), triplets.end());
345  }
346  }

378 template <typename MatrixT>
379 intx
380 BinaryInputStream::readMatrix(bool read_block_header, MatrixT & m, Codec const & codec)
381 {
382  Codec::BlockHeader bh_obj, * bh = nullptr;
383  if (read_block_header)
384  {
385  bh_obj.read(*this);
386  bh = &bh_obj;
387  }
388 
389  if (codec == Codec_AUTO())
390  {
391  // Default codecs
392  static CodecCSV<MatrixT> const codec_csv;
393  static MatrixCodec<MatrixT> const * default_codecs[] = { &codec_csv };
394 
395  for (size_t i = 0; i < sizeof(default_codecs) / sizeof(MatrixCodec<MatrixT> const *); ++i)
396  if (default_codecs[i]->matches(*this, bh))
397  {
398  default_codecs[i]->readMatrix(m, *this, bh);
399  return 0; // always
400  }
401 
402  throw Error(getNameStr() + ": Could not autodetect a suitable codec to read the matrix");
403  }
404  else
405  {
406  MatrixCodec<MatrixT> const * mat_codec = dynamic_cast< MatrixCodec<MatrixT> const * >(&codec);
407  if (!mat_codec)
408  throw Error(getNameStr() + ": Codec is not a matrix codec for this matrix type");
409 
410  if (bh && bh->magic != mat_codec->getMagic())
411  throw Error(getNameStr() + ": Magic string mismatch");
412 
413  mat_codec->readMatrix(m, *this, bh);
414  }
415 
416  return 0; // always
417 }
