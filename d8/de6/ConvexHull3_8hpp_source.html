88  template <typename Mesh> void computeApprox(Mesh & mesh) const
89  {
90  updateApprox();
91 
92  typedef Graphics::IncrementalMeshBuilder<Mesh> Builder;
93  Builder builder(&mesh);
94  builder.begin();
95 
96  Array<typename Builder::VertexHandle> approx_vrefs(approx_vertices.size());
97  for (size_t i = 0; i < approx_vertices.size(); ++i)
98  approx_vrefs[i] = builder.addVertex(approx_vertices[i]);
99 
100  typename Builder::VertexHandle face[3];
101  for (size_t i = 0; i < approx_indices.size(); i += 3)
102  {
103  face[0] = approx_vrefs[approx_indices[i ]];
104  face[1] = approx_vrefs[approx_indices[i + 1]];
105  face[2] = approx_vrefs[approx_indices[i + 2]];
106 
107  builder.addFace(face, face + 3);
108  }
109 
110  builder.end();
111  }
