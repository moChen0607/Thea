33 Plugin *
34 PluginManager::load(std::string const & path)
35 {
36  if (dynlibs.find(path) != dynlibs.end()) return nullptr;
37 
38  DynLib * dynlib = Application::getDynLibManager().load(path);
39 
40  THEA_DLL_START_PLUGIN start_func = (THEA_DLL_START_PLUGIN)dynlib->getSymbol("dllStartPlugin");
41  Plugin * plugin = start_func(this);
42 
43  if (!plugin)
44  {
45  Application::getDynLibManager().unload(dynlib);
46 
47  THEA_ERROR << "PluginManager: Could not initialize plugin '" << path << '\'';
48  return nullptr;
49  }
50 
51  install(plugin, dynlib, path);
52 
53  return plugin;
54 }

92 void
93 PluginManager::unloadAllPlugins()

95  shutdownAllPlugins();

97  for (PluginList::iterator pi = plugins.begin(); pi != plugins.end(); ++pi)
98  {
99  THEA_LOG << "PluginManager: Uninstalling plugin '" << (*pi)->getName() << '\'';
100  (*pi)->uninstall();

102 
103  plugins.clear();

105  for (DynLibMap::iterator di = dynlibs.begin(); di != dynlibs.end(); ++di)

107  THEA_DLL_STOP_PLUGIN stop_func = (THEA_DLL_STOP_PLUGIN)di->second.dynlib->getSymbol("dllStopPlugin");

111  }

113  dynlibs.clear();

134 void
135 PluginManager::install(Plugin * plugin, DynLib * dynlib, std::string const & path)
136 {
137  alwaysAssertM(plugin, "PluginManager: Plugin pointer cannot be null");
138  alwaysAssertM(!dynlib || !path.empty(), "PluginManager: Plugin loaded from a dynamic library must have a non-empty path");
139 
140  for (PluginList::iterator pi = plugins.begin(); pi != plugins.end(); ++pi)
141  if (*pi == plugin) // already installed
142  return;
143 
144  THEA_LOG << "PluginManager: Installing plugin '" << plugin->getName() << '\'';
145 
146  plugins.push_back(plugin);
147  if (dynlib)
148  dynlibs[path] = PluginDynLib(plugin, dynlib);
149 
150  plugin->install();
151 }

153 void
154 PluginManager::uninstall(Plugin * plugin)
155 {
156  if (!plugin) return;
157 
158  THEA_LOG << "PluginManager: Uninstalling plugin '" << plugin->getName() << '\'';
159  plugin->uninstall();
160 
161  for (PluginList::iterator pi = plugins.begin(); pi != plugins.end(); ++pi)
162  if (*pi == plugin)
163  {
164  plugins.erase(pi);
165  break;
166  }
167 
168  for (DynLibMap::iterator di = dynlibs.begin(); di != dynlibs.end(); ++di)
169  if (di->second.plugin == plugin)
170  {
171  dynlibs.erase(di);
172  break;
173  }
174 }
