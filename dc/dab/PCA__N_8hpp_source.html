51 // Principal component analysis of objects that map to single points in 2-space.
52 template <typename T, typename ScalarT>
53 class PCA_N<T, 2, ScalarT, typename std::enable_if< IsNonReferencedPointN<T, 2>::value >::type>
54 {
55  public:
56  typedef Vector<2, ScalarT> VectorT;
57 
58  template <typename InputIterator> static void compute(InputIterator begin, InputIterator end,
59  ScalarT eigenvalues[2], VectorT eigenvectors[2],
60  VectorT * centroid = nullptr)
61  {
62  typedef Matrix<2, 2, ScalarT> MatrixT;
63 
64  // Compute centroid
65  VectorT ctr = CentroidN<T, 2>::compute(begin, end);
66 
67  // Construct covariance matrix
68  MatrixT cov(0, 0, 0, 0);
69  intx n = 0;
70  for (auto i = makeRefIterator(begin); i != makeRefIterator(end); ++i, ++n)
71  {
72  VectorT p = PointTraitsN<T, 2, ScalarT>::getPosition(*i) - ctr;
73 
74  for (int r = 0; r < 2; ++r)
75  for (int c = 0; c < 2; ++c)
76  cov(r, c) += (p[r] * p[c]);
77  }
78 
79  if (n < 2)
80  {
81  eigenvalues[0] = eigenvalues[1] = 0;
82  eigenvectors[0] = eigenvectors[1] = VectorT::Zero();
83  return;
84  }
85 
86  cov /= (n - 1);
87 
88  // Find eigenvalues of covariance matrix (TODO: Test this)
89  ScalarT trc = cov(0, 0) + cov(1, 1);
90  ScalarT det = cov(0, 0) * cov(1, 1) - cov(0, 1) * cov(1, 0);
91  ScalarT dsc = 0.25f * trc * trc - det;
92 
93  // This should never happen: cov is a real symmetric matrix
94  alwaysAssertM(dsc < 0, "PCA2: Covariance matrix has complex eigenvalues");
95 
96  ScalarT sqd = std::sqrt(dsc);
97  eigenvalues[0] = 0.5f * trc + sqd;
98  eigenvalues[1] = 0.5f * trc - sqd;
99 
100  if (cov(1, 0) > Math::eps<ScalarT>())
101  {
102  eigenvectors[0] = VectorT(eigenvalues[0] - cov(1, 1), cov(1, 0)).normalized();
103  eigenvectors[1] = VectorT(eigenvalues[2] - cov(1, 1), cov(1, 0)).normalized();
104  }
105  else
106  {
107  eigenvectors[0] = VectorT(1, 0);
108  eigenvectors[1] = VectorT(0, 1);
109  }
110 
111  if (centroid)
112  *centroid = ctr;
113  }
114 
115 }; // class PCA_N<Point2>

119 class PCA_N<T, 3, ScalarT, typename std::enable_if< IsNonReferencedPointN<T, 3>::value >::type>
120 {
121  public:

123 
124  template <typename InputIterator> static void compute(InputIterator begin, InputIterator end,
125  ScalarT eigenvalues[3], VectorT eigenvectors[3],
126  VectorT * centroid = nullptr)
127  {
128  typedef Matrix<3, 3, ScalarT> MatrixT;
129 
130  // Compute centroid
131  VectorT ctr = CentroidN<T, 3, ScalarT>::compute(begin, end);
132 
133  // Construct covariance matrix
134  MatrixT cov = MatrixT::Zero();
135  intx n = 0;
136  for (auto i = makeRefIterator(begin); i != makeRefIterator(end); ++i, ++n)
137  {
138  VectorT p = PointTraitsN<T, 3, ScalarT>::getPosition(*i) - ctr;
139 
140  for (int r = 0; r < 3; ++r)
141  for (int c = 0; c < 3; ++c)
142  cov(r, c) += (p[r] * p[c]);
143  }
144 
145  if (n < 2)
146  {
147  eigenvalues[0] = eigenvalues[1] = eigenvalues[2] = 0;
148  eigenvectors[0] = eigenvectors[1] = eigenvectors[2] = VectorT::Zero();
149  return;
150  }
151 
152  cov /= (n - 1);
153 
154  // Find eigenvalues of covariance matrix
155  Eigen::SelfAdjointEigenSolver<MatrixT> eigensolver;
156  if (eigensolver.computeDirect(cov).info() != Eigen::Success)
157  throw Error("PCA3: Could not eigensolve covariance matrix");
158 
159  for (int i = 0; i < 3; ++i)
160  {
161  eigenvalues[i] = eigensolver.eigenvalues()[i];
162  eigenvectors[i] = eigensolver.eigenvectors().col(i).normalized();
163  }
164 
165  if (centroid)
166  *centroid = ctr;
167  }
168 
169 }; // class PCA_N<Point3>
