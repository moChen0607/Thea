138  GeneralMeshEdge * nextAroundEndpoint(int i)
139  {
140  debugAssertM(i == 0 || i == 1, "GeneralMeshEdge: Invalid endpoint index");
141 
142  if (numFaces() > 2) // non-manifold
143  return nullptr;
144 
145  // Find which incident face has this endpoint as the origin of the edge when stepping round the face. The required edge
146  // is then the predecessor of this edge around the face.
147  for (FaceIterator fi = facesBegin(); fi != facesEnd(); ++fi)
148  {
149  Face * face = *fi;
150  GeneralMeshEdge * prev = face->getPredecessor(this);
151  if (prev->hasEndpoint(endpoints[i])) // found it!
152  return prev;
153  }
154 
155  return nullptr;
156  }
