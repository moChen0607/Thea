91  void cacheBinom() const
92  {
93  if (binom.rows() > 0)
94  return;
95 
96  intx n = (intx)ctrl[0].size();
97  binom.resize(n, n);
98 
99  // From https://www.geometrictools.com/GTEngine/Include/Mathematics/GteBezierCurve.h
100  //
101  // Compute combinatorial values binom(n, k). The values binom(r, c) are invalid for r < c; that is, we use only the
102  // entries for r >= c.
103  binom(0, 0) = 1;
104  binom(1, 0) = 1;
105  binom(1, 1) = 1;
106  for (intx i = 2; i < n; ++i)
107  {
108  binom(i, 0) = 1;
109  binom(i, i) = 1;
110  for (intx j = 1; j < i; ++j)
111  binom(i, j) = binom(i - 1, j - 1) + binom(i - 1, j);
112  }
113  }

115  void update() const
116  {
117  if (!this->isChanged()) return;
118 
119  size_t n = ctrl[0].size();
120 
121  // Compute first-order differences
122  ctrl[1].resize(n - 1);
123  for (size_t i = 0; i < n - 1; ++i)
124  ctrl[1][i] = ctrl[0][i + 1] - ctrl[0][i];
125 
126  // Compute second-order differences
127  if (n > 2)
128  {
129  ctrl[2].resize(n - 2);
130  for (size_t i = 0; i < n - 2; ++i)
131  ctrl[2][i] = ctrl[1][i + 1] - ctrl[1][i];
132  }
133 
134  // Compute third-order differences
135  if (n > 3)
136  {
137  ctrl[3].resize(n - 3);
138  for (size_t i = 0; i < n - 3; ++i)

146  VectorT eval(T const & t, intx deriv_order) const
147  {
148  alwaysAssertM(t >= -0.00001 && t <= 1.00001, format("BezierN: Curve parameter %lf out of range", static_cast<double>(t)));
149  alwaysAssertM(deriv_order >= 0, format("BezierN: Invalid derivative order %ld", deriv_order));
150 
151  intx order = getOrder();
152  if (deriv_order > order) return VectorT::Zero();
153 
154  update();
155 
156  T omt = 1 - t;
157  VectorT result = omt * ctrl[deriv_order][0];
158 
159  T tpow = t;
160  intx isup = order - deriv_order;
161  for (intx i = 1; i < isup; ++i)
162  {
163  T c = static_cast<T>(binom(isup, i) * tpow);
164  result = (result + c * ctrl[deriv_order][i]) * omt;
165  tpow *= t;
166  }
167  result = (result + tpow * ctrl[deriv_order][isup]);
168 
169  intx multiplier = 1;
170  for (intx i = 0; i < deriv_order; ++i)
171  multiplier *= (order - i);
172 
173  result *= multiplier;
174 
175  return result;
176  }
