82 // Iterate over a circular loop of edges, derefencing to some property of the edge accessed by a pointer. EdgeT should be either
83 // DCELHalfedge or DCELHalfedge const. This tries to map a circular loop to an iterator with a well-defined begin and end, by
84 // keeping track of where the iteration began, and allowing only forward progress. There are clear caveats with this approach
85 // but it should fit a majority of use cases. Use with care.
86 //
87 // DerefFunctorT should be callable with the signature: ValueT * operator()(EdgeT *)
88 // IncrementFunctorT should be callable with the signature: void operator()(EdgeT **)
89 template <typename EdgeT, typename ValueT, typename DerefFunctorT, typename IncrementFunctorT>
90 class FwdIterator
91 : public std::iterator<std::forward_iterator_tag, ValueT, std::ptrdiff_t, ValueT *, ValueT &>
92 {
93  public:
94  // Constructor.
95  FwdIterator(EdgeT * e_ = nullptr, bool first_ = true) : initial(e_), e(e_), first(first_) {}
96 
97  // Construct from a compatible iterator, typically a non-const to const conversion. Compatibility is the responsibility of
98  // the caller.
99  template <typename E2, typename V2, typename D2, typename I2> explicit FwdIterator(FwdIterator<E2, V2, D2, I2> const & src)
100  : initial(src.getInitial()), e(src.getCurrent()), first(src.isAtInitial()) {}
101 
102  // Assign from a compatible iterator, typically a non-const to const conversion. Compatibility is the responsibility of
103  // the caller.
104  template <typename OtherFwdIteratorT> FwdIterator & operator=(OtherFwdIteratorT const & src)
105  {
106  initial = src.initial;
107  e = src.e;
108  first = src.first;
109 
110  return *this;
111  }
112 
113  // Equality comparison.
114  bool operator==(FwdIterator const & other)
115  {
116  // True if the current edges match AND (both are null OR neither is at its initial position).
117  return e == other.e && (!e || ((e != initial || !first) && (other.e != other.initial || !other.first)));
118  }
119 
120  // Inequality comparison.
121  bool operator!=(FwdIterator const & other) { return !operator==(other); }
122 
123  // Dereference.
124  ValueT * operator*() const { return DerefFunctorT()(e); }
125 
126  // Arrow operator.
127  ValueT const * const * operator->() const
128  { alwaysAssertM(false, "FwdIterator: Can't call '->' on iterator-over-pointers"); }
129 
130  // Pre-increment. */
131  FwdIterator & operator++()
132  {
133  if (e)
134  {
135  IncrementFunctorT()(&e);
136  first = false;
137  }
138 
139  return *this;
140  }
141 
142  // Post-increment.
143  FwdIterator operator++(int)
144  {
145  FwdIterator ret = *this;
146  if (e)
147  {
148  IncrementFunctorT()(&e);
149  first = false;
150  }
151 
152  return ret;
153  }
154 
155  // Get the initial edge from which the iterator was created.
156  EdgeT * getInitial() const { return initial; }
157 
158  // Get the current edge the iterator is at.
159  EdgeT * getCurrent() const { return e; }
160 
161  // Check if the iterator is at the initial position without ever having moved away from it, or not.
162  bool isAtInitial() const { return first; }
163 
164  private:
165  EdgeT * initial; // The edge used to construct the iterator.
166  EdgeT * e; // The current edge.
167  bool first; // Is this the the first time we're seeing the edge the iterator was initialized with?
168 
169 }; // class FwdIterator
