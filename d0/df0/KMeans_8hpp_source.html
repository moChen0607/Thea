150  typename std::enable_if< std::is_base_of< AbstractAddressableMatrix<typename AddressableMatrixT::Value>,
151  AddressableMatrixT >::value, int >::type = 0 >
152  bool cluster(intx num_clusters, AddressableMatrixT const & points, intx * labeling = nullptr,
153  double * squared_distances = nullptr)
154  {
155  static intx const DEFAULT_MAX_ITERS = 1000;
156  static double const DEFAULT_MAX_TIME = 60; // seconds
157 
158  alwaysAssertM(num_clusters > 0, "KMeans: Number of clusters must be at least 1");
159 
160  intx num_points = points.rows();
161  intx num_features = points.cols();
162 
163  alwaysAssertM(num_points > 0, "KMeans: Number of points must be at least 1");
164  alwaysAssertM(num_features > 0, "KMeans: Number of features must be at least 1");
165 
166  num_clusters = std::min(num_clusters, num_points);
167 
168  if (options.verbose)
169  THEA_CONSOLE << "KMeans: Clustering " << num_points << " point(s) into " << num_clusters << " group(s)";
170 
171  //=======================================================================================================================
172  // Select initial centers
173  //=======================================================================================================================
174 
175  double start_time = System::time();
176  Stopwatch timer;
177  if (options.verbose)
178  timer.tick();
179 
180  if (!selectInitialCenters(num_clusters, points))
181  return false;
182 
183  if (options.verbose)
184  {
185  timer.tock();
186  THEA_CONSOLE << "KMeans: Selected initial center(s) in " << timer.elapsedTime() << 's';
187  timer.tick();
188  }
189 
190  //=======================================================================================================================
191  // Find initial assignment of points to clusters
192  //=======================================================================================================================
193 
194  Array<intx> labeling_local;
195  if (!labeling)
196  {
197  labeling_local.resize((size_t)num_points, -1);
198  labeling = &labeling_local[0];
199  }
200 
201  Array<double> sqdist_local;
202  if (!squared_distances)
203  {
204  sqdist_local.resize((size_t)num_points);
205  squared_distances = &sqdist_local[0];
206  }
207 
208  mapToClusters(num_clusters, points, labeling, squared_distances);
209 
210  if (options.verbose)
211  {
212  timer.tock();
213  THEA_CONSOLE << "KMeans: Made initial assignment of points to clusters in " << timer.elapsedTime() << 's';
214  timer.tick();
215  }
216 
217  //=======================================================================================================================
218  // Iterate
219  //=======================================================================================================================
220 
221  intx max_iters = (options.max_iterations >= 0 ? options.max_iterations : DEFAULT_MAX_ITERS);
222  double max_time = (options.max_time >= 0 ? options.max_time : DEFAULT_MAX_TIME);
223 
224  bool converged = false;
225  intx num_iterations = 0;
226  double last_time = System::time();
227  while (!converged)
228  {
229  if (max_iters >= 0 && num_iterations >= max_iters)
230  break;
231 
232  double curr_time = System::time();
233  if (max_time >= 0 && curr_time - start_time > max_time)
234  break;
235 
236  // Print an update every 3 seconds or every iteration, whichever is longer
237  if (options.verbose && curr_time - last_time > 3)
238  {
239  THEA_CONSOLE << "KMeans: -- " << num_iterations << " iteration(s)";
240  last_time = curr_time;
241  }
242 
243  updateClusters(points, labeling, squared_distances); // FIXME: Check for reassignments and || with converged? Unclear.
244  converged = !mapToClusters(num_clusters, points, labeling, squared_distances);
245  num_iterations++;
246  }
247 
248  if (options.verbose)
249  {
250  timer.tock();
251 
252  if (converged)
253  THEA_CONSOLE << "KMeans: Clusters found after " << num_iterations << " iteration(s) in " << timer.elapsedTime()
254  << 's';
255  else
256  THEA_CONSOLE << "KMeans: Did not converge after " << num_iterations << " iteration(s) in " << timer.elapsedTime()
257  << 's';
258  }
259 
260  return converged;
261  }

314  template <typename AddressableMatrixT>
315  bool selectInitialCenters(intx num_clusters, AddressableMatrixT const & points)
316  {
317  intx num_points = (intx)points.rows();
318  intx num_features = (intx)points.cols();
319 
320  alwaysAssertM(num_clusters > 0, "KMeans: Must select at least one center");
321  alwaysAssertM(num_points >= num_clusters, "KMeans: Cannot select more centers than points");
322  alwaysAssertM(num_features > 0, "KMeans: Cannot select centers without any point features");
323 
324  if (options.seeding == Seeding::K_MEANS_PLUS_PLUS)
325  {
326  // Choose initial cluster centers by k-means++ [Arthur/Vassilvitskii '07]:
327  // 1. Choose one center uniformly at random from among the data points.
328  // 2. For each data point x, compute D(x), the distance between x and the nearest center that has already been chosen.
329  // 3. Choose one new data point at random as a new center, using a weighted probability distribution where a point x is
330  // chosen with probability proportional to D(x)^2.
331  // 4. Repeat steps 2 and 3 until k centers have been chosen.
332 
333  centers.resize(num_clusters, num_features);
334  centers.fill(0);
335 
336  // First center is randomly chosen
337  intx index = Random::common().integer(0, num_points - 1);
338  addPointToCenter(points, index, 0);
339 
340  // Subsequent centers by k-means++
341  Array<double> sqdist((size_t)num_points);
342  double start_time = System::time();
343  for (intx i = 1; i < num_clusters; ++i)
344  {
345  mapToClusters(i, points, nullptr, &sqdist[0]);
346 
347  // Sample next center from points with probability proportional to sqdist
348  double sum_sqdist = 0;
349  for (size_t j = 0; j < sqdist.size(); ++j)
350  sum_sqdist += sqdist[j];
351 
352  double r = Random::common().uniform(0, sum_sqdist);
353  sum_sqdist = 0;
354  index = num_points - 1; // to compensate for numerical error when r is approximately = sum_sqdist
355  for (size_t j = 0; j < sqdist.size(); ++j)
356  {
357  sum_sqdist += sqdist[j];
358  if (sum_sqdist >= r)
359  {
360  index = (intx)j;
361  break;
362  }
363  }
364 
365  addPointToCenter(points, index, i);
366 
367  double curr_time = System::time();
368  if (curr_time - start_time > 3) // print an update every 3 seconds or every iteration, whichever is longer
369  {
370  THEA_CONSOLE << "KMeans: -- selected " << i + 1 << " center(s)";
371  start_time = curr_time;
372  }
373  }
374  }
375  else if (options.seeding == Seeding::RANDOM)
376  {
377  centers.resize(num_clusters, num_features);
378  centers.fill(0);
379 
380  Array<intx> indices((size_t)num_points);
381  for (intx i = 0; i < num_points; ++i)
382  indices[(size_t)i] = i;
383 
384  // Select num_clusters random points as initial centers
385  Random::common().randomShuffle((int32)num_points, (int32)num_clusters, &indices[0]);
386 
387  for (intx i = 0; i < num_clusters; ++i)
388  addPointToCenter(points, indices[(size_t)i], i);
389  }
390  else
391  {
392  THEA_ERROR << "KMeans: Unsupported seeding strategy " << options.seeding.toString();
393  return false;
394  }
395 
396  return true;
397  }

453  template <typename AddressableMatrixT>
454  bool mapToClusters(intx num_clusters, AddressableMatrixT const & points, intx * cluster_indices,
455  double * cluster_sqdists = nullptr) const
456  {
457  intx num_points = (intx)points.rows();
458  unsigned int concurrency = std::thread::hardware_concurrency();
459  flag = 0;
460 
461  if (options.parallelize && concurrency > 1 && num_points > (intx)(2 * concurrency))
462  {
463  ThreadGroup pool;
464  double points_per_thread = num_points / (double)concurrency;
465 
466  intx points_begin = 0;
467  for (unsigned int i = 0; i < concurrency; ++i)
468  {
469  intx points_end = (i + 1 == concurrency ? num_points
470  : std::min((intx)std::ceil(points_begin + points_per_thread), num_points));
471  if (points_begin >= points_end) // more threads than points
472  continue;
473 
474  pool.addThread(new std::thread(ClusterMapper<AddressableMatrixT>(this,
475  num_clusters,
476  &points,
477  points_begin,
478  points_end,
479  cluster_indices,
480  cluster_sqdists)));
481  points_begin = points_end;
482  }
483 
484  pool.joinAll();
485  }
486  else
487  {
488  ClusterMapper<AddressableMatrixT> mapper(this, num_clusters, &points, 0, num_points, cluster_indices, cluster_sqdists);
489  mapper();
490  }
491 
492  return (flag.value() > 0);
493  }

506  void mapToCluster(intx num_clusters, double const * point, intx & cluster_index, double & cluster_sqdist) const
507  {
508  intx num_features = centers.cols();
509  cluster_index = -1;
510  cluster_sqdist = -1;
511  for (intx i = 0; i < num_clusters; ++i)
512  {
513  // Compute squared distance to this center
514  double sqdist = 0;
515  for (intx j = 0; j < num_features; ++j)
516  {
517  double diff = point[j] - centers(i, j);
518  sqdist += (diff * diff);
519  }
520 
521  if (i == 0 || sqdist < cluster_sqdist)
522  {
523  cluster_index = i;
524  cluster_sqdist = sqdist;
525  }
526  }
527  }

534  template <typename AddressableMatrixT>
535  bool updateClusters(AddressableMatrixT const & points, intx * cluster_indices, double * cluster_sqdists)
536  {
537  intx num_clusters = centers.rows();
538  intx num_points = points.rows();
539  intx num_features = points.cols();
540 
541  centers.fill(0);
542 
543  Array<intx> num_assigned((size_t)num_clusters, 0);
544  for (intx i = 0; i < num_points; ++i)
545  {
546  intx cc_index = cluster_indices[i];
547 
548  addPointToCenter(points, i, cc_index);
549  num_assigned[(size_t)cc_index]++;
550  }
551 
552  // If a cluster is empty then:
553  // 1. Find the biggest cluster.
554  // 2. Find the farthest from the center point in the biggest cluster.
555  // 3. Exclude the farthest point from the biggest cluster and form a new 1-point cluster.
556  //
557  // (follows OpenCV)
558 
559  bool reassigned = false;
560 
561  for (intx i = 0; i < num_clusters; ++i)
562  {
563  if (num_assigned[(size_t)i] <= 0)
564  {
565  intx max_cluster = (intx)(std::max_element(num_assigned.begin(), num_assigned.end()) - num_assigned.begin());
566  alwaysAssertM(num_assigned[(size_t)max_cluster] > 1, "KMeans: Maximum cluster has < 2 points");
567 
568  intx farthest = -1;
569  for (intx j = 0; j < num_points; ++j)
570  if (cluster_indices[j] == max_cluster && (farthest < 0 || cluster_sqdists[j] > cluster_sqdists[farthest]))
571  farthest = j;
572 
573  alwaysAssertM(farthest >= 0, "KMeans: Farthest point in maximum cluster not found");
574 
575  // Reassign the point
576  cluster_indices[farthest] = i;
577  cluster_sqdists[farthest] = 0.0;
578 
579  addPointToCenter(points, farthest, i);
580  subtractPointFromCenter(points, farthest, max_cluster);
581 
582  num_assigned[(size_t)i]++;
583  num_assigned[(size_t)max_cluster]--;
584 
585  reassigned = true;
586  }
587  }
588 
589  for (intx i = 0; i < num_clusters; ++i)
590  {
591  intx n = num_assigned[(size_t)i];
592  if (n > 0)
593  {
594  for (intx j = 0; j < num_features; ++j)
595  centers(i, j) /= n;
596  }
597  }
598 
599  return reassigned;
600  }
