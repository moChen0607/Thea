109 static bool
110 Polygon3_insideTriangle2(Vector2 const & A, Vector2 const & B, Vector2 const & C, Vector2 const & P)
111 {
112  Real ax, ay, bx, by, cx, cy, apx, apy, bpx, bpy, cpx, cpy;
113  Real cCROSSap, bCROSScp, aCROSSbp;
114 
115  ax = C.x() - B.x(); ay = C.y() - B.y();
116  bx = A.x() - C.x(); by = A.y() - C.y();
117  cx = B.x() - A.x(); cy = B.y() - A.y();
118  apx = P.x() - A.x(); apy = P.y() - A.y();
119  bpx = P.x() - B.x(); bpy = P.y() - B.y();
120  cpx = P.x() - C.x(); cpy = P.y() - C.y();
121 
122  aCROSSbp = ax * bpy - ay * bpx;
123  cCROSSap = cx * apy - cy * apx;
124  bCROSScp = bx * cpy - by * cpx;
125 
126  return ((aCROSSbp >= 0) && (bCROSScp >= 0) && (cCROSSap >= 0));
127 }

129 bool
130 Polygon3::snip(size_t u, size_t v, size_t w, size_t n, Array<size_t> const & indices,
131  Real epsilon) const
132 {
133  Vector2 const & A = proj_vertices[indices[u]];
134  Vector2 const & B = proj_vertices[indices[v]];
135  Vector2 const & C = proj_vertices[indices[w]];
136 
137  if (epsilon > (((B.x() - A.x()) * (C.y() - A.y())) - ((B.y() - A.y()) * (C.x() - A.x()))))
138  {
139  // THEA_DEBUG << "Polygon3: Epsilon test failed: A = " << A << ", B = " << B << ", C = " << C << ", eps = " << EPSILON;
140  return false;
141  }
142 
143  for (size_t p = 0; p < n; ++p)
144  {
145  if ((p == u) || (p == v) || (p == w))
146  continue;
147 
148  Vector2 const & P = proj_vertices[indices[p]];
149  if (Polygon3_insideTriangle2(A, B, C, P))
150  {
151  // THEA_DEBUG << "Polygon3: Triangle test failed: A = " << A << ", B = " << B << ", C = " << C << ", P = " << P;
152  return false;
153  }
154  }
155 
156  return true;
157 }

159 // Original comment:
160 // Triangulation happens in 2d. We could inverse transform the polygon around the normal direction, or we just use the two
161 // most signficant axes. Here we find the two longest axes and use them to triangulate. Inverse transforming them would
162 // introduce more doubling point error and isn't worth it.
163 //
164 // SC says:
165 // This doesn't work: the vertices can be collinear when projected onto the plane of the two longest axes of the bounding box.
166 // Example (from real data):
167 //
168 // v[0] = (-13.7199, 4.45725, -8.00059)
169 // v[1] = (-0.115787, 12.3116, -4.96109)
170 // v[2] = (0.88992, 12.8922, -3.80342)
171 // v[3] = (-0.115787, 12.3116, -2.64576)
172 // v[4] = (-13.7199, 4.45725, 0.393742)
173 // v[5] = (-13.7199, 4.45725, -0.856258)
174 // v[6] = (-12.5335, 5.14221, -3.80342)
175 // v[7] = (-13.7199, 4.45725, -6.75059)
176 //
177 // Instead, we will project onto the plane of the polygon.
178 intx
179 Polygon3::triangulate(Array<intx> & tri_indices, Real epsilon) const
180 {
181  if (epsilon < 0)
182  epsilon = Math::eps<Real>();
183 
184  if (vertices.size() < 3)
185  {
186  tri_indices.clear();
187  }
188  else if (vertices.size() == 3)
189  {
190  tri_indices.resize(3);
191  tri_indices[0] = vertices[0].index;
192  tri_indices[1] = vertices[1].index;
193  tri_indices[2] = vertices[2].index;
194  }
195  else if (vertices.size() > 3)
196  {
197  tri_indices.clear();
198 
199  size_t n = vertices.size();
200  proj_vertices.resize(n);
201 
202  Vector3 normal = computeNormal();
203  Matrix3 basis = Math::orthonormalBasis(normal);
204  Vector3 axis0 = basis.col(0);
205  Vector3 axis1 = basis.col(1);
206 
207  Vector3 v0 = vertices[0].position; // a reference point for the plane of the polygon
208  for (size_t i = 0; i < n; ++i)
209  {
210  Vector3 v = vertices[i].position - v0;
211  proj_vertices[i] = Vector2(v.dot(axis0), v.dot(axis1));
212  }
213 
214  Array<size_t> indices(n);
215  bool flipped = false;
216  if (projArea() > 0)
217  {
218  for (size_t v = 0; v < n; ++v)
219  indices[v] = v;
220  }
221  else
222  {
223  for (size_t v = 0; v < n; ++v)
224  indices[v] = (n - 1) - v;
225 
226  flipped = true;
227  }
228 
229  size_t nv = n;
230  size_t count = 2 * nv;
231  for (size_t v = nv - 1; nv > 2; )
232  {
233  if ((count--) <= 0)
234  break;
235 
236  size_t u = v;
237  if (nv <= u) u = 0;
238 
239  v = u + 1;
240  if (nv <= v) v = 0;
241 
242  size_t w = v + 1;
243  if (nv <= w) w = 0;
244 
245  if (snip(u, v, w, nv, indices, epsilon))
246  {
247  size_t a = indices[u];
248  size_t b = indices[v];
249  size_t c = indices[w];
250  if (flipped)
251  {
252  tri_indices.push_back(vertices[c].index);
253  tri_indices.push_back(vertices[b].index);
254  tri_indices.push_back(vertices[a].index);
255  }
256  else
257  {
258  tri_indices.push_back(vertices[a].index);
259  tri_indices.push_back(vertices[b].index);
260  tri_indices.push_back(vertices[c].index);
261  }
262 
263  size_t s = v, t = v + 1;
264  for ( ; t < nv; ++s, ++t)
265  indices[s] = indices[t];
266 
267  nv--;
268  count = 2 * nv;
269  }
270  }
271  }
272 
273  return (intx)tri_indices.size() / 3;
274 }

276 Real
277 Polygon3::projArea() const
278 {
279  size_t n = proj_vertices.size();
280  Real a = 0;
281  for (size_t p = n - 1, q = 0; q < n; p = q++)
282  {
283  Vector2 const & pval = proj_vertices[p];
284  Vector2 const & qval = proj_vertices[q];
285  a += pval.x() * qval.y() - qval.x() * pval.y();
286  }
287 
288  return 0.5f * a;
289 }
