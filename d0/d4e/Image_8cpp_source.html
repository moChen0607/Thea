96 static Image::Type
97 typeFromFreeImageTypeAndBPP(FREE_IMAGE_TYPE fi_type, WORD fi_bpp)
98 {
99  switch (fi_type)
100  {
101  case FIT_BITMAP:
102  {
103  switch (fi_bpp)
104  {
105  case 1: return Image::Type::LUMINANCE_1U;
106  case 2: return Image::Type::LUMINANCE_2U;
107  case 4: return Image::Type::LUMINANCE_4U;
108  case 8: return Image::Type::LUMINANCE_8U;
109  case 24: return Image::Type::RGB_8U;
110  case 32: return Image::Type::RGBA_8U;
111  }
112  break;
113  }
114  case FIT_INT16: return Image::Type::LUMINANCE_16;
115  case FIT_UINT16: return Image::Type::LUMINANCE_16U;
116  case FIT_INT32: return Image::Type::LUMINANCE_32;
117  case FIT_UINT32: return Image::Type::LUMINANCE_32U;
118  case FIT_FLOAT: return Image::Type::LUMINANCE_32F;
119  case FIT_DOUBLE: return Image::Type::LUMINANCE_64F;
120  case FIT_RGB16: return Image::Type::RGB_16U;
121  case FIT_RGBA16: return Image::Type::RGBA_16U;
122  case FIT_RGBF: return Image::Type::RGB_32F;
123  case FIT_RGBAF: return Image::Type::RGBA_32F;
124  case FIT_COMPLEX: return Image::Type::COMPLEX_64F;
125  default: break; // to eliminate a GCC warning about unhandled FIT_UNKNOWN
126  }
127 
128  return Image::Type::UNKNOWN;
129 }

131 // Returns bytes consumed by an aligned row
132 int64
133 scanWidth(int64 width, int bpp, int32 alignment)
134 {
135  int64 width_bits = width * bpp;
136  int64 row_bytes = width_bits / 8 + (width_bits % 8 == 0 ? 0 : 1);
137  return row_bytes + (alignment - (row_bytes % alignment)) % alignment;
138 }

220 void \
221 codec::readImage(Image & image, BinaryInputStream & input, bool read_block_header) const \
222 { \
223  /* Get the size of the image block in bytes */ \
224  uint64 size = (read_block_header ? BlockHeader(input).data_size : input.size()); \
225  \
226  /* Read the image block into a memory buffer (optimization possible when the data has already been buffered within the */ \
227  /* input stream?) */ \
228  Array<uint8> img_block((size_t)size); \
229  input.readBytes((int64)size, img_block.data()); \
230  \
231  /* Decode the image */ \
232  fipMemoryIO mem((BYTE *)img_block.data(), (DWORD)size); \
233  FIBITMAP * bitmap = mem.load(fip_format, flags); \
234  if (!bitmap) \
235  throw Error(std::string(getName()) + ": Could not decode image from memory stream"); \
236  \
237  fipImage * fip_img = image._getFreeImage(); \
238  debugAssertM(fip_img, std::string(getName()) + ": Image does not wrap a valid FreeImage bitmap"); \
239  \
240  *fip_img = bitmap; /* the FIP object will now manage the destruction of the bitmap */ \
241  \
242  Image::Type type = ImageInternal::typeFromFreeImageTypeAndBPP(fip_img->getImageType(), fip_img->getBitsPerPixel()); \
243  if (type == Image::Type::UNKNOWN) \
244  { \
245  image.clear(); \
246  throw Error(std::string(getName()) \
247  + ": Image was successfully decoded but it has a format for which this library does not provide an interface"); \
248  } \
249  image._setType(type); \
250 }

347 void
348 Codec3BM::readImage(Image & image, BinaryInputStream & input, bool read_block_header) const
349 {
350  // Get the size of the image block in bytes
351  uint64 prefixed_size = (read_block_header ? BlockHeader(input).data_size : 0);
352 
353  BinaryInputStream::EndiannessScope scope(input, Endianness::LITTLE);
354 
355  int64 start_position = input.getPosition();
356 
357  // Read file header (32 bytes):
358  // 4 bytes: Magic string "3BM\0"
359  // 4 bytes: Version
360  // 8 bytes: Size of the whole file in bytes
361  // 8 bytes: Reserved
362  // 8 bytes: Starting offset (from the beginning of the file) of the bitmap image data
363 
364  uint8 magic[4];
365  input.readBytes(4, &magic);
366  if (magic[0] != (uint8)'3' || magic[1] != (uint8)'B' || magic[2] != (uint8)'M' || magic[3] != (uint8)'\0')
367  throw Error(std::string(getName()) + ": Image is not in 3BM format");
368 
369  input.skip(4);
370  uint64 size = input.readUInt64();
371  if (prefixed_size > 0 && size != prefixed_size)
372  throw Error(std::string(getName()) + ": Prefixed size does not match image size from header");
373 
374  input.skip(8);
375  uint64 bitmap_offset = input.readUInt64();
376 
377  // Read info header (72 bytes):
378  // 8 bytes: Size of this header (72 bytes)
379  // 8 bytes: Width of the bitmap in voxels
380  // 8 bytes: Height of the bitmap in voxels
381  // 8 bytes: Depth of the bitmap in voxels
382  // 4 bytes: Number of bits per pixel (1, 4, 8, 16, 24, 32), currently only 8, 24 and 32 are supported
383  // 4 bytes: Compression method (0 for no compression, currently this is the only one supported)
384  // 8 bytes: Size of raw image data
385  // 8 bytes: Width resolution in voxels/meter
386  // 8 bytes: Height resolution in voxels/meter
387  // 8 bytes: Depth resolution in voxels/meter
388 
389  input.skip(8);
390  int64 width = (int64)input.readUInt64();
391  int64 height = (int64)input.readUInt64();
392  int64 depth = (int64)input.readUInt64();
393 
394  int bpp = (int)input.readUInt32();
395  if (bpp != 8 && bpp != 24 && bpp != 32)
396  throw Error(std::string(getName()) + ": Only 8, 24 and 32-bit bitmaps currently supported");
397 
398  uint32 compression = input.readUInt32();
399  if (compression != 0)
400  throw Error(std::string(getName()) + ": Unsupported compression method");
401 
402  int64 stream_scan_width = ImageInternal::scanWidth(width, bpp, 16); // 16-byte row alignment for SSE compatibility
403  uint64 data_size = input.readUInt64();
404  if (data_size != (uint64)stream_scan_width * (uint64)height * (uint64)depth)
405  throw Error(std::string(getName()) + ": Pixel data block size does not match image dimensions");
406 
407  // Read image data as L, BGR or BGRA voxels, one 16-byte aligned scanline at a time
408  input.setPosition(start_position + (int64)bitmap_offset);
409 
410  Image::Type type = Image::Type::UNKNOWN;
411  switch (bpp)
412  {
413  case 8 : type = Image::Type::LUMINANCE_8U; break;
414  case 24 : type = Image::Type::RGB_8U; break;
415  case 32 : type = Image::Type::RGBA_8U; break;
416  default : throw Error(format("%s: Bit depth %d not supported", getName(), bpp));
417  }
418 
419  image.resize(type, width, height, depth);
420  if (width > 0 && height > 0 && depth > 0 && bpp > 0)
421  {
422  int bytes_pp = bpp / 8;
423  Array<uint8> row_buf((size_t)stream_scan_width);
424  for (intx i = 0; i < depth; ++i)
425  for (intx j = 0; j < height; ++j)
426  {
427  input.readBytes(stream_scan_width, &row_buf[0]);
428 
429  uint8 const * in_pixel = &row_buf[0];
430  uint8 * out_pixel = (uint8 *)image.getScanLine(j, i);
431 
432  switch (bytes_pp)
433  {
434  case 1: Algorithms::fastCopy(in_pixel, in_pixel + width * bytes_pp, out_pixel); break;
435  case 3:
436  {
437  for (intx k = 0; k < width; ++k, in_pixel += bytes_pp, out_pixel += bytes_pp)
438  {
439  out_pixel[Image::Channel::RED ] = in_pixel[2];
440  out_pixel[Image::Channel::GREEN] = in_pixel[1];
441  out_pixel[Image::Channel::BLUE ] = in_pixel[0];
442  }
443  break;
444  }
445  default: // 4
446  {
447  for (intx k = 0; k < width; ++k, in_pixel += bytes_pp, out_pixel += bytes_pp)
448  {
449  out_pixel[Image::Channel::RED ] = in_pixel[2];
450  out_pixel[Image::Channel::GREEN] = in_pixel[1];
451  out_pixel[Image::Channel::BLUE ] = in_pixel[0];
452  out_pixel[Image::Channel::ALPHA] = in_pixel[3];
453  }
454  break;
455  }
456  }
457  }
458  }
459 }

461 void
462 Codec3BM::writeImage(Image const & image, BinaryOutputStream & output, bool write_block_header) const
463 {
464  if (!image.isValid())
465  throw Error(std::string(getName()) + ": Cannot write an invalid image");
466 
467  Image::Type type(image.getType());
468  int64 width = image.getWidth();
469  int64 height = image.getHeight();
470  int64 depth = image.getDepth();
471 
472  if (type != Image::Type::LUMINANCE_8U
473  && type != Image::Type::RGB_8U
474  && type != Image::Type::RGBA_8U)
475  throw Error(std::string(getName()) + ": Can only write 8-bit luminance, RGB or RGBA images");
476 
477  int bpp = image.getBitsPerPixel();
478  int64 stream_scan_width = ImageInternal::scanWidth(width, bpp, 16); // 16-byte row alignment for SSE compatibility
479 
480  static uint64 const FILE_HEADER_SIZE = 32; // Remember to change these if the format changes!
481  static uint64 const INFO_HEADER_SIZE = 72;
482  uint64 data_size = (uint64)stream_scan_width * (uint64)height * (uint64)depth;
483  uint64 size_in_bytes = FILE_HEADER_SIZE + INFO_HEADER_SIZE + data_size;
484 
485  if (write_block_header)
486  BlockHeader(getMagic(), size_in_bytes).write(output);
487 
488  BinaryOutputStream::EndiannessScope scope(output, Endianness::LITTLE);
489 
490  // Write file header (32 bytes):
491  // 4 bytes: Magic string "3BM\0"
492  // 4 bytes: Version
493  // 8 bytes: Size of the whole file in bytes
494  // 8 bytes: Reserved
495  // 8 bytes: Starting offset (from the beginning of the file) of the bitmap image data
496 
497  static uint8 const MAGIC[4] = { (uint8)'3', (uint8)'B', (uint8)'M', (uint8)'\0' };
498  output.writeBytes(4, MAGIC);
499  output.writeUInt32(0x0001);
500  output.writeUInt64(size_in_bytes);
501  output.writeUInt64(0);
502  output.writeUInt64(FILE_HEADER_SIZE + INFO_HEADER_SIZE);
503 
504  // Write info header (72 bytes):
505  // 8 bytes: Size of this header (72 bytes)
506  // 8 bytes: Width of the bitmap in voxels
507  // 8 bytes: Height of the bitmap in voxels
508  // 8 bytes: Depth of the bitmap in voxels
509  // 4 bytes: Number of bits per pixel (1, 4, 8, 16, 24, 32), currently only 8, 24 and 32 are supported
510  // 4 bytes: Compression method (0 for no compression, currently this is the only one supported)
511  // 8 bytes: Size of raw image data
512  // 8 bytes: Width resolution in voxels/meter
513  // 8 bytes: Height resolution in voxels/meter
514  // 8 bytes: Depth resolution in voxels/meter
515 
516  output.writeUInt64(INFO_HEADER_SIZE);
517  output.writeUInt64((uint64)width);
518  output.writeUInt64((uint64)height);
519  output.writeUInt64((uint64)depth);
520  output.writeUInt32((uint32)bpp);
521  output.writeUInt32(0);
522  output.writeUInt64(data_size);
523  output.writeUInt64(3937); // 100dpi
524  output.writeUInt64(3937); // 100dpi
525  output.writeUInt64(3937); // 100dpi
526 
527  // Write image data as L, BGR or BGRA voxels, one 16-byte aligned scanline at a time
528  if (width <= 0 || height <= 0 || depth <= 0 || bpp <= 0)
529  return;
530 
531  int bytes_pp = bpp / 8;
532  Array<uint8> row_buf((size_t)stream_scan_width, 0);
533  for (intx i = 0; i < depth; ++i)
534  for (intx j = 0; j < height; ++j)
535  {
536  uint8 const * in_pixel = (uint8 const *)image.getScanLine(j, i);
537  uint8 * out_pixel = &row_buf[0];
538 
539  switch (bytes_pp)
540  {
541  case 1: Algorithms::fastCopy(in_pixel, in_pixel + width * bytes_pp, out_pixel); break;
542  case 3:
543  {
544  for (intx k = 0; k < width; ++k, in_pixel += bytes_pp, out_pixel += bytes_pp)
545  {
546  out_pixel[0] = in_pixel[Image::Channel::BLUE ];
547  out_pixel[1] = in_pixel[Image::Channel::GREEN];
548  out_pixel[2] = in_pixel[Image::Channel::RED ];
549  }
550  break;
551  }
552  default: // 4
553  {
554  for (intx k = 0; k < width; ++k, in_pixel += bytes_pp, out_pixel += bytes_pp)
555  {
556  out_pixel[0] = in_pixel[Image::Channel::BLUE ];
557  out_pixel[1] = in_pixel[Image::Channel::GREEN];
558  out_pixel[2] = in_pixel[Image::Channel::RED ];
559  out_pixel[3] = in_pixel[Image::Channel::ALPHA];
560  }
561  break;
562  }
563  }
564 
565  output.writeBytes((int64)stream_scan_width, &row_buf[0]);
566  }

787 double
788 Image::getNormalizedValue(void const * pixel, int channel) const
789 {
790  switch (type)
791  {
792  case Image::Type::LUMINANCE_8U : return channel == Channel::ALPHA ? *((uint8 const *)pixel) / 255.0 : 0.0;
793 
794  case Image::Type::LUMINANCE_16 : return channel == Channel::ALPHA ? *((int16 const *)pixel) / 32768.0 : 0.0;
795  case Image::Type::LUMINANCE_16U : return channel == Channel::ALPHA ? *((uint16 const *)pixel) / 65535.0 : 0.0;
796 
797  case Image::Type::LUMINANCE_32 : return channel == Channel::ALPHA ? *((int32 const *)pixel) / 2147483648.0 : 0.0;
798  case Image::Type::LUMINANCE_32U : return channel == Channel::ALPHA ? *((uint32 const *)pixel) / 4294967295.0 : 0.0;
799 
800  case Image::Type::LUMINANCE_32F : return channel == Channel::ALPHA ? *((float32 const *)pixel) : 0.0;
801  case Image::Type::LUMINANCE_64F : return channel == Channel::ALPHA ? *((float64 const *)pixel) : 0.0;
802 
803  case Image::Type::RGB_8U : return channel != Channel::ALPHA ? ((uint8 const *)pixel)[channel] / 255.0 : 0.0;
804  case Image::Type::RGB_16U : return channel != Channel::ALPHA ? ((uint16 const *)pixel)[channel] / 65535.0 : 0.0;
805  case Image::Type::RGB_32F : return channel != Channel::ALPHA ? ((float32 const *)pixel)[channel] : 0.0;
806 
807  case Image::Type::RGBA_8U : return ((uint8 const *)pixel)[channel] / 255.0;
808  case Image::Type::RGBA_16U : return ((uint16 const *)pixel)[channel] / 65535.0;
809  case Image::Type::RGBA_32F : return ((float32 const *)pixel)[channel];
810 
811  case Image::Type::COMPLEX_64F :
812  {
813  if (channel == Channel::ALPHA)
814  {
815  float64 re = ((float64 const *)pixel)[0];
816  float64 im = ((float64 const *)pixel)[1];
817  return std::sqrt(re * re + im * im);
818  }
819  else
820  return 0.0;
821  }
822 
823  default : return 0.0;
824  }

974 void
975 Image::read_AUTO(BinaryInputStream & input, bool read_block_header)
976 {
977  // Read the block header, if present. We will only use the size info and autodetect the codec from the image block itself.
978  int64 size = (read_block_header ? (int64)Codec::BlockHeader(input).data_size : input.size());
979  if (size <= 0)
980  throw Error("No image data found");
981 
982  // Read the image block into a memory buffer (optimization possible when the data has already been buffered within the input
983  // stream?)
984  Array<uint8> img_block((size_t)size);
985  input.readBytes((int64)size, &img_block[0]);
986 
987  ImageCodec const * detected_codec = ImageInternal::codecFromMagic((int64)size, &img_block[0]);
988  if (detected_codec)
989  {
990  BinaryInputStream mem_stream(&img_block[0], (int64)size, Endianness::LITTLE, /* copy_memory = */ false);
991  detected_codec->readImage(*this, mem_stream, false);
992  }
993  else
994  {
995  // Decode via FreeImage
996  if (!fip_img)
997  fip_img = new fipImage;
998 
999  fipMemoryIO mem((BYTE *)&img_block[0], (DWORD)size);
1000  if (!fip_img->loadFromMemory(mem))
1001  throw Error("Could not load image from memory stream");
1002

1004  if (type == Type::UNKNOWN)
1005  {
1006  clear();
1007  throw Error("Image was successfully decoded but it has a format for which this library does not provide an interface");
1008  }
1009 
1010  width = fip_img->getWidth();
1011  height = fip_img->getHeight();
1012  depth = 1;
1013  }

1054  {
1055  if (&dst != this) dst = *this;
1056  status = true;
1057  }

1063  }
1064  else

1073  {
1074  case 4:
1075  {

1083 
1084  case 8:

1087

1089  status = (dst.fip_img->convertToType(FIT_BITMAP) == TRUE);
