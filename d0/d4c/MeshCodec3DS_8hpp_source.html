222  void convert3DSSubtree(Lib3dsMesh * meshes, Lib3dsNode * nodes, MeshGroup * mesh_group, Matrix4 const & accum_transform,
223  Builder * flat_builder, ReadCallback * callback) const
224  {
225  for (Lib3dsMesh * m = meshes; m; m = m->next)
226  {
227  intx num_vertices = (intx)m->points;
228  if (num_vertices <= 0 && read_opts.skip_empty_meshes)
229  continue;
230 
231  intx num_faces = (intx)m->faces;
232  if (read_opts.verbose)
233  {
234  THEA_CONSOLE << getName() << ": Mesh " << m->name << " has " << num_vertices << " vertices and " << num_faces
235  << " faces";
236  }
237 
238  MeshPtr mesh;
239  BuilderPtr local_builder;
240  Builder * builder = flat_builder;
241  if (!builder)
242  {
243  std::string name = m->name;
244  mesh = MeshPtr(new Mesh(std::string(mesh_group->getName()) + '/' + name));
245 
246  local_builder = BuilderPtr(new Builder(mesh));
247  builder = local_builder.get();
248  builder->begin();
249  }
250 
251  Matrix4 transform;
252  if (read_opts.use_transforms)
253  transform = accum_transform
254  * (Matrix4() << m->matrix[0][0], m->matrix[1][0], m->matrix[2][0], m->matrix[3][0],
255  m->matrix[0][1], m->matrix[1][1], m->matrix[2][1], m->matrix[3][1],
256  m->matrix[0][2], m->matrix[1][2], m->matrix[2][2], m->matrix[3][2],
257  m->matrix[0][3], m->matrix[1][3], m->matrix[2][3], m->matrix[3][3]).finished();
258  else
259  transform = Matrix4::Identity();
260 
261  // Read list of vertices
262  Array<typename Builder::VertexHandle> vrefs;
263  typename Builder::VertexHandle vref;
264  for (size_t i = 0; i < (size_t)num_vertices; ++i)
265  {
266  Vector3 vertex(m->pointL[i].pos[0], m->pointL[i].pos[1], m->pointL[i].pos[2]);
267 
268  intx vindex = (read_opts.store_vertex_indices ? vertex_count : -1);
269  if (!read_opts.ignore_texcoords && (intx)i < (intx)m->texels)
270  {
271  Vector2 texcoord(m->texelL[i][0], m->texelL[i][1]);
272  vref = builder->addVertex(read_opts.use_transforms ? Math::hmul(transform, vertex)
273  : vertex, vindex, nullptr, nullptr, &texcoord);
274  }
275  else
276  vref = builder->addVertex(read_opts.use_transforms ? Math::hmul(transform, vertex) : vertex, vindex);
277 
278  if (callback)
279  callback->vertexRead(mesh.get(), vertex_count, vref);
280 
281  vrefs.push_back(vref);
282  vertex_count++;
283  }
284 
285  // Read list of faces
286  typename Builder::VertexHandle face[3];
287  intx indices[3];
288  for (intx i = 0; i < num_faces; ++i)
289  {
290  indices[0] = (intx)m->faceL[i].points[0];
291  indices[1] = (intx)m->faceL[i].points[1];
292  indices[2] = (intx)m->faceL[i].points[2];
293 
294  if (indices[0] < 0 || indices[0] >= num_vertices
295  || indices[1] < 0 || indices[1] >= num_vertices
296  || indices[2] < 0 || indices[2] >= num_vertices)
297  {
298  if (read_opts.strict)
299  throw Error(getName() + format(": Vertex index out of bounds in face %ld: (%ld, %ld, %ld)",
300  i, indices[0], indices[1], indices[2]));
301  else
302  {
303  THEA_WARNING << getName() << ": Skipping face " << i << ", vertex index out of bounds: ("
304  << indices[0] << ", " << indices[1] << ", " << indices[2] << ')';
305  face_count++; // increment in any case to keep face indexing correct
306  continue;
307  }
308  }
309 
310  if (indices[0] == indices[1] || indices[1] == indices[2] || indices[2] == indices[0])
311  {
312  if (read_opts.strict)
313  throw Error(getName() + format(": Repeated vertices in face %ld: (%ld, %ld, %ld)",
314  i, indices[0], indices[1], indices[2]));
315  else
316  {
317  THEA_WARNING << getName() << ": Skipping face " << i << " with repeated vertices: ("
318  << indices[0] << ", " << indices[1] << ", " << indices[2] << ')';
319  face_count++; // increment in any case to keep face indexing correct
320  continue;
321  }
322  }
323 
324  face[0] = vrefs[(size_t)indices[0]];
325  face[1] = vrefs[(size_t)indices[1]];
326  face[2] = vrefs[(size_t)indices[2]];
327 
328  typename Builder::FaceHandle fref = builder->addFace(face, face + 3,
329  (read_opts.store_face_indices ? face_count : -1));
330  if (callback)
331  callback->faceRead(mesh.get(), face_count, fref);
332 
333  face_count++;
334  }
335 
336  if (!flat_builder)
337  {
338  builder->end();
339  mesh_group->addMesh(mesh);
340  }
341  }
342 
343  // Recurse over child nodes, if any
344  int i = 0;
345  for (Lib3dsNode * n = nodes; n && i < 10; n = n->next, ++i)
346  {
347  if (n->type == LIB3DS_OBJECT_NODE)
348  {
349  Matrix4 transform;
350  if (read_opts.use_transforms)
351  transform = accum_transform
352  * (Matrix4() << n->matrix[0][0], n->matrix[1][0], n->matrix[2][0], n->matrix[3][0],
353  n->matrix[0][1], n->matrix[1][1], n->matrix[2][1], n->matrix[3][1],
354  n->matrix[0][2], n->matrix[1][2], n->matrix[2][2], n->matrix[3][2],
355  n->matrix[0][3], n->matrix[1][3], n->matrix[2][3], n->matrix[3][3]).finished();
356  else
357  transform = Matrix4::Identity();
358 
359  if (flat_builder)
360  convert3DSSubtree(n->user.mesh, n->childs, nullptr, transform, flat_builder, callback);
361  else
362  {
363  typename MeshGroup::Ptr child_group(new MeshGroup(mesh_group->getName() + format("/Child%d", i)));
364  convert3DSSubtree(n->user.mesh, n->childs, child_group.get(), transform, nullptr, callback);
365  mesh_group->addChild(child_group);
366  }
367  }
368  }
369  }

372  static intx seekInput(void * self, intx offset, Lib3dsIoSeek origin)
373  {
374  if (!self) return 0;
375  BinaryInputStream & in = *static_cast<BinaryInputStream *>(self);
376 
377  int64 abs_pos;
378  switch (origin)
379  {
380  case LIB3DS_SEEK_SET: abs_pos = (int64)offset; break;
381  case LIB3DS_SEEK_CUR: abs_pos = (int64)(in.getPosition() + offset); break;
382  case LIB3DS_SEEK_END: abs_pos = (int64)(in.size() + offset); break;
383  default: throw Error(std::string(_getName()) + ": Invalid enum value specified to seekInput");
384  }
385 
386  if (abs_pos >= 0 && abs_pos <= in.size())
387  {
388  in.setPosition(abs_pos);
389  return 0;
390  }
391  else
392  return -1;
393  }
