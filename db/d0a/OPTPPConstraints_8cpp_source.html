25 template <typename T, typename I2, typename I1>
26 void
27 toNEWMATMatrix(MatrixWrapper<T, I2, I1> const & src, NEWMAT::Matrix & dst)
28 {
29  typedef MatrixWrapper<T, I2, I1> MW;
30 
31  dst.ReSize(src.rows(), src.cols());
32 
33  switch (src.getFormat())
34  {
35  case MatrixFormat::DENSE_ROW_MAJOR:
36  {
37  typename MW::DenseRowMatrix const & drm = src.getDenseRowMatrix();
38  for (int r = 0; r < drm.rows(); ++r)
39  for (int c = 0; c < drm.cols(); ++c)
40  dst(r, c) = static_cast<NEWMAT::Real>(drm(r, c));
41 
42  break;
43  }
44 
45  case MatrixFormat::DENSE_COLUMN_MAJOR:
46  {
47  typename MW::DenseColumnMatrix const & dcm = src.getDenseColumnMatrix();
48  for (int c = 0; c < dcm.cols(); ++c)
49  for (int r = 0; r < dcm.rows(); ++r)
50  dst(r, c) = static_cast<NEWMAT::Real>(dcm(r, c));
51 
52  break;
53  }
54 
55  case MatrixFormat::SPARSE_ROW_MAJOR:
56  {
57  typename MW::SparseRowMatrix const & srm = src.getSparseRowMatrix();
58  Array<I1> const & irow = srm.getRowIndices();
59  Array<I2> const & icol = srm.getColumnIndices();
60  Array<T> const & val = srm.getValues();
61 
62  dst = 0; // set all entries to zero
63  for (size_t r = 0; r < (size_t)srm.rows(); ++r)
64  for (I1 i = irow[r]; i < irow[r + 1]; ++i)
65  {
66  int col = (int)icol[(size_t)i];
67  dst(r, col) = static_cast<NEWMAT::Real>(val[(size_t)i]);
68  }
69 
70  break;
71  }
72 
73  case MatrixFormat::SPARSE_COLUMN_MAJOR:
74  {
75  typename MW::SparseColumnMatrix const & scm = src.getSparseColumnMatrix();
76  Array<I1> const & icol = scm.getColumnIndices();
77  Array<I2> const & irow = scm.getRowIndices();
78  Array<T> const & val = scm.getValues();
79 
80  dst = 0; // set all entries to zero
81  for (size_t c = 0; c < (size_t)scm.cols(); ++c)
82  for (I1 i = icol[c]; i < icol[c + 1]; ++i)
83  {
84  int row = (int)irow[(size_t)i];
85  dst(row, c) = static_cast<NEWMAT::Real>(val[(size_t)i]);
86  }
87 
88  break;
89  }
90 
91  default:
92  throw Error("OPTPPNumericalOptimizer: Can't convert to NEWMAT matrix -- unknown source format");
93  }
94 }

105 OPTPP::LinearConstraint *
106 toOPTPPConstraint(LinearConstraint const & constraint)
107 {
108  alwaysAssertM(Math::isSquare(constraint.getCoefficients())
109  && constraint.getCoefficients().rows() == (int)constraint.getConstants().size(),
110  "OPTPPNumericalOptimizer: Dimension mismatch in constraint");
111 
112  NEWMAT::Matrix lhs;
113  toNEWMATMatrix(constraint.getCoefficients(), lhs);
114 
115  NEWMAT::ColumnVector rhs;
116  toNEWMATColumnVector(constraint.getConstants(), rhs);
117 
118  switch (constraint.getCompareOp())
119  {
120  case CompareOp::EQUAL:
121  return new OPTPP::LinearEquation(lhs, rhs);
122 
123  case CompareOp::GEQUAL:
124  return new OPTPP::LinearInequality(lhs, rhs); // standard form
125 
126  case CompareOp::LEQUAL:
127  {
128  // Negate LHS
129  for (int r = 0; r < lhs.Nrows(); ++r)
130  for (int c = 0; c < lhs.Ncols(); ++c)
131  lhs(r, c) = -lhs(r, c);
132 
133  // Negate RHS
134  for (int i = 0; i < rhs.Ncols(); ++i)
135  rhs(i) = -rhs(i);
136 
137  return new OPTPP::LinearInequality(lhs, rhs);
138  }
139 
140  default:
141  throw Error("OPTPPNumericalOptimizer: Unsupported comparison operator in constraint");
142  }
143 }
