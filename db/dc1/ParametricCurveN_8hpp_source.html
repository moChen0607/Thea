103  VectorT getNormal(T const & t) const
104  {
105  if (N < 2 || !hasDeriv(2)) return VectorT::Zero();
106 
107  VectorT d1 = eval(t, 1);
108  VectorT d2 = eval(t, 2);

112  d2 -= (d2.dot(d1) / d1_sqlen) * d1; // sqrt in normalizing d1 avoided because of repeated d1

127  VectorT getFrenetVector(T const & t, intx deriv_order) const
128  {
129  alwaysAssertM(deriv_order > 0, format("ParametricCurveN: Frenet vector of order %ld does not exist", (long)deriv_order));
130 
131  if (deriv_order > N || !hasDeriv(deriv_order)) return VectorT::Zero();
132 
133  if (deriv_order == 1) return getTangent(t);
134  if (deriv_order == 2) return getNormal(t);
135  if (deriv_order == 3) return static_cast<ParametricCurveT>(this)->getBinormal(t);
136 
137  // TODO: Is there a faster alternative to this recursion? Probably not very important since deriv_order is unlikely to be
138  // more than 2 or 3.
139  VectorT d = eval(t, deriv_order);
140  VectorT f = d;
141  for (intx i = 1; i < deriv_order; ++i)
142  {
143  VectorT g = getFrenetVector(t, i);
144  f -= d.dot(g) * g;
145  }
146 
147  return f.normalized();

161  virtual void getEvenlySpacedPoints(intx num_points, VectorT * pts_begin = nullptr, T * params_begin = nullptr,
162  intx num_arc_samples = -1) const
163  {
164  alwaysAssertM(num_points >= 2, "ParametricCurveNBase: At least two evenly-spaced points must be sampled");
165 
166  if (num_arc_samples < 0)
167  {
168  intx curve_complexity = std::max(2L, std::max(numControls() - 1, getOrder()));
169  num_arc_samples = curve_complexity * 50;
170  }
171 
172  alwaysAssertM(num_arc_samples >= 2, "ParametricCurveNBase: At least two samples must be used to approximate arc lengths");
173 
174  // Generate num_points samples with uniform parameter spacing
175  Array<VectorT> arc_samples((size_t)num_arc_samples);
176  T arc_scaling = (max_param - min_param) / static_cast<T>(num_arc_samples - 1);
177  T t;
178  for (intx i = 0; i < num_arc_samples; ++i)
179  {
180  t = i * arc_scaling + min_param;
181  arc_samples[(size_t)i] = eval(t, 0);
182  }
183 
184  // Compute the normalized arc length for each sampled point, by a piecewise linear approximation. This is a monotonically
185  // increasing sorted array with first element 0 and last element 1.
186  Array<double> arclen;
187  chordLengthParametrize(arc_samples.begin(), arc_samples.end(), arclen, 0.0, 1.0);
188 
189  // Generate a uniform distribution of normalized arc lengths, map each arc length to the corresponding interval in the
190  // generated sequence, and use linear interpolation within the interval to find the corresponding curve parameter
191  typedef Array<double>::iterator DoubleIterator;
192  DoubleIterator last = arclen.begin();
193  for (size_t i = 0; i < (size_t)num_points; ++i)
194  {
195  if (i == 0) // short-circuit for first and last points and save two binary searches
196  t = min_param;
197  else if (i == (size_t)num_points - 1)
198  t = max_param;
199  else
200  {
201  double s = i / (double)(num_points - 1); // target arc-length
202  DoubleIterator seg_stop = std::upper_bound(last, arclen.end(), s);
203  if (seg_stop == arclen.end()) // in degenerate cases
204  t = max_param;
205  else if (seg_stop != arclen.begin())
206  {
207  DoubleIterator seg_start = seg_stop - 1;
208  double f = (s - *seg_start) / (*seg_stop - *seg_start); // denom won't be zero because of upper_bound spec
209  size_t index = (seg_start - arclen.begin());
210  t = static_cast<T>(index + f) * arc_scaling + min_param;
211  }
212 
213  last = seg_stop;
214  }
215 
216  if (pts_begin) pts_begin[i] = eval(t, 0);
217  if (params_begin) params_begin[i] = t;
218  }
219  }

241  template <typename InputIterator>
242  static void chordLengthParametrize(InputIterator begin, InputIterator end, Array<double> & u,
243  double min_param_ = 0, double max_param_ = 1)
244  {
245  using namespace Algorithms;
246  typedef typename std::iterator_traits<InputIterator>::value_type PointT;
247 
248  u.clear();
249 
250  if (begin == end)
251  return;
252 
253  u.push_back(0.0);
254 
255  VectorT last = PointTraitsN<PointT, N, T>::getPosition(*begin);
256  InputIterator pi = begin;
257  double sum_u = 0.0;
258  for (++pi; pi != end; ++pi)
259  {
260  VectorT curr = PointTraitsN<PointT, N, T>::getPosition(*pi);
261  sum_u += static_cast<double>((curr - last).norm());
262  u.push_back(sum_u);
263  last = curr;
264  }
265 
266  double umax = u[u.size() - 1];
267  if (umax <= 0)
268  return;
269 
270  double scaling = (max_param_ - min_param_) / umax;
271  for (size_t i = 0; i < u.size(); ++i)
272  u[i] = u[i] * scaling + min_param_;
273  }
