76 double
77 TextInputStream::parseNumber(std::string const & _string)
78 {
79  std::string s = toLower(_string);
80 
81  if (s == "-1.#ind00" || s == "nan")
82  {
83  return Math::nan<double>();
84  }
85 
86  if (s == "1.#inf00" || s == "inf" || s == "+inf")
87  {
88  return Math::inf<double>();
89  }
90 
91  if (s == "-1.#inf00" || s == "-inf")
92  {
93  return -Math::inf<double>();
94  }
95 
96  double n;
97 
98  if ((_string.length() > 2) &&
99  (_string[0] == '0') &&
100  (_string[1] == 'x'))
101  {
102  // Hex
103  unsigned int i;
104  std::sscanf(_string.c_str(), "%x", &i);
105  n = (double)i;
106  }
107  else
108  {
109  std::sscanf(_string.c_str(), "%lg", &n);
110  }
111 
112  return n;
113 }

176 std::string
177 TextInputStream::readLine()
178 {
179  // Go to the front of the next token
180  Token t = read();
181 
182  // Reset the position to the start of this token
183  currentCharOffset = (size_t)t.bytePosition();
184  stack.clear();
185 
186  if (currentCharOffset == buffer.size())
187  {
188  // End of file
189  return "";
190  }
191 
192  std::string s;
193 
194  // Read until newline or eof
195  char c = '\0';
196  do
197  {
198  c = buffer[currentCharOffset];
199 
200  if (c == '\r' || c == '\n')
201  {
202  // Done
203  break;
204  }
205  else
206  {
207  s += c;
208  ++currentCharOffset;
209  }
210  }
211  while (currentCharOffset < buffer.size());
212 
213  return s;
214 }

255 int
256 TextInputStream::eatInputChar()
257 {
258  // Don't go off the end
259  if (currentCharOffset >= buffer.size())
260  {
261  return EOF;
262  }
263 
264  char c = buffer[currentCharOffset];
265  ++currentCharOffset;
266 
267  // update lineNumber and charNumber to reflect the location of the *next*
268  // character which will be read.
269 
270  // increment line number for \r, \n and \r\n which matches Token::Type::NEWLINE parsing
271  if (c == '\r')
272  {
273  ++lineNumber;
274  charNumber = 1;
275 
276  // check for \r\n
277  if (currentCharOffset < buffer.size())
278  {
279  char c2 = buffer[currentCharOffset];
280 
281  if (c2 == '\n')
282  {
283  c = c2;
284  ++currentCharOffset;
285  }
286  }
287  }
288  else if (c == '\n')
289  {
290  ++lineNumber;
291  charNumber = 1;
292  }
293  else
294  {
295  ++charNumber;
296  }
297 
298  return c;
299 }

314 Token
315 TextInputStream::nextToken()
316 {
317  Token t;
318  t._bytePosition = currentCharOffset;
319  t._line = lineNumber;
320  t._character = charNumber;
321  t._type = Token::Type::END;
322  t._extendedType = Token::ExtendedType::END;
323  int c = peekInputChar();
324 
325  if (c == EOF)
326  {
327  return t;
328  }
329 
330  // loop through white space, newlines and comments
331  // found before other tokens
332  bool whitespaceDone = false;
333 
334  while (! whitespaceDone)
335  {
336  whitespaceDone = true;
337 
338  // generate newlines tokens for '\n' and '\r' and '\r\n'
339  while (isWhitespace(c))
340  {
341  if (options.generateNewlineTokens && isNewline(c))
342  {
343  t._type = Token::Type::NEWLINE;
344  t._extendedType = Token::ExtendedType::NEWLINE;
345  t._bytePosition = currentCharOffset;
346  t._line = lineNumber;
347  t._character = charNumber;
348  t._string = c;
349  int c2 = peekInputChar(1);
350 
351  if (c == '\r' && c2 == '\n')
352  {
353  t._string += c2;
354  }
355 
356  eatInputChar();
357  return t;
358  }
359  else
360  {
361  // Consume the single whitespace
362  c = eatAndPeekInputChar();
363  }
364  }
365 
366  // update line and character number to include discarded whitespace
367  t._line = lineNumber;
368  t._character = charNumber;
369  t._bytePosition = (uint64)currentCharOffset;
370  int c2 = peekInputChar(1);
371 
372  // parse comments and generate tokens if enabled
373  std::string commentString;
374 
375  // check for line comments first
376  bool isLineComment = false;
377 
378  if (options.cppLineComments && (c == '/' && c2 == '/'))
379  {
380  // set start of line comment and eat markers
381  isLineComment = true;
382  eatInputChar();
383  eatInputChar();
384  }
385  else if ( options.otherCommentCharacter &&
386  (options.otherCommentCharacter != '\0' && c == options.otherCommentCharacter) )
387  {
388  // set start of line comment and eat markers
389  isLineComment = true;
390  eatInputChar();
391  }
392  else if ( options.otherCommentCharacter &&
393  (options.otherCommentCharacter2 != '\0' && c == options.otherCommentCharacter2) )
394  {
395  // set start of line comment and eat markers
396  isLineComment = true;
397  eatInputChar();
398  }
399 
400  if (isLineComment)
401  {
402  // consume line comment to newline or EOF
403  c = peekInputChar();
404 
405  while (! isNewline(c) && c != EOF)
406  {
407  // build comment string for token
408  commentString += c;
409  c = eatAndPeekInputChar();
410  }
411 
412  if (options.generateCommentTokens)
413  {
414  t._type = Token::Type::COMMENT;
415  t._extendedType = Token::ExtendedType::LINE_COMMENT;
416  t._string = commentString;
417  return t;
418  }
419  else
420  {
421  // There is whitespace after the comment (in particular, the
422  // newline that terminates the comment). There might also be
423  // whitespace at the start of the next line.
424  whitespaceDone = false;
425  }
426  }
427  else if (options.cppBlockComments && (c == '/' && c2 == '*'))
428  {
429  // consume block comment to end-marker or EOF
430  // consume both start-comment chars, can't let the trailing one
431  // help close the comment.
432  eatInputChar();
433  eatInputChar();
434  // c is the next character we'll read, c2 is the one after *that*
435  c = peekInputChar();
436  c2 = peekInputChar(1);
437 
438  while (! ((c == '*') && (c2 == '/')) && (c != EOF))
439  {
440  commentString += c;
441  // Eat input char may consume more than one character if there is a newline
442  eatInputChar();
443  c = peekInputChar();
444  c2 = peekInputChar(1);
445  }
446 
447  eatInputChar(); // eat closing '*'
448  eatInputChar(); // eat closing '/'
449  c = peekInputChar();
450 
451  if (options.generateCommentTokens)
452  {
453  t._type = Token::Type::COMMENT;
454  t._extendedType = Token::ExtendedType::BLOCK_COMMENT;
455  t._string = commentString;
456  return t;
457  }
458  else
459  {
460  // There is whitespace after the comment (in particular, the
461  // newline that terminates the comment). There might also be
462  // whitespace at the start of the next line.
463  whitespaceDone = false;
464  }
465  }
466  } // while (! whitespaceDone)
467 
468  t._line = lineNumber;
469  t._character = charNumber;
470  t._bytePosition = (uint64)currentCharOffset;
471 
472  // handle EOF
473  if (c == EOF)
474  {
475  return t;
476  }
477 
478  // Extended ASCII parses as itself, except for EOF
479  if (c > 127 && c < 255)
480  {
481  t._type = Token::Type::SYMBOL;
482  t._extendedType = Token::ExtendedType::SYMBOL;
483  t._string = c;
484  c = eatAndPeekInputChar();
485  }
486 
487  // Perform appropriate setup for a symbol (including setting up the token
488  // string to start with c), eat the input character, and overwrite
489  // 'c' with the peeked next input character.
490 #define THEA_SETUP_SYMBOL(c) \
491  { \
492  t._type = Token::Type::SYMBOL; \
493  t._extendedType = Token::ExtendedType::SYMBOL; \
494  t._string = c; \
495  c = eatAndPeekInputChar(); \
496  }
497 
498  switch (c)
499  {
500  case '@': // Simple symbols -> just themselves.
501  case '(':
502  case ')':
503  case ',':
504  case ';':
505  case '{':
506  case '}':
507  case '[':
508  case ']':
509  case '#':
510  case '$':
511  case '?':
512  case '%':
513  THEA_SETUP_SYMBOL(c);
514  return t;
515 
516  case '-': // negative number, -, --, -=, or ->
517  THEA_SETUP_SYMBOL(c);
518 
519  switch (c)
520  {
521  case '>': // ->
522  case '-': // --
523  case '=': // -=
524  t._string += c;
525  eatInputChar();
526  return t;
527  }
528 
529  if (options.signedNumbers)
530  {
531  if (isDigit(c) || (c == '.' && isDigit(peekInputChar(1))))
532  {
533  // Negative number. 'c' is still the first digit, and is
534  // the next input char.
535  goto numLabel;
536  }
537  else
538  {
539  char terminal = peekInputChar(3);
540 
541  if (options.simpleFloatSpecials && (c == 'i') && (peekInputChar(1) == 'n') && (peekInputChar(2) == 'f') &&
542  ! isAlpha(terminal) && (terminal != '_'))
543  {
544  // negative infinity
545  t._type = Token::Type::NUMBER;
546  t._extendedType = Token::ExtendedType::FLOATING_POINT;
547  t._string = "-inf";
548  eatInputChar(); // i
549  eatInputChar(); // n
550  eatInputChar(); // f
551  return t;
552  }
553  }
554  }
555 
556  // plain -
557  return t;
558 
559  case '+': // positive number, +, ++, or +=
560  THEA_SETUP_SYMBOL(c);
561 
562  switch (c)
563  {
564  case '+': // ++
565  case '=': // +=
566  t._string += c;
567  eatInputChar();
568  return t;
569  }
570 
571  if (options.signedNumbers)
572  {
573  if (isDigit(c) || (c == '.' && isDigit(peekInputChar(1))))
574  {
575  // Positive number. 'c' is still the first digit, and is
576  // the next input char.
577  goto numLabel;
578  }
579  else
580  {
581  char terminal = peekInputChar(3);
582 
583  if (options.simpleFloatSpecials && (c == 'i') && (peekInputChar(1) == 'n') && (peekInputChar(2) == 'f') &&
584  ! isAlpha(terminal) && (terminal != '_'))
585  {
586  // positive infinity
587  t._type = Token::Type::NUMBER;
588  t._extendedType = Token::ExtendedType::FLOATING_POINT;
589  t._string = "+inf";
590  eatInputChar(); // i
591  eatInputChar(); // n
592  eatInputChar(); // f
593  return t;
594  }
595  }
596  }
597 
598  return t;
599 
600  case ':': // : or :: or ::> or ::= or := or :>
601  THEA_SETUP_SYMBOL(c);
602 
603  if (c == ':')
604  {
605  t._string += c;
606  eatInputChar();
607 
608  if (options.proofSymbols)
609  {
610  c = peekInputChar(0);
611 
612  if ((c == '>') || (c == '='))
613  {
614  t._string += c;
615  eatInputChar();
616  }
617  }
618  }
619  else if (options.proofSymbols && (c == '=' || c == '>'))
620  {
621  t._string += c;
622  eatInputChar();
623  }
624 
625  return t;
626 
627  case '=': // = or == or =>
628  THEA_SETUP_SYMBOL(c);
629 
630  if (c == '=')
631  {
632  t._string += c;
633  eatInputChar();
634  return t;
635  }
636  else if (options.proofSymbols && (c == '>'))
637  {
638  t._string += c;
639  eatInputChar();
640  return t;
641  }
642 
643  return t;
644 
645  case '*': // * or *=
646  case '/': // / or /=
647  case '!': // ! or !=
648  case '~': // ~ or ~=
649  case '^': // ^ or ^=
650  THEA_SETUP_SYMBOL(c);
651 
652  if (c == '=')
653  {
654  t._string += c;
655  eatInputChar();
656  return t;
657  }
658 
659  return t;
660 
661  case '>': // >, >>,or >=
662  case '<': // <<, <<, or <= or <- or <:
663  case '|': // ||, ||, or |= or |-
664  case '&': // &, &&, or &=
665  {
666  int orig_c = c;
667  THEA_SETUP_SYMBOL(c);
668 
669  if ((c == '=') || (orig_c == c))
670  {
671  t._string += c;
672  eatInputChar();
673  return t;
674  }
675  else if (options.proofSymbols)
676  {
677  if ((orig_c == '<') && (c == '-'))
678  {
679  t._string += c;
680  eatInputChar();
681  }
682  else if ((orig_c == '|') && (c == '-'))
683  {
684  t._string += c;
685  eatInputChar();
686  }
687  else if ((orig_c == '<') && (c == ':'))
688  {
689  t._string += c;
690  c = eatAndPeekInputChar();
691 
692  if (c == ':')
693  {
694  t._string += c;
695  eatInputChar();
696  }
697  }
698  }
699  }
700 
701  return t;
702 
703  case '\\': // backslash or escaped comment char.
704  THEA_SETUP_SYMBOL(c);
705 
706  if ((options.otherCommentCharacter != '\0'
707  && c == options.otherCommentCharacter)
708  || (options.otherCommentCharacter2 != '\0'
709  && c == options.otherCommentCharacter2))
710  {
711  // Escaped comment character. Return the raw comment char (no backslash).
712  t._string = c;
713  eatInputChar();
714  return t;
715  }
716 
717  return t;
718 
719  case '.': // number, ., .., or ...
720  if (isDigit(peekInputChar(1)))
721  {
722  // We're parsing a float that began without a leading zero
723  goto numLabel;
724  }
725 
726  THEA_SETUP_SYMBOL(c);
727 
728  if (c == '.') // .. or ...
729  {
730  t._string += c;
731  c = eatAndPeekInputChar();
732 
733  if (c == '.') // ...
734  {
735  t._string += c;
736  eatInputChar();
737  }
738 
739  return t;
740  }
741 
742  return t;
743  } // switch (c)
744 
745 #undef THEA_SETUP_SYMBOL
746 numLabel:
747 
748  if (isDigit(c) || (c == '.'))
749  {
750  // A number. Note-- single dots have been
751  // parsed already, so a . indicates a number
752  // less than 1 in floating point form.
753 
754  // [0-9]*(\.[0-9][f]) or [0-9]+ or 0x[0-9,A-F]+
755  if (t._string != "-")
756  {
757  // If we picked up a leading "-" sign above, keep it,
758  // otherwise drop the string parsed thus far
759  t._string = "";
760  }
761 
762  t._type = Token::Type::NUMBER;
763 
764  if (c == '.')
765  {
766  t._extendedType = Token::ExtendedType::FLOATING_POINT;
767  }
768  else
769  {
770  t._extendedType = Token::ExtendedType::INTEGER;
771  }
772 
773  if ((c == '0') && (peekInputChar(1) == 'x'))
774  {
775  // Hex number
776  t._string += "0x";
777  // skip the 0x
778  eatInputChar();
779  eatInputChar();
780  c = peekInputChar();
781 
782  while (isDigit(c) || ((c >= 'A') && (c <= 'F')) || ((c >= 'a') && (c <= 'f')))
783  {
784  t._string += c;
785  c = eatAndPeekInputChar();
786  }
787  }
788  else
789  {
790  // Non-hex number
791 
792  // Read the part before the decimal.
793  while (isDigit(c))
794  {
795  t._string += c;
796  c = eatAndPeekInputChar();
797  }
798 
799  // True if we are reading a floating-point special type
800  bool isSpecial = false;
801 
802  // Read the decimal, if one exists
803  if (c == '.')
804  {
805  t._extendedType = Token::ExtendedType::FLOATING_POINT;
806  // The '.' character was a decimal point, not the start of a
807  // method or range operator
808  t._string += c;
809  c = eatAndPeekInputChar();
810 
811  // Floating point specials (msvc format only)
812  if (options.msvcFloatSpecials && (c == '#'))
813  {
814  isSpecial = true;
815  // We are reading a floating point special value
816  // of the form -1.#IND00, -1.#INF00, or 1.#INF00
817  c = eatAndPeekInputChar();
818  char test = c;
819 
820  if (! options.caseSensitive)
821  {
822  test = toupper(c);
823  }
824 
825  if (test != 'I')
826  {
827  throw BadMSVCSpecial
828  (
829  "Incorrect floating-point special (inf or nan) format",
830  t.line(), charNumber);
831  }
832 
833  c = eatAndPeekInputChar();
834  test = c;
835 
836  if (! options.caseSensitive)
837  {
838  test = toupper(c);
839  }
840 
841  if (test != 'N')
842  {
843  throw BadMSVCSpecial
844  (
845  "Incorrect floating-point special (inf or nan) format",
846  t.line(), charNumber);
847  }
848 
849  t._string += "#IN";
850  c = eatAndPeekInputChar();
851  test = c;
852 
853  if (! options.caseSensitive)
854  {
855  test = toupper(c);
856  }
857 
858  if ((test != 'F') && (test != 'D'))
859  {
860  throw BadMSVCSpecial
861  (
862  "Incorrect floating-point special (inf or nan) format",
863  t.line(), charNumber);
864  }
865 
866  t._string += c;
867 
868  for (int j = 0; j < 2; ++j)
869  {
870  c = eatAndPeekInputChar();
871 
872  if (c != '0')
873  {
874  throw BadMSVCSpecial
875  (
876  "Incorrect floating-point special (inf or nan) format",
877  t.line(), charNumber);
878  }
879 
880  t._string += (char)c;
881  }
882  }
883  else
884  {
885  // Read the part after the decimal
886  while (isDigit((char)c))
887  {
888  t._string += (char)c;
889  c = eatAndPeekInputChar();
890  }
891  }
892  }
893 
894  if (! isSpecial && ((c == 'e') || (c == 'E')))
895  {
896  // Read exponent
897  t._extendedType = Token::ExtendedType::FLOATING_POINT;
898  t._string += c;
899  c = eatAndPeekInputChar();
900 
901  if ((c == '-') || (c == '+'))
902  {
903  t._string += c;
904  c = eatAndPeekInputChar();
905  }
906 
907  while (isDigit(c))
908  {
909  t._string += c;
910  c = eatAndPeekInputChar();
911  }
912  }
913 
914  if (! isSpecial && (t._extendedType == Token::ExtendedType::FLOATING_POINT) && (c == 'f'))
915  {
916  // Trailing f on a float
917  t._string += c;
918  c = eatAndPeekInputChar();
919  }
920  }
921 
922  return t;
923  }
924  else if (isAlpha(c) || (c == '_'))
925  {
926  // Identifier or keyword
927  // [A-Za-z_][A-Za-z_0-9]*
928  t._type = Token::Type::SYMBOL;
929  t._extendedType = Token::ExtendedType::SYMBOL;
930  t._string = "";
931 
932  do
933  {
934  t._string += c;
935  c = eatAndPeekInputChar();
936  }
937  while (isAlpha(c) || isDigit(c) || (c == '_'));
938 
939  // See if this symbol is actually a boolean
940  if ((options.trueSymbols.size() > 0) || (options.falseSymbols.size() > 0))
941  {
942  std::string str = t._string;
943 
944  if (! options.caseSensitive)
945  {
946  str = toUpper(str);
947  }
948 
949  if (options.trueSymbols.find(str) != options.trueSymbols.end())
950  {
951  t._type = Token::Type::BOOLEAN;
952  t._extendedType = Token::ExtendedType::BOOLEAN;
953  t._bool = true;
954  }
955  else if (options.falseSymbols.find(str) != options.falseSymbols.end())
956  {
957  t._type = Token::Type::BOOLEAN;
958  t._extendedType = Token::ExtendedType::BOOLEAN;
959  t._bool = false;
960  }
961  }
962 
963  if (options.simpleFloatSpecials && ((t._string == "nan") || (t._string == "inf")))
964  {
965  t._type = Token::Type::NUMBER;
966  t._extendedType = Token::ExtendedType::FLOATING_POINT;
967  }
968 
969  return t;
970  }
971  else if (c == '\"')
972  {
973  // Discard the double-quote.
974  eatInputChar();
975  // Double quoted string
976  parseQuotedString('\"', t);
977  return t;
978  }
979  else if (c == options.singleQuoteCharacter)
980  {
981  // Discard the single-quote.
982  eatInputChar();
983 
984  if (options.singleQuotedStrings)
985  {
986  // Single quoted string
987  parseQuotedString(options.singleQuoteCharacter, t);
988  }
989  else
990  {
991  t._string = c;
992  t._type = Token::Type::SYMBOL;
993  t._extendedType = Token::ExtendedType::SYMBOL;
994  }
995 
996  return t;
997  } // end of special case tokens
998 
999  if (c == EOF)
1000  {
1001  t._type = Token::Type::END;
1002  t._extendedType = Token::ExtendedType::END;
1003  t._string = "";
1004  return t;
1005  }
1006

1008  debugAssertM(false,
1009  format("Unrecognized token type beginning with character '%c' (ASCII %d)",
1010  c, (int)c));
1011  return t;

1014 void
1015 TextInputStream::parseQuotedString(char delimiter, Token & t)
1016 {
1017  t._type = Token::Type::STRING;
1018 
1019  if (delimiter == options.singleQuoteCharacter)
1020  {
1021  t._extendedType = Token::ExtendedType::SINGLE_QUOTED;
1022  }
1023  else
1024  {
1025  t._extendedType = Token::ExtendedType::DOUBLE_QUOTED;
1026  }
1027 
1028  while (true)
1029  {
1030  // We're definitely going to consume the next input char, so we get
1031  // it right now. This makes the condition handling below a bit easier.
1032  int c = eatInputChar();
1033 
1034  if (c == EOF)
1035  {
1036  // Type::END inside a quoted string. (We finish the string.)
1037  break;
1038  }
1039 
1040  if (options.escapeSequencesInStrings && (c == '\\'))
1041  {
1042  // An escaped character. We're definitely going to consume it,
1043  // so we get it (and consume it) now.
1044  c = eatInputChar();
1045 
1046  switch (c)
1047  {
1048  case 'r':
1049  t._string += '\r';
1050  break;
1051 
1052  case 'n':
1053  t._string += '\n';
1054  break;
1055 
1056  case 't':
1057  t._string += '\t';
1058  break;
1059 
1060  case '0':
1061  t._string += '\0';
1062  break;
1063 
1064  case '\\':
1065  case '\"':
1066  t._string += (char)c;
1067  break;
1068 
1069  default:
1070  if (c == options.singleQuoteCharacter)
1071  {
1072  t._string += (char)c;
1073  break;
1074  }
1075 
1076  if (((c == options.otherCommentCharacter) &&
1077  (options.otherCommentCharacter != '\0')) ||
1078  ((c == options.otherCommentCharacter2) &&
1079  (options.otherCommentCharacter2 != '\0')))
1080  {
1081  t._string += c;
1082  }
1083 
1084  // otherwise, some illegal escape sequence; skip it.
1085  break;
1086  } // switch
1087  }
1088  else if (c == delimiter)
1089  {
1090  // End of the string. Already consumed the character.
1091  break;
1092  }
1093  else
1094  {
1095  // All other chars, go on to the string. Already consumed the
1096  // character.
1097  t._string += (char)c;
1098  }
1099  }
1100 }
