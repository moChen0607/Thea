77  void * getVertex(Vector3 const & position) const
78  {
79  typedef std::pair<VertexMap::const_iterator, VertexMap::const_iterator> IteratorRange;
80  Long3 base_key = toGrid(position);
81 
82  // Every vertex within the welding distance of the given point must be inside a neighbor of the grid cell containing the
83  // point
84  for (int dx = -1; dx <= 1; ++dx)
85  for (int dy = -1; dy <= 1; ++dy)
86  for (int dz = -1; dz <= 1; ++dz)
87  {
88  Long3 key(base_key.x + dx, base_key.y + dy, base_key.z + dz);
89  IteratorRange nbrs = vertex_map.equal_range(key);
90  for (VertexMap::const_iterator ni = nbrs.first; ni != nbrs.second; ++ni)
91  if ((ni->second.position - position).squaredNorm() <= squared_weld_radius)
92  return ni->second.vertex;
93  }
94 
95  return nullptr;
96  }
