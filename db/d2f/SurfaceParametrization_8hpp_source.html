90  template <typename FaceIteratorT, typename FixedVertexParameterMapT, typename FreeVertexParameterMapT>
91  static bool parametrize(FaceIteratorT faces_begin, FaceIteratorT faces_end, WeightType weight_type,
92  FixedVertexParameterMapT const & fixed_vertex_params, FreeVertexParameterMapT & free_vertex_params)
93  {
94  // Quick check to catch the case when the user has forgotten to provide any boundary parameters. Partial boundary
95  // initialization is detected by more expensive means below.
96  if (fixed_vertex_params.empty() && faces_begin != faces_end)
97  {
98  THEA_ERROR << "SurfaceParametrization: Free-boundary parametrization not currently implemented";
99  return false;
100  }
101 
102  typedef typename std::remove_cv<
103  typename std::remove_pointer<
104  typename std::remove_cv< typename FreeVertexParameterMapT::key_type >::type >::type >::type VertexT;
105  typedef typename VertexT::Edge EdgeT;
106  typedef typename VertexT::Face FaceT;
107  typedef typename FreeVertexParameterMapT::mapped_type FreeParamT; // some 2D vector
108 
109  // Cache the set of faces so we can quickly check if a face is in the patch or not (and also avoid
110  // dereferencing/incrementing the input iterators multiple times, in case this is expensive)
111  typedef UnorderedSet<FaceT const *> FaceSet;
112  FaceSet faces(makePtrIterator(faces_begin), makePtrIterator(faces_end));
113 
114  typedef UnorderedMap<VertexT const *, intx> VertexIndexMap;
115  typedef UnorderedSet<EdgeT const *> EdgeSet;
116  VertexIndexMap interior_verts;
117  EdgeSet interior_edges;
118 
119  // First build the set of all interior edges
120  for (auto fi = faces.begin(); fi != faces.end(); ++fi)
121  {
122  for (auto fei = (*fi)->edgesBegin(); fei != (*fi)->edgesEnd(); ++fei)
123  {
124  auto e = *fei;
125  if (e->numFaces() > 2)
126  {
127  THEA_ERROR << "SurfaceParametrization: Mesh is non-manifold (edge is shared by > 2 faces)";
128  return false;
129  }
130 
131  // If this is a boundary edge, make sure all its vertices have fixed parameters
132  bool is_boundary = (e->numFaces() < 2);
133  if (!is_boundary)
134  {
135  for (auto efi = e->facesBegin(); efi != e->facesEnd(); ++efi)
136  if (faces.find(*efi) == faces.end()) // this is a boundary edge
137  {
138  is_boundary = true;
139  break;
140  }
141  }
142 
143  if (is_boundary)
144  {
145  if (fixed_vertex_params.find(e->getEndpoint(0)) == fixed_vertex_params.end()
146  || fixed_vertex_params.find(e->getEndpoint(1)) == fixed_vertex_params.end())
147  {
148  THEA_ERROR << "SurfaceParametrization: Boundary vertex lacks fixed parameters";
149  return false;
150  }
151  }
152  else
153  interior_edges.insert(e);
154  }
155  }
156 
157  // Now build and index the set of all interior vertices
158  for (auto e : interior_edges)
159  for (intx i = 0; i < 2; ++i)
160  {
161  auto v = e->getEndpoint(i);
162  if (fixed_vertex_params.find(v) == fixed_vertex_params.end() // already checked every boundary vertex is parametrized
163  && interior_verts.find(v) == interior_verts.end())
164  {
165  intx index = (intx)interior_verts.size();
166  interior_verts[v] = index;
167  }
168  }
169 
170  // For every edge (i, j) leaving every interior vertex i, there is a weight w_ij
171  intx num_vars = (intx)interior_verts.size();
172  Array<Triplet> triplets;
173  VectorXd u_consts(num_vars); u_consts.setZero();
174  VectorXd v_consts(num_vars); v_consts.setZero();
175  bool weight_error = false;
176  for (auto v : interior_verts)
177  {
178  // The diagonal is all 1's
179  triplets.push_back(Triplet(v.second, v.second, 1));
180 
181  // Add a weight for each neighbor
182  double sum_weights = 0;
183  size_t tbase = triplets.size();
184  for (auto vei = v.first->edgesBegin(); vei != v.first->edgesEnd(); ++vei)
185  {
186  auto e = *vei;
187  auto v1 = e->getOtherEndpoint(v.first);
188  double w_ij = getWeight(weight_type, v.first, v1, e, weight_error);
189  sum_weights += w_ij;
190 
191  auto other = interior_verts.find(v1);
192  if (other != interior_verts.end())
193  triplets.push_back(Triplet(v.second, other->second, -w_ij));
194  else
195  {
196  auto p = fixed_vertex_params.find(v1);
197  debugAssertM(p != fixed_vertex_params.end(), "SurfaceParametrization: Boundary vertex lacks fixed parameters");
198  u_consts[v.second] += w_ij * p->second[0];
199  v_consts[v.second] += w_ij * p->second[1];
200  }
201  }
202 
203  // Normalization
204  for (size_t i = tbase; i < triplets.size(); ++i)
205  triplets[i].val /= sum_weights;
206 
207  u_consts[v.second] /= sum_weights;
208  v_consts[v.second] /= sum_weights;
209  }
210 
211  // Build the sparse coefficient matrix
212  SparseMatrix<double> coeffs(num_vars, num_vars);
213  coeffs.setFromTriplets(triplets.begin(), triplets.end());
214 
215  // Solve the system for U and V
216  StdLinearSolver u_solver(StdLinearSolver::Method::SPARSE_LU);
217  StdLinearSolver v_solver(StdLinearSolver::Method::SPARSE_LU);
218  if (!u_solver.solve(coeffs, u_consts.data()) || !v_solver.solve(coeffs, v_consts.data()))
219  {
220  THEA_ERROR << "SurfaceParametrization: Couldn't solve sparse linear system";
221  return false;
222  }
223 
224  float64 const * u_sol = u_solver.getSolution();
225  float64 const * v_sol = v_solver.getSolution();
226  for (auto v : interior_verts)
227  {
228  // This is the only place the free vertices are updated, and fixed_vertex_params is not accessed here or later. Hence,
229  // the two references can point to the same underlying container.
230  free_vertex_params[v.first][0] = static_cast<typename FreeParamT::value_type>(u_sol[v.second]);
231  free_vertex_params[v.first][1] = static_cast<typename FreeParamT::value_type>(v_sol[v.second]);
232  }
233 
234  return true;
235  }

242  template <typename VertexT, typename EdgeT>
243  static double getWeight(WeightType weight_type, VertexT const * vi, VertexT const * vj, EdgeT const * e, bool & error)
244  {
245  error = false;
246  switch (weight_type)
247  {
248  case WeightType::UNIFORM: return 1;
249  case WeightType::COTANGENT: return getCotangentWeight(vi, vj, e, error);
250  case WeightType::MEAN_VALUE: return getMeanValueWeight(vi, vj, e, error);
251  default:
252  THEA_ERROR << "SurfaceParametrization: Unsupported weight type";
253  error = true; return -1;
254  }

261  template <typename VertexT, typename EdgeT>
262  static double getCotangentWeight(VertexT const * vi, VertexT const * vj, EdgeT const * e, bool & error)
263  {
264  typename EdgeT::Face const * f[2];
265  if (getFaces(e, f) != 2) { error = true; return -1; }
266 
267  double w = 0;
268  for (intx n = 0; n < 2; ++n)
269  {
270  auto vk = getThirdVertex(f[n], vi, vj); // assuming triangles makes search faster
271  if (!vk) { error = true; return -1; }
272 
273  w += cot(vk, vi, vj);
274  }
275 
276  return w;
277  }

283  template <typename VertexT, typename EdgeT>
284  static double getMeanValueWeight(VertexT const * vi, VertexT const * vj, EdgeT const * e, bool & error)
285  {
286  typename EdgeT::Face const * f[2];
287  if (getFaces(e, f) != 2) { error = true; return -1; }
288 
289  double len_ij = (vj->getPosition() - vi->getPosition()).norm();
290  if (Math::fuzzyEq(len_ij, 0.0))
291  return 0;
292 
293  double w = 0;
294  for (intx n = 0; n < 2; ++n)
295  {
296  auto vk = getThirdVertex(f[n], vi, vj); // assuming triangles makes search faster
297  if (!vk) { error = true; return -1; }
298 
299  w += tanHalf(vi, vj, vk, len_ij);
300  }
301 
302  return w / len_ij;
303  }

309  template <typename EdgeT, typename FaceT>
310  static intx getFaces(EdgeT const * e, FaceT const ** f)
311  {
312  // Currently, this will only be called for interior edges, which must have exactly two incident faces
313  if (e->numFaces() != 2)
314  {
315  THEA_ERROR << "SurfaceParametrization: Interior edge doesn't have exactly two incident faces";
316  return -1;
317  }
318 
319  auto efi = e->facesBegin();
320  f[0] = *efi;
321  f[1] = *(++efi);
322 
323  return 2;
324  };

327  template <typename FaceT, typename VertexT>
328  static VertexT const * getThirdVertex(FaceT const * f, VertexT const * vi, VertexT const * vj)
329  {
330  if (f->numVertices() != 3)
331  {
332  THEA_ERROR << "SurfaceParametrization: Selected weight type requires triangular faces";
333  return nullptr;
334  }
335 
336  for (auto fvi = f->verticesBegin(); fvi != f->verticesEnd(); ++fvi)
337  if (*fvi != vi && *fvi != vj)
338  return *fvi;
339 
340  return nullptr;

344  template <typename VertexT>
345  static double cot(VertexT const * v0, VertexT const * v1, VertexT const * v2)
346  {
347  Vector3 const & e01 = v1->getPosition() - v0->getPosition();
348  Vector3 const & e02 = v2->getPosition() - v0->getPosition();
349 
350  double c = e01.dot(e02); // NOTE: not normalized
351  double s = e01.cross(e02).norm(); // NOTE: not normalized
352 
353  return c / s;
354  }

361  template <typename VertexT>
362  static double tanHalf(VertexT const * v0, VertexT const * v1, VertexT const * v2, double e01_len)
363  {
364  Vector3 const & e01 = v1->getPosition() - v0->getPosition();
365  Vector3 const & e02 = (v2->getPosition() - v0->getPosition()).normalized();
366 
367  // Make use of the formula tan(x/2) = sqrt((1 - cos(x)) / (1 + cos(x))), for x < pi
368  double c = e01.dot(e02); // is larger than cos(x) by a factor of e01_len
369  return std::sqrt((e01_len - c) / (e01_len + c));
370  }
