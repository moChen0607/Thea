144  {
145  intx nn_index = precomp_kdtree ? precomp_kdtree->template closestElement<MetricL2>(position)
146  : kdtree->template closestElement<MetricL2>(position);
147  if (nn_index < 0)
148  {

170  double compute(Vector3 const & position, Vector3 const & normal, bool only_hit_interior_surfaces = true) const
171  {
172  Vector3 in = -normal.normalized();
173  Matrix3 rot = Math::orthonormalBasis(in);
174  Vector3 offset = 0.001f * scale * in;
175 
176  static int const NUM_RAYS = 30;
177  static Vector3 const CONE_DIRS[NUM_RAYS] = {
178  Vector3( 0.407497f, 0.134086f, 0.903309f),
179  Vector3( 0.128722f, -0.822715f, 0.553688f),
180  Vector3( 0.359298f, 0.686129f, 0.632560f),
181  Vector3( 0.813318f, -0.153956f, 0.561081f),
182  Vector3(-0.707169f, 0.296987f, 0.641647f),
183  Vector3(-0.478783f, -0.543441f, 0.689520f),
184  Vector3( 0.561083f, 0.523634f, 0.641087f),
185  Vector3(-0.380898f, -0.655789f, 0.651811f),
186  Vector3( 0.722588f, 0.472155f, 0.504912f),
187  Vector3(-0.535205f, 0.421235f, 0.732200f),
188  Vector3( 0.577593f, -0.477896f, 0.661817f),
189  Vector3( 0.393652f, -0.144484f, 0.907834f),
190  Vector3( 0.141508f, -0.660777f, 0.737122f),
191  Vector3(-0.274967f, -0.141703f, 0.950954f),
192  Vector3(-0.304935f, 0.474085f, 0.825989f),
193  Vector3( 0.711150f, 0.301383f, 0.635164f),
194  Vector3(-0.696119f, -0.112341f, 0.709082f),
195  Vector3(-0.248327f, 0.381035f, 0.890588f),
196  Vector3(-0.0772394f, 0.0462501f, 0.995939f),
197  Vector3( 0.438024f, 0.656463f, 0.614159f),
198  Vector3(-0.483384f, -0.542477f, 0.687065f),
199  Vector3( 0.498569f, -0.697466f, 0.514752f),
200  Vector3(-0.128902f, -0.0173429f, 0.991506f),
201  Vector3( 0.338759f, -0.318772f, 0.885227f),
202  Vector3(-0.574364f, -0.463887f, 0.674474f),
203  Vector3(-0.459207f, 0.260125f, 0.849390f),
204  Vector3(-0.628057f, -0.528928f, 0.570771f),
205  Vector3( 0.232916f, -0.328273f, 0.915416f),
206  Vector3( 0.231667f, -0.796882f, 0.557951f),
207  Vector3(-0.270612f, -0.809654f, 0.520797f),
208  };
209 
210  double values[NUM_RAYS];
211  double weights[NUM_RAYS];
212  int num_values = 0;
213  for (int i = 0; i < NUM_RAYS; ++i)
214  {
215  Vector3 dir = rot * CONE_DIRS[i];
216  Ray3 ray(position + offset, dir);
217  RayStructureIntersection3 isec = precomp_kdtree
218  ? precomp_kdtree->template rayStructureIntersection<RayIntersectionTester>(ray)
219  : kdtree->template rayStructureIntersection<RayIntersectionTester>(ray);
220 
221  if (isec.isValid() && (!only_hit_interior_surfaces || isec.getNormal().dot(dir) >= 0))
222  {
223  values[num_values] = isec.getTime();
224  weights[num_values] = CONE_DIRS[i][2]; // cos(angle) is just the z-component
225  num_values++;
226  }
227  }
228 
229  if (num_values <= 0)
230  return -1.0; // either the normal is in the wrong direction or this is a 2D surface
231 
232  // Outlier rejection: reject all values more than one standard deviation from the median
233  int mid = num_values / 2; // integer division takes floor
234  std::nth_element(values, values + mid, values + num_values);
235 
236  // Compute variance
237  double sum_values = 0, sum_squares = 0;
238  for (int i = 0; i < num_values; ++i)
239  {
240  sum_values += values[i];
241  sum_squares += (values[i] * values[i]);
242  }
243 
244  double avg = sum_values / num_values;
245  double var = sum_squares / num_values - avg * avg;
246 
247  sum_values = 0;
248  double sum_weights = 0;
249  for (int i = 0; i < num_values; ++i)
250  if (Math::square(values[i] - values[mid]) <= var)
251  {
252  sum_values += (weights[i] * values[i]);
253  sum_weights += weights[i];
254  }
255 
256  if (sum_weights > 0)
257  return Math::clamp((double)(sum_values / (sum_weights * scale)), 0.0, 1.0);
258  else // should never happen
259  return Math::clamp((double)(values[mid] / scale), 0.0, 1.0);
260  }
